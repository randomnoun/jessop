6   import java.io.ByteArrayOutputStream;
7   import java.io.PrintWriter;
8   import java.io.StringReader;
9   
10  import com.randomnoun.common.jexl.ast.TopLevelExpression;
11  import com.randomnoun.common.jexl.parser.ExpressionParser;
12  import com.randomnoun.common.jexl.parser.ParseException;
13  import com.randomnoun.common.jexl.parser.TokenMgrError;
14  import com.randomnoun.common.jexl.visitor.TreeDumper;
15  
16  /**
17   * Couple of methods to convert things to and from TopLevelExpressions.
18   *
19   * 
20   * @author knoxg
21   */
22  public class ExpressionUtil
23  {
24  
25      /** Convert Java expression String to a TopLevelExpression */
26      public static TopLevelExpression stringToExpression(String expressionString)
27          throws java.text.ParseException
28      {
29          StringReader reader = new StringReader(expressionString);
30          ExpressionParser parser = new ExpressionParser(reader);
31          TopLevelExpression root = null;
32  
33          try {
34              root = parser.TopLevelExpression();
35          } catch (ParseException pe) {
36              throw new java.text.ParseException(pe.getMessage(), -1);
37          } catch (TokenMgrError tme) {
38              throw new java.text.ParseException(tme.getMessage(), -1);
39          }
40  
41          return root;
42      }
43  
44      /** Convert TopLevelExpression to a Java expression String */
45      public static String expressionToString(TopLevelExpression expression) {
46          ByteArrayOutputStream baos = new ByteArrayOutputStream();
47          PrintWriter out = new PrintWriter(baos);
48          TreeDumper dumper = new TreeDumper(out);
49          dumper.visit(expression);
50          out.flush();
51          return baos.toString();
52      }
53  
54  }
