26  /**
27   * Deliver a page containing multiple javascript files
28   *
29   * Parameters:
30   * js - csv list of javascript files. 
31   *      in this list of files the ".js" extension is optional
32   *      files will be retrieved from "/js" relative directory 
33   * 
34   */
35  public class MultiJavascriptServlet extends javax.servlet.http.HttpServlet implements javax.servlet.Servlet {
36      
37       /** Generated serialVersionUID */
38  	private static final long serialVersionUID = -5425445186806002874L;
39  
40  	/** Cache of combined js */
41       @SuppressWarnings("unchecked")
42  	public static Map<String, String> cache = new MRUCache(100, 0, null);
43       
44       /** Logger for this class */
45       public static final Logger logger = Logger.getLogger(MultiJavascriptServlet.class);
46       
47       
48  	public MultiJavascriptServlet() {
49  		super();
50  	}
51  	
52  	public void init(ServletConfig config) throws ServletException {
53  		super.init(config);
54  		logger.debug("Calling init");
55  	}
56  	
57  	/** Post method; just defers to get
58  	 * 
59  	 * @see javax.servlet.http.HttpServlet#doPost(HttpServletRequest request, HttpServletResponse response)
60  	 */
61  	protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
62  		doGet(request, response);
63  	}
64      
65  	/** Lets get this turkey stand on the road
66  	 * 
67  	 * @see javax.servlet.http.HttpServlet#doGet(HttpServletRequest request, HttpServletResponse response)
68  	 */
69      protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException 
70      {
71  		// dump the classpath
72  		/*
73  		System.out.println("*************************");
74  		System.out.println(getClassLoaderTrace(MultiJavascriptServlet.class));
75  		System.out.println("*************************");
76  		*/
77  		String result = null;
78  		// could cache these results, or respond to If-Modified-Since headers
79  		// TODO: Content-Length headers
80  		// TODO: maybe store result as byte arrays rather than strings
81  		// TODO: try/catch & log everything
82  		
83          String js = request.getParameter("js");
84          if (js==null || js.equals("")) { return; }
85          
86          PrintWriter pw = response.getWriter();
87          
88          result = cache.get(js);
89          if (result!=null) { 
90          	pw.println(result);
91          	return;
92          }
93          
94          ByteArrayOutputStream os = new ByteArrayOutputStream();
95          List<String> files;
96  		try {
97  			files = Text.parseCsv(js);
98  		} catch (ParseException e) {
99  			logger.error("Invalid js parameter '" + js + "'", e);
100 			throw new IllegalArgumentException("Invalid js parameter");
101 		}
102         for (String file : files) {
103         	if (!file.endsWith(".js")) {
104         		file+=".js";
105         	}
106         	// @TODO get relative base from somewhere. referrer ?
107         	// request param ? servler init param ?
108         	/*InputStream is = MultiJavascriptServlet.class.getClassLoader().
109         	  getResourceAsStream("./stocktake/js/" + file);*/
110         	InputStream is = getServletContext().getResourceAsStream("/js/" + file);
111         	if (is==null) {
112         		logger.error("Unknown file '" + file + "'");
113         		throw new IllegalArgumentException("Unknown file '" + file + "'");
114         	}
115         	StreamUtil.copyStream(is, os);
116         	is.close();
117         }
118         os.flush();
119         result = os.toString();
120         cache.put(js, result);
121         pw.println(result);
122 	}  	
123 }
