21  /**
22   * Reads the contents of the build.properties file in the 
23   * current application and sends it to the browser in a 
24   * variety of forms. Probably JSON for starters.
25   *
26   * <p>The properties file must be located on the classpath at the
27   * location "/build.properties".
28   * 
29   * <p>For this to work, the application needs to have been built
30   * through maven (and possibly bamboo), and there should be a 
31   * <tt>build.properties</tt> file in the application being created, 
32   * with the contents:
33   * 
34   * <pre>
35   *  bamboo.buildKey=${bambooBuildKey}
36  	bamboo.buildNumber=${bambooBuildNumber}
37  	bamboo.buildPlanName=${bambooBuildPlanName}
38  	bamboo.buildTimeStamp=${bambooBuildTimeStamp}
39  	bamboo.buildForceCleanCheckout=${bambooForceCleanCheckout}
40  	bamboo.custom.cvs.last.update.time=${bambooCustomCvsLastUpdateTime}
41  	bamboo.custom.cvs.last.update.time.label=${bambooCustomCvsLastUpdateTimeLabel}
42  	
43  	maven.pom.name=${pom.name}
44      maven.pom.version=${pom.version}
45      maven.pom.groupId=${pom.groupId}
46      maven.pom.artifactId=${pom.artifactId}
47     </pre>
48   *
49   * and the bamboo plan should have the following in the bamboo plan's
50   * stage's job's task's maven goal specification:
51   * 
52   * <pre>
53   *  "-DbambooBuildKey=${bamboo.buildKey}" 
54      "-DbambooBuildNumber=${bamboo.buildNumber}" 
55      "-DbambooBuildPlanName=${bamboo.buildPlanName}" 
56      "-DbambooBuildTimeStamp=${bamboo.buildTimeStamp}" 
57      "-DbambooBuildForceCleanCheckout=${bamboo.buildForceCleanCheckout}" 
58      "-DbambooCustomCvsLastUpdateTime=${bamboo.custom.cvs.last.update.time}" 
59      "-DbambooCustomCvsLastUpdateTimeLabel=${bamboo.custom.cvs.last.update.time.label}" 
60      -DuniqueVersion=false 
61   * </pre>
62   *
63   * <hr/>
64   * 
65   * This script will also look for a build.properties file in /etc/build.properties; 
66   * this should exist on docker containers built by bamboo. If this exists, it is also 
67   * included in the JSON output under the "/etc/build.properties" key.
68   * 
69   * 
70   * @author knoxg
71   * @blog http://www.randomnoun.com/wp/2013/09/24/webapp-versions-v1-0/
72   * 
73   */
74  @SuppressWarnings("serial")
75  public class VersionServlet extends HttpServlet {
76      
77  	
78      
79  
80  	/** Logger for this class */
81      public static final Logger logger = Logger.getLogger(VersionServlet.class);
82  
83      
84  	/** Post method; just defers to get
85  	 * 
86  	 * @see javax.servlet.http.HttpServlet#doPost(HttpServletRequest request, HttpServletResponse response)
87  	 */
88  	protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
89  		doGet(request, response);
90  	}
91  
92  	/** If a property has not been populated through the maven resource filter
93  	 * mechanism, then remove it from the Properties object.
94  	 *   
95  	 * @param props Properties object
96  	 * @param key key to check
97  	 * @param value default value
98  	 */
99  	private void removeDefaultProperty(Properties props, String key, String value) {
100 		if (("${" + value + "}").equals(props.get(key))) {
101 			props.remove(key);
102 		}
103 	}
104 	
105 	/** Removes build properties that were not set during the build process
106 	 *  (i.e. are still set to '${xxx}' placeholders).
107 	 *   
108 	 * @param props Properties object
109 	 */
110 	private void removeDefaultProperties(Properties props) {
111 		// projects not built in bamboo won't set these
112     	removeDefaultProperty(props, "bamboo.buildKey", "bambooBuildKey");
113     	removeDefaultProperty(props, "bamboo.buildNumber", "bambooBuildNumber");
114     	removeDefaultProperty(props, "bamboo.buildPlanName", "bambooBuildPlanName");
115     	removeDefaultProperty(props, "bamboo.buildTimeStamp", "bambooBuildTimeStamp");
116     	removeDefaultProperty(props, "bamboo.buildForceCleanCheckout", "bambooForceCleanCheckout");
117     	// bamboo.custom.cvs.last.update.time == "${bambooCustomCvsLastUpdateTime}" 
118     	// if there is no -Dbamboo.custom.cvs.last.update.time property on mvn build cmdline
119     	removeDefaultProperty(props, "bamboo.custom.cvs.last.update.time", "bambooCustomCvsLastUpdateTime");
120     	removeDefaultProperty(props, "bamboo.custom.cvs.last.update.time.label", "bambooCustomCvsLastUpdateTimeLabel");
121         // bamboo.custom.cvs.last.update.time == "${bamboo.custom.cvs.last.update.time}"
122     	// if there IS a -Dbamboo.custom.cvs.last.update.time property on mvn build cmdline
123     	// but it's not set by bamboo (because it's now a git project, but the mvn cmdline hasn't been changed)  
124     	removeDefaultProperty(props, "bamboo.custom.cvs.last.update.time", "bamboo.custom.cvs.last.update.time");
125     	removeDefaultProperty(props, "bamboo.custom.cvs.last.update.time.label", "bamboo.custom.cvs.last.update.time.label");
126     	
127     	
128     	// bamboo git projects have some extra properties we could include here; 
129     	//   bamboo.repository.git.branch
130     	//   bamboo.repository.git.repositoryUrl
131     	// see https://confluence.atlassian.com/bamboo/bamboo-variables-289277087.html
132     	
133     	// projects in cvs repositories won't set this
134     	removeDefaultProperty(props, "git.buildNumber", "buildNumber");
135 	}
136 	
137 	/** See class documentation
138 	 * 
139 	 * @see javax.servlet.http.HttpServlet#doGet(HttpServletRequest request, HttpServletResponse response)
140 	 */
141     protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException 
142     {
143     	InputStream is = VersionServlet.class.getClassLoader().getResourceAsStream("/build.properties");
144     	Properties props = new Properties();
145     	if (is==null) {
146     		props.put("error", "Missing build.properties");
147     	} else {
148 	    	props.load(is);
149 	    	is.close();
150     	}
151     	
152     	removeDefaultProperties(props);
153     	File etcPropsFile = new File("/etc/build.properties");
154     	if (etcPropsFile.exists()) {
155     		Properties etcProps = new Properties();
156     		is = new FileInputStream(etcPropsFile);
157     		etcProps.load(is);
158     		is.close();
159     		removeDefaultProperties(etcProps);
160     		props.put("/etc/build.properties", etcProps);
161     	}
162     	
163     	response.setHeader("Content-Type", "application/json");
164     	response.setStatus(HttpServletResponse.SC_OK);
165     	response.getWriter().println(new JSONObject(props).toString());
166     }
167 }
