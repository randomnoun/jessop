7   import java.util.HashMap;
8   import java.util.Map;
9   
10  /** Case-preserving hashmap that allows case-insensitive lookups.
11   * 
12   * @author knoxg
13   * 
14   * 
15   */
16  public class HashMapIgnoreCase<K, V> extends HashMap<String, V> {
17      
18  	/** generated  serialVersionUID */
19  	private static final long serialVersionUID = -6469387933015736361L;
20  	
21  	/** an extra map of lower-case keys to case-sensitive keys */
22  	private Map<String, String> lowercase = new HashMap<String, String>();
23  	
24  	/**
25  	 * Retrieve a value from the cache.
26  	 *
27  	 * @param key key whose associated value is to be returned
28  	 *
29  	 * @return the value to which this map maps the specified key, or
30  	 *   null if the map contains no mapping for this key.
31  	 */
32  	@Override
33  	public V get(Object key)
34  	{
35  		// System.out.println("getting " + lowercase.get(key.toLowerCase()));
36  		return super.get(lowercase.get(((String)key).toLowerCase()));
37  	}
38  	
39  	/**
40  	 * Associates the specified value with the specified key in this map.
41  	 * If the map previously contained a mapping for this key, the old
42  	 * value is replaced.
43  	 *
44  	 * @param key key with which the specified value is to be associated.
45  	 * @param value value to be associated with the specified key.
46  	 * 
47  	 * @return previous value associated with specified key, or <tt>null</tt>
48  	 *	       if there was no mapping for key.  A <tt>null</tt> return can
49  	 *	       also indicate that the HashMap previously associated
50  	 *	       <tt>null</tt> with the specified key.
51  	 *
52  	 * @throws IllegalArgumentException if the key parameter is not a String
53  	 */
54  	@Override
55  	public V put(String key, V value) {
56  		if (!(key instanceof String)) {
57  			throw new IllegalArgumentException("HashMapIgnoreCase keys must be Strings");
58  		}
59  		
60  		String lowercaseKey = ((String)key).toLowerCase();
61  		V previousEntry = null;
62  		
63  		// remove any existing entry with the same case-insensitive value
64  		if (lowercase.containsKey(lowercaseKey)) {
65  			//System.out.println("Removing " + lowercase.get(lowercaseKey));
66  			previousEntry = super.remove(lowercase.get(lowercaseKey));
67  		}
68  		lowercase.put(lowercaseKey, key);
69  		//System.out.println("putting " + key + "/" + value);
70  		super.put(key, value);
71  		return previousEntry;
72  	}
73  	
74  	/** Returns true if this map contains a mapping for the specified key. 
75  	 * More formally, returns true if and only if this map contains at 
76  	 * a mapping for a key <tt>k</tt> such that 
77  	 * (<tt>key==null ? k==null : key.equals(k)</tt>). (There can be at 
78  	 * most one such mapping.)
79  	 *  
80  	 * @param key key whose presence in this map is to be tested. 
81  	 * 
82  	 * @return true if this map contains a mapping for the specified key. 
83  	 */
84  	@Override
85  	public boolean containsKey(Object key) {
86  		if (!(key instanceof String)) {
87  			return false;
88  		}
89  		String lowercaseKey = ((String)key).toLowerCase();
90  		return lowercase.containsKey(lowercaseKey);
91  	}
92  	
93  
94  	/**
95  	 * Removes the mapping for this key from this map if present.
96  	 *
97  	 * @param  key key whose mapping is to be removed from the map.
98  	 * @return previous value associated with specified key, or <tt>null</tt>
99  	 *	       if there was no mapping for key.  A <tt>null</tt> return can
100 	 *	       also indicate that the map previously associated <tt>null</tt>
101 	 *	       with the specified key.
102 	 */
103 	@Override
104 	public V remove(Object key) {
105 		V previousEntry;
106 		String lowercaseKey = ((String) key).toLowerCase();
107 		previousEntry = super.remove(lowercase.get(lowercaseKey));
108 		lowercase.remove(lowercaseKey);
109 		return previousEntry;
110 	}
111 	
112 
113 }
