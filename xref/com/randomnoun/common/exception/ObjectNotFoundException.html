3   /** An exception which is thrown by DAOs when an objectId is supplied that doesn't
4    * have a record in the database.
5    * 
6    * <p>Would typically be caused by a user deleting an object, then hitting the back button on their browser
7    * to an object editor page showing that object, and then hitting reload.
8    * 
9    * <p>I used to use IllegalArgumentExceptions for this, but now these are being caught and acted upon
10   * in the Action classes, I thought it better to use a new exception type.
11   * 
12   * @author knoxg
13   */
14  public class ObjectNotFoundException extends RuntimeException {
15  
16  	/** generated serialVersionUID */
17  	private static final long serialVersionUID = -2560326697104634124L;
18  
19  	public ObjectNotFoundException(String message) {
20  		super(message);
21  	}
22  
23  	public ObjectNotFoundException(String message, Throwable cause) {
24  		super(message, cause);
25  	}
26  
27  }
