12  /** A ResultSetExtractor that uses a RowMapper to return a List of values from a ResultSet, but
13   * will not include nulls in the returned List. This could be used to return
14   * structured objects, where some records are used to modify existing results rather
15   * than returning new objects. 
16   *  
17   * @author knoxg
18   *
19   * @param <T> The type of object that the supplied RowMapper will return
20   */
21  public class SkipNullResultSetExtractor<T> implements ResultSetExtractor<List<T>> {
22  
23  	RowMapper<T> rowMapper;
24  	
25  	public SkipNullResultSetExtractor(RowMapper<T> rowMapper) {
26  		this.rowMapper = rowMapper;
27  	}
28  	
29  	@Override
30  	public List<T> extractData(ResultSet rs) throws SQLException, DataAccessException {
31  		List<T> results = new ArrayList<T>();
32          int rowNum = 0;
33          while (rs.next()) {
34          	T rowObj = (T) rowMapper.mapRow(rs, rowNum++);
35              if (rowObj!=null) { results.add(rowObj); }
36          }
37          return results;
38  	}
