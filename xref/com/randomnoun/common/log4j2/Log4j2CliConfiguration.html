12  /** Configures log4j for command-line interface programs.
13   * 
14   * <p>This class configured the log4j2 (two!) framework, using log4j1 (one!) properties.
15   * 
16   * <p>It uses a modified Log4j1ConfigurationParser (from the log4j2 framework) to do this, 
17   * which maps some well-known log4j1 appenders to log4j2 appenders, and does a pretty ham-fisted job of everything else.
18   * 
19   * <p>By default, everything's sent to stdout, using the following log4j initialisation properties:
20   * 
21   * <pre>
22  log4j.rootCategory=DEBUG, CONSOLE
23  log4j.appender.CONSOLE=org.apache.log4j.ConsoleAppender
24  log4j.appender.CONSOLE.layout=org.apache.log4j.PatternLayout
25  log4j.appender.CONSOLE.layout.ConversionPattern=%d{ABSOLUTE} %-5p %c - %m %n
26  
27  log4j.logger.org.springframework=INFO
28  
29  # log4j.appender.FILE=com.randomnoun.common.log4j.CustomRollingFileAppender
30  # log4j.appender.FILE.File=c:\\another.log
31  # log4j.appender.FILE.MaxBackupIndex=100
32  # log4j.appender.FILE.layout=org.apache.log4j.PatternLayout
33  # log4j.appender.FILE.layout.ConversionPattern=%d{dd/MM/yy HH:mm:ss.SSS} %-5p %c - %m %n
34     </pre>
35   * 
36   * <p>with an optional line prefix before the <code>%d{ABSOLUTE}</code> in the ConversionPattern.
37   * 
38   * @see http://logging.apache.org/log4j/1.2/manual.html
39   * 
40   * @blog http://www.randomnoun.com/wp/2013/01/13/logging/
41   * @author knoxg
42   * 
43   */
44  public class Log4j2CliConfiguration {
45      
46  	/** Create a new Log4jCli2Configuration instance */
47  	public Log4j2CliConfiguration() {
48  	}
49  	
50  	/** Initialise log4j.
51  	 *
52  	 * <p>The properties file supplied is in log4j format.
53  	 * 
54  	 * @param logFormatPrefix a string prefixed to each log. Useful for program identifiers;
55  	 *   e.g. "[programName] "
56  	 * @param override if non-null, additional log4j properties. Any properties contained in 
57  	 *   this object will override the defaults.
58  	 * 
59  	 */
60  	public void init(String logFormatPrefix, Properties override) {
61  		
62  		if (logFormatPrefix==null) { 
63  			logFormatPrefix = ""; 
64  		} else {
65  			logFormatPrefix += " ";
66  		}
67  		
68  		Properties props = new Properties();
69  		props.put("log4j.rootCategory", "DEBUG, CONSOLE");
70  		
71  		// log4j1 class names - the Log4j1ConfigurationParser recognises a limited set of these
72  		// and converts them into the log4j2 equivalents. 
73  		// Unknown appenders are ignored, which is obviously what people want to happen.
74  		props.put("log4j.appender.CONSOLE", "org.apache.log4j.ConsoleAppender");
75  		props.put("log4j.appender.CONSOLE.layout", "org.apache.log4j.PatternLayout");
76  
77  		// log4j2 class names - the Log4j1ConfigurationParser doesn't recognise these
78  		// props.put("log4j.appender.CONSOLE", "org.apache.logging.log4j.core.appender.ConsoleAppender");
79  		// props.put("log4j.appender.CONSOLE.layout", "org.apache.logging.log4j.core.layout.PatternLayout");
80  		props.put("log4j.appender.CONSOLE.layout.ConversionPattern", logFormatPrefix + "%d{ABSOLUTE} %-5p %c - %m%n");
81  		props.put("log4j.logger.org.springframework", "INFO"); // since Spring is a bit too verbose for my liking at DEBUG level
82  		if (override!=null) { 
83  			props.putAll(override);
84  		}
85  
86  		/*
87  		// to use the bundled Log41ConfigurationParser, had to do this:
88  		try {
89  			ByteArrayOutputStream baos = new ByteArrayOutputStream();
90  			props.store(baos,  null);
91  			ByteArrayInputStream bais = new ByteArrayInputStream(baos.toByteArray());
92  			
93  			// in log4j-1.2.jar 
94  			// Log4j1ConfigurationParser lcp = new Log4j1ConfigurationParser();
95  			
96  			Log4j1ConfigurationParser lcp = new Log4j1ConfigurationParser();
97  			ConfigurationBuilder<?> builder = lcp.buildConfigurationBuilder(bais);
98  		} catch (IOException e) {
99  			throw new IllegalStateException("IOException in log4j initiialisation", e);
100 		}
101 		*/
102 
103 		// our Log4j1ConfigurationParser can take a Properties object instead
104 		Log4j1ConfigurationParser lcp = new Log4j1ConfigurationParser();
105 		ConfigurationBuilder<?> builder = lcp.buildConfigurationBuilder(props);
106 		
107 		// this needs to happen before Configurator.initialize()
108 		// LogManager.shutdown();
109 		Configuration config = builder.build();
110 
111 		// so this doesn't work, because the (ctx.getState() == LifeCycle.State.INITIALIZED)) condition
112 		// in Log4jContextFactory.getContext() prevents the configuration from taking effect 
113 		/*LoggerContext ctx; =*/ // Configurator.initialize(config);
114 		
115 		// so you need to do this,
116 		// which is not what https://logging.apache.org/log4j/2.x/manual/customconfig.html 
117 		// fucking tells you to do. This took about two hours to figure out.
118 		/*LoggerContext ctx; =*/  Configurator.reconfigure(config);
119 
120 		
121 		// incidentally, did you know that every time you create a Logger, it's introspecting the stack for no good reason ?
122 		// *slow hand clap*, log4j.
123 		
124 		
125 		
126 		// log4j doesn't have ThrowableRenderers, it has ExtendedStackTraceElements instead
127 		// which I can't find a working example of anywhere
128 		/*
129 		String highlightPrefix = props.getProperty("log4j.revisionedThrowableRenderer.highlightPrefix");
130 		if (highlightPrefix != null) {
131 			LoggerRepository repo = LogManager.getLoggerRepository();
132 			if (repo instanceof ThrowableRendererSupport) {
133 	            // if null, log4j will use a DefaultThrowableRenderer
134 	            // ThrowableRenderer renderer = ((ThrowableRendererSupport) repo).getThrowableRenderer();
135 	            ((ThrowableRendererSupport) repo).setThrowableRenderer(new RevisionedThrowableRenderer(highlightPrefix));
136 	        }
137 		}
138 		
139 		PropertyConfigurator.configure(props);
140         */
141 	}
142 }
