3   import java.io.IOException;
4   import java.io.InputStreamReader;
5   import java.io.OutputStream;
6   import java.io.PipedInputStream;
7   import java.io.PipedOutputStream;
8   import java.nio.charset.Charset;
9   
10  import org.apache.logging.log4j.Logger;
11  
12  /** An outputstream that writes to a log4j Logger
13   *
14   * @author knoxg
15   */
16  public class LoggingOutputStream extends OutputStream {
17  	Logger logger;
18  	StringBuffer line = new StringBuffer();
19  	PipedOutputStream pos;
20  	PipedInputStream pis;
21  	InputStreamReader isr;
22  	
23  	public LoggingOutputStream(Logger logger, Charset charset) throws IOException {
24  		this.logger = logger;
25  		this.pos = new PipedOutputStream();
26  		this.pis = new PipedInputStream(pos);
27  		this.isr = new InputStreamReader(pis, charset);
28  	}
29  	public LoggingOutputStream(Logger logger, String charsetName) throws IOException {
30  		this(logger, Charset.forName(charsetName));
31  	}
32  	public LoggingOutputStream(Logger logger) throws IOException {
33  		this(logger, Charset.defaultCharset());
34  	}
35  	
36  	@Override
37  	public void write(int b) throws IOException {
38  		
39  		pos.write(b);
40  		int ch = isr.read();
41  		if (ch == -1) {
42  			// not a full character
43  		} else if (ch == 10) {
44  			logger.info(line); line.setLength(0);
45  		} else { 
46  			line.append((char) b);
47  		}
48  	}
49  }
