27  /**
28   * An implementation of the {@link com.randomnoun.common.security.SecurityLoader}
29   * class, using the Spring framework to populate the SecurityContext from a JDBC
30   * datasource.
31   * 
32   * <p>This security context used to optionally take a customerId (used to partition
33   * users across separate SaaS contexts), or an applicationId (used to partition
34   * permissions across separate software products). If not defined, then these 
35   * columns did not need to be present in the database. This functionality
36   * has been deprecated.
37   * 
38   * <p>I think we had String userids at one point as well, so look out for that.
39   *
40   * <p>This class has a number of initialisation properties that are specific
41   * to this class (in addition to those initialisation properties that are
42   * set by the SecurityContext itself):
43   *
44   * <attributes>
45   * {@link #INIT_JDBCTEMPLATE} - The Spring JdbcTemplate class used to retrieve
46   *   information from a database.
47   * {@link #INIT_USERNAME_MASK} - Used to mangle usernames before they are looked
48   *   up in an LDAP repository. A mask does a simple text replacment, with the
49   *   text "<tt>{username}</tt>" being replaced by the real username being passed
50   *   to the authenticate method. <i>e.g.</i> a username mask of
51   *   "{username}@abc.com" will convert the username "bob" to "bob@abc.com"
52   *   before authentication is performed.
53   * {@link #INIT_TABLE_SUFFIX} - A String appended to each table used in queries
54   *   against the database. Used to separate the 'LIVE' area from the 'WORK' staging
55   *   area.
56   * {@link #INIT_DATABASE_VENDOR} - Set to one of the SqlGenerator.DATABASE_* constants,
57   *   which specifies what syntax of SQL to use (DB2, Oracle or SqlServer).
58   * {@link #INIT_AUDIT_ENABLED} - (Boolean) If false, then disables auditting security changes
59   * </attributes>
60   * 
61   * @TODO this class does not internationalise text added to the audit log
62   *   (it should be in a form that can be parsed and internationalised depending on the
63   *   user viewing it).
64   *
65   * 
66   * @author knoxg
67   */
68  public class SpringSecurityLoaderImpl

177 	/** Return the sql for a string comparison for a username field, taking
178 	 * case sensitivity into account.
179 	 * 
180 	 * @param lhs expression to lowercase (if required) 
181 	 *
182 	 * @return
183 	 */
184 	private String lowerSql(String lhs) {
185 		if (getCaseInsensitive()) {
186 			String vendor = getDatabaseVendor();
187 			if (vendor.equals(SqlGenerator.DATABASE_DB2)) {
188 				return "LOWER(" + lhs + ")";
189 			} else if (vendor.equals(SqlGenerator.DATABASE_ORACLE)) {
190 				return "LOWER(" + lhs + ")";
191 			} else if (vendor.equals(SqlGenerator.DATABASE_SQLSERVER)) {
192 				return "LOWER(" + lhs + ")";
193 			} else if (vendor.equals(SqlGenerator.DATABASE_MYSQL)) {
194 				return "LOWER(" + lhs + ")";
195 			} else {
196 				throw new IllegalStateException("Case-insensitivity security contexts" +
197 				  " not supported for database type '" + vendor + "'");
198 			}
199 		} else {
200 			return lhs;
201 		}
202 	}

204 	/** Converts a value to lowercase, but only if case insensitivity is required */
205 	private String lower(String value) {
206 		if (getCaseInsensitive()) {
207 			return value.toLowerCase();
208 		} else {
209 			return value;
210 		}

284 	 *
285 	 * {@inheritDoc}
286 	 *
287 	 * @return A List of Permission objects for all roles in the current application. Does
288 	 * not return Permissions that are not explicitly associated with a role in the security table.
289 	 *
290 	 * @throws IOException if an error occured loading from the database. This IOException

293 	 *

297 		throws IOException

300 		String loadAllRolePermissionsSql =

466 	/** Private method to convert a list of PERMISSION rows (as returned by Spring)
467 	 *  into Permission Objects.
468 	 *
469 	 * <p>On reflection, probably should have used a Spring RowMapper for this. Oh well. It works.
470 	 *
471 	 * @param list  The list to convert
472 	 * @return  A List of Permission objects
473 	 */
474 	private List<Permission> convertPermissionList(List<Map<String, Object>> list, int permissionType)
475 	{
476 		String resourceType;
477 		String expressionString;
478 		Permission permission;
479 		List<Permission> result = new ArrayList<Permission>(list.size());
480         
481 		for (Iterator<Map<String, Object>> i = list.iterator(); i.hasNext(); ) {
482 			Map<String, Object> map = i.next();
483 			resourceType = (String) map.get("resourceName");
484 			expressionString = (String) map.get("resourceCriteria");
485 			ResourceCriteriaImpl resourceCriteriaImpl = null;
486 			if (expressionString != null && !expressionString.equals("")) {
487 				try {
488 					resourceCriteriaImpl = new ResourceCriteriaImpl(expressionString);
489 				} catch (Exception ce) {
490 					throw new DataIntegrityViolationException(
491 						"Invalid criteria found in SECURITY.RESOURCECRITERIA: '" +
492 						expressionString + "'", ce);
493 				}
494 			}
495 			
496 			if (permissionType == PERMISSION_ROLE) {
497 				permission = new Permission((String)map.get("roleName"),
498 				  (String)map.get("activityName"), resourceType, resourceCriteriaImpl);
499 			} else if (permissionType == PERMISSION_USER) {
500 				User user = new User();
501 				user.setUsername((String) map.get("userId"));
502 				permission = new Permission( user,
503 				  (String)map.get("activityName"), resourceType, resourceCriteriaImpl);
504 			} else if (permissionType == PERMISSION_NONE) {
505 				permission = new Permission((String)map.get("activityName"), resourceType);
506 				
507 			} else {
508 				throw new IllegalArgumentException("Unknown permission type '" + permissionType + "'");
509 			}
510 			result.add(permission);
511 		}
512 		return result;
513 	}
