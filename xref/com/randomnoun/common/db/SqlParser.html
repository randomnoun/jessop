4   import java.io.InputStream;

6   import java.io.Reader;

13  /** A class to read an InputStream containing SQL statements (e.g. a MySQL input file) and split it into 
14   * individual SQL statements.
15   * 
16   * <p>Some MySQL command-line directives (e.g. 'DELIMITER') are processed whilst parsing. 
17   */
18  public class SqlParser {
19  	
20  	private static Pattern MYSQL_DELIMITER_COMMAND = Pattern.compile("delimiter\\s+(.*)$", Pattern.CASE_INSENSITIVE);
21  	
22  	// could have a ParseState that contains things like the current delimiter, or current database
23  	// (for 'use' commands), which can be modified by the InputStream, or the database vendor
24  	// if we wanted to support escape sequences, which appear to be database-specific.
25  	
26  	/** Convert an InputStream of SQL statements into a List of individual
27  	 * statements. Statements are delimited by ";" strings that occur outside of strings or comments.
28  	 * The delimiter string is not included in the returned list of statements.
29  	 * 
30  	 * <p>The delimiter string may be changed using the 'delimiter str' command. 
31  	 * This command may end with a newline instead of a delimiter.
32  	 * 
33  	 * <p>Comments may also be returned in the result.
34  	 * <p>Comments are defined by '-- to-end-of-line' or '/* within slash-star star-slash *&#8288;/' syntax.
35  	 * Comments that are created with '--' and that occur within a statement are returned before that statement
36  	 * has finished parsing.
37  	 * 
38  	 * <p>NB: Does not handle escape sequences found within double or single quotes
39  	 * (i.e. escape sequences are handled by the underlying database, not by this Parser)
40  	 * 
41  	 * @param includeComments include comment strings in result
42  	 * 
43  	 * @throws IOException 
44  	 * @throws ParseException unclosed /*-style comment or single/double-quoted string
45  	 */
46  	public List<String> parseStatements(InputStream is, boolean includeComments) throws IOException, ParseException {
47  		return parseStatements(new InputStreamReader(is), includeComments);
48  	}
49  	
50  	public List<String> parseStatements(Reader is, boolean includeComments) 
51  		throws IOException, ParseException 
52  	{
53  		final List<String> allSql = new ArrayList<String>();
54  		consumeStatements(is, includeComments, new Consumer<String>() {
55  			@Override
56  			public void consume(String s) {
57  				allSql.add(s);
58  			}
59  		});
60  		return allSql;
61  	}
62  
63  	public static interface Consumer<T> {
64  		public void consume(T s);
65  	}
66  	
67  	/** convert a text file of SQL statements into a List of individual
68  	 * statements. Comments may also be returned in the result.
69  	 * 
70  	 * <p>NB: Does not handle escape sequences found within double or single quotes
71  	 * 
72  	 * @param includeComments include comment strings in result
73  	 * 
74  	 * @throws IOException 
75  	 * @throws ParseException unclosed /*-style comment or single/double-quoted string
76  	 */
77  	public void consumeStatements(InputStream is, boolean includeComments, Consumer<String> consumer) 
78  			throws IOException, ParseException {
79  		consumeStatements(new InputStreamReader(is), includeComments, consumer);
80  	}
81  	
82  	public void consumeStatements(Reader is, boolean includeComments, Consumer<String> consumer) 
83  		throws IOException, ParseException 
84  	{
85  		// @TODO some error handling
86  		// List allSql = new ArrayList<String>();
87  		
88  		int state = 0; // command parsing [to states 0, 1, 2, 4, 7, 8]
89  		// 1 // parsed double quote, in double quotes [to states 0, 1, 9]
90  		// 2 // parsed - [to states 0, 3]
91  		// 3 // parsed -- [ to states 0, 3]
92  		// 4 // parsed /  [ to states 0, 5]
93  		// 5 // parsed /* [ to states 6, 5]
94  		// 6 // parsed * from state 5  [ to states 0, 5]
95  		// 7 // parse single quote, in single quotes [to states 0, 7, 10]
96  		// 8 // parsed backslash in SQL, next character will be emitted without changing state [then to state 0]
97  		// 9 // parsed backslash in double quote, next char emitted without changing state [then to state 1]
98  		// 10 // parsed backslash in single quote, next char emitted without changing state [then to state 7]
99  
100 		String s = ""; // current statement
101 		String c = ""; // current comment
102 		String delimiter = ";"; // default delimiter
103 
104 		int intch = is.read();
105 		int delimIdx = 0; // number of delimiter characters read
106 		while (intch!=-1) {
107 			char ch = (char) intch;
108 			
109 			if (state==0) {
110 				if (ch == delimiter.charAt(delimIdx)) {
111 					delimIdx++;
112 					if (delimiter.length()==delimIdx) {
113 						s = s.trim(); delimIdx = 0; // was s.trim() + ";"
114 						// could check for all commands at http://dev.mysql.com/doc/refman/5.7/en/mysql-commands.html
115 						// but let's just implement 'DELIMITER' for now
116 						Matcher m = MYSQL_DELIMITER_COMMAND.matcher(s);
117 						if (m.matches()) {
118 							delimiter = m.group(1); // set the new delimiter
119 						} else {
120 							consumer.consume(s); 
121 						}
122 						s="";
123 					} else {
124 						state = 0; 
125 					}
126 				} else {
127 					if (delimIdx > 0) {
128 						// could push these back onto the inputStream in case the delimiter startsWith a " or -, but that seems a bit fiddly
129 						s = s + delimiter.substring(0, delimIdx); delimIdx = 0; 
130 					}
131 					switch(ch) {
132 						case '"' : state = 1; s = s + ch; break;
133 						case '-' : state = 2; break;
134 						case '/' : state = 4; break;
135 						case '\'' : state = 7; s = s + ch; break;
136 						case '\r' :
137 						case '\n' :
138 							// if this is a delimiter command, process it
139 							Matcher m = MYSQL_DELIMITER_COMMAND.matcher(s.trim());
140 							if (m.matches()) {
141 								delimiter = m.group(1); // set the new delimiter
142 								s = ""; delimIdx = 0;
143 							} else {
144 								s = s + ch;
145 							}
146 							break;
147 						case '\\': state = 8; s = s + ch; break;							
148 						default: s = s + ch;
149 					}
150 				}
151 				
152 			} else if (state==1) {
153 				switch(ch) {
154 					case '"' : state = 0; s = s + ch; break;
155 					case '\\': state = 9; s = s + ch; break;
156 					default: s = s + ch;
157 				}
158 			} else if (state==2) {
159 				switch(ch) {
160 					case '-' : state = 3; break;
161 					default: state = 0; s = s + "-" + ch;
162 				}
163 			} else if (state==3) {
164 				switch(ch) {
165 					case '\r' : 
166 					case '\n' : 
167 						state = 0; 
168 						if (includeComments) { consumer.consume("-- " + c.trim()); } 
169 						c=""; break;
170 					default :
171 						c = c + ch;
172 				}
173 			} else if (state==4) {
174 				switch(ch) {
175 					case '*' : state = 5; break;
176 					default: state = 0; s = s + "/" + ch;
177 				}
178 			} else if (state==5) {
179 				switch(ch) {
180 					case '*' : state = 6; break;
181 					default: c = c + ch;
182 				}
183 			} else if (state==6) {
184 				switch(ch) {
185 					case '/' : 
186 						state = 0; 
187 						if (includeComments) { s += ("/* " + c.trim() + " */"); } // was consumer.consume(...) 
188 						c=""; break;
189 					default: c = c + "*" + ch;
190 				}
191 			} else if (state==7) {
192 				switch(ch) {
193 					case '\'' : state = 0; s = s + ch; break;
194 					case '\\' : state = 10; s = s + ch; break;
195 					default: s = s + ch;
196 				}
197 			} else if (state==8) {
198 				// could check ch is printable here, but probably ok
199 				s = s + ch;
200 				state = 0;
201 			} else if (state==9) {
202 				s = s + ch;
203 				state = 1;
204 			} else if (state==10) {
205 				s = s + ch;
206 				state = 7;
207 			}
208 			
209 			intch = is.read();
210 		}
211 		
212 		if (state==5) {
213 			// unclosed /*-style comment
214 			// ignore for the time being
215 			throw new ParseException("Unclosed /*-style comment before EOF", -1);
216 		} else if (state==1) {
217 			// unclosed quoted string
218 			// ignore for the time being
219 			throw new ParseException("Unclosed double quoted string before EOF", -1);
220 		} else if (state==7) {
221 			// unclosed quoted string
222 			// ignore for the time being
223 			throw new ParseException("Unclosed single quoted string before EOF", -1);
224 		} else if (state==0) {
225 			if (!s.trim().equals("")) {
226 				// unterminated statement at end of InputStream; add to list
227 				consumer.consume(s.trim());
228 			}
229 		}
230 		
231 		// return allSql;
232 	}
