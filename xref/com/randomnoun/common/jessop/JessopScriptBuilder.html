7   import java.io.PrintWriter;
8   
9   import javax.script.ScriptException;
10  
11  /** Each target language we intend to support within jessop must have an implementation of this interface. 
12   * Developers wishing to implement this interface should use the {@link AbstractJessopScriptBuilder} abstract class.
13   * 
14   * <p>Implementations of this class should return the language it supports (and the default script engine name) via the
15   * {@link #getLanguage()} and {@link #getDefaultScriptEngineName()} methods.
16   * 
17   * <p>These values will be used to select the correct JessopScriptBuilder identified in the 
18   * <tt>&lt;%@ jessop language="xxx"%&gt;</tt>
19   * declaration in the jessop source file.
20   * 
21   * <p>When this class is instantiated, the caller will invoke {@link #setPrintWriter(PrintWriter)} and 
22   * {@link #setTokeniserAndDeclarations(Tokeniser, JessopDeclarations)}.
23   * 
24   * <p>As the Tokeniser parses the jessop source file, it will invoke emit() methods on this class.
25   * As these methods are called, this class should generate code in the target language via the printWriter.
26   * 
27   * <p>Care should be taken to ensure that code in the target language script is on the same line number
28   * as the corresponding code on the input script, to make error messages more developer-friendly.
29   * 
30   * @author knoxg
31   */
32  public interface JessopScriptBuilder {
33  	// lines are source line numbers; try to keep these intact in the generated script
34  	
35  	/** Returns the name of the language that this scriptBuilder can parse; e.g. "javascript" or "python2".
36  	 * 
37  	 * <p>This is used to register this JessopScriptBuilder in the registry and is used to lookup the JessopScriptBuilder
38  	 * from the language defined in the jessop script declaration.
39  	 * 
40  	 * @return the name of the language that this scriptBuilder can parse; e.g. "javascript" or "python2".
41  	 */
42  	String getLanguage();                                             
43  	
44  	/** Returns the name of the script engine that is used to evaluate this script; e.g. "rhino" or "jython"
45  	 * 
46  	 * @return the name of the script engine that is used to evaluate this script; e.g. "rhino" or "jython"
47  	 */
48  	String getDefaultScriptEngineName();                             
49  
50  	/** Returns the name of the default exceptionConverter class that should be used for the default script engine,
51  	 * or null if no converter is required.
52  	 * 
53  	 * <p>Changing the engine in the jessop declaration will reset the the converter to null. 
54  	 * 
55  	 * @return the name of the default exceptionConverter class that should be used for the default script engine
56  	 */
57  	String getDefaultExceptionConverterClassName();                             
58  
59  	/** Returns the name of the default bindingsConverter class that should be used for the default script engine,
60  	 * or null if no converter is required.
61  	 * 
62  	 * <p>Changing the engine in the jessop declaration will reset the the converter to null. 
63  	 * 
64  	 * @return the name of the default bindingsConverter class that should be used for the default script engine
65  	 */
66  	String getDefaultBindingsConverterClassName();                             
67  
68  	
69  	/** Sets the printWriter that this class will write to during emit() methods
70  	 * 
71  	 * @param pw the printWriter that this class will write to during emit() methods
72  	 */
73  	void setPrintWriter(PrintWriter pw);
74  	
75  	/** Sets the tokeniser and declaration that are in effect at the start of parsing. This allows this
76  	 * ScriptBuilder to switch ScriptBuilders.
77  	 * 
78  	 * @param t tokeniser that is processing the jessop source
79  	 * @param declarations jessop declarations that are in effect
80  	 */
81  	void setTokeniserAndDeclarations(Tokeniser t, JessopDeclarations declarations);
82  	
83  	/** Called by the tokeniser and requests that this JessopScriptBuilder generate code to generate some text output.
84  	 * 
85  	 * <p>This method should escape the output and generate code that outputs this text in the implementation language.
86  	 *  
87  	 * @param line jessop source line number that begins this text output
88  	 * @param s the text to output; may include newlines.
89  	 */
90  	void emitText(int line, String s);        
91  	
92  	/** Called by the tokeniser and requests that this JessopScriptBuilder generate code to generate some evaluated output.
93  	 * i.e. process a <tt>&lt;%= ... %&gt;</tt> expression.
94  	 * 
95  	 * <p>This method should generate the code to evaluate and output the contents of the <tt>&lt;%= ... %&gt;</tt> expression.
96  	 * 
97  	 * <p>This method should attempt to preserve line numbers between the jessop source file and the generated code file.
98  	 * 
99  	 * @param line jessop source line number that begins this text output.
100 	 * @param s the contents of the <tt>&lt;%= ... %&gt;</tt> expression.
101 	 */
102 	void emitExpression(int line, String s);  // <%= ... %>
103 	
104 
105 	/** Called by the tokeniser and requests that this JessopScriptBuilder copy the included code into the generated script.
106  	 * i.e. process a <tt>&lt;% ... %&gt;</tt> scriptlet. (The 'scriptlet' term is the term used in the JSP specification)
107  	 * 
108 	 * <p>This method should generate the same code that is contained within the <tt>&lt;% ... %&gt;</tt> scriptlet.
109 	 * 
110 	 * <p>This method should attempt to preserve line numbers between the jessop source file and the generated code file.
111 	 * 
112 	 * <p>This scriptlet may not be syntactically correct by itself and may require further scriptlets to function, e.g.
113 	 * <pre>
114 	 * &lt;% if (a>b) { %&gt;
115 	 * something
116 	 * &lt;% } %&gt;
117 	 * </pre>
118 	 * 
119 	 * @param line jessop source line number that begins this scriptlet.
120 	 * @param s the contents of the <tt>&lt;%= ... %&gt;</tt> expression.
121 	 */
122 	void emitScriptlet(int line, String s);
123 
124 	/** Called by the tokeniser and requests that this JessopScriptBuilder process a <tt>&lt;%@ ... %&gt;</tt> directive.
125 	 * 
126 	 * <p>The exact semantics of these declarations can be defined per-language, but 'jessop' declarations 
127 	 * should be processed by the AbstractJessopScriptBuilder, i.e. declarations that are in the form
128 	 * <tt>&lt;%@ jessop language="xxx" %&gt;</tt>
129 	 * 
130 	 * <p>Ideally, there's only one declaration, at the top of the source script.
131 	 * 
132 	 * @param line jessop source line number that begins this directive
133 	 * @param s the contents of the <tt>&lt;%@ ... %&gt;</tt> directive
134 	 * @throws ScriptException
135 	 */
136 	void emitDeclaration(int line, String s) throws ScriptException; 
137 
138 	/** Returns the value of any jessop declarations that are in effect.
139 	 * 
140 	 * <p>Used to maintain declarations when switching JessopScriptBuilders, and to retrieve declarations after the script is built.
141 	 */
142 	JessopDeclarations getDeclarations();
143 	
144 }
