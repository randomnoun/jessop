7   import javax.script.ScriptContext;
8   import javax.script.ScriptException;
9   
10  /** The only reason this thing exists is because the 'bsh' and 'luaj' script engines 
11   * don't throw ScriptExceptions properly
12   *  
13   * @author knoxg
14   */
15  public interface JessopExceptionConverter {
16  
17  	/** Convert runtime exceptions generated by this language's ScriptEngine into ScriptExceptions.
18  	 * 
19  	 * <p>(Most languages do this properly already, except for lua).
20  	 * 
21  	 * <p>This is called at runtime, not target script generation time, and should be 
22  	 * associated with the target source engine (e.g. rhino), not the target language (e.g. javascript).
23  	 */
24  	ScriptException toScriptException(ScriptContext scriptContext, Throwable t);
25  
26  }
