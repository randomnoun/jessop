12  public class LuaJessopScriptBuilder extends AbstractJessopScriptBuilder implements JessopScriptBuilder {
13  
14  	Logger logger = Logger.getLogger(LuaJessopScriptBuilder.class);
15  	int outputLine = 1;        // current line number in the target script;
16  	int lastScriptletLine = 1; // the last line number of the last scriptlet (used for suppressEol)
17  
18  	public LuaJessopScriptBuilder() {
19  	}
20  	private void skipToLine(int line) {
21  		while (outputLine < line) { print("\n"); }
22  	}
23  	private void print(String s) {
24  		// logger.info("** PRINT " + s);
25  		pw.print(s);
26  		for (int i=0; i<s.length(); i++) {
27  			if (s.charAt(i)=='\n') { outputLine++; } 
28  		}
29  	}
30  	private static String escapeLua(String string) {
31  		/* valid escapes ( https://www.lua.org/pil/2.4.html )
32  \a	bell
33  \b	back space
34  \f	form feed
35  \n	newline
36  \r	carriage return
37  \t	horizontal tab
38  \v	vertical tab
39  \\	backslash
40  \"	double quote
41  \'	single quote
42  \[	left square bracket
43  \]	right square bracket
44  		 */
45  		
46      	StringBuilder sb = new StringBuilder(string.length());
47      	String escapeChars = "\u0007" + "\u0008" + "\u000f" + "\n" + "\r" + "\u0009" + "\u000b" + "\\" + "\"" + "'" + "[" + "]";
48      	String backslashChars = "abfnrtv\\\"'[]";
49  		for (int i = 0; i<string.length(); i++) {
50  			char ch = string.charAt(i);
51  			int pos = escapeChars.indexOf(ch);
52  			if (pos !=- 1) {
53  			   sb.append("\\" + backslashChars.charAt(pos));
54  			
55  			// so apparently lua allows any character in a string whatsoever. looking forward to seeing this breaking.
56  			// have seen some examples of, e.g. \006 to represent chars, but nothing in the lua spec
57  			   
58  			//} else if (ch=='\\' || ch=='"' || ch=='\'' || ch<32 && ch>126) {
59  			//	String hex = Integer.toString(ch, 16);
60  			//	// sb.append("\\u" + "0000".substring(0, 4-hex.length()) + hex);
61  			//	sb.append(ch); 
62  			} else {
63  				sb.append(ch);
64  			}
65  		}
66          return sb.toString();
67      }
68  	
69  	@Override
70  	public void emitText(int line, String s) {
71  		skipToLine(line);
72  		s = suppressEol(s, declarations.isSuppressEol() && lastScriptletLine == line);
73  		print("out:write(\"" + escapeLua(s) + "\")");
74  		lastScriptletLine = 0; // don't suppress eols on this line
75  	}
76  	@Override
77  	public void emitExpression(int line, String s) {
78  		skipToLine(line);
79  		print("out:write((" + s + ") .. \"\")"); // coerce to String
80  		lastScriptletLine = 0; // don't suppress eols on this line
81  	}
82  	@Override
83  	public void emitScriptlet(int line, String s) {
84  		skipToLine(line);
85  		print(s);
86  		lastScriptletLine = line;
87  		for (int i=0; i<s.length(); i++) { if (s.charAt(i)=='\n') { lastScriptletLine++; } }
88  	}
89  	@Override
90  	public String getLanguage() {
91  		return "lua";
92  	}
93  	@Override
94  	public String getDefaultScriptEngineName() {
95  		return "luaj";
96  	}
97  	@Override
98  	public String getDefaultExceptionConverterClassName() {
99  		return "com.randomnoun.common.jessop.engine.LuajExceptionConverter";
100 	}
