9   import com.randomnoun.common.jessop.AbstractJessopScriptBuilder;

12  public class Python2JessopScriptBuilder extends AbstractJessopScriptBuilder implements JessopScriptBuilder {
13  
14  	Logger logger = Logger.getLogger(Python2JessopScriptBuilder.class);
15  	int outputLine = 1;        // current line number in the target script;
16  	int outputCol = 1;         // current output column
17  	int lastScriptletLine = 1; // the last line number of the last scriptlet (used for suppressEol)
18  	int indent = 0;            // current number of spaces at start of line (we use 4-space indents)
19  	public Python2JessopScriptBuilder() {
20  	}
21  	private void skipToLine(int line, int indent) {
22  		if (outputLine > line) {
23  			// could allow, but then that'll open another can of worms 
24  			// throw new IllegalArgumentException("cannot generate output on same line as starting new python block");
25  			logger.warn("can't go back to line " + line + " (outputLine=" + outputLine + "); line numbers may be inaccurate");
26  		}
27  		while (outputLine < line) { print("\n"); }
28  		while (outputCol < indent) { print(" "); }
29  		// for (int i=0; i<indent; i++) { print(" "); }
30  	}
31  	private void print(String s) {
32  		// logger.info("** PRINT " + s);
33  		pw.print(s);
34  		for (int i=0; i<s.length(); i++) {
35  			if (s.charAt(i)=='\n') { outputLine++; outputCol = 1; }
36  			else { outputCol++; }
37  		}
38  	}
39  	private static String escapePython(String string) {
40  		/* valid escapes ( https://docs.python.org/2.0/ref/strings.html )
41  \a	bell
42  \b	back space
43  \f	form feed
44  \n	newline
45  \r	carriage return
46  \t	horizontal tab
47  \v	vertical tab
48  \\	backslash
49  \"	double quote
50  \'	single quote
51  		 */
52  		
53      	StringBuilder sb = new StringBuilder(string.length());
54      	String escapeChars = "\u0007" + "\u0008" + "\u000f" + "\n" + "\r" + "\u0009" + "\u000b" + "\\" + "\"" + "'";
55      	String backslashChars = "abfnrtv\\\"'";
56  		for (int i = 0; i<string.length(); i++) {
57  			char ch = string.charAt(i);
58  			int pos = escapeChars.indexOf(ch);
59  			if (pos !=- 1) {
60  			   sb.append("\\" + backslashChars.charAt(pos));
61  			
62  			} else if (ch<32 || (ch>126 && ch <= 255)) {
63  				String hex = Integer.toString(ch, 16);
64  				sb.append("\\x" + "00".substring(0, 2-hex.length()) + hex);
65  				sb.append(ch);
66  				
67  			} else if (ch<=255) {
68  				sb.append(ch);
69  				
70  			} else {
71  				throw new IllegalArgumentException("Cannot escape characters > 0xFF in python2 (found char '" + ch + "'; code=" + ((int) ch) + ")");
72  			}
73  		}
74          return sb.toString();
75      }

77  	@Override
78  	public void emitText(int line, String s) {
79  		skipToLine(line, indent);
80  		s = suppressEol(s, declarations.isSuppressEol() && lastScriptletLine == line);
81  		print("out.write(\"" + escapePython(s) + "\");");
82  		lastScriptletLine = 0; // don't suppress eols on this line
83  	}
84  	@Override
85  	public void emitExpression(int line, String s) {
86  		skipToLine(line, indent);
87  		print("out.write((str) (" + s + "));"); // coerce to String
88  		lastScriptletLine = 0; // don't suppress eols on this line
89  	}
90  	@Override
91  	public void emitScriptlet(int line, String s) {
92  		if (outputLine > line) {
93  			throw new IllegalArgumentException("cannot generate scriptlet output on same line as starting new python block");
94  		}
95  
96  		skipToLine(line, indent);
97  		// if there's content before the first newline, remove whitespace from the beginning
98  		// so that we maintain our line/indentation position
99  		int pos = s.indexOf("\n"); if (pos==-1) { pos = s.length(); }
100 		int i=0;
101 		while (i < s.length() && s.charAt(i)==' ') { i++; }
102 		if (s.charAt(i)=='\t') { throw new IllegalArgumentException("tab indentation for python scriplets not supported"); }
103 		s = s.substring(i);
104 		
105 		logger.debug("scriptlet is '" + s + "'");
106 		print(s);
107 		/* 
108 		  <%
109 		      for i=1..10:
110 		        something
111 		        somethingElse
112 		  %>no idea whether this is in the for loop or not
113 		  
114  		  <%
115  		      for i=1..10:
116 		  %>presumably this is in the for loop. not sure how to terminate it though.
117 		  <%
118 		      pass;  # empty statement on a line could indicate end of block
119 		  %>
120 		   
121 		      
122 		 * This sort of thing is why whitespace-significant languages should burn in hellfire,
123 		 * (I'm looking at you, yaml), and why things like jinja are apparently necessary
124 		 */
125 		
126 		// get the amount of indentation on the last non-blank line
127 		int endLinePos = s.length();
128 		int startLinePos = s.lastIndexOf("\n");
129 		while (s.substring(startLinePos+1).trim().equals("")) { 
130 			endLinePos = startLinePos; 
131 			startLinePos = s.lastIndexOf("\n", startLinePos-1); 
132 		}
133 		startLinePos++; // don't include the '\n'
134 		
135 		i = 0;
136 		while (startLinePos + i < endLinePos && s.charAt(i)==' ') { i++; }
137 		logger.debug("python last newline pos=" + startLinePos + ", indent on last line=" + i);
138 		if (s.charAt(startLinePos + i)=='\t') { 
139 			// could support this later, perhaps
140 			throw new IllegalArgumentException("tab indentation for python scriplets not supported");
141 		}
142 		indent = i;
143 		if (s.substring(startLinePos, endLinePos).trim().endsWith(":")) {
144 			logger.debug("last non-blank line endsWith ':', indenting by 4");
145 			// this may mean some loop constructs will now have inaccurate line numbers
146 			// e.g. <% for i in range (0,10): %><%= something %>
147 			print("\n"); 
148 			indent += 4;
149 		}
150 
151 		// this should probably go before the indent processing above. maybe.
152 		lastScriptletLine = line;
153 		for (i=0; i<s.length(); i++) { if (s.charAt(i)=='\n') { lastScriptletLine++; } }
154 	}
155 	@Override
156 	public String getLanguage() {
157 		return "python2";
158 	}
159 	@Override
160 	public String getDefaultScriptEngineName() {
161 		return "jython";
162 	}
