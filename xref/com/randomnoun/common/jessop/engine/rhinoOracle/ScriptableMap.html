25  import java.util.Map;

28  /**
29   * ScriptableMap is a wrapper for java.util.Map instances that allows developers
30   * to interact with them as if it were a native JavaScript object.
31   */
32  public class ScriptableMap extends NativeJavaObject {
33  
34      boolean reflect;
35      Map map;
36      final static String CLASSNAME = "ScriptableMap";
37  
38      // Set up a custom constructor, for this class is somewhere between a host class and
39      // a native wrapper, for which no standard constructor class exists
40      public static void init(Scriptable scope) throws NoSuchMethodException {
41          BaseFunction ctor = new BaseFunction(scope, ScriptableObject.getFunctionPrototype(scope)) {
42              @Override
43              public Scriptable construct(Context cx, Scriptable scope, Object[] args) {
44                  boolean reflect = false;
45                  if (args.length > 2) {
46                      throw new EvaluatorException("ScriptableMap() called with too many arguments");
47                  } if (args.length == 2) {
48                      reflect = ScriptRuntime.toBoolean(args[1]);
49                  }
50                  return new ScriptableMap(scope, args.length == 0 ? null : args[0], reflect);
51              }
52              @Override
53              public Object call(Context cx, Scriptable scope, Scriptable thisObj, Object[] args) {
54                  return construct(cx, scope, args);
55              }
56          };
57          ScriptableObject.defineProperty(scope, CLASSNAME, ctor,
58                  ScriptableObject.DONTENUM | ScriptableObject.READONLY);
59      }
60  
61      @SuppressWarnings("unchecked")
62      private ScriptableMap(Scriptable scope, Object obj, boolean reflect) {
63          this.parent = scope;
64          this.reflect = reflect;
65          if (obj instanceof Wrapper) {
66              obj = ((Wrapper) obj).unwrap();
67          }
68          if (obj instanceof Map) {
69              this.map = (Map) obj;
70          } else if (obj == null || obj == Undefined.instance) {
71              this.map = new HashMap();
72          } else if (obj instanceof Scriptable) {
73              this.map = new HashMap();
74              Scriptable s = (Scriptable) obj;
75              Object[] ids = s.getIds();
76              for (Object id: ids) {
77                  if (id instanceof String) {
78                      map.put(id, s.get((String)id, s));
79                  } else if (id instanceof Number) {
80                      map.put(id, s.get(((Number)id).intValue(), s));
81                  }
82              }
83          } else {
84              throw new EvaluatorException("Invalid argument to ScriptableMap(): " + obj);
85          }
86          this.javaObject = this.map;
87          this.staticType = this.map.getClass();
88          initMembers();
89          initPrototype(scope);
90  
91      }
92  
93      public ScriptableMap(Scriptable scope, Map map) {
94          super(scope, map, map.getClass());
95          this.map = map;
96          initPrototype(scope);
97      }
98  
99      /**
100      * Set the prototype to the Object prototype so we can use object methods such as
101      * getOwnPropertyNames, hasOwnProperty, keys etc.
102      * @param scope the global scope for looking up the Object constructor
103      */
104     protected void initPrototype(Scriptable scope) {
105         Scriptable objectProto = ScriptableObject.getClassPrototype(scope, "Object");
106         if (objectProto != null) {
107             this.setPrototype(objectProto);
108         }
109     }
110 
111     public Object get(String name, Scriptable start) {
112         if (map == null || (reflect && super.has(name, start))) {
113             return super.get(name, start);
114         }
115         return getInternal(name);
116     }
117 
118     public Object get(int index, Scriptable start) {
119         if (map == null) {
120             return super.get(index, start);
121         }
122         return getInternal(new Integer(index));
123     }
124 
125     private Object getInternal(Object key) {
126         Object value = map.get(key);
127         if (value == null) {
128             return Scriptable.NOT_FOUND;
129         }
130         return ScriptUtils.javaToJS(value, getParentScope());
131     }
132 
133     public boolean has(String name, Scriptable start) {
134         if (map == null || (reflect && super.has(name, start))) {
135             return super.has(name, start);
136         } else {
137             return map.containsKey(name);
138         }
139     }
140 
141     public boolean has(int index, Scriptable start) {
142         if (map == null) {
143             return super.has(index, start);
144         } else {
145             return map.containsKey(new Integer(index));
146         }
147     }
148 
149     public void put(String name, Scriptable start, Object value) {
150         if (map == null || (reflect && super.has(name, start))) {
151             super.put(name, start, value);
152         } else {
153             putInternal(name, value);
154         }
155     }
156 
157     public void put(int index, Scriptable start, Object value) {
158         if (map == null) {
159              super.put(index, start, value);
160          } else {
161              putInternal(new Integer(index), value);
162         }
163     }
164 
165     @SuppressWarnings("unchecked")
166     private void putInternal(Object key, Object value) {
167         try {
168             map.put(key, ScriptUtils.jsToJava(value));
169         } catch (RuntimeException e) {
170             Context.throwAsScriptRuntimeEx(e);
171         }
172     }
173 
174     public void delete(String name) {
175         if (map != null) {
176             try {
177                 map.remove(name);
178             } catch (RuntimeException e) {
179                 Context.throwAsScriptRuntimeEx(e);
180             }
181         } else {
182             super.delete(name);
183         }
184     }
185 
186     public void delete(int index) {
187         if (map != null) {
188             try {
189                 map.remove(new Integer(index));
190             } catch (RuntimeException e) {
191                 Context.throwAsScriptRuntimeEx(e);
192             }
193         } else {
194             super.delete(index);
195         }
196     }
197 
198     public Object[] getIds() {
199         if (map == null) {
200             return super.getIds();
201         } else {
202             return map.keySet().toArray();
203         }
204     }
205 
206     public String toString() {
207         if (map == null)
208             return super.toString();
209         return map.toString();
210     }
211 
212     public Object getDefaultValue(Class typeHint) {
213         return toString();
214     }
215 
216     public Object unwrap() {
217         return map;
218     }
219 
220     public Map getMap() {
221         return map;
222     }
223 
224     public String getClassName() {
225         return CLASSNAME;
226     }
227 }
