21  	@Override
22  	public Bindings toScriptBindings(ScriptEngine engine,
23  			ScriptContext newContext, Bindings bindings, int engineScope) {
24  
25  		// return the same bindings and add a few graaljs-specific properties
26  		// see https://github.com/graalvm/graaljs/blob/master/docs/user/ScriptEngine.md
27  		bindings.put("polyglot.js.allowAllAccess", true);
28  		
29  		// return bindings;
30  		// Bindings newBindings = engine.createBindings();
31  		// bindings.put("polyglot.js.allowAllAccess", true);
32  		// this fails with 'failed to set graal-js option "polyglot.js.allowAllAccess": js context is already initialized'
33  		
34  		Set<String> keySet = new TreeSet<String>(bindings.keySet());
35  		for (String k : keySet) {
36  			Object v = bindings.get(k);
37  			bindings.put(k, toScriptObject(v)); 
38  		}
39  		return bindings;
40  	}
41  	
42  	// wrap Maps and Lists in ProxyObject
43  	@SuppressWarnings({ "rawtypes", "unchecked" })
44  	public Object toScriptObject(Object o) {
45  		Object result;
46  		
47  		if (o == null) {
48  			return null;
49  		
50  		} else if (o instanceof Map) {
51  			Map javaMap = (Map) o;
52  			Map jsMap = new HashMap<String, Object>();
53  			
54  			for (Object entryObj : javaMap.entrySet()) {
55  				Map.Entry entry = (Map.Entry) entryObj;
56  				jsMap.put((String) entry.getKey(), toScriptObject(entry.getValue()));
57  			}
58  			result = ProxyObject.fromMap(jsMap);
59  
60  		} else if (o instanceof List) {
61  			// return o;
62  			List javaList = (List) o;
63  			List newList = new ArrayList(javaList.size()); 
64  			for (int i=0; i < javaList.size(); i++) {
65  				newList.add(toScriptObject(javaList.get(i)));
66  			}
67  			result = newList;
68  
69  		// possibly array types here
70  		} else if (o instanceof String) { 
71  			result = o;
72  		
73  		} else {
74  			result = o; 
75  		}
76  		return result;
77  	}
