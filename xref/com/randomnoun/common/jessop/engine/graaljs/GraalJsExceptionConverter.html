3   import javax.script.ScriptContext;
4   import javax.script.ScriptException;
5   
6   import com.randomnoun.common.jessop.JessopExceptionConverter;
7   
8   public class GraalJsExceptionConverter implements JessopExceptionConverter {
9   
10  	@Override
11  	public ScriptException toScriptException(ScriptContext scriptContext, Throwable t) {
12  		String filename = null;
13  		Integer lineNumber = null;
14  		String innerMessage = null;
15  		if (t instanceof ScriptException) {
16  			// try to find a stack trace element with "<js>" in it
17  			Throwable chain = t;
18  			while (chain != null && filename == null) {
19  				for (StackTraceElement ste : chain.getStackTrace()) {
20  					if (ste.getClassName().equals("<js>")) {
21  						innerMessage = chain.getMessage();
22  						filename = ste.getFileName();
23  						lineNumber = ste.getLineNumber();
24  						break;
25  					}
26  				}
27  				chain = chain.getCause();
28  			}
29  			if (filename != null) {
30  				return (ScriptException) new ScriptException(innerMessage, filename, lineNumber == null ? -1 : lineNumber).initCause(t);
31  			} else {
32  				return (ScriptException) t;
33  			}
34  		}
35  		return (ScriptException) new ScriptException(t.getMessage()).initCause(t);
36  	}
37  
38  }
