14  	@Override
15  	public Bindings toScriptBindings(ScriptEngine engine,
16  			ScriptContext newContext, Bindings bindings, int engineScope) {
17  		Context cx = Context.enter();
18  		try {
19  			// so hopefully initStandardObjects() is less expensive than cloning all these maps/lists
20  			// this will be a different scope to that used at runtime, which hopefully won't be a problem later
21  			ScriptableObject newScope = cx.initStandardObjects();
22  			Bindings newBindings = engine.createBindings();
23  			for (String k : bindings.keySet()) {
24  				Object v = bindings.get(k);
25  				newBindings.put(k, ScriptUtils.javaToJS(v, newScope)); 
26  				// newBindings.put(k, toScriptObject(newScope, v));
27  			}
28  			return newBindings;
29  		} finally {
30  			Context.exit();
31  		}
32  	}
