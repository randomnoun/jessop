017    public HashMapIgnoreCaseTest(String name) {

020
021    /**
022     * Test the map methods :)
023     */
024    public void testMap() {
025                HashMapIgnoreCase<String, String> map = new HashMapIgnoreCase<String, String>();
026        map.put("abc", "def");
027        map.put("AbC", "ghi");
028        assertEquals(1, map.size());
029        assertEquals(true, map.containsKey("abc"));
030                assertEquals(true, map.containsKey("ABc"));
031                assertEquals(false, map.containsKey("ghi"));
032                
033        assertEquals("ghi", map.get("abc"));
034                assertEquals("ghi", map.get("AbC"));
035                assertEquals("ghi", map.get("ABC"));
036                assertEquals(null, map.get("ghi"));
037                
038                assertEquals("ghi", map.remove("ABC"));
039                assertEquals(0, map.size());
040    }
041    
042        public static void main(String[] args) {
043                junit.textui.TestRunner.run(new HashMapIgnoreCaseTest("testMap"));
044        }
