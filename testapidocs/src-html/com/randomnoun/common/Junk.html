015public class Junk {
016    
017        // fake CVS revision id. Even though we're on git, don't remove this or the unit tests will fail
018    public static final String _revision = "$Id: Junk.java,v 1.0 2017/10/06 00:00:00 knoxg Exp $";
019        
020    static void a() throws HighLevelException {
021        try {
022            b();
023        } catch(MidLevelException e) {
024            throw new HighLevelException(e);
025        }
026    }
027     
028    static void b() throws MidLevelException {
029        c();
030    }
031     
032    static void c() throws MidLevelException {
033        try {
034            d();
035        } catch(LowLevelException e) {
036            throw new MidLevelException(e);
037        }
038    }
039     
040    static void d() throws LowLevelException {
041       e();
042    }
043     
044    static void e() throws LowLevelException {
045        throw new LowLevelException();
046    }
047    
048    public static class InnerClassB1 {
049        // we normally wouldn't revision inner classes
050        // public static String _revision = "Test revision B1";
051
052        public static void f() throws LowLevelException {
053            InnerClassB2.g();
054        }
055
056        public static class InnerClassB2 {
057                // we normally wouldn't revision inner classes
058            // public static String _revision = "Test revision B2";
059
060            public static void g() throws LowLevelException {
061                throw new LowLevelException();
062            }
063        }
064    }
065}

067/** Utility exception class */
068class HighLevelException extends Exception {
069    HighLevelException(Throwable cause) { super(cause); }
070}
071
072/** Utility exception class */
073class MidLevelException extends Exception {
074    MidLevelException(Throwable cause)  { super("<middle>", cause); }
075}
