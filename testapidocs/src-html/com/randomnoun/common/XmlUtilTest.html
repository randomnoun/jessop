038        private String getResourceAsString(String name) throws IOException {
039                ByteArrayOutputStream baos = new ByteArrayOutputStream();
040                InputStream is = XmlUtilTest.class.getResourceAsStream("./" + name);
041                if (is==null) { 
042                        is = XmlUtilTest.class.getResourceAsStream(name);
043                        if (is==null) {
044                                throw new IllegalStateException("Missing resource"); 
045                        }
046                }
047                int ch = is.read(); while (ch!=-1) { baos.write(ch); ch=is.read(); }
048                is.close();
049                return baos.toString();
050        }

052        public void testGetCleanXml() throws IOException, SAXException {
053                String input, output;           
054                input = "<html>So here's <br>some HTML with<p>some <b>unclosed <i>formatting</p>tags.";
055                output = XmlUtil.getCleanXml(input, false); // apply HTML rules
056                assertEquals("<html xmlns=\"http://www.w3.org/1999/xhtml\"><body>So here's <br clear=\"none\"></br>some HTML with<p>some <b>unclosed <i>formatting</i></b></p><b><i>tags.</i></b></body></html>", output.trim());
057                
058                output = XmlUtil.getCleanXml(input, true); // apply XML rules
059                assertEquals("<html xmlns=\"http://www.w3.org/1999/xhtml\"><body>So here's <br clear=\"none\">some HTML with<p>some <b>unclosed <i>formatting</i></b></p><b><i>tags.</i></b></body></html>", output.trim());
060                
061                // clean the 'extreme' HTML from http://ccil.org/~cowan/XML/tagsoup/extreme.html
062                input = getResourceAsString("/tagsoupInput.txt");
063                output = XmlUtil.getCleanXml(input, true);
064                String expected = getResourceAsString("/tagsoupOutput.txt"); expected=expected.replaceAll("\r\n",  "\n");
065                assertEquals(expected, output);

080        public void testGetTextPreserveElements() throws SAXException {
081                String input, output;
082                Document d;
083                
084                input = "<p>Here is some <b>bold text</b> and <i>some <u>underlined</u> italics</i> text</p>";
085                d = XmlUtil.toDocument(input);
086                Element paraEl = d.getDocumentElement(); // top-level element is a paragraph element
087                output = XmlUtil.getTextPreserveElements(paraEl, new String[] { "b", "i", "u" } );
088                assertEquals("Here is some <b>bold text</b> and <i>some <u>underlined</u> italics</i> text", output);
089                
090                output = XmlUtil.getTextPreserveElements(paraEl, new String[] { "b", "i" } );
091                assertEquals("Here is some <b>bold text</b> and <i>some underlined italics</i> text", output);
092                
093                output = XmlUtil.getTextPreserveElements(paraEl, new String[] { "i" } );
094                assertEquals("Here is some bold text and <i>some underlined italics</i> text", output);
095
096                // attributes are preserved, but may be reordered
097                // self-closing elements are expanded to an opening and closing element
098                input = "<p>And this would be a paragraph with two images in it: <img src=\"src1.png\" /> and <img src=\"src2.png\"></img></p>";
099                d = XmlUtil.toDocument(input);
100                paraEl = d.getDocumentElement(); // top-level element is a paragraph element
101                output = XmlUtil.getTextPreserveElements(paraEl, new String[] { "img" } );
102                assertEquals("And this would be a paragraph with two images in it: <img src=\"src1.png\"></img> and <img src=\"src2.png\"></img>", output);
103                
104        }

122        public void testToDocumentInputStream() throws SAXException, TransformerException {
123                String input, output;
124                InputStream is;
125                Document d;
126                
127                // DOCTYPEs are discarded
128                input = "<!DOCTYPE something><p>XML with a DOCTYPE</p>";
129                is = new ByteArrayInputStream(input.getBytes());
130                d = XmlUtil.toDocument(is);
131                output = XmlUtil.getXmlString(d.getDocumentElement(), false);
132                assertEquals("<?xml version=\"1.0\" encoding=\"UTF-8\"?><p>XML with a DOCTYPE</p>", output);
133
134                // XML with namespaces
135                input = "<root xmlns:dc=\"http://purl.org/dc/terms/\"" +
136          " xmlns:html=\"http://www.w3.org/1999/xhtml\">" +
137                  "<html:p><dc:something>dublin core dublin core</dc:something>" +
138                  "Well this should be aware of namespaces then</html:p></root>";
139                is = new ByteArrayInputStream(input.getBytes());
140                d = XmlUtil.toDocument(is); 
141                output = XmlUtil.getXmlString(d.getDocumentElement(), false);
142                assertEquals("<?xml version=\"1.0\" encoding=\"UTF-8\"?>" + input, output);
143                
144        }

164        public void testCompact() throws ParserConfigurationException, SAXException, TransformerException {
165                String input =
166                        "<body>\n" +
167                    "  <el>  This is an element\n" +
168                        "    <el2> This is another one</el2>\n" +
169                    "    <el3>and another <!-- with some comments --></el3>\n" +
170                        "  </el>\n" +
171                    "</body>";
172                Document d = XmlUtil.toDocument(input);
173                String output = XmlUtil.getXmlString(d.getDocumentElement(), true);
174                assertEquals(input.replaceAll("\n", System.getProperty("line.separator")), output);
175                XmlUtil.compact(d.getDocumentElement());
176                output = XmlUtil.getXmlString(d.getDocumentElement(), true);
177                assertEquals("<body><el>This is an element<el2>This is another one</el2>" +
178                  "<el3>and another<!-- with some comments --></el3></el></body>", output);
179                                
180        }

199        /** A test class that extends XmlUtil.AbstractStackContentHandler.
200         * The "real" DeviceContentHandler populates DeviceTO and DevicePropertyTO objects here, 
201         * but for the purposes of this unit test, I'm just storing the result in 
202         * structured Lists of Maps.   
203         */
204        public static class DeviceContentHandler extends XmlUtil.AbstractStackContentHandler {
205                List<Map<String, Object>> result = new ArrayList<Map<String, Object>>();
206                Map<String, Object> d = null;  // current device
207                Map<String, String> prop = null;  // current property
208                
209                // the 'path' variable is maintained by the AbstractStackContentHandler as a 
210                // slash-delimited path of the current element within the XML document. 
211
212                /** paths that match this pattern define attributes of a device */
213                Pattern p1 = Pattern.compile("^devices/device/(name|className|type|active|universeNumber)$");
214                
215                /** stacks that match this pattern define attributes of a deviceProperty */
216                Pattern p2 = Pattern.compile("^devices/device/deviceProperties/deviceProperty/(key|value)$");
217                
218                /** process the start of an XML element */
219                public void element(String path) throws SAXException {
220                        if (path.equals("devices/device")) {
221                                d = new HashMap<String, Object>();
222                                d.put("deviceProperties", new ArrayList<Map<String, String>>());;
223                                result.add(d);
224                        } else if (path.equals("devices/device/deviceProperties")) {
225                                // 
226                        } else if (path.equals("devices/device/deviceProperties/deviceProperty")) {
227                                prop = new HashMap<String, String>();
228                                ((List) d.get("deviceProperties")).add(prop);
229                        }
230                }
231                
232                /** process the text of an XML element */
233                public void elementText(String path, String content) throws SAXException {
234                        Matcher m1 = p1.matcher(path);
235                        if (m1.matches()) {
236                                d.put(m1.group(1), content);
237                        } else {
238                                Matcher m2 = p2.matcher(path);
239                                if (m2.matches()) {
240                                        prop.put(m2.group(1), content);
241                                }
242                        }
243                }
244        }

247                String input = getResourceAsString("/device.xml");
