032        public static void listEngines(){ // Note 1
033        ScriptEngineManager mgr = new ScriptEngineManager();
034        List<ScriptEngineFactory> factories = mgr.getEngineFactories();
035        for (ScriptEngineFactory factory: factories) {
036            System.out.println("ScriptEngineFactory Info");
037            String engName = factory.getEngineName();
038            String engVersion = factory.getEngineVersion();
039            String langName = factory.getLanguageName();
040            String langVersion = factory.getLanguageVersion();
041            System.out.println("\tScript Engine: " + engName + ":" + engVersion);
042            List<String> engNames = factory.getNames();
043            for(String name: engNames) {
044                System.out.println("\tEngine Alias: " + name);
045            }
046            System.out.println("\tLanguage: " + langName + ":" + langVersion);
047        }
048    }
