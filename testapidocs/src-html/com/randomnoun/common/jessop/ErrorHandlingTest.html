119        public void testJessopCompile() throws ScriptException {
120                ScriptEngine engine = new ScriptEngineManager().getEngineByName("jessop");
121                if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
122                engine.put(JessopScriptEngine.FILENAME, "test.jessop");
123                Bindings b = engine.createBindings();
124                b.put("name", "Baron von Count");
125                b.put("maxCount", 3);
126
127                Compilable compilableEngine = (Compilable) engine;
128                CompiledScript script;
129                try {
130                        script = compilableEngine.compile(JAVSCRIPT_COUNTING_SCRIPT);
131                } catch (ScriptException se) { // java.scriptx engine 'rhino' not found
132                        script = compilableEngine.compile(JAVSCRIPT_COUNTING_SCRIPT_NASHORN);
133                }
134                
135                // compilation should still succeed
136                JessopCompiledScript jessopScript = (JessopCompiledScript) script;
137                logger.info("Start source");
138                logger.info(jessopScript.getSource());
139                logger.info("End source");
140                
141        }

143        public void _testScript(String input, int expectedLine) {
144                ScriptEngine engine = new ScriptEngineManager().getEngineByName("jessop");
145                if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
146                engine.put(JessopScriptEngine.FILENAME, "test.jessop");
147                Bindings b = engine.createBindings();
148                b.put(JessopScriptEngine.FILENAME, "test.jessop"); // ScriptContext.ENGINE_SCOPE
149                b.put("name", "Baron von Count");
150                b.put("maxCount", 3);
151                try {
152                        logger.info("Start eval");
153                        engine.eval(input, b);
154                        fail("script runtime exception expected");
155                } catch (ScriptException e) {
156                        logger.info("caught ScriptException");
157                        assertEquals("test.jessop", e.getFileName());
158                        assertEquals(expectedLine, e.getLineNumber());
159                }
160        }

163                boolean hasRhino = false;
164                try { 
165                        hasRhino = new ScriptEngineManager().getEngineByName("rhino") != null;
166                } catch (Exception e) {
167                        // ignore
168                }
169                if (hasRhino) {
170                        _testScript(JAVSCRIPT_COUNTING_SCRIPT, 6);
171                } else {
172                        _testScript(JAVSCRIPT_COUNTING_SCRIPT_NASHORN, 6);
173                }
174        }
