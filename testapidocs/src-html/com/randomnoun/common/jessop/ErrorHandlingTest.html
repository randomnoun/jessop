144        public void _testScript(String input, int expectedLine) {
145                ScriptEngine engine = new ScriptEngineManager().getEngineByName("jessop");
146                if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
147                engine.put(JessopScriptEngine.FILENAME, "test.jessop");
148                Bindings b = engine.createBindings();
149                b.put(JessopScriptEngine.FILENAME, "test.jessop"); // ScriptContext.ENGINE_SCOPE
150                b.put("name", "Baron von Count");
151                b.put("maxCount", 3);
152                try {
153                        logger.info("Start eval");
154                        engine.eval(input, b);
155                        fail("script runtime exception expected");
156                } catch (ScriptException e) {
157                        logger.info("caught ScriptException", e);
158                        // e.printStackTrace();
159                        assertEquals("test.jessop", e.getFileName());
160                        assertEquals(expectedLine, e.getLineNumber());
161                }
162        }

164        // @TODO better support for multiple js engines in the same JVM
165        public void testJessopJavascript() throws ScriptException {
166                boolean hasGraalvm = false;
167                boolean hasNashorn = false;
168                try { 
169                        hasGraalvm = new ScriptEngineManager().getEngineByName("graal-js") != null;
170                } catch (Exception e) {
171                        // ignore
172                }
173                try { 
174                        hasNashorn = new ScriptEngineManager().getEngineByName("nashorn") != null;
175                } catch (Exception e) {
176                        // ignore
177                }
178                if (hasGraalvm) {
179                        _testScript(JAVSCRIPT_COUNTING_SCRIPT_GRAALVM, 6);
180                } else if (hasNashorn) {
181                        _testScript(JAVSCRIPT_COUNTING_SCRIPT_NASHORN, 6);
182                } else {
183                        _testScript(JAVSCRIPT_COUNTING_SCRIPT_RHINO, 6);
184                }
185        }
