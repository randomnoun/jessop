124        private String getSource(ScriptEngine engine, String jessopSource) throws ScriptException {
125                Compilable compilable = (Compilable) engine;
126                JessopCompiledScript compiledScript = (JessopCompiledScript) compilable.compile(jessopSource);
127                return compiledScript.getSource();
128        }

130    /**
131     * Returns the HTML-escaped form of a string. The <tt>&amp;</tt>,
132     * <tt>&lt;</tt>, <tt>&gt;</tt>, and <tt>"</tt> characters are converted to
133     * <tt>&amp;amp;</tt>, <tt>&amp;lt;<tt>, <tt>&amp;gt;<tt>, and
134     * <tt>&amp;quot;</tt> respectively.
135     *
136     * @param string the string to convert
137     *
138     * @return the HTML-escaped form of the string
139     */
140    static public String escapeHtml(String string) {
141        if (string == null) {
142            return "";
143        }
144
145        char c;
146        StringBuilder sb = new StringBuilder(string.length());
147
148        for (int i = 0; i < string.length(); i++) {
149            c = string.charAt(i);
150
151            switch (c) {
152                case '&':
153                    sb.append("&amp;");
154                    break;
155                case '<':
156                    sb.append("&lt;");
157                    break;
158                case '>':
159                    sb.append("&gt;");
160                    break;
161                case '\"':
162                    // interestingly, &quote; (with the e) works fine for HTML display,
163                    // but not inside hidden field values
164                    sb.append("&quot;");
165                    break;
166                default:
167                    sb.append(c);
168            }
169        }
170
171        return sb.toString();
172    }

196    public void _testScript(String input) throws ScriptException {
197                ScriptEngine engine = new ScriptEngineManager().getEngineByName("jessop");
198                if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
199                Bindings b = getBindings(engine);
200                
201                logger.info("jessop input: " + escapeHtml(input));
202                logger.info("target language source: " + getSource(engine, input));
203                logger.info("Start eval");
204                engine.eval(input, b);
205                logger.info("End eval");
206    }
