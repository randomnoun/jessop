051        public final static String JAVASCRIPT_COUNTING_SCRIPT = 
052          // "<%@ jessop language=\"javascript\" engine=\"rhino\" %>\n" +
053          "Hello, <%= name %>\n" +
054          "<% for (var i = 1; i < maxCount; i++) { %>\n" +
055          "<%= i %>\n" +
056          "<% } %>";

058        
059        private String getSource(ScriptEngine engine, String jessopSource) throws ScriptException {
060                Compilable compilable = (Compilable) engine;
061                JessopCompiledScript compiledScript = (JessopCompiledScript) compilable.compile(jessopSource);
062                return compiledScript.getSource();
063        }
064        
065    /**
066     * Returns the HTML-escaped form of a string. The <tt>&amp;</tt>,
067     * <tt>&lt;</tt>, <tt>&gt;</tt>, and <tt>"</tt> characters are converted to
068     * <tt>&amp;amp;</tt>, <tt>&amp;lt;<tt>, <tt>&amp;gt;<tt>, and
069     * <tt>&amp;quot;</tt> respectively.
070     *
071     * @param string the string to convert
072     *
073     * @return the HTML-escaped form of the string
074     */
075    static public String escapeHtml(String string) {
076        if (string == null) {
077            return "";
078        }
079
080        char c;
081        StringBuilder sb = new StringBuilder(string.length());
082
083        for (int i = 0; i < string.length(); i++) {
084            c = string.charAt(i);
085
086            switch (c) {
087                case '&':
088                    sb.append("&amp;");
089                    break;
090                case '<':
091                    sb.append("&lt;");
092                    break;
093                case '>':
094                    sb.append("&gt;");
095                    break;
096                case '\"':
097                    // interestingly, &quote; (with the e) works fine for HTML display,
098                    // but not inside hidden field values
099                    sb.append("&quot;");
100                    break;
101                default:
102                    sb.append(c);
103            }
104        }
105
106        return sb.toString();
107    }

110        public void testJessopJavascript() throws ScriptException {
111                String input = JAVASCRIPT_COUNTING_SCRIPT; 
112                ScriptEngine engine = new ScriptEngineManager().getEngineByName("jessop");
113                if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
114                Bindings b = engine.createBindings();
115                b.put("name", "Baron von Count");
116                b.put("maxCount", 4);
117                
118                logger.info("jessop input: " + escapeHtml(input));
119                logger.info("target language source: " + getSource(engine, input));
120                logger.info("Start eval");
121                engine.eval(input, b);
122                logger.info("End eval");
123        }

125        public void testJessopCompile() throws ScriptException {
126                ScriptEngine engine = new ScriptEngineManager().getEngineByName("jessop");
127                if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
128                Bindings b = engine.createBindings();
129                b.put("name", "Baron von Count");
130                b.put("maxCount", 4);
131
132                Compilable compilableEngine = (Compilable) engine;
133                CompiledScript script = compilableEngine.compile(JAVASCRIPT_COUNTING_SCRIPT);
134                
135                JessopCompiledScript jessopScript = (JessopCompiledScript) script;
136                logger.info("Start source");
137                logger.info(jessopScript.getSource());
138                logger.info("End source");
139        }
