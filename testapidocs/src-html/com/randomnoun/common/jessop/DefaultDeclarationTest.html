042        public final static String JAVASCRIPT_COUNTING_SCRIPT = 
043          // "<%@ jessop language=\"javascript\" engine=\"rhino\" %>\n" +
044          "Hello, <%= name %>\n" +
045          "<% for (var i = 1; i < maxCount; i++) { %>\n" +
046          "<%= i %>\n" +
047          "<% } %>";
048
049        
050        private String getSource(ScriptEngine engine, String jessopSource) throws ScriptException {
051                Compilable compilable = (Compilable) engine;
052                JessopCompiledScript compiledScript = (JessopCompiledScript) compilable.compile(jessopSource);
053                return compiledScript.getSource();
054        }
055        
056    /**
057     * Returns the HTML-escaped form of a string. The <tt>&amp;</tt>,
058     * <tt>&lt;</tt>, <tt>&gt;</tt>, and <tt>"</tt> characters are converted to
059     * <tt>&amp;amp;</tt>, <tt>&amp;lt;<tt>, <tt>&amp;gt;<tt>, and
060     * <tt>&amp;quot;</tt> respectively.
061     *
062     * @param string the string to convert
063     *
064     * @return the HTML-escaped form of the string
065     */
066    static public String escapeHtml(String string) {
067        if (string == null) {
068            return "";
069        }
070
071        char c;
072        StringBuilder sb = new StringBuilder(string.length());
073
074        for (int i = 0; i < string.length(); i++) {
075            c = string.charAt(i);
076
077            switch (c) {
078                case '&':
079                    sb.append("&amp;");
080                    break;
081                case '<':
082                    sb.append("&lt;");
083                    break;
084                case '>':
085                    sb.append("&gt;");
086                    break;
087                case '\"':
088                    // interestingly, &quote; (with the e) works fine for HTML display,
089                    // but not inside hidden field values
090                    sb.append("&quot;");
091                    break;
092                default:
093                    sb.append(c);
094            }
095        }
096
097        return sb.toString();
098    }
099
100    
101        public void testJessopJavascript() throws ScriptException {
102                String input = JAVASCRIPT_COUNTING_SCRIPT; 
103                ScriptEngine engine = new ScriptEngineManager().getEngineByName("jessop");
104                if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
105                Bindings b = engine.createBindings();
106                b.put("name", "Baron von Count");
107                b.put("maxCount", 4);
108                
109                logger.info("jessop input: " + escapeHtml(input));
110                logger.info("target language source: " + getSource(engine, input));
111                logger.info("Start eval");
112                engine.eval(input, b);
113                logger.info("End eval");
114        }

116        public void testJessopCompile() throws ScriptException {
117                ScriptEngine engine = new ScriptEngineManager().getEngineByName("jessop");
118                if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
119                Bindings b = engine.createBindings();
120                b.put("name", "Baron von Count");
121                b.put("maxCount", 4);
122
123                Compilable compilableEngine = (Compilable) engine;
124                CompiledScript script = compilableEngine.compile(JAVASCRIPT_COUNTING_SCRIPT);
125                
126                JessopCompiledScript jessopScript = (JessopCompiledScript) script;
127                logger.info("Start source");
128                logger.info(jessopScript.getSource());
129                logger.info("End source");
130        }
