001package com.randomnoun.common.jessop;
002import javax.script.*;
003
004
005// adapted from JSR223Example in the abcl distribution
006
007public class LispTest {
008
009  public static void main(String[] args) {
010      //Script Engine instantiation using ServiceProvider - this will
011      //look in the classpath for a file
012      //  /META-INF/services/javax.script.ScriptEngineFactory
013      //where the AbclScriptEngineFactory is registered
014              ScriptEngine lispEngine = new ScriptEngineManager().getEngineByExtension("lisp");
015        
016              //Alternatively, you can directly instantiate the script engine:
017             
018              //ScriptEngineManager scriptManager = new ScriptEngineManager();
019              //scriptManager.registerEngineExtension("lisp", new AbclScriptEngineFactory());
020              //ScriptEngine lispEngine = scriptManager.getEngineByExtension("lisp");
021        
022              //(thanks to Peter Tsenter for suggesting this)
023               
024            //Accessing variables
025            System.out.println();
026            System.out.println("*package* = " + lispEngine.get("*package*"));
027            Object someValue = new Object();
028            lispEngine.put("someVariable", someValue);
029            System.out.println("someVariable = " + lispEngine.get("someVariable"));
030            try {
031              // simple loop in lisp
032              // https://www.tutorialspoint.com/lisp/lisp_loops.htm
033          lispEngine.eval(
034                "(loop for a from 10 to 20\n" +
035                "do (print a)\n" +
036                ")");
037
038          lispEngine.eval(
039                "(loop for a from 10 to 20\n" +
040                "do (format t \"~a\" a )\n" +
041                "(format t \"!\")\n" +  // also in the loop
042                ")");
043                          
044                
045              // simple condition in lisp
046              // lispEngine.eval("however that works");
047          lispEngine.eval(
048                "(setq a 100)\n" +
049                        "(if (> a 20)\n" +
050                        "   (format t \"~% a is greater than 20\")\n" + 
051                        "   (format t \"~% a is less than 20\"))\n" +
052                        "(format t \"~% value of a is ~d \" a)\n");
053
054          // platform dependent
055          lispEngine.eval("(format t \"line 1 ending with lisp newline ~%\")\n");
056          System.out.println("===");
057          // syntax from https://common-lisp.net/project/armedbear/releases/1.3.3/abcl-1.3.3.pdf, p41, doesn't work
058          lispEngine.eval("(format t \"line 2 ending with inline unicode D #\\U000D\")\n");
059          System.out.println("===");
060          lispEngine.eval("(format t \"line 3 ending with inline unicode D A #\\U000D#\\U000A\")\n");
061          System.out.println("===");
062          lispEngine.eval("(format t \"line 3b without the hash D A \\U000D\\U000A\")\n");
063          // syntax from https://rosettacode.org/wiki/Terminal_control/Unicode_output
064          System.out.println("===");
065          lispEngine.eval("(format t \"line 4 ending with code-chars D A ~a~a\" (code-char #x000d) (code-char #x000a))\n");
066          System.out.println("===");
067          // this seems to work ok
068          lispEngine.eval("(format t \"line 5 ending with unicode chars D A ~a~a\" '#\\U000D '#\\U000A)\n");
069          System.out.println("===");
070          lispEngine.eval("(format t \"and\")(format t \"so\")");
071          System.out.println("===");
072          lispEngine.eval("(princ \"this\" *standard-output*)(princ \"that\" *standard-output*)");
073          System.out.println("===");
074                        
075          
076          // need to use ~~ to output a '~'
077          // probably some weirdness require to output a '#' as well
078          // so this'll probably be format out \"whatever\" in jessop then
079          // don't like the ~a syntax, would prefer to keep chars inline
080          
081          
082          
083          lispEngine.eval("(format t \"line 4 with \\\" embedded quote ~%\")\n");
084              
085              
086                
087              //Interpretation (also from streams)
088              lispEngine.eval("(defun hello (arg) (print (list arg someVariable)) (terpri))");
089             
090              //Direct function invocation
091              ((Invocable) lispEngine).invokeFunction("hello", "world");
092             
093              //Implementing a Java interface in Lisp
094              lispEngine.eval("(defun compare-to (&rest args) 42)");
095              Comparable<?> c = ((Invocable) lispEngine).getInterface(java.lang.Comparable.class);
096              System.out.println("compareTo: " + c.compareTo(null));
097             
098              //Compilation!
099              lispEngine.eval("(defmacro slow-compiling-macro (arg) (dotimes (i 1000000) (incf i)) `(print ,arg))");
100             
101              long millis = System.currentTimeMillis();
102              lispEngine.eval("(slow-compiling-macro 42)");
103              millis = System.currentTimeMillis() - millis;
104              System.out.println("interpretation took " + millis);
105             
106              millis = System.currentTimeMillis();
107              CompiledScript cs = ((Compilable) lispEngine).compile("(slow-compiling-macro 42)");
108              millis = System.currentTimeMillis() - millis;
109              System.out.println("compilation took " + millis);
110             
111              millis = System.currentTimeMillis();
112              cs.eval();
113              millis = System.currentTimeMillis() - millis;
114              System.out.println("evaluation took " + millis);
115        
116              millis = System.currentTimeMillis();
117              cs.eval();
118              millis = System.currentTimeMillis() - millis;
119              System.out.println("evaluation took " + millis);
120        
121              //Ecc. ecc.
122            } catch (NoSuchMethodException e) {
123              e.printStackTrace();
124            } catch (ScriptException e) {
125              e.printStackTrace();
126            }
127          }
128         
129        }
