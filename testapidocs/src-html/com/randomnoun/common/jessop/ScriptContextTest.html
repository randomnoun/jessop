095        private String getSource(ScriptEngine engine, String jessopSource) throws ScriptException {
096                Compilable compilable = (Compilable) engine;
097                JessopCompiledScript compiledScript = (JessopCompiledScript) compilable.compile(jessopSource);
098                return compiledScript.getSource();
099        }

101    /**
102     * Returns the HTML-escaped form of a string. The <tt>&amp;</tt>,
103     * <tt>&lt;</tt>, <tt>&gt;</tt>, and <tt>"</tt> characters are converted to
104     * <tt>&amp;amp;</tt>, <tt>&amp;lt;<tt>, <tt>&amp;gt;<tt>, and
105     * <tt>&amp;quot;</tt> respectively.
106     *
107     * @param string the string to convert
108     *
109     * @return the HTML-escaped form of the string
110     */
111    static public String escapeHtml(String string) {
112        if (string == null) {
113            return "";
114        }
115
116        char c;
117        StringBuilder sb = new StringBuilder(string.length());
118
119        for (int i = 0; i < string.length(); i++) {
120            c = string.charAt(i);
121
122            switch (c) {
123                case '&':
124                    sb.append("&amp;");
125                    break;
126                case '<':
127                    sb.append("&lt;");
128                    break;
129                case '>':
130                    sb.append("&gt;");
131                    break;
132                case '\"':
133                    // interestingly, &quote; (with the e) works fine for HTML display,
134                    // but not inside hidden field values
135                    sb.append("&quot;");
136                    break;
137                default:
138                    sb.append(c);
139            }
140        }
141
142        return sb.toString();
143    }

145        public Bindings getBindings(ScriptEngine engine) {
146                Bindings b = engine.createBindings();
147                b.put("name", "Baron von Count");
148                b.put("maxCount", 4);
149                return b;

152        public void _testScript(String input) throws ScriptException {
153                ScriptEngine engine = new ScriptEngineManager().getEngineByName("jessop");
154                if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
155                Bindings b = getBindings(engine);
156                
157                logger.info("jessop input: " + escapeHtml(input));
158                logger.info("target language source: " + getSource(engine, input));
159                logger.info("Start eval");
160                engine.eval(input, b);
161                logger.info("End eval");
162        }

164        public void testJessopCompile() throws ScriptException {
165                ScriptEngine engine = new ScriptEngineManager().getEngineByName("jessop");
166                if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
167                /*Bindings b =*/ getBindings(engine);
168
169                Compilable compilableEngine = (Compilable) engine;
170                CompiledScript script = compilableEngine.compile(JAVASCRIPT_COUNTING_SCRIPT);
171                
172                JessopCompiledScript jessopScript = (JessopCompiledScript) script;
173                logger.info("Start source");
174                logger.info(jessopScript.getSource());
175                logger.info("End source");
176                
177        }
