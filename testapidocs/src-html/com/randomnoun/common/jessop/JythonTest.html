027        public static void listEngines(){ // Note 1
028        ScriptEngineManager mgr = new ScriptEngineManager();
029        List<ScriptEngineFactory> factories =
030                mgr.getEngineFactories();
031        for (ScriptEngineFactory factory: factories) {
032            System.out.println("ScriptEngineFactory Info");
033            String engName = factory.getEngineName();
034            String engVersion = factory.getEngineVersion();
035            String langName = factory.getLanguageName();
036            String langVersion = factory.getLanguageVersion();
037            System.out.println("\tScript Engine: " + engName + ":" +
038                               engVersion);
039            List<String> engNames = factory.getNames();
040            for(String name: engNames) {
041                System.out.println("\tEngine Alias: " + name);
042            }
043            System.out.println("\tLanguage: " + langName + ":" +
044                               langVersion);
045        }
046    }
047        public void testJython() throws ScriptException {
048                // -Dpython.console.encoding=UTF-8
049                // see http://stackoverflow.com/questions/30443537/how-do-i-fix-unsupportedcharsetexception-in-eclipse-kepler-luna-with-jython-pyde
050                System.setProperty("python.console.encoding", "UTF-8");
051        
052                listEngines(); // Note 1
053                
054                // this is the same test code as on https://wiki.python.org/jython/LearningJython#using-script-engine-support
055                ScriptEngineManager mgr = new ScriptEngineManager(); // Note 2
056            ScriptEngine eng = mgr.getEngineByName("python");  
057
058            // except that eng here is null.
059            System.out.println("eng: " + String.valueOf(eng));
060            eng.put("var1", new Integer(257)); // Note 3
061            eng.eval("print 'var1: %s' % var1");
062            eng.eval("import sys"); // Note 4
063            eng.eval("print sys.version");
064        }
