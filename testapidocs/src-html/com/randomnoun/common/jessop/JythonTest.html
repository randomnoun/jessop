031        public static void listEngines(){ // Note 1
032        ScriptEngineManager mgr = new ScriptEngineManager();
033        List<ScriptEngineFactory> factories =
034                mgr.getEngineFactories();
035        for (ScriptEngineFactory factory: factories) {
036            System.out.println("ScriptEngineFactory Info");
037            String engName = factory.getEngineName();
038            String engVersion = factory.getEngineVersion();
039            String langName = factory.getLanguageName();
040            String langVersion = factory.getLanguageVersion();
041            System.out.println("\tScript Engine: " + engName + ":" +
042                               engVersion);
043            List<String> engNames = factory.getNames();
044            for(String name: engNames) {
045                System.out.println("\tEngine Alias: " + name);
046            }
047            System.out.println("\tLanguage: " + langName + ":" +
048                               langVersion);
049        }
050    }
051        public void testJython() throws ScriptException {
052                // -Dpython.console.encoding=UTF-8
053                // see http://stackoverflow.com/questions/30443537/how-do-i-fix-unsupportedcharsetexception-in-eclipse-kepler-luna-with-jython-pyde
054                System.setProperty("python.console.encoding", "UTF-8");
055        
056                listEngines(); // Note 1

058                // this is the same test code as on https://wiki.python.org/jython/LearningJython#using-script-engine-support
