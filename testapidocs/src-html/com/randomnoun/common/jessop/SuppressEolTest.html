037        public void setUp() {
038                String logFormatPrefix = "[SuppressEolTest] ";
039                Properties lp = new Properties();
040                lp.put("log4j.rootCategory", "INFO, CONSOLE");
041                lp.put("log4j.appender.CONSOLE", "org.apache.log4j.ConsoleAppender");
042                lp.put("log4j.appender.CONSOLE.layout", "org.apache.log4j.PatternLayout");
043                lp.put("log4j.appender.CONSOLE.layout.ConversionPattern", logFormatPrefix + "%d{ABSOLUTE} %-5p %c - %m%n");
044                lp.put("log4j.logger.org.springframework", "INFO"); // since Spring is a bit too verbose for my liking at DEBUG level

096        private String getSource(ScriptEngine engine, String jessopSource) throws ScriptException {
097                Compilable compilable = (Compilable) engine;
098                JessopCompiledScript compiledScript = (JessopCompiledScript) compilable.compile(jessopSource);
099                return compiledScript.getSource();
100        }
101        
102        public void testJessop1() throws ScriptException {
103                ScriptEngine engine = new ScriptEngineManager().getEngineByName("jessop");
104                if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
105                logger.info("Start eval");
106                engine.eval(JAVASCRIPT_COUNTING_SCRIPT);
107                logger.info("End eval");
108        }
109        
110        public void testJessopCompile() throws ScriptException {
111                Compilable engine = (Compilable) new ScriptEngineManager().getEngineByName("jessop");
112                if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
113                
114                CompiledScript script = engine.compile(JAVASCRIPT_COUNTING_SCRIPT);
115                
116                JessopCompiledScript jessopScript = (JessopCompiledScript) script;
117                logger.info("Start source");
118                logger.info(jessopScript.getSource());
119                logger.info("End source");
120                
121        }

124                ScriptEngine engine = new ScriptEngineManager().getEngineByName("jessop");
125                if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
126                logger.info("lua source: " + ((JessopCompiledScript) (((Compilable) engine).compile(LUA_COUNTING_SCRIPT))).getSource());

133        public void testJessopPython1() throws ScriptException {
134                // -Dpython.console.encoding=UTF-8
135                // see http://stackoverflow.com/questions/30443537/how-do-i-fix-unsupportedcharsetexception-in-eclipse-kepler-luna-with-jython-pyde
136                System.setProperty("python.console.encoding", "UTF-8");
137                
138                // can either specify the language here (e.g. jessop-rhino), or just 'jessop' to get language from the script itself
139                ScriptEngine engine = new ScriptEngineManager().getEngineByName("jessop");
140                if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
141                logger.info("python source: " + getSource(engine, PYTHON_COUNTING_SCRIPT_1));
142                
143                logger.info("Start eval");
144                engine.eval(PYTHON_COUNTING_SCRIPT_1);
145                logger.info("End eval");
146        }
