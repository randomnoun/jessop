121        private String getSource(ScriptEngine engine, String jessopSource) throws ScriptException {
122                Compilable compilable = (Compilable) engine;
123                JessopCompiledScript compiledScript = (JessopCompiledScript) compilable.compile(jessopSource);
124                return compiledScript.getSource();
125        }
126        
127        public void testJessopCompile() throws ScriptException {
128                Compilable engine = (Compilable) new ScriptEngineManager().getEngineByName("jessop");
129                if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
130                
131                CompiledScript script = engine.compile(JAVASCRIPT_COUNTING_SCRIPT);
132                
133                JessopCompiledScript jessopScript = (JessopCompiledScript) script;
134                logger.info("Start source");
135                logger.info(jessopScript.getSource());
136                logger.info("End source");
137                
138        }
139
140        public void _testScript(String input) throws ScriptException {
141                ScriptEngine engine = new ScriptEngineManager().getEngineByName("jessop");
142                if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
143                logger.info("jessop input: " + input);
144                logger.info("target language source: " + getSource(engine, input));
145                logger.info("Start eval");
146                engine.eval(input);
147                logger.info("End eval");
148        }
