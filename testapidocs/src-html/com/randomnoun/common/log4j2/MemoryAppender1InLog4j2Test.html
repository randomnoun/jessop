096        public void testMemoryAppenderViaObjectModel() {
097                Logger logger = Logger.getLogger("testLogger");
098                
099                
100                LoggerContext ctx = (LoggerContext) LogManager.getContext(false);
101        Configuration config = ctx.getConfiguration();
102        
103                config.getRootLogger().setLevel(org.apache.logging.log4j.Level.INFO);
104                
105                // one of those adapter thingies
106                MemoryAppender log4j1MemoryAppender = new MemoryAppender(1000);
107                // Log4j1WrapperAppender memoryAppender = Log4j1WrapperAppender.createAppender("Memory", log4j1MemoryAppender);  
108                
109                Log4j1WrapperAppender memoryAppender = Log4j1WrapperAppender.createAppender("Memory", log4j1MemoryAppender);  
110                
111                // MemoryAppender memoryAppender = MemoryAppender.createAppender("Memory");
112                // memoryAppender.start();
113        // config.addAppender(memoryAppender);
114        
115                
116        AppenderRef ref = AppenderRef.createAppenderRef("MEMORY", null, null);
117        AppenderRef[] refs = new AppenderRef[] {ref};
118        LoggerConfig loggerConfig = LoggerConfig.createLogger(false, org.apache.logging.log4j.Level.INFO, "testLogger", "true", refs, null, config, null );
119        loggerConfig.addAppender(memoryAppender, null, null);
120        config.addLogger("testLogger", loggerConfig);
121        ctx.updateLoggers();
122        
123                List<LoggingEvent> logEvents;
124
125                long start = System.currentTimeMillis();
126                logger.info("info message");
127                long end = System.currentTimeMillis();
128                
129                logEvents = log4j1MemoryAppender.getLoggingEvents();
130                assertTrue("info message in memoryAppender", logEvents.size()==1);
131                assertEquals("info message", logEvents.get(0).getMessage().toString());
132                assertEquals("info message", logEvents.get(0).getRenderedMessage());
133                assertNotNull(logEvents.get(0).getLoggerName());
134                assertEquals(Level.INFO, logEvents.get(0).getLevel());
135                
136                // the LoggingEvent class provided in the log4j-1.2-api jar 
137                // has a 'final' implementation of getTimeStamp() which returns 0, 
138                // so the LogEventAdapter class also returns zero.
139                // I mean, why would anyone want to know when a log message was generated
140                // Surprised they didn't have a final getMessage() that returned null as well,
141                // which would make logging all that more informative.
142                
143                // anyway, I've created my own LogEvent class that actually stores the TimeStamp, this will need to be 
144                // include ahead of log4j-1.2-api to take effect.
145                System.out.println(logEvents.get(0).getTimeStamp()); // was zero
146                System.out.println(start);
147                assertTrue("timestamp of loggingEvent >= start", logEvents.get(0).getTimeStamp() >= start);
148                assertTrue("timestamp of loggingEvent <= end", logEvents.get(0).getTimeStamp() <= end);
149                
150                
151                logger.debug("debug message");
152                logEvents = log4j1MemoryAppender.getLoggingEvents();
153                assertTrue("debug message suppressed from memoryAppender", logEvents.size()==1);
154        }

158        public void testLogWindowSize() {
159
160                // create a MemoryLogger capable of holding 10 entries
161                
162                Properties props = new Properties();
163                props.put("log4j.rootCategory", "INFO, MEMORY");
164                props.put("log4j.appender.MEMORY", "com.randomnoun.common.log4j.MemoryAppender");
165                props.put("log4j.appender.MEMORY.MaximumLogSize", "10");
166                // PropertyConfigurator.configure(props);
167                
168
169                Log4j1ConfigurationParser lcp = new Log4j1ConfigurationParser();
170                ConfigurationBuilder<?> builder = lcp.buildConfigurationBuilder(props);
171                LoggerContext ctx = Configurator.initialize(builder.build());
172
173                Logger logger = Logger.getLogger("testLogger");
174                Log4j1WrapperAppender wrapperAppender = ctx.getConfiguration().getAppender("MEMORY");
175                MemoryAppender memoryAppender = (MemoryAppender) wrapperAppender.getLog4j1Appender();
176                
177                List<LoggingEvent> logEvents;
178
179                // write five messages. 
180                // the 0th message in MemoryAppender should be the most recent (i.e. message #5)
181                for (int i=1; i<=5; i++) {
182                        logger.info("message number " + i);
183                }
184                logEvents = memoryAppender.getLoggingEvents();
185                assertTrue("5 info messages in memoryAppender", logEvents.size()==5);
186                for (int i=0; i<5; i++) {
187                        assertEquals("message number " + (5 - i), logEvents.get(i).getRenderedMessage());
188                }
189                
190                // write another five messages
191                for (int i=6; i<=10; i++) {
192                        logger.info("message number " + i);
193                }
194                logEvents = memoryAppender.getLoggingEvents();
195                assertTrue("10 info messages in memoryAppender", logEvents.size()==10);
196                for (int i=0; i<10; i++) {
197                        assertEquals("message number " + (10 - i), logEvents.get(i).getRenderedMessage());
198                }
199                
200                // write another five messages (memoryAppender should just contain messages 6-15)
201                for (int i=11; i<=15; i++) {
202                        logger.info("message number " + i);
203                }
204                logEvents = memoryAppender.getLoggingEvents();
205                //System.out.println(loggingEvents.size());
206                assertTrue("10 info messages in memoryAppender", logEvents.size()==10);
207                for (int i=0; i<10; i++) {
208                        assertEquals("message number " + (15 - i), logEvents.get(i).getRenderedMessage());
209                }
210                
211        }
