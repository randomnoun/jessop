047        public void testMemoryAppenderViaProperties_Log2Logger() {
048
049                Properties props = new Properties();
050                props.put("log4j.rootCategory", "INFO, MEMORY");
051                props.put("log4j.appender.MEMORY", "com.randomnoun.common.log4j2.MemoryAppender");
052                props.put("log4j.appender.MEMORY.MaximumLogSize", 100);
053                
054                // layouts have no effect on MemoryAppenders
055                //props.put("log4j.appender.MEMORY.layout", "org.apache.log4j.PatternLayout");
056                //props.put("log4j.appender.MEMORY.layout.ConversionPattern", "%d{dd/MM/yy HH:mm:ss.SSS} %-5p %c - %m%n");
057                // PropertyConfigurator.configure(props);
058                
059                Log4j2CliConfiguration lcc = new Log4j2CliConfiguration();
060                lcc.init("[MemoryAppenderTest]", props);
061                
062                Log4j1ConfigurationParser lcp = new Log4j1ConfigurationParser();
063                ConfigurationBuilder<?> builder = lcp.buildConfigurationBuilder(props);
064                LoggerContext ctx = Configurator.initialize(builder.build());
065
066                org.apache.logging.log4j.Logger logger = LogManager.getLogger("testLogger");
067                MemoryAppender memoryAppender = ctx.getConfiguration().getAppender("MEMORY");
068                
069                List<LogEvent> logEvents;
070
071                long start = System.currentTimeMillis();
072                logger.info("info message");
073                long end = System.currentTimeMillis();
074                
075                logEvents = memoryAppender.getLogEvents();
076                assertTrue("info message in memoryAppender", logEvents.size()==1);
077                assertEquals("info message", logEvents.get(0).getMessage().getFormattedMessage());
078                // assertEquals("info message", logEvents.get(0).getRenderedMessage());
079                assertNotNull(logEvents.get(0).getLoggerName());
080                assertEquals(Level.INFO, logEvents.get(0).getLevel());
081                // NB: timestamp field requires log4j 1.2.15
082                assertTrue("timestamp of loggingEvent >= start", logEvents.get(0).getTimeMillis() >= start);
083                assertTrue("timestamp of loggingEvent <= end", logEvents.get(0).getTimeMillis() <= end);
084                
085                logger.debug("debug message");
086                logEvents = memoryAppender.getLogEvents();
087                assertTrue("debug message suppressed from memoryAppender", logEvents.size()==1);
088                
089        }

134        public void testLogWindowSize() {
135
136                // create a MemoryLogger capable of holding 10 entries
137                
138                Properties props = new Properties();
139                props.put("log4j.rootCategory", "INFO, MEMORY");
140                props.put("log4j.appender.MEMORY", "com.randomnoun.common.log4j2.MemoryAppender");
141                props.put("log4j.appender.MEMORY.MaximumLogSize", "10");
142                // PropertyConfigurator.configure(props);
143                Log4j2CliConfiguration lcc = new Log4j2CliConfiguration();
144                lcc.init("[MemoryAppenderTest]", props);
145                
146                Log4j1ConfigurationParser lcp = new Log4j1ConfigurationParser();
147                ConfigurationBuilder<?> builder = lcp.buildConfigurationBuilder(props);
148                LoggerContext ctx = Configurator.initialize(builder.build());
149
150                Logger logger = Logger.getLogger("testLogger");
151                // MemoryAppender memoryAppender = (MemoryAppender) Logger.getRootLogger().getAppender("MEMORY");
152                MemoryAppender memoryAppender = ctx.getConfiguration().getAppender("MEMORY");
153                
154                List<LogEvent> logEvents;
155
156                // write five messages. 
157                // the 0th message in MemoryAppender should be the most recent (i.e. message #5)
158                for (int i=1; i<=5; i++) {
159                        logger.info("message number " + i);
160                }
161                logEvents = memoryAppender.getLogEvents();
162                assertTrue("5 info messages in memoryAppender", logEvents.size()==5);
163                for (int i=0; i<5; i++) {
164                        assertEquals("message number " + (5 - i), logEvents.get(i).getMessage().getFormattedMessage());
165                }
166                
167                // write another five messages
168                for (int i=6; i<=10; i++) {
169                        logger.info("message number " + i);
170                }
171                logEvents = memoryAppender.getLogEvents();
172                assertTrue("10 info messages in memoryAppender", logEvents.size()==10);
173                for (int i=0; i<10; i++) {
174                        assertEquals("message number " + (10 - i), logEvents.get(i).getMessage().getFormattedMessage());
175                }
176                
177                // write another five messages (memoryAppender should just contain messages 6-15)
178                for (int i=11; i<=15; i++) {
179                        logger.info("message number " + i);
180                }
181                logEvents = memoryAppender.getLogEvents();
182                //System.out.println(loggingEvents.size());
183                assertTrue("10 info messages in memoryAppender", logEvents.size()==10);
184                for (int i=0; i<10; i++) {
185                        assertEquals("message number " + (15 - i), logEvents.get(i).getMessage().getFormattedMessage());
186                }
187                
188        }
