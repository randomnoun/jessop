020        @Test
021        public void testParseStatements() throws IOException, ParseException {
022                SqlParser p = new SqlParser();
023                String s1 = "statement1; statement2; statement3; /* a comment */ statement /* inside */ 4;";
024                String[] result = p.parseStatements(new ByteArrayInputStream(s1.getBytes()), true).toArray(new String[]{});
025                String[] expected = new String[] { 
026                        "statement1", 
027                        "statement2", 
028                        "statement3", 
029                        "/* a comment */ statement /* inside */ 4" };
030                assertArrayEquals(expected,  result);
031                
032                result = p.parseStatements(new ByteArrayInputStream(s1.getBytes()), false).toArray(new String[]{});
033                expected = new String[] { 
034                        "statement1", 
035                        "statement2", 
036                        "statement3", 
037                        "statement  4" };
038                assertArrayEquals(expected,  result);
039                // System.out.println(Struct.structuredListToString("result",p.parseStatements(new ByteArrayInputStream(s1.getBytes()),false)));
040                
041                
042                String s2 = "--things\n" +
043                        "a multi-line -- more things\n" +
044                        "statement with -- yet more things\n" +
045                        "eol comments \"-- not this one /* or this */ of course--\" embedded within -- things?\n" +
046                        "it;";
047                result = p.parseStatements(new ByteArrayInputStream(s2.getBytes()), true).toArray(new String[]{});
048                expected = new String[] { 
049                        "-- things", 
050                        "-- more things", 
051                        "-- yet more things", 
052                        "-- things?", 
053                        "a multi-line statement with eol comments \"-- not this one /* or this */ of course--\" embedded within it" };
054                assertArrayEquals(expected,  result);
055
056                result = p.parseStatements(new ByteArrayInputStream(s2.getBytes()), false).toArray(new String[]{});
057                expected = new String[] { 
058                        "a multi-line statement with eol comments \"-- not this one /* or this */ of course--\" embedded within it" };
059                assertArrayEquals(expected,  result);
060
061                
062                String s3 = "a statement; followed by a statement not terminated with a semi-colon";
063                result = p.parseStatements(new ByteArrayInputStream(s3.getBytes()), true).toArray(new String[]{});
064                expected = new String[] { 
065                        "a statement",
066                        "followed by a statement not terminated with a semi-colon" };
067                assertArrayEquals(expected,  result);
068                
069                
070                try {
071                        String s4 = "a statement; followed by a statement with an unclosed \"quoted string";
072                        System.out.println(Struct.structuredListToString("result",p.parseStatements(new ByteArrayInputStream(s4.getBytes()), true)));
073                        fail("Expected ParseException");
074                } catch (ParseException pe) {
075                        // fine
076                }
077
078                try {
079                        String s5 = "a statement; followed by a statement with an unclosed /* comment";
080                        System.out.println(Struct.structuredListToString("result",p.parseStatements(new ByteArrayInputStream(s5.getBytes()), true)));
081                        fail("Expected ParseException");
082                } catch (ParseException pe) {
083                        // fine
084                }
085                
086                String s6 = "ending with semi-colon; delimiter $$; and; now $$ things are delimited by dollars $$ delimiter ;$$ back to semicolon; hopefully\n";
087                result = p.parseStatements(new ByteArrayInputStream(s6.getBytes()), true).toArray(new String[]{});
088                expected = new String[] { 
089                        "ending with semi-colon",
090                        "and; now",
091                        "things are delimited by dollars",
092                        "back to semicolon",
093                        "hopefully"
094                };
095                assertArrayEquals(expected,  result);
096                
097                
098                // the 'delimiter' command can also end with newlines without it's own delimiter
099                String s7 = "ending with semi-colon; delimiter $$\n" +
100                  "and; now $$ things are delimited by dollars $$ delimiter ;\n" +
101                  "back to semicolon; hopefully\n";
102                result = p.parseStatements(new ByteArrayInputStream(s7.getBytes()), true).toArray(new String[]{});
103                expected = new String[] { 
104                        "ending with semi-colon",
105                        "and; now",
106                        "things are delimited by dollars",
107                        "back to semicolon",
108                        "hopefully"
109                };
110                assertArrayEquals(expected,  result);
111                
112                String s8 = "statement1; a='escaped quotes (\\' and \\\") in single quotes'; statement3;";
113                result = p.parseStatements(new ByteArrayInputStream(s8.getBytes()), true).toArray(new String[]{});
114                expected = new String[] { 
115                        "statement1", 
116                        "a='escaped quotes (\\' and \\\") in single quotes'", 
117                        "statement3"};
118                assertArrayEquals(expected,  result);
119
120                String s9 = "statement1; a=\"escaped quotes (\\' and \\\") in double quotes\"; statement3;";
121                result = p.parseStatements(new ByteArrayInputStream(s9.getBytes()), true).toArray(new String[]{});
122                expected = new String[] { 
123                        "statement1", 
124                        "a=\"escaped quotes (\\' and \\\") in double quotes\"", 
125                        "statement3"};
126                assertArrayEquals(expected,  result);
127
128                
129        }
