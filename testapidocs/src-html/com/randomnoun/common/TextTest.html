131    {
132        
133
134        // single char replacement        
135        assertEquals("replaceString('', 'b', 'x')", "", Text.replaceString("", "b", "x"));
136        assertEquals("replaceString('abc', 'b', 'x')", "axc",
137            Text.replaceString("abc", "b", "x"));
138        assertEquals("replaceString('abc', 'd', 'x')", "abc",
139            Text.replaceString("abc", "d", "x"));
140        assertEquals("replaceString('abc', 'bc', 'x')", "ax",
141            Text.replaceString("abc", "bc", "x"));
142        assertEquals("replaceString('abababa', 'b', 'x')", "axaxaxa",
143            Text.replaceString("abababa", "b", "x"));
144        assertEquals("replaceString('abababa', 'bab', 'x')", "axaba",
145            Text.replaceString("abababa", "bab", "x"));
146
147        // multiple char replacement
148        assertEquals("replaceString('', 'b', 'fish')", "",
149            Text.replaceString("", "b", "x"));
150        assertEquals("replaceString('abc', 'b', 'fish')", "afishc",
151            Text.replaceString("abc", "b", "fish"));
152        assertEquals("replaceString('abc', 'd', 'fish')", "abc",
153            Text.replaceString("abc", "d", "fish"));
154        assertEquals("replaceString('abc', 'bc', 'fish')", "afish",
155            Text.replaceString("abc", "bc", "fish"));
156        assertEquals("replaceString('abababa', 'b', 'fish')", "afishafishafisha",
157            Text.replaceString("abababa", "b", "fish"));
158        assertEquals("replaceString('abababa', 'bab', 'fish')", "afishaba",
159            Text.replaceString("abababa", "bab", "fish"));
160
161        // replace with text already in original string
162        assertEquals("replaceString('', 'b', 'a')", "", Text.replaceString("", "b", "a"));
163        assertEquals("replaceString('abc', 'b', 'a')", "aac",
164            Text.replaceString("abc", "b", "a"));
165        assertEquals("replaceString('abc', 'd', 'a')", "abc",
166            Text.replaceString("abc", "d", "a"));
167        assertEquals("replaceString('abc', 'bc', 'a')", "aa",
168            Text.replaceString("abc", "bc", "a"));
169        assertEquals("replaceString('abababa', 'b', 'a')", "aaaaaaa",
170            Text.replaceString("abababa", "b", "a"));
171        assertEquals("replaceString('abababa', 'bab', 'a')", "aaaba",
172            Text.replaceString("abababa", "bab", "a"));
173    }

232    public void testCsvEscape(){
233        
234        assertEquals("\"'=CMD()\"", Text.escapeCsv("=CMD()"));
235        assertEquals("\"'=12345\"", Text.escapeCsv("=12345"));
236        assertEquals("\"abc,def\"", Text.escapeCsv("abc,def"));
237        assertEquals("\"'=DDE(\"\"cmd\"\";\"\"/C calc\"\";\"\"__DdeLink_60_870516294\"\")\"", Text.escapeCsv("=DDE(\"cmd\";\"/C calc\";\"__DdeLink_60_870516294\")"));
238        assertEquals("\"'=cmd|' /C calc'!A0\"", Text.escapeCsv("=cmd|' /C calc'!A0"));
239        assertEquals("\"'+cmd|' /C calc'!A0\"", Text.escapeCsv("+cmd|' /C calc'!A0"));
240        assertEquals("\"'-cmd|' /C calc'!A0\"", Text.escapeCsv("-cmd|' /C calc'!A0"));
241
242        // numbers shouldn't be escaped
243        assertEquals("123.456", Text.escapeCsv("123.456"));
244        assertEquals("123", Text.escapeCsv("123"));
245        assertEquals("-123.456", Text.escapeCsv("-123.456"));
246        assertEquals("-123", Text.escapeCsv("-123"));
247        
248        // and with E notation
249        assertEquals("123.456E+5", Text.escapeCsv("123.456E+5"));
250        assertEquals("123E+5", Text.escapeCsv("123E+5"));
251        assertEquals("-123.456E+5", Text.escapeCsv("-123.456E+5"));
252        assertEquals("-123E+5", Text.escapeCsv("-123E+5"));

305    public void testSplitEscapedPath() {
306        assertArrayEquals(new String[] { "abc" }, Text.splitEscapedPath("abc"));
307        assertArrayEquals(new String[] { "abc", "def" }, Text.splitEscapedPath("abc/def"));
308        assertArrayEquals(new String[] { "a:bc", "def" }, Text.splitEscapedPath("a:bc/def"));
309        assertArrayEquals(new String[] { "a:bc", "def" }, Text.splitEscapedPath("a\\:bc/def")); // /: unescapes to :
310        
311        assertArrayEquals(new String[] { "a:bc\\/def" }, Text.splitEscapedPath("a\\:bc\\\\\\/def")); // does not unescape or split escaped '/' characters
312        assertArrayEquals(new String[] { "a:bcd\\efg" }, Text.splitEscapedPath("a\\:bcd\\\\efg")); // does not unescape other characters
313        assertArrayEquals(new String[] { "a:bcdefg\\" }, Text.splitEscapedPath("a\\:bcdefg\\\\")); // does not unescape other characters
314
315        try {
316                Text.splitEscapedPath(null);
317                fail("Expected NullPointerException");
318        } catch (NullPointerException iae) {
319                // OK
320        }
321        assertArrayEquals(new String[] { "a:bc/def" }, Text.splitEscapedPath("a:bc\\/def"));
322        try {
323                Text.splitEscapedPath("a:bcd\\efg"); // unknown escape '\e'
324                fail("Expected IllegalArgumentException");
325        } catch (IllegalArgumentException iae) {
326                // OK
327        }
328        try {
329                Text.splitEscapedPath("a:bcdefg\\"); // unclosed escape
330        } catch (IllegalArgumentException iae) {
331                // OK
332        }

348    public void testParseCsv() throws ParseException {
349        assertEquals(Arrays.asList(new String[] { "abc" }), Text.parseCsv("abc"));
350        assertEquals(Arrays.asList(new String[] { "abc","def" }), Text.parseCsv("abc,def"));
351        assertEquals(Arrays.asList(new String[] { "", "abc", "" }), Text.parseCsv(",abc,"));
352        assertEquals(Arrays.asList(new String[] { "" }), Text.parseCsv(""));
353        assertEquals(Arrays.asList(new String[] { "quotes" }), Text.parseCsv("\"quotes\""));
354        assertEquals(Arrays.asList(new String[] { "q", "a,b", "c" }), Text.parseCsv("\"q\",\"a,b\",c"));
355        assertEquals(Arrays.asList(new String[] { "q", "a,\"b\",c", "d" }), Text.parseCsv("\"q\",\"a,\"\"b\"\",c\",d"));
356        assertEquals(Arrays.asList(new String[] { "nq", "q", "a,\"b\",c", "d" }), Text.parseCsv("nq,\"q\",\"a,\"\"b\"\",c\",d"));
357        assertEquals(Arrays.asList(new String[] { "nq", "q", "a,\"b\",c", "d" }), Text.parseCsv("nq   ,   \"q\",   \"a,\"\"b\"\",c\"   ,d"));
358        // whitespace-sensitive
359        assertEquals(Arrays.asList(new String[] { "nq", "q", "a,\"b\",c", "d" }), Text.parseCsv("nq,\"q\",\"a,\"\"b\"\",c\",d", true));
360        assertEquals(Arrays.asList(new String[] { "nq   ", "   \"q\"", "  \"a  " }), Text.parseCsv("nq   ,   \"q\",  \"a  ", true));
361        
362        // test failure modes
363        try {
364                Text.parseCsv("\"q\",\"a"); // unclosed quote
365                fail("expected ParseException");
366        } catch (ParseException e) { }
367        try {
368                Text.parseCsv("\"q\",\"a\"  ",true); // whitespace after close quote in whitespace-sensitive mode
369                fail("expected ParseException");
370        } catch (ParseException e) { }
371
372    }
