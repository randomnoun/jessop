017/** Simple jessop example  
018 * 
019 * @author knoxg
020 * @version $Id$
021 */
022public class JessopExample {
023
024    /** A revision marker to be used in exception stack traces. */
025    public static final String _revision = "$Id$";
026
027        public void main(String args[]) throws FileNotFoundException, ScriptException {
028                String filename = args[1];
029                Scanner scanner = new Scanner(new File(filename));
030                String input = scanner.useDelimiter("\\Z").next();
031                scanner.close();
032                ScriptEngine engine = new ScriptEngineManager().getEngineByName("jessop");
033                if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
034                // maybe put a function in here, to be called by the script
035                // or a compound object
036                Bindings b = engine.createBindings();
037                b.put("name", "Count von Count"); // I have been told it is actually "Count von Count"
038                b.put("maxCount", 4);
039                engine.eval(input);
040        }
