50  	private String getSource(ScriptEngine engine, String jessopSource) throws ScriptException {
51  		Compilable compilable = (Compilable) engine;
52  		JessopCompiledScript compiledScript = (JessopCompiledScript) compilable.compile(jessopSource);
53  		return compiledScript.getSource();
54  	}

56      /**
57       * Returns the HTML-escaped form of a string. The <tt>&amp;</tt>,
58       * <tt>&lt;</tt>, <tt>&gt;</tt>, and <tt>"</tt> characters are converted to
59       * <tt>&amp;amp;</tt>, <tt>&amp;lt;<tt>, <tt>&amp;gt;<tt>, and
60       * <tt>&amp;quot;</tt> respectively.
61       *
62       * @param string the string to convert
63       *
64       * @return the HTML-escaped form of the string
65       */
66      static public String escapeHtml(String string) {
67          if (string == null) {
68              return "";
69          }
70  
71          char c;
72          StringBuilder sb = new StringBuilder(string.length());
73  
74          for (int i = 0; i < string.length(); i++) {
75              c = string.charAt(i);
76  
77              switch (c) {
78                  case '&':
79                      sb.append("&amp;");
80                      break;
81                  case '<':
82                      sb.append("&lt;");
83                      break;
84                  case '>':
85                      sb.append("&gt;");
86                      break;
87                  case '\"':
88                      // interestingly, &quote; (with the e) works fine for HTML display,
89                      // but not inside hidden field values
90                      sb.append("&quot;");
91                      break;
92                  default:
93                      sb.append(c);
94              }
95          }
96  
97          return sb.toString();
98      }

101 	public void testJessopJavascript() throws ScriptException {
102 		String input = JAVASCRIPT_COUNTING_SCRIPT; 
103 		ScriptEngine engine = new ScriptEngineManager().getEngineByName("jessop");
104 		if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
105 		Bindings b = engine.createBindings();
106 		b.put("name", "Baron von Count");
107 		b.put("maxCount", 4);
108 		
109 		logger.info("jessop input: " + escapeHtml(input));
110 		logger.info("target language source: " + getSource(engine, input));
111 		logger.info("Start eval");
112 		engine.eval(input, b);
113 		logger.info("End eval");
114 	}
