27  	public static void listEngines(){ // Note 1
28          ScriptEngineManager mgr = new ScriptEngineManager();
29          List<ScriptEngineFactory> factories =
30                  mgr.getEngineFactories();
31          for (ScriptEngineFactory factory: factories) {
32              System.out.println("ScriptEngineFactory Info");
33              String engName = factory.getEngineName();
34              String engVersion = factory.getEngineVersion();
35              String langName = factory.getLanguageName();
36              String langVersion = factory.getLanguageVersion();
37              System.out.println("\tScript Engine: " + engName + ":" +
38                                 engVersion);
39              List<String> engNames = factory.getNames();
40              for(String name: engNames) {
41                  System.out.println("\tEngine Alias: " + name);
42              }
43              System.out.println("\tLanguage: " + langName + ":" +
44                                 langVersion);
45          }
46      }
