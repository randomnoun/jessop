28  	public void setUp() {
29  		String logFormatPrefix = "[SuppressEolTest] ";
30  		Properties lp = new Properties();
31  		lp.put("log4j.rootCategory", "INFO, CONSOLE");
32  		lp.put("log4j.appender.CONSOLE", "org.apache.log4j.ConsoleAppender");
33  		lp.put("log4j.appender.CONSOLE.layout", "org.apache.log4j.PatternLayout");
34  		lp.put("log4j.appender.CONSOLE.layout.ConversionPattern", logFormatPrefix + "%d{ABSOLUTE} %-5p %c - %m%n");
35  		lp.put("log4j.logger.org.springframework", "INFO"); // since Spring is a bit too verbose for my liking at DEBUG level
36  		PropertyConfigurator.configure(lp);

87  	private String getSource(ScriptEngine engine, String jessopSource) throws ScriptException {
88  		Compilable compilable = (Compilable) engine;
89  		JessopCompiledScript compiledScript = (JessopCompiledScript) compilable.compile(jessopSource);
90  		return compiledScript.getSource();
91  	}

93  	public void testJessop1() throws ScriptException {
94  		ScriptEngine engine = new ScriptEngineManager().getEngineByName("jessop");
95  		if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
96  		logger.info("Start eval");
97  		engine.eval(JAVASCRIPT_COUNTING_SCRIPT);
98  		logger.info("End eval");
99  	}

101 	public void testJessopCompile() throws ScriptException {
102 		Compilable engine = (Compilable) new ScriptEngineManager().getEngineByName("jessop");
103 		if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
104 		
105 		CompiledScript script = engine.compile(JAVASCRIPT_COUNTING_SCRIPT);
106 		
107 		JessopCompiledScript jessopScript = (JessopCompiledScript) script;
108 		logger.info("Start source");
109 		logger.info(jessopScript.getSource());
110 		logger.info("End source");
111 		
112 	}

124 	public void testJessopPython1() throws ScriptException {
125 		// -Dpython.console.encoding=UTF-8
126 		// see http://stackoverflow.com/questions/30443537/how-do-i-fix-unsupportedcharsetexception-in-eclipse-kepler-luna-with-jython-pyde
127 		System.setProperty("python.console.encoding", "UTF-8");
128 		
129 		// can either specify the language here (e.g. jessop-rhino), or just 'jessop' to get language from the script itself
130 		ScriptEngine engine = new ScriptEngineManager().getEngineByName("jessop");
131 		if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
132 		logger.info("python source: " + getSource(engine, PYTHON_COUNTING_SCRIPT_1));
133 		
134 		logger.info("Start eval");
135 		engine.eval(PYTHON_COUNTING_SCRIPT_1);
136 		logger.info("End eval");
137 	}
