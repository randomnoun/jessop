028/** This class will generate history table and triggers. 
029 * 
030 * If you get these error messages:
031 * <pre>
032 * Access denied; you need the SUPER privilege for this operation
033 * </pre>
034 * 
035 * then try this:
036 * <pre>
037 * GRANT ALL PRIVILEGES ON ON schema_name.* TO 'ON schema_name'@'%' WITH GRANT OPTION;
038 * </pre>
039 * 
040 * I've got a sqlserver version of this somewhere. Good luck finding that again.
041 * 
042 * <p>This class is similar to the old HistoryTableGenerator, but uses templates instead.
043 * 
044 * @TODO add other db types
045 * @TODO add stored procs to roll back table(s) to a given point in time
046 * @TODO add history partitions, if mysql supports it 
047 **/
048public class HistoryTableGenerator {
049
050        Logger logger = Logger.getLogger(HistoryTableGenerator.class);
051                
052        Logger scriptLogger = Logger.getLogger("com.randomnoun.common.db.HistoryTableGenerator2.script");
053        
054        private DataSource ds;
055        
056        private String jessopScript;
057        private String jessopScriptFilename;
058        private String schemaName; 
059        private Map<String, Object> options = new HashMap<String, Object>();
060        
061        public HistoryTableGenerator(DataSource ds) {
062                this.ds = ds;
063        }
064
065        /** Set the options for the history generator.
066         * 
067         * <p>History options are specific to the generator being used, but typical options are:
068         * 
069         * <ul>
070         * <li>undoEnabledTableNames - List<String>
071     * <li>dropTables - Boolean
072         * <li>existingDataUserActionId - Boolean
073         * <li>alwaysRecreateTriggers - Boolean
074         * <li>alwaysRecreateStoredProcedures - Boolean
075         * <li>includeUpdateBitFields - Boolean 
076         * <li>includeCurrentUser - Boolean
077         * </ul>
078         * 
079         * <p>Refer to the source code of the generator as to which options are supported
080         * ( e.g. src/main/resources/common/db/mysql/mysql-historyTable-2.sql.jessop )
081         * 
082         * @param options
083         */
084        public void setOptions(Map<String, Object> options) {
085                this.options = options;
086        }
087
088        
089        /** When the logger in this class is set to log at DEBUG level, then this method returns the
090         * jessop script transpiled to whichever language it's supposed to be in.
091         * 
092         * @param engine
093         * @param jessopSource
094         * @return
095         * @throws ScriptException
096         */
097        private String getSource(ScriptEngine engine, String jessopSource) throws ScriptException {
098                Compilable compilable = (Compilable) engine;
099                JessopCompiledScript compiledScript = (JessopCompiledScript) compilable.compile(jessopSource);
100                return compiledScript.getSource();
101        }
102        
103        /** Returns the SQL that will create history tables, triggers and stored procedures.
104         * 
105         * <p>To break this String back into individual SQL statements, use {@link com.randomnoun.common.db.SqlParser}
106         * 
107         * @return SQL that will create history tables, triggers and stored procedures 
108         * 
109         * @throws ScriptException
110         */
111        public String generateHistoryTableSql() throws ScriptException {
112   
113                DatabaseReader dr = new MysqlDatabaseReader(ds);
114                SchemaTO schema = dr.getSchema(schemaName);
115                
116                ByteArrayOutputStream baos = new ByteArrayOutputStream();
117                PrintWriter pw = new PrintWriter(baos);
118                
119                ScriptEngine engine = new ScriptEngineManager().getEngineByName("jessop");
120                if (engine==null) { throw new IllegalStateException("Missing engine 'jessop'"); }
121                engine.put(ScriptEngine.FILENAME, jessopScriptFilename);
122                
123                ScriptContext sc = new SimpleScriptContext();
124                sc.setWriter(pw);
125                sc.setAttribute("schema", schema, ScriptContext.ENGINE_SCOPE);
126                
127                sc.setAttribute("logger", scriptLogger, ScriptContext.ENGINE_SCOPE);
128                sc.setAttribute("options", options, ScriptContext.ENGINE_SCOPE);
129
130                logger.info("Start eval");
131                jessopScript = Text.replaceString(jessopScript, "\r", ""); // jessop has issues with \r\n EOLs
132                if (logger.isDebugEnabled()) { logger.debug(getSource(engine, jessopScript)); }
133                
134                
135                // table loop is within the generator now
136                engine.eval(jessopScript, sc);
137                
138                String sql = baos.toString();
139                return sql;
140        }
141
142        public String getJessopScript() {
143                return jessopScript;
144        }
145
146        public void setJessopScript(String jessopScript) {
147                this.jessopScript = jessopScript;
148        }
149
150        public String getJessopScriptFilename() {
151                return jessopScriptFilename;
152        }
153
154        public void setJessopScriptFilename(String jessopScriptFilename) {
155                this.jessopScriptFilename = jessopScriptFilename;
156        }
157
158        public String getSchemaName() {
159                return schemaName;
160        }
161
162        public void setSchemaName(String schemaName) {
163                this.schemaName = schemaName;
164        }
165
166        public Map<String, Object> getOptions() {
167                return options;
168        }
