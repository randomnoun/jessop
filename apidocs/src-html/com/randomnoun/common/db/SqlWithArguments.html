001package com.randomnoun.common.db;
002
003/** A container class for SQL with positional placeholders, and the arguments to be substituted 
004 * into those placeholders, to be used in a JdbcTemplate.
005 * 
006 * @author knoxg
007 */
008public class SqlWithArguments {
009        
010        private String sql;
011        private Object[] args;
012        private int[] argTypes;
013        
014        /** Create a new SqlWithArguments object.
015         * 
016         * @param sql the SQL, containing '?' positional placeholders
017         * @param args the arguments to be substituted into those placeholders
018         * @param argTypes the SQL datatypes of the arguments (as java.sql.Types constants)   
019         */
020        public SqlWithArguments(String sql, Object[] args, int[] argTypes) {
021                this.sql = sql;
022                this.args = args;
023                this.argTypes = argTypes;
024        }
025
026        /** Return the SQL, containing '?' positional placeholders
027         * 
028         * @return the SQL, containing '?' positional placeholders
029         */
030        public String getSql() {
031                return sql;
032        }
033
034        
035        /** Return the arguments to be substituted into the SQL placeholders
036         * 
037         * @return the arguments to be substituted into the SQL placeholders
038         */
039        public Object[] getArgs() {
040                return args;
041        }
042        
043        /** Return the SQL datatypes of the arguments (as java.sql.Types constants)
044         *  
045         * @return the SQL datatypes of the arguments (as java.sql.Types constants)
046         */
047        public int[] getArgTypes() {
048                return argTypes;
049                
050        }
051
052        // setters
053        
054        public void setSql(String sql) {
055                this.sql = sql;
056        }
057
058        public void setArgs(Object[] args) {
059                this.args = args;
060        }
061
062        public void setArgTypes(int[] argTypes) {
063                this.argTypes = argTypes;
064        }
065}
