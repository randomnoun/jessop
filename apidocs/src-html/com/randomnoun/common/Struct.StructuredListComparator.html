035/**
036 * Encoder/decoder of JavaBeans and 'structured' maps and lists. A structured
037 * map or list is any Map or List that satisfies certain conventions, listed below.
038 *
039 * <p>A structured map is any implementation of the {@link java.util.Map}
040 * interface, in which every key is a {@link String}, and every value is either a
041 * primitive wrapper type ({@link String}, {@link Long},
042 * {@link Integer}, etc...), a structured map or a structured list.
043 * A structured list is any implementation of the {@link java.util.List} interface,
044 * of which every value is a primitive wrapper type, a structured map or a structured list.
045 *
046 * <p>In this way, arbitrarily complex objects can be created and passed between
047 * Struts code and the business layer, or Struts code and the JSP layer,
048 * without resorting to the creation of application-specific
049 * datatypes. These datatypes are also used by the Spring JdbcTemplate framework to
050 * return values from a database, so it is useful to have some generic functions
051 * that operate on them.
052 *
053 *

055 */

127    /** A Comparator which performs comparisons between two rows in a structured list (used
128     *  in sorting). This comparator is equivalent to an 'ORDER BY' on a single field
129     *  returned by the Spring JdbcTemplate method.
130     */
131    public static class StructuredListComparator implements Comparator {
132        /** The key field to sort on */
133        private String fieldName;
134
135        /** Create a new StructuredListComparator object, which will sort on the
136         *  supplied field.
137         *
138         * @param fieldName The name of the field to sort on
139         */
140        public StructuredListComparator(String fieldName) {
141            this.fieldName = fieldName;
142        }
143
144        /** Compare two structured list elements
145         *
146         * @see java.util.Comparator#compare(java.lang.Object, java.lang.Object)
147         */
148        public int compare(Object a, Object b)
149            throws IllegalArgumentException {
150            if (!(a instanceof Map)) {
151                throw new IllegalArgumentException("List must be composed of Maps");
152            }
153            if (!(b instanceof Map)) {
154                throw new IllegalArgumentException("List must be composed of Maps");
155            }
156
157            Map mapA = (Map) a;
158            Map mapB = (Map) b;
159            if (!mapA.containsKey(fieldName)) {
160                throw new IllegalArgumentException("keyField '" + fieldName + "' not found in Map");
161            }
162            if (!mapB.containsKey(fieldName)) {
163                throw new IllegalArgumentException("keyField '" + fieldName + "' not found in Map");
164            }
165
166            Object objectA = mapA.get(fieldName);
167            if (!(objectA instanceof Comparable)) {
168                throw new IllegalArgumentException("keyField '" + fieldName + "' element must implement Comparable");
169            }
170            return ((Comparable) objectA).compareTo(mapB.get(fieldName));
171        }
172    }

255    /** Set all fields in object 'obj' using values in request. Each request parameter
256     *  is checked against the object to see if a bean setter method exists for that
257     *  parameter name. If it does, then it is invoked, using the parameter value
258     *  as the setter argument. This method uses the
259     * {@link Struct#setValue(Object, String, Object, boolean, boolean, boolean)} method
260     *  to dynamically create map and list elements as required if the object being
261     *  set implements the Map interface.
262     *
263     *  <p>e.g. if passed a Map object, and the HttpServletRequest has a parameter named
264     *  "<tt>table[12].id</tt>" with the value "<tt>1234</tt>", then:
265     *  <ul><li>a List object named "table" is created
266     *  in the Map,
267     *  <li>the list is increased to allow at least 12 elements,
268     *  <li> that element is set to a new Map object ...
269     *  <li> ... which contains the String key "id", which is assigned the  value "1234".
270     *  </ul>
271     *  <p>This processing allows developers to pass arbitrarily complex structures through
272     *  from HttpServletRequest name/value pairs.
273     *
274     *  <p>This method operates on both structured objects (Maps/Lists) or 
275     *  data-transfer objects (DTOs). For example, if passed a bean-like Object which had a getTable() method that
276     *  returned a List, then this would be used to perform the processing above. (If the
277     *  List returned a Map at index 12, then an 'id' key would be created as before; if
278     *  the List returned another Object at index 12, then the setId() method would be
279     *  called instead; if the list returned null at index 12, then a Map would be created as before).
280     *
281     *  <p>NB: If an object is not a Map, and does not have the appropriate setter() method,
282     *  then this function will *not* raise an exception. This is intentional behaviour -
283     *  (use {@link #setValue(Object, String, Object, boolean, boolean, boolean)} if you
284     *  need more fine-grained control.
285     *
286     *  <p>You can limit the setters that are invoked by using the
287     *  {@see #setFromRequest(Object, HttpServletRequest, String[])} form of this method.
288     *
289     * @param obj The object being set.
290     * @param request The request containing the source data
291     *
292     * @throws RuntimeException if an invocation target exception occurred whilst
293     *   calling the object's set() methods.
294     */
295    public static void setFromRequest(Object obj, HttpServletRequest request) {
296        setFromRequest(obj, request, null);
297    }

299    /** Set all fields in object 'obj' using values in request. Only the request
300     *  parameter names passed in through the 'fields' argument will be used
301     *  to populate the object. If the named parameter does not exist in the request,
302     *  then the setter for that parameter is not invoked.
303     *
304     *  <p>See {@see #setFromRequest(Object, HttpServletRequest)} for more information
305     *  on how this method operates.
306     *
307     * @param obj The object being set.
308     * @param request The request containing the source data
309     * @param fields An array of strings, denoting the fields that are sourced from
310     *   the request. All other parameters in the request are ignored.
311     *
312     * @throws RuntimeException if an invocation target exception occurred whilst
313     *   calling the object's set() methods.
314     */
315    public static void setFromRequest(Object obj, HttpServletRequest request, String[] fields) {
316        Iterator paramListIter;
317
318        if (fields == null) {
319            paramListIter = request.getParameterMap().keySet().iterator();
320        } else {
321            paramListIter = Arrays.asList(fields).iterator();
322        }
323
324        while (paramListIter.hasNext()) {
325            String parameter = (String) paramListIter.next();
326            if (parameter.endsWith("[]")) {
327                parameter = parameter.substring(0, parameter.length()-2);
328                String values[] = request.getParameterValues(parameter);
329                if (values!=null) {
330                        for (int i=0; i<values.length; i++) {
331                                values[i] = Text.replaceString(values[i], "\015\012", "\n");
332                        }
333                }
334                setValue(obj, parameter, values, true, true, true);
335            } else {
336                    String value = request.getParameter(parameter);
337                    
338                    // convert CR-LFs into java newlines 
339                    value = Text.replaceString(value, "\015\012", "\n");
340        
341                    // The null check below is commented out because we need to be able to pass
342                    // null values through in order to set booleans values for checkboxes
343                    // that are unchecked (these are represented in HTTP by missing (null) parameters)
344        
345                    //if (value != null) { 
346                    setValue(obj, parameter, value, true, true, true);
347                    //}
348            }
349        }
350    }

428    /** Return a getter Method for a class. 
429     *
430     * <p>(Getters are cached by this class)
431     *
432     * @param clazz The class we are retrieving the getter Method for
433     * @param propertyName The name of the property on this class; e.g. "asd" will return the
434     *   method "getAsd()"
435     * 
436     * @return the getter Method for the supplied class.
437     */
438    private static Method getGetterMethod(Class<?> clazz, String propertyName) {
439        Map<String,Method> getters = (Map<String,Method>) gettersCache.get(clazz);
440
441        if (getters == null) {
442            getters = new HashMap<String,Method>();
443
444            Method[] methods = clazz.getMethods();
445            Method method;
446
447            // get a list of setters this object contains
448            for (int i = 0; i < methods.length; i++) {
449                method = methods[i];
450
451                if (method.getName().startsWith("get") && method.getParameterTypes().length == 1) {
452                    String property = method.getName().substring(3);
453                    getters.put(property.toLowerCase(), method);
454                }
455            }
456            gettersCache.put(clazz, getters);
457        }
458
459        return (Method) getters.get(propertyName.toLowerCase());
460    }

497    /** Call setter method, converting from String */
498    private static void callSetterMethodWithString(Object target, Method setter, String value) {
499        Class<?> clazz = setter.getParameterTypes()[0];
500        Object realValue;
501
502        if (clazz.equals(String.class)) {
503            realValue = value;
504        } else if (clazz.equals(boolean.class)) {
505            realValue = Boolean.valueOf(value);
506        } else if (clazz.equals(long.class)) {
507                realValue = value.equals("") ? new Long(0) : new Long(value);
508        } else if (clazz.equals(int.class)) {
509                realValue = value.equals("") ? new Integer(0) : new Integer(value);
510        } else if (clazz.equals(double.class)) {
511                realValue = value.equals("") ? new Double(0) : new Double(value);
512        } else if (clazz.equals(float.class)) {
513                realValue = value.equals("") ? new Float(0) : new Float(value);
514        } else if (clazz.equals(short.class)) {
515                realValue = value.equals("") ? new Short((short) 0) : new Short(value);
516        // missing char.class here
517
518        } else if (clazz.equals(Long.class)) {
519                realValue = value.equals("") ? null : new Long(value);
520        } else if (clazz.equals(Integer.class)) {
521                realValue = value.equals("") ? null : new Integer(value);
522        } else if (clazz.equals(Double.class)) {
523                realValue = value.equals("") ? null : new Double(value);
524        } else if (clazz.equals(Float.class)) {
525                realValue = value.equals("") ? null : new Float(value);
526        } else if (clazz.equals(Short.class)) {
527                realValue = value.equals("") ? null : new Short(value);
528        // missing Character.class here
529        
530        
531        } else {
532            throw new IllegalArgumentException("Cannot convert string value to " + "'" + clazz.getName() + "' required for setter method '" + setter.getName() + "'");
533        }
534
535        callSetterMethod(target, setter, realValue);
536    }

552    /** Checks whether the an instance of the value class can be assigned to a variable
553     * of 'variableClass' class via a
554     * {@see java.lang.reflect.Method#invoke(java.lang.Object, java.lang.Object[])}
555     * method. This is different to the normal
556     * {@see java.lang.Class#isAssignableFrom(java.lang.Class)} method since it
557     * also takes primitive to wrapper conversions into account.
558     *
559     * @param variableClass The class representing the variable we are assigning to
560     *   (e.g. the 'x' in 'x = y')
561     * @param value The class representing the value we are assigning to the variable
562     *   (e.g. the 'y' in 'x = y')
563     *
564     * @return true if the assignment is compatible, false otherwise.
565     */
566    private static boolean isAssignmentSafe(Class variableClass, Class value) {
567        // check for instance or wrapper type
568        return variableClass.isAssignableFrom(value) || 
569          (variableClass.equals(boolean.class) && value.equals(Boolean.class)) || 
570          (variableClass.equals(char.class) && value.equals(Character.class)) || 
571          (variableClass.equals(byte.class) && value.equals(Byte.class)) || 
572          (variableClass.equals(short.class) && value.equals(Short.class)) || 
573          (variableClass.equals(int.class) && value.equals(Integer.class)) || 
574          (variableClass.equals(long.class) && value.equals(Long.class)) || 
575          (variableClass.equals(float.class) && value.equals(Float.class)) || 
576          (variableClass.equals(double.class) && value.equals(Double.class));
577    }

579    /** Call setter method */
580    private static void callSetterMethod(Object target, Method setter, Object value) {
581
582        Object[] methodArgs = new Object[1];
583        methodArgs[0] = value;
584        
585        if (value!=null) {
586                Class<?> setterParamClass = setter.getParameterTypes()[0];
587                Class<?> valueClass = value.getClass();
588                
589                // special case for assigning Numbers (including BigDecimals) to longs
590                        if (setterParamClass.equals(Long.class) && Number.class.isAssignableFrom(valueClass)) { // valueClass.equals(BigDecimal.class)
591                                methodArgs[0] = new Long(((Number)value).longValue()); 
592                        } else if (setterParamClass.equals(long.class) && Number.class.isAssignableFrom(valueClass)) { // valueClass.equals(BigDecimal.class)

648     * <p>There is a possible denial-of-service attack that can be used against

653     * 
654     * <p>It should also be noted that even if an exception is thrown within this
655     * method, the data structure underneath it may still have been modified, e.g.
656     * setting "<tt>abc.def.ghi</tt>" may successfully construct 'def' but later
657     * fail on 'ghi'.

668     *   performs conversions to that type automatically. (Useful when setting
669     *   values sourced from a HttpServletRequest, which will always be of String type).

671     *   that does not exist, will create the required Map keys or extend the List
672     *   so that the value can be set.

674     * @throws IllegalArgumentException if
675     * <ul><li>An null map or list is navigated down
676     * <li>A non-List object is invoked with the '[]' operator
677     * <li>A non-Map object is invoked with the '.' operator, or the object
678     *   has no getter/setter method with the appropriate name
679     * <li>An empty mapped property is supplied e.g. bob..something
680     * <li>An invalid key was supplied (e.g. 'bob[123' -- no closing square bracket)
681     * </ul>
682     * @throws NumberFormatException if an array index cannot be converted to
683     *   an integer via {@see java.lang.Integer#parseInt(java.lang.String)}.
684     */
685    static public void setValue(Object object, String key, Object value, boolean ignoreMissingSetter, boolean convertStrings, boolean createMissingElements)
686        throws NumberFormatException 
687    {
688        if (object == null) { throw new NullPointerException("null object"); }
689        if (key == null) { throw new NullPointerException("null key"); }
690
691        // parse state: 
692        //   0=searching for new value (at start of line) 
693        //   1=consuming list index (after '[')
694        //   2=consuming map index (after start of line or after '.')
695        //   3=search for new value (after ']')
696        Object ref = object; // reference into object
697
698        int parseState = 0;
699        int length = key.length();
700        String element;
701        int elementInt = -1;
702        List lastList = null;
703        String parentName = null;
704        Object parentRef = null;
705        
706        if (object instanceof List) { lastList = (List) object; }
707
708        char ch;
709        StringBuffer buffer = new StringBuffer();
710
711        for (int pos = 0; pos < length; pos++) {
712            ch = key.charAt(pos);
713
714            // System.out.println("pos " + pos + ", state=" + parseState + ", nextchar=" + ch + ", buf=" + buffer);
715            switch (parseState) {
716                
717                case 0: 
718                    if (ch == '[') {
719                        buffer.setLength(0);
720                        parseState = 1;
721                    } else { // could check for legal map index characters here
722                        buffer.setLength(0);
723                        buffer.append(ch);
724                        parseState = 2;
725                    }
726                    break;
727                    
728                case 3:
729                    if (ch == '[') {
730                        buffer.setLength(0);
731                        parseState = 1;
732                    } else if (ch == '.') {
733                        buffer.setLength(0);
734                        parseState = 2;
735                    } else {
736                        throw new IllegalArgumentException("Expecting '[' or '.' after ']' in key '" + key + "'; found '" + ch + "'");
737                    }
738                    break;
739                
740                case 1:
741                    if (ch >= '0' && ch <= '9') {
742                        buffer.append(ch);
743                    } else if (ch == ']') {
744                        element = buffer.toString();
745                        elementInt = Integer.parseInt(element);
746
747                        if (ref == null) {
748                            // create list if parent was a map
749                            if ((parentRef instanceof Map) && createMissingElements) {
750                                ref = new ArrayList();
751                                ((Map) parentRef).put(parentName, ref);
752                            } else if ((parentRef instanceof List) && createMissingElements) {
753                                ref = new ArrayList();
754                                ((List) parentRef).set(Integer.parseInt(parentName), ref);
755                            } else {
756                                throw new IllegalArgumentException("Could not retrieve indexed property " + element + " in key '" + key + "' from null object");
757                            }
758                        }
759
760                        if (ref instanceof List) {
761                            lastList = (List) ref;
762
763                            // XXX: denial of service attacks possible here
764                            // (large list number may exceed available memory)
765                            if (lastList.size() <= elementInt) {
766                                setListElement(lastList, elementInt, null);
767                            }
768
769                            parentName = element;
770                            parentRef = ref;
771                            ref = lastList.get(elementInt);
772                        } else {
773                            // could attempt to reflect on a get(int)-style method,
774                            // but this seems like overkill; anything that has List-like
775                            // semantics should implement the List interface.
776                            throw new IllegalArgumentException("Could not retrieve indexed property " + element + " in key '" + key + "' from object of type '" + ref.getClass().getName() + "'");
777                        }
778
779                        parseState = 3;
780                    } else {
781                        throw new IllegalArgumentException("Illegal character '" + ch + "'" + " found in list index");
782                    }
783                    break;
784                
785                case 2:
786                    if (ch != '.' && ch != '[') {
787                        buffer.append(ch);
788                    } else {
789                        // navigate map
790                        element = buffer.toString();
791
792                        if (ref == null) {
793                            if ((parentRef instanceof List) && createMissingElements) {
794                                ref = new HashMap();
795                                ((List) parentRef).set(Integer.parseInt(parentName), ref);
796                            } else if ((parentRef instanceof Map) && createMissingElements) {
797                                ref = new HashMap();
798                                ((Map) parentRef).put(parentName, ref);
799                            } else {
800                                throw new IllegalArgumentException("Could not retrieve mapped property '" + element + "' in key '" + key + "' from null object");
801                            }
802                        }
803
804                        if (element.equals("")) {
805                            throw new IllegalArgumentException("Could not retrieve empty mapped property '" + element + "' in key '" + key + "'");
806                        } else if (ref instanceof Map) {
807                            parentName = element;
808                            parentRef = ref;
809                            ref = ((Map) ref).get(element);
810                        } else {
811                            // look for getter method
812                            Method getter = getGetterMethod(ref.getClass(), element);
813
814                            if (getter == null) {
815                                throw new IllegalArgumentException("Could not retrieve mapped property '" + element + "' in key '" + key + "' for class '" + ref.getClass().getName() + "': no getter method found");
816                            }
817
818                            parentName = null; // can't dynamically create these
819                            parentRef = null;
820                            ref = callGetterMethod(ref, getter);
821                        }
822
823                        buffer.setLength(0);
824
825                        if (ch == '[') {
826                            parseState = 1;
827                        } else if (ch == '.') {
828                            parseState = 2;
829                        } else {
830                            throw new IllegalStateException("Unexpected character '" + ch + "' parsing key '" + key + "'");
831                        }
832                    }
833                    break;
834                
835                default:
836                    throw new IllegalStateException("Unexpected state " + parseState + " parsing key '" + key + "'");
837            }
838        }
839
840        if (parseState == 0) {
841            throw new IllegalStateException("Unexpected error after parsing key '" + key + "'");
842        } else if (parseState == 1) {
843            throw new IllegalStateException("Missing ']' in key '" + key + "'");
844        } else if (parseState == 2) {
845            // set mapped property
846            element = buffer.toString();
847
848            if (ref == null) {
849                if ((parentRef instanceof Map) && createMissingElements) {
850                    ref = new HashMap();
851                    ((Map) parentRef).put(parentName, ref);
852                } else if ((parentRef instanceof List) && createMissingElements) {
853                    ref = new HashMap();
854                    ((List) parentRef).set(Integer.parseInt(parentName), ref);
855                } else {
856                    throw new IllegalArgumentException("Could not set mapped property '" + element + "' in key '" + key + "' for null object");
857                }
858            }
859
860            if (element.equals("")) {
861                throw new IllegalArgumentException("Could not set empty mapped property '" + element + "' in key '" + key + "'");
862            } else if (ref instanceof Map) {
863                ((Map) ref).put(element, value);
864            } else {
865                Method setter = getSetterMethod(ref.getClass(), element);
866
867                if (setter == null) {
868                    // System.out.println("Missing setter '" + element + "'");
869                    if (!ignoreMissingSetter) {
870                        throw new IllegalArgumentException("Could not set mapped property '" + element + "' in key '" + key + "' for class '" + ref.getClass().getName() + "': no setter method found");
871                    }
872                } else {
873                    // System.out.println("Setting '" + element + "' with value " + value);
874                    try {
875                            if (convertStrings && ((value == null) || (value instanceof String))) {
876                                if (value==null) { value = ""; }
877                                callSetterMethodWithString(ref, setter, (String) value);
878                            } else {
879                                callSetterMethod(ref, setter, value);
880                            }
881                    } catch (Exception e) {
882                        throw (IllegalArgumentException) new IllegalArgumentException("Could not set field '" + key + "' with value '" + value + "'").initCause(e);
883                    }
884                }
885            }
886        } else if (parseState == 3) {
887            // set list property
888            // ref currently points to the current list element; we want to set the 
889            // value of the last list we saw, contained in lastList.
890            ref = lastList;
891
892            if (ref == null) {
893                throw new IllegalArgumentException("Could not set indexed property '" + elementInt + "' in key '" + key + "' for null object");
894            }
895
896            if (ref instanceof List) {
897                ((List) ref).set(elementInt, value);
898            } else {
899                throw new IllegalArgumentException("Could not set indexed property " + elementInt + " in key '" + key + "' in object of type '" + ref.getClass().getName() + "'");
900            }
901        }
902    }

2008    /** Create a structured Map out of key / value pairs. Keys must be Strings.
2009     * 
2010     * <p>So the value of <code>Struct.newStructuredMap("thing", 1L, "otherThing", "value");</code> is a map
2011     * with two entries:
2012     * <ul><li>an entry with key "thing" and value new Long(1), and

2014     * </ul> 
2015     * 
2016     * @param keyValuePairs a list of key / value pairs 
2017     * 
2018     * @return a Map as described above
2019     * 
2020     * @throws ClassCastException if any of the supplied keys is not a String

2022    // looking forward to when statically defined Maps become a Java language construct
2023    static public Map<String, Object> newStructuredMap(Object... keyValuePairs) {
2024        if (keyValuePairs.length%2==1) { throw new IllegalArgumentException("keyValuePairs must have even number of elements"); }
2025        Map<String, Object> map = new HashMap();
2026        for (int i=0; i<keyValuePairs.length; i+=2) {

2051                }
