045        /** Clean some HTML text through the tagsoup filter. The returned string is guaranteed to be 
046         * well-formed XML (and can therefore be used by other tools that expect valid XML). 
047         * 
048         * @param inputXml input XML document
049         * @param isHtml if true, uses the HTML schema, omits the XML declaration, and uses the html method
050         * 
051         * @throws SAXException if the tagsoup library could not parse the input string
052         * @throws IllegalStateException if an error occurred reading from a string (should never occur)
053         */ 
054        public static String getCleanXml(String inputXml, boolean isHtml) throws SAXException {
055                return getCleanXml(new ByteArrayInputStream(inputXml.getBytes()), isHtml);
056        }

058        /** Clean a HTML inputStream through the tagsoup filter. The returned string is guaranteed to be 
059         * well-formed XML (and can therefore be used by other tools that expect valid XML). 
060         * 
061         * @param is input XML stream
062         * @param isHtml if true, uses the HTML schema, omits the XML declaration, and uses the html method
063         * 
064         * @throws SAXException if the tagsoup library could not parse the input string
065         * @throws IllegalStateException if an error occurred reading from a string (should never occur)
066         */ 
067        public static String getCleanXml(InputStream inputStream, boolean isHtml) throws SAXException {
068                try {
069                        ByteArrayOutputStream baos = new ByteArrayOutputStream();
070                        InputSource is = new InputSource();
071                        is.setByteStream(inputStream); // could use raw inputstream here later
072
073                        XMLReader xmlReader = new Parser();
074                        Writer w = new OutputStreamWriter(baos);
075                        XMLWriter tagsoupXMLWriter = new XMLWriter(w);
076                        tagsoupXMLWriter.setOutputProperty(XMLWriter.OMIT_XML_DECLARATION, "yes");
077                        if (isHtml) {
078                                HTMLSchema theSchema = new HTMLSchema();
079                                xmlReader.setProperty(Parser.schemaProperty, theSchema);
080        
081                                tagsoupXMLWriter.setOutputProperty(XMLWriter.METHOD, "html");
082                                tagsoupXMLWriter.setPrefix(theSchema.getURI(), "");
083                        }
084                        
085                        xmlReader.setContentHandler(tagsoupXMLWriter);
086                        xmlReader.parse(is);
087                        return baos.toString("UTF-8");
088                } catch (IOException ioe) {
089                        throw (IllegalStateException) new IllegalStateException("IO Exception reading from string").initCause(ioe);             
090                }
091        }

094        /**
095         * Iterates through the child nodes of the specified element, and returns the contents
096         * of all Text and CDATA elements among those nodes, concatenated into a string.
097         *
098         * <p>Elements are recursed into.
099         *
100         * @param element the element that contains, as child nodes, the text to be returned.
101         * @return the contents of all the CDATA children of the specified element.
102         */
103        public static String getText(Element element)
104        {
105                if (element == null) { throw new NullPointerException("null element"); }
106                StringBuffer buf = new StringBuffer();
107                NodeList children = element.getChildNodes();
108                for (int i = 0; i < children.getLength(); ++i) {
109                        org.w3c.dom.Node child = children.item(i);
110                        short nodeType = child.getNodeType();
111                        if (nodeType == org.w3c.dom.Node.CDATA_SECTION_NODE) {
112                                buf.append(((org.w3c.dom.Text) child).getData());                       
113                        } else if (nodeType == org.w3c.dom.Node.TEXT_NODE) {
114                                buf.append(((org.w3c.dom.Text) child).getData());
115                        } else if (nodeType == org.w3c.dom.Node.ELEMENT_NODE) {
116                                buf.append(getText((Element) child));
117                        }
118                }
119                return buf.toString();
120        }

122        /**
123         * Iterates through the child nodes of the specified element, and returns the contents
124         * of all Text and CDATA elements among those nodes, concatenated into a string. 
125         * Any elements with tagNames that are included in the tagNames parameter of this
126         * method are also included. 
127         * 
128         * <p>Attributes of these tags are also included in the result, but may be reordered.
129         * 
130         * <p>Self-closing elements (e.g. <code>&lt;br/&gt;</code>)
131         * are expanded into opening and closing elements (e.g. <code>&lt;br&gt;&lt;/br&gt;</code>)
132         *
133         * <p>Elements are recursed into.
134         *
135         * @param element the element that contains, as child nodes, the text to be returned.
136         * @return the contents of all the CDATA children of the specified element.
137         */
138        public static String getTextPreserveElements(Element element, String[] tagNames) {
139                if (element == null) { throw new NullPointerException("null element"); }
140                Set<String> tagNamesSet = new HashSet<String>(Arrays.asList(tagNames));
141                StringBuffer buf = new StringBuffer();
142                NodeList children = element.getChildNodes();
143                for (int i = 0; i < children.getLength(); ++i) {
144                        org.w3c.dom.Node child = children.item(i);
145                        short nodeType = child.getNodeType();
146                        if (nodeType == org.w3c.dom.Node.CDATA_SECTION_NODE) {
147                                buf.append(((org.w3c.dom.Text) child).getData());                       
148                        } else if (nodeType == org.w3c.dom.Node.TEXT_NODE) {
149                                buf.append(((org.w3c.dom.Text) child).getData());
150                        } else if (nodeType == org.w3c.dom.Node.ELEMENT_NODE) {
151                                String tagName = ((Element) child).getTagName();
152                                boolean includeEl = tagNamesSet.contains(tagName);
153                                if (includeEl) {
154                                        buf.append('<');
155                                        buf.append(tagName);
156                                        NamedNodeMap nnm = ((Element) child).getAttributes();
157                                        for (int j = 0; j < nnm.getLength(); j++) {
158                                                Attr attr = (Attr) nnm.item(j);
159                                                buf.append(" " + attr.getName());
160                                                if (attr.getValue()!=null) {
161                                                        buf.append("=\"" + attr.getValue() + "\"");
162                                                }
163                                        }
164                                        buf.append('>');
165                                }
166                                buf.append(getTextPreserveElements((Element) child, tagNames));
167                                if (includeEl) {
168                                        buf.append("</" + tagName + ">");
169                                }
170                        }
171                }
172                return buf.toString();
173        }

213        /** Return a DOM document object from an InputStream
214         * 
215         * @param is the InputStream containing the XML to parse 
216         */
217        public static Document toDocument(InputStream is) throws SAXException {
218                try {
219                        DocumentBuilderFactory docBuilderFactory = DocumentBuilderFactory.newInstance();
220                        DocumentBuilder docBuilder = docBuilderFactory.newDocumentBuilder();
221                        Document doc = docBuilder.parse(is);
222                        doc.getDocumentElement().normalize(); // Collapses adjacent text nodes into one node.
223                        return doc;
224                } catch (ParserConfigurationException pce) {
225                        // this can never happen 
226                        throw (IllegalStateException) new IllegalStateException("Error creating DOM parser").initCause(pce);
227                } catch (IOException ioe) {
228                        // this can also never happen
229                        throw (IllegalStateException) new IllegalStateException("Error retrieving information").initCause(ioe);
230                } 
231        }

233        /** Converts a document node subtree back into an XML string 
234         * 
235         * @param node a DOM node 
236         * @param omitXmlDeclaration if true, omits the XML declaration from the returned result
237         * 
238         * @return the XML for this node
239         * 
240         * @throws TransformerException if the transformation to XML failed
241         * @throws IllegalStateException if the transformer could not be initialised 
242         */
243        public static String getXmlString(Node node, boolean omitXmlDeclaration) 
244                throws TransformerException 
245        {
246                try {
247                        ByteArrayOutputStream baos = new ByteArrayOutputStream();
248                        TransformerFactory transformerFactory = TransformerFactory.newInstance();
249                        Transformer transformer = transformerFactory.newTransformer();
250                        DOMSource source = new DOMSource(node);
251                        StreamResult result = new StreamResult(baos);
252                        transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, omitXmlDeclaration ? "yes": "no");
253                        transformer.transform(source, result);
254                        return baos.toString("UTF-8");
255                } catch (TransformerConfigurationException tce) {
256                        throw (IllegalStateException) new IllegalStateException("Could not initialise transformer").initCause(tce);
257                } catch (UnsupportedEncodingException uee) {
258                        throw (IllegalStateException) new IllegalStateException("Unknown charset UTF-8").initCause(uee);
259                }
260        }

263        /** Remove leading/trailing whitespace from all text nodes in this nodeList.
264         * Will iterate through subnodes recursively.
265         * 
266         * @param nodeList
267         */
268        public static void compact(Node node) {
269                if (node.getNodeType()==Node.TEXT_NODE) {
270                        org.w3c.dom.Text el = (org.w3c.dom.Text) node;
271                        if (el.getNodeValue()!=null) {
272                                el.setNodeValue(el.getNodeValue().trim());
273                        }
274                } else if (node.getNodeType()==Node.ELEMENT_NODE) {
275                        NodeList childNodes = node.getChildNodes();
276                        if (childNodes != null && childNodes.getLength() > 0) {
277                                int len = childNodes.getLength();
278                                for (int i=0; i<len; i++) {
279                                        Node childNode = childNodes.item(i);
280                                    compact(childNode);
281                                }
282                        }
283                }
284        }

287        /** Parse a string of XML text using a SAX contentHandler. Nothing is returned by this method - it 
288         * is assumed that the contentHandler supplied maintains it's own state as it parses the XML supplied,
289         * and that this state can be extracted from this object afterwards.
290         * 
291         * @param contentHandler a SAX content handler 
292         * @param xmlText an XML document (or part thereof)
293         * 
294         * @throws SAXException if the document could not be parsed
295         * @throws IllegalException if the parser could not be initialised, or an I/O error occurred 
296         *   (should not happen since we're just dealing with strings)
297         */
298        public static void processContentHandler(ContentHandler contentHandler, String xmlText) throws SAXException {
299                 SAXParserFactory factory = SAXParserFactory.newInstance();
300                 try {
301                         // Parse the input
302                         SAXParser saxParser = factory.newSAXParser();
303                         XMLReader xmlReader = saxParser.getXMLReader();
304                         xmlReader.setContentHandler(contentHandler);
305                         xmlReader.parse(new InputSource(new ByteArrayInputStream(xmlText.getBytes())));
306                 } catch (IOException ioe) {
307                        throw (IllegalStateException) new IllegalStateException("IO Exception reading from string").initCause(ioe);
308                 } catch (ParserConfigurationException pce) {
309                        throw (IllegalStateException) new IllegalStateException("Could not initialise parser").initCause(pce);                          
310                 }
311        }

313        /** Convert a table into a List of Lists (each top-level list represents a table row,
314         * each second-level list represents a table cell). Only contents are returned; attributes
315         * and formatting are ignored.
316         * 
317         * <p>This class will probably not work when tables are embedded within other tables
318         */
319        public static class SimpleTableContentHandler
320                implements ContentHandler 
321        {
322                /** Logger instance for this class */
323                public static final Logger logger = Logger.getLogger(SimpleTableContentHandler.class);
324
325                /** Current table */
326                List<List<String>> thisTable = null;
327                /** Current row in table */
328                List<String> thisRow = null;
329                /** Current cell in row */
330                String thisCell = "";
331
332                /** The state of this parser */
333                private enum State {
334                        /** start of doc, expecting 'table' */
335                        START,
336                        /** in table element, expecting 'tr' */
337                        IN_TABLE,
338                        /** in tr element, expecting 'td' (or other ignored elements) */
339                        IN_TR,
340                        /** in td element, capturing to closing tag */
341                        IN_TD
342                }
343
344                State state = State.START;
345                
346                // unused interface methods
347                public void setDocumentLocator(Locator locator) { }
348                public void startDocument() throws SAXException { }
349                public void endDocument() throws SAXException { }
350                public void startPrefixMapping(String prefix, String uri) throws SAXException { }
351                public void endPrefixMapping(String prefix) throws SAXException { }
352                public void ignorableWhitespace(char[] ch, int start, int length) throws SAXException { }
353                public void processingInstruction(String target, String data) throws SAXException { }
354                public void skippedEntity(String name) throws SAXException { }
355
356
357                public void startElement(String uri, String localName, String qName, Attributes atts)
358                        throws SAXException 
359                {
360                        switch (state) {
361                                case START: 
362                                        if (qName.equals("table")) {
363                                                thisTable = new ArrayList<List<String>>(); 
364                                                state = State.IN_TABLE; 
365                                        } else {
366                                                logger.warn("Warning: top-level element '" + qName + "' found (expected 'table')");
367                                        }
368                                        break;
369                                
370                                case IN_TABLE:
371                                        if (qName.equals("tr")) {
372                                                thisRow = new ArrayList<String>();
373                                                thisTable.add(thisRow);
374                                                state = State.IN_TR;
375                                        }
376                                        break;
377                                        
378                                case IN_TR: 
379                                        if (qName.equals("td")) {
380                                                thisCell = "";
381                                                state = State.IN_TD;
382                                        }
383                                        break;
384                                        
385                                case IN_TD:
386                                        break;
387                                        
388                                default:
389                                        throw new IllegalStateException("Illegal state " + state + " in SimpleTableContentHandler");
390                                
391                        }
392                }
393
394                public void characters(char[] ch, int start, int length)
395                        throws SAXException {
396                        if (state==State.IN_TD) {
397                                thisCell += new String(ch, start, length);
398                        }
399                }
400
401                public void endElement(String uri, String localName, String qName)
402                        throws SAXException 
403                {
404                        if (state == State.IN_TD && qName.equals("td")) {
405                                thisRow.add(thisCell);
406                                state = State.IN_TR;
407                        } else if (state == State.IN_TR && qName.equals("tr")) {
408                                state = State.IN_TABLE;
409                        }
410                }
411        
412                public List<List<String>> getTable() {
413                        return thisTable;
414                }
415        }

417        /** An abstract stack-based XML parser. Similar to the apache digester, but without
418         * the dozen or so dependent JARs.
419         * 
420         * <p>Only element text is captured 
421         * <p>Element attributes are not parsed by this class.
422         * <p>Mixed text/element nodes are not parsed by this class.
423         * 
424         */
425        public abstract static class AbstractStackContentHandler implements ContentHandler 
426        {
427                /** Logger instance for this class */
428                public static final Logger logger = Logger.getLogger(AbstractStackContentHandler.class);
429
430                /** Location in stack */
431                protected String stack = "";
432                protected String text = null;     // text captured so far
433                
434                // unused interface methods
435                public void setDocumentLocator(Locator locator) { }
436                public void startDocument() throws SAXException { }
437                public void endDocument() throws SAXException { }
438                public void startPrefixMapping(String prefix, String uri) throws SAXException { }
439                public void endPrefixMapping(String prefix) throws SAXException { }
440                public void ignorableWhitespace(char[] ch, int start, int length) throws SAXException { }
441                public void processingInstruction(String target, String data) throws SAXException { }
442                public void skippedEntity(String name) throws SAXException { }
443
444                public void startElement(String uri, String localName, String qName, Attributes atts)
445                        throws SAXException 
446                {
447                        stack = stack.equals("") ? qName : stack + "/" + qName;
448                        text = "";
449                        element(stack);
450                }
451                public void characters(char[] ch, int start, int length) throws SAXException {
452                        text += new String(ch, start, length);
453                }
454                public void endElement(String uri, String localName, String qName)
455                        throws SAXException 
456                {
457                        elementText(stack, text);
458                        text = ""; // probably not necessary
459                        stack = stack.contains("/") ? stack.substring(0, stack.lastIndexOf("/")) : "";
460                }
461
462                // abstract methods to be implemented by subclasses
463                public abstract void element(String path) throws SAXException;
464                public abstract void elementText(String path, String content) throws SAXException;
465        }

468        /** Convert a NodeList into something that Java1.5 can treat as Iterable,
469         * so that it can be used in <tt>for (Node node : nodeList) { ... }</tt> style
470         * constructs.
471         * 
472         * <p>(org.w3c.dom.traversal.NodeListIterator doesn't currently implement Iterable)
473         * 
474         */
475        public static class NodeListIterator implements Iterable<org.w3c.dom.Node> {
476                private final NodeList nodeList;
477                public NodeListIterator(NodeList nodeList) {
478                        this.nodeList = nodeList;
479                }
480                public Iterator<org.w3c.dom.Node> iterator() {
481                        return new Iterator<org.w3c.dom.Node>() {
482                                private int index = 0;
483                                public boolean hasNext() {
484                                        return index < nodeList.getLength();
485                                }
486                                public org.w3c.dom.Node next() {
487                                        return nodeList.item(index++);
488                                }
489                                public void remove() {
490                                        throw new UnsupportedOperationException("remove() not allowed in NodeList");
491                                }
492                        };
493                }
494        }
