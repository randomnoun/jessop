021/**
022 * Provides a simple, one-class wrapper around the Java Mail API. Use the
023 * {@link #emailTo(String, String, String, String, String, String, String)} 
024 * method to send a mail in a single method call, or 
025 * {@link #emailToNoEx(String, String, String, String, String, String, String)} 
026 * to send an email ignoring exceptions.
027 *
028 * <p>SMTP authentication is supported.
029 * 
030 * <p>S/MIME encryption isn't supported, but would be relatively easy to add;
031 *
032 * <p>Use the (somewhat more complex) {@see #emailAttachmentTo(Map)} method to send emails
033 * with attachments sourced from the file system, byte arrays or the classpath,
034 * or to modify the email headers.
035 *
036 * @author knoxg
037 */
038public class EmailWrapper {

045    private static boolean isBlank(String string) {
046        return (string==null || "".equals(string));
047    }

049    /**
050     * A static method which provides the facility to easily send off an
051     * email using the JavaMail API. Mail-generated exceptions are sent to
052     * logger.error.
053     *
054     * @param to A comma-separated list of recipients
055     * @param from The address to place in the From: field of the email
056     * @param subject The subject text
057     * @param msgText The message text
058     */
059    public static void emailToNoEx(String to, String from, String host, String subject, 
060      String bodyText, String username, String password) {
061        try {
062            emailTo(to, from, host, subject, bodyText, username, password);
063        } catch (MessagingException me) {
064            logger.error("A messaging exception occurred whilst sending an email", me);
065        }
066    }

070     *
071     * @param to A comma-separated list of recipients
072     * @param from The address to place in the From: field of the email
073     * @param host The SMTP host to use
074     * @param subject The subject text
075     * @param bodyText The message text
076     * @param username The SMTP user to send from (null if not authenticated)
077     * @param password The SMTP password to use (null if not authenticated)
078     *
079     */
080    public static void emailTo(String to, String from, String host, String subject, String bodyText,
081        String username, String password)
082        throws MessagingException {
083        Properties props;
084
085        props = new Properties();
086        props.put("to", to);
087        props.put("from", from);
088        props.put("host", host);
089        props.put("subject", subject);
090        props.put("bodyText", bodyText);
091        if (username!=null) { props.put("username", username); } 
092        if (password!=null) { props.put("password", password); } 
093        emailAttachmentTo(props);
094    }

128    /** Return an array of InternetAddress objects from a comma-separated list,
129     *  and a header list. The headerList object passed to the
130     *  {@link #emailAttachmentTo(java.util.Map)} object is passed into this
131     *  object; if a header exists in this list with the name _headerName_ passed
132     *  to this method, then the email addresses contained within it will also
133     *  be appended to the returned array.
134     *
135     *  @param stringList A comma-separated list of names which will be appended
136     *    to the array
137     *  @param headerName The header name that will be used to search for
138     *    more email addresses
139     *  @param headerList A structured list, describing a list of custom
140     *    headers to be added to an email, as passed to emailAttachmentTo()
141     *
142     *  @return an InternetAddress[] structure, suitable for use in various
143     *    JavaMail API calls
144     */
145    private static InternetAddress[] getAddressList(String stringList, String headerName, List headerList)
146        throws MessagingException {
147        String[] addresses;
148        List addressList = new ArrayList();
149        int i;
150
151        // add addresses in 'stringList' to addressList array
152        if (!isBlank(stringList)) {
153            addresses = stringList.split(",");
154
155            for (i = 0; i < addresses.length; i++) {
156                addressList.add(new InternetAddress(addresses[i]));
157            }
158        }
159
160        // do the same for any headers matching the headerName supplied
161        if (!isBlank(headerName) && (headerList != null)) {
162            for (Iterator j = headerList.iterator(); j.hasNext();) {
163                Map map = (Map) j.next();
164
165                if (headerName.equals(map.get("name"))) {
166                    addresses = ((String) map.get("value")).split(",");
167
168                    for (i = 0; i < addresses.length; i++) {
169                        addressList.add(new InternetAddress(addresses[i]));
170                    }
171                }
172            }
173        }
174
175        return (InternetAddress[]) addressList.toArray(new InternetAddress[] {  });
176    }

178    /** Versatile extendable email wrapper method.
179     *
180     * This method takes a structured map as input, representing the
181     * message to be sent. It supports multiple to, from, cc, bcc, and replyto
182     * fields. It supports custom headers. It supports attachments from
183     * files on the filesystem, classpath resources, and passed in directly.
184     *
185     * The following attributes are accepted by this method
186     *
187     * <attributes>
188     * to - To addresses, comma-separated. Will also include any
189     *      custom headers supplied with the headername of 'to'
190     * from - From address
191     * subject - The subject of the email
192     * bodyText - The body text of the email
193     * bodyHtml - The body text of the email, in HTML format
194     * username - If not null, the username to authenticate to the SMTP server
195     * password - If not null, the password with which to authenticate to the SMTP 
196     * cc - CC addresses, comma-separated. Will also include any
197     *                   custom headers supplied with the headername of 'cc'
198     * bcc - BCC addresses, comma-separated. Will also include any
199     *                   custom headers supplied with the headername of 'bcc'
200     * replyTo - Reply-To addresses, comma-separated. Will also include any
201     *                   custom headers supplied with the headername of 'replyTo'
202     * client - email client to use in generated MessageID (defaults to "JavaMail")
203     * suffix - suffix to use in generated MessageID (defaults to "username@host")
204     *          (both client & suffix must be specified together)
205     * sessionProperties - A map containing additional JavaMail session properties
206     * headers - A structured list of custom headers
207     * <attributes>
208     *   name - The header name
209     *   value - The header value
210     * </attributes>
211     *
212     * attachFiles   Attachments to this email, sourced from the filesystem
213     * <attributes>
214     *   filename - The file on the local filesystem which contains the  data to send
215     *   attachFilename - The name of the file visible in the email
216     *   contentType - The content-type to assign to this file. Will default to
217     *                   <tt>application/octet-stream</tt>
218     * </attributes>
219     *
220     * attachResources Attachments to this email, sourced from the classLoader
221     * <attributes>
222     *   resource - The resource name
223     *   attachFilename - The name of the file visible in the email
224     *   contentType - The content-type to assign to this file. Will default to
225     *                   <tt>application/octet-stream</tt>
226     *   classLoader - Any class which will indicate which class loader to
227     *                   use to find this resource, or a ClassLoader instance.
228     *                   If missing, will default to the class loader of the
229     *                   EmailWrapper class.
230     * </attributes>
231     *
232     * attachData    Attachments to this email, passed in directly
233     * <attributes>
234     *   data - The data comprising the attachment. Can be either a
235     *                   byte array, or a string. If the object is of any other
236     *                   type, then it is converted to a string using it's
237     *                   .toString() method.
238     *   attachFilename - The name of the file visible in the email
239     *   contentType - The content-type to assign to this file. Will default
240     *                   to <tt>application/octet-stream</tt>
241     * </attributes>
242     * </attributes>
243     *
244     * This method returns no values
245     *
246     * @throws MessagingException An error occurred sending the email
247     */
248    public static void emailAttachmentTo(Map params)
249        throws MessagingException {
250        logger.debug("Inside emailAttachmentTo method with params");
251
252        // get parameters out of map
253        String to = (String) params.get("to");
254        String from = (String) params.get("from");
255        String cc = (String) params.get("cc");
256        String bcc = (String) params.get("bcc");
257        String replyTo = (String) params.get("replyTo");
258        String bodyText = (String) params.get("bodyText");
259        String bodyHtml = (String) params.get("bodyHtml");
260        String subject = (String) params.get("subject");
261        String host = (String) params.get("host");
262        String username = (String) params.get("username");
263        String password = (String) params.get("password");
264        String client = (String) params.get("client");
265        String suffix = (String) params.get("suffix");
266        
267        List attachFiles = (List) params.get("attachFiles"); // list of filenames to retrieve from disk
268        List attachResources = (List) params.get("attachResources"); // list of resources to retrieve from classpath
269        List attachData = (List) params.get("attachData"); // list of attachment data
270        List headers = (List) params.get("headers");
271        Map sessionProperties = (Map) params.get("sessionProperties");
272        
273        boolean isMultipart = false;
274        boolean isAltContent = false;  // true if both text and html
275
276        // validate / set defaults
277        if (isBlank(to)) { throw new IllegalArgumentException("Empty 'to' address"); }
278        if (isBlank(host)) { host = "127.0.0.1"; }
279
280        // set up mail session
281        Properties props = new Properties();
282        props.put("mail.smtp.host", host);
283        if (username != null) { props.put("mail.smtp.user", username); }
284        // props.put("mail.host", host);
285        if (sessionProperties != null) {
286                props.putAll(sessionProperties);
287        }
288        
289        
290        // could make this another parameter whether to reuse the default session or not
291        // breaks having per-message bounce addresses though.
292        // Session session = Session.getDefaultInstance(props, null);
293        Session session = Session.getInstance(props, null);
294        Message msg;
295        if (isBlank(client) || isBlank(suffix)) {
296                msg = new MimeMessage(session);
297        } else {
298                msg = new CustomMimeMessage(session, client, suffix);
299        }
300
301        if (!isBlank(from)) {
302            msg.setFrom(new InternetAddress(from));
303        }
304        if (!isBlank(subject)) {
305            msg.setSubject(subject);
306        }
307
308        msg.setSentDate(new Date());
309        
310        if (bodyHtml!=null && bodyText!=null) { 
311                isMultipart = true; 
312                isAltContent = true;
313        }
314
315        // add receipient information
316        InternetAddress[] toArray = getAddressList(to, "to", headers);
317        if (toArray.length > 0) { msg.setRecipients(Message.RecipientType.TO, toArray); }
318
319        InternetAddress[] ccArray = getAddressList(cc, "cc", headers);
320        if (ccArray.length > 0) { msg.setRecipients(Message.RecipientType.CC, ccArray); }
321
322        InternetAddress[] bccArray = getAddressList(bcc, "bcc", headers);
323        if (bccArray.length > 0) { msg.setRecipients(Message.RecipientType.BCC, bccArray); }
324
325        InternetAddress[] replyToArray = getAddressList(replyTo, "replyTo", headers);
326        if (replyToArray.length > 0) { msg.setReplyTo(replyToArray); }
327
328        // add other headers
329        if (headers != null) {
330            for (Iterator i = headers.iterator(); i.hasNext();) {
331                Map map = (Map) i.next();
332                String headerName = (String) map.get("name");
333
334                if (headerName != null && !headerName.equals("to") && !headerName.equals("cc") && !headerName.equals("bcc")) {
335                    msg.addHeader(headerName, (String) map.get("value"));
336                }
337            }
338        }
339
340        // create multipart message. Beware of non us-ascii charsets.
341        MimeMultipart multiPart;
342        if (isAltContent) {
343                multiPart = new MimeMultipart("alternative");
344                MimeBodyPart bodyPart = new MimeBodyPart();
345                bodyPart.setText(bodyText /*, "us-ascii", "plain" */);
346                multiPart.addBodyPart(bodyPart);
347                bodyPart = new MimeBodyPart();
348                bodyPart.setText(bodyHtml, "us-ascii", "html" );  /* creates text/html Content-type */
349                multiPart.addBodyPart(bodyPart);
350        } else {
351                multiPart = new MimeMultipart("mixed");
352                MimeBodyPart bodyPart = new MimeBodyPart();
353                if (bodyText!=null) {
354                        bodyPart.setText(bodyText /*, "us-ascii", "text/plain" */);
355                } else if (bodyHtml!=null) {
356                        bodyPart.setText(bodyHtml, "us-ascii", "html");
357                }
358            multiPart.addBodyPart(bodyPart);
359        }
360
361        // create attachments from files on disk
362        if (attachFiles != null) {
363            for (Iterator i = attachFiles.iterator(); i.hasNext();) {
364                Map map = (Map) i.next();
365                String filename = (String) map.get("filename");
366                String attachFilename = (String) map.get("attachFilename");
367                DataSource dataSource = new FileDataSource(filename);
368                MimeBodyPart attachment = new MimeBodyPart();
369
370                attachment.setDataHandler(new DataHandler(dataSource));
371                attachment.setFileName(attachFilename); //attachFile);
372                multiPart.addBodyPart(attachment);
373                isMultipart = true;
374            }
375        }
376
377        // create attachments from a list of classpath resources
378        try {
379            if (attachResources != null) {
380                for (Iterator i = attachResources.iterator(); i.hasNext();) {
381                    Map map = (Map) i.next();
382                    String resource = (String) map.get("resource");
383                    String attachFilename = (String) map.get("attachFilename");
384                    Object classLoaderObject = (Object) map.get("classloader");
385                    String contentType = (String) map.get("contentType");
386
387                    if (isBlank(contentType)) {
388                        contentType = "application/octet-stream";
389                    }
390
391                    ClassLoader classLoader;
392
393                    if (classLoaderObject == null) {
394                        classLoaderObject = EmailWrapper.class;
395                    }
396
397                    if (classLoaderObject instanceof Class) {
398                        classLoader = ((Class) classLoaderObject).getClassLoader();
399                    } else if (classLoaderObject instanceof ClassLoader) {
400                        classLoader = (ClassLoader) classLoaderObject;
401                    } else {
402                        classLoader = classLoaderObject.getClass().getClassLoader();
403                    }
404
405                    InputStream inputStream = classLoader.getResourceAsStream(resource);
406                    byte[] attachmentData = StreamUtil.getByteArray(inputStream);
407                    DataSource dataSource = new ByteArrayDataSource(attachmentData, attachFilename, contentType);
408                    MimeBodyPart attachment = new MimeBodyPart();
409
410                    attachment.setDataHandler(new DataHandler(dataSource));
411                    attachment.setFileName(attachFilename);
412                    multiPart.addBodyPart(attachment);
413                    isMultipart = true;
414                }
415            }
416        } catch (IOException ioe) {
417            throw new MessagingException("Error reading resource", ioe);
418        }
419
420        // create attachments from data passed in to this method
421        if (attachData != null) {
422            for (Iterator i = attachData.iterator(); i.hasNext();) {
423                Map map = (Map) i.next();
424                String attachFilename = (String) map.get("attachFilename");
425                Object data = map.get("data");
426                String contentType = (String) map.get("contentType");
427                String contentId = (String) map.get("contentId");
428                String disposition = (String) map.get("disposition");
429                if (isBlank(contentType)) {
430                    contentType = "application/octet-stream";
431                }
432                
433
434                byte[] attachmentData;
435
436                if (data instanceof byte[]) {
437                    attachmentData = (byte[]) data;
438                } else if (data instanceof String) {
439                    attachmentData = ((String) data).getBytes();
440                } else {
441                    attachmentData = data.toString().getBytes();
442                }
443
444                DataSource dataSource = new ByteArrayDataSource(attachmentData, attachFilename, contentType);
445                MimeBodyPart attachment = new MimeBodyPart();
446
447                attachment.setDataHandler(new DataHandler(dataSource));
448                attachment.setFileName(attachFilename);
449                multiPart.addBodyPart(attachment);
450                if (contentId!=null) {
451                        attachment.addHeader("Content-ID", contentId); // because this makes perfect sense
452                }
453                // sing with me: one of these APIs is not like the other
454                if ("inline".equals(disposition)) {
455                        attachment.setDisposition(Part.INLINE);
456                } else {
457                        attachment.setDisposition(Part.ATTACHMENT);
458                }
459                
460                
461                isMultipart = true;
462            }
463        }
464
465        // only make this a multi-part message if we have attachments
466        if (isMultipart) {
467                
468            msg.setContent(multiPart);
469        } else {
470                if (bodyText!=null) {
471                        msg.setText(bodyText);
472                } else if (bodyHtml!=null) {
473                        throw new UnsupportedOperationException("HTML text supplied without plain text. Test this before using.");
474                        // msg.setContent(bodyHtml, "us-ascii", "html");
475                        // or perhaps setText() then modify the Content-Type header. Yeah. That'll work.
476                }
477                /*
478                if (!isBlank(bodyText)) {
479                msg.setText(bodyText);
480            }
481            */
482        }
483        
484        // authentication is performed here. We can supply a 
485        // password details at the mail Session creation stage, but I could
486        // never get that to work properly :(
487        
488        Transport tr = session.getTransport("smtp");
489        if (username!=null && password!=null) {
490            tr.connect(host, username, password);
491        } else {
492            tr.connect();
493        }
494        msg.saveChanges(); 
495        tr.sendMessage(msg, msg.getAllRecipients());
496        tr.close();
497    }
