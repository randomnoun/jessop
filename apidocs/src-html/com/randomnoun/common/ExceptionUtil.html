021/**
022 * Exception utilities class.
023 *
024 * <p>This class contains static utility methods for handling and manipulating exceptions;
025 * the only one you're likely to call being 
026 * {@link #getStackTraceWithRevisions(Throwable, ClassLoader, int, String)},
027 * which extracts CVS revision information from classes to produce an annotated (and highlighted)
028 * stack trace.
029 * 
030 * <p>When passing in {@link java.lang.ClassLoader}s to these methods, you may want to try one of 
031 * <ul>
032 * <li>this.getClass().getClassLoader()
033 * <li>Thread.currentThread().getContextClassLoader()
034 * </ul>
035 * 
036 * @blog http://www.randomnoun.com/wp/2012/12/17/marginally-better-stack-traces/
037 * 
038 * @author knoxg

066    /**
067     * Converts an exception's stack trace to a string. If the exception passed
068     * to this function is null, returns the empty string.
069     *
070     * @param e exception
071     * 
072     * @return string representation of the exception's stack trace
073     */
074    public static String getStackTrace(Throwable e) {
075        if (e == null) {
076            return "";
077        }
078        StringWriter writer = new StringWriter();
079        e.printStackTrace(new PrintWriter(writer));
080        return writer.toString();
081    }

083    /**
084     * Converts an exception's stack trace to a string. Each stack trace element
085     * is annotated to include the CVS revision Id, if it contains a public static
086     * String element containing this information. 
087     * 
088     * <p>Stack trace elements whose classes begin with the specified highlightPrefix
089     * are also marked, to allow easier debugging. Highlights used can be text
090     * (which will insert the string "=>" before relevent stack trace elements), or 
091     * HTML (which will render the stack trace element between &lt;b&gt; and &lt;/b&gt; tags.
092     * 
093     * <p>If HTML highlighting is enabled, then the exception message is also HTML-escaped.
094     *
095     * @param e exception
096     * @param loader ClassLoader used to read stack trace element revision information
097     * @param highlight One of the HIGHLIGHT_* constants in this class
098     * @param highlightPrefix A prefix used to determine which stack trace elements are
099     *   rendered as being 'important'. (e.g. "<tt>com.randomnoun.common.</tt>"). Multiple
100     *   prefixes can be specified, if separated by commas.   
101     * 
102     * @return string representation of the exception's stack trace
103     */
104    public static String getStackTraceWithRevisions(Throwable e, ClassLoader loader, int highlight, String highlightPrefix) {
105        if (e == null) {
106            return "(null)";
107        }
108        StringBuffer sb = new StringBuffer();
109        String additionalMessageSource = null;
110        String additionalMessage = null; // because some Exceptions are <strike>retarded</strike> of questionable quality
111
112        // was replacing 'e' here with e.getTarget() if e was a bsh.TargetError, 
113        // but should probably just tack this onto the .getCause() chain stead
114                
115        try {
116                if (e.getClass().getName().equals("com.spotify.docker.client.exceptions.DockerRequestException")) {
117                        Method m;
118                                m = e.getClass().getMethod("message");
119                                additionalMessageSource = "DockerRequestException.message()=";
120                                additionalMessage = (String) m.invoke(e);
121                                if (additionalMessage!=null) { additionalMessage = additionalMessage.trim(); }
122                }
123                } catch (SecurityException e1) {
124                        // ignore - just use original exception
125                } catch (NoSuchMethodException e1) {
126                        // ignore - just use original exception
127                } catch (IllegalArgumentException e1) {
128                        // ignore - just use original exception
129                } catch (IllegalAccessException e1) {
130                        // ignore - just use original exception
131                } catch (InvocationTargetException e1) {
132                        // ignore - just use original exception
133                }
134        
135        String s = e.getClass().getName();
136        String message = e.getLocalizedMessage();
137        if (highlight==HIGHLIGHT_HTML || highlight==HIGHLIGHT_HTML_WITH_CSS) {
138                sb.append(escapeHtml((message != null) ? (s + ": " + message) : s ));
139        } else {
140                sb.append((message != null) ? (s + ": " + message) : s );
141        }
142        
143        if (additionalMessage!=null) {
144                sb.append('\n');
145            if (highlight==HIGHLIGHT_HTML || highlight==HIGHLIGHT_HTML_WITH_CSS) {
146                sb.append(escapeHtml(additionalMessageSource));
147                sb.append(escapeHtml(additionalMessage));
148            } else {
149                sb.append(additionalMessageSource);
150                sb.append(additionalMessage);
151            }
152        }
153        sb.append('\n');
154        
155        // dump the stack trace for the top-level exception
156        StackTraceElement[] trace = null;
157        trace = e.getStackTrace();
158        
159        Map<String, JarMetadata> jarMetadataCache = new HashMap<String, JarMetadata>();
160        
161        for (int i=0; i < trace.length; i++) {
162            sb.append(getStackTraceElementWithRevision(jarMetadataCache, trace[i], loader, highlight, highlightPrefix) + "\n");
163        }
164        Throwable cause = getCause(e);
165        if (cause != null) {
166            sb.append(getStackTraceWithRevisionsAsCause(jarMetadataCache, cause, trace, loader, highlight, highlightPrefix));
167        }
168        return sb.toString();
169    }

171    /** Returns the 'Caused by...' exception text for a chained exception, performing the
172     * same stack trace element reduction as performed by the built-in {@link java.lang.Throwable#printStackTrace()}
173     * class.
174     * 
175     * <p>Note that the notion of 'Suppressed' exceptions introduced in Java 7 is not 
176     * supported by this implementation.
177     *  
178     * @param e the cause of the original exception
179     * @param causedTrace the original exception trace
180     * @param loader ClassLoader used to read stack trace element revision information
181     * @param highlight One of the HIGHLIGHT_* constants in this class
182     * @param highlightPrefix A prefix used to determine which stack trace elements are
183     *   rendered as being 'important'. (e.g. "<tt>com.randomnoun.common.</tt>"). Multiple
184     *   prefixes can be specified, if separated by commas.
185     *   
186     * @return the 'caused by' component of a stack trace
187     */
188    private static String getStackTraceWithRevisionsAsCause(Map<String, JarMetadata> jarMetadataCache, Throwable e, StackTraceElement[] causedTrace, ClassLoader loader, int highlight, String highlightPrefix) {
189        
190        StringBuffer sb = new StringBuffer();
191        
192        // Compute number of frames in common between this and caused
193        StackTraceElement[] trace = e.getStackTrace();
194        int m = trace.length-1;
195        int n = causedTrace.length-1;
196        while (m >= 0 && n >=0 && trace[m].equals(causedTrace[n])) {
197            m--; n--;
198        }
199        int framesInCommon = trace.length - 1 - m;
200
201        String s = e.getClass().getName();
202        String message = e.getLocalizedMessage();
203        if (message==null) {  // org.json.simple.parser.ParseException doesn't set message 
204                message = e.toString(); 
205                if (s.equals(message)) { message = null; }
206        } 
207        sb.append("Caused by: ");
208        if (highlight==HIGHLIGHT_HTML || highlight==HIGHLIGHT_HTML_WITH_CSS) {
209                sb.append(escapeHtml((message != null) ? (s + ": " + message) : s ));
210        } else {
211                sb.append((message != null) ? (s + ": " + message) : s );
212        }
213        sb.append("\n");
214        
215        for (int i=0; i <= m; i++) {
216            sb.append(getStackTraceElementWithRevision(jarMetadataCache, trace[i], loader, highlight, highlightPrefix) + "\n");
217        }
218        if (framesInCommon != 0)
219            sb.append("\t... " + framesInCommon + " more\n");
220
221        // Recurse if we have a cause
222        Throwable ourCause = getCause(e);
223        if (ourCause != null) {
224            sb.append(getStackTraceWithRevisionsAsCause(jarMetadataCache, ourCause, trace, loader, highlight, highlightPrefix));
225        }
226        return sb.toString();
227    }

229    /** Returns a single stack trace element as a String, with highlighting
230     * 
231     * @param jarMetadataCache jar-level metadata cached during the traversal of the stackTraceElements.
232     * @param ste the StackTraceElement
233     * @param loader ClassLoader used to read stack trace element revision information
234     * @param highlight One of the HIGHLIGHT_* constants in this class
235     * @param highlightPrefix A prefix used to determine which stack trace elements are
236     *   rendered as being 'important'. (e.g. "<tt>com.randomnoun.common.</tt>"). Multiple
237     *   prefixes can be specified, separated by commas.
238     *   
239     * @return the stack trace element as a String, with highlighting
240     */
241    private static String getStackTraceElementWithRevision(Map<String, JarMetadata> jarMetadataCache, 
242        StackTraceElement ste, ClassLoader loader,
243        int highlight, String highlightPrefix) 
244    {
245        // s should be something like:
246        // javax.servlet.http.HttpServlet.service(HttpServlet.java:740) or
247        // java.net.Socket.<init>(Socket.java:425), which should be HTML escaped
248        String s;
249        if (highlightPrefix==null || highlight==HIGHLIGHT_NONE) {
250                s = "    at " + ste.toString();
251        } else {
252                boolean isHighlighted = (highlight!=HIGHLIGHT_NONE && isHighlighted(ste.getClassName(), highlightPrefix));
253                if (isHighlighted) {
254                        if (highlight==HIGHLIGHT_HTML) {
255                            s = "    at <b>" + escapeHtml(ste.toString()) + "</b>";  
256                        } else if (isHighlighted && highlight==HIGHLIGHT_HTML_WITH_CSS) {
257                                s = "    at <span class=\"stackTrace-highlight\">" + escapeHtml(ste.toString()) + "</span>";
258                        } else if (isHighlighted && highlight==HIGHLIGHT_TEXT) {
259                            s = " => at " + ste.toString();
260                        } else { 
261                            throw new IllegalArgumentException("Unknown highlight " + highlight);
262                        }
263                } else { // !isHighlighted
264                        if (highlight==HIGHLIGHT_HTML || highlight==HIGHLIGHT_HTML_WITH_CSS) {
265                                s = "    at " + escapeHtml(ste.toString());
266                        } else {
267                                s = "    at " + ste.toString();
268                        }
269                } 
270        }
271        
272        int openBracketPos = s.lastIndexOf("(");
273        int closeBracketPos = s.lastIndexOf(")");
274        if (openBracketPos!=-1 && closeBracketPos!=-1) {
275            try {
276                // add maven, git and cvs revision info
277                String className = ste.getClassName();
278                JarMetadata jarMetadata = getJarMetadata(jarMetadataCache, loader, className);
279                String revision = getClassRevision(loader, className);
280                if (revision != null) {
281                        if (highlight==HIGHLIGHT_HTML_WITH_CSS) {
282                                s = s.substring(0, closeBracketPos) + "<span class=\"stackTrace-revision\">" + revision + "</span> " + s.substring(closeBracketPos);
283                        } else {
284                                s = s.substring(0, closeBracketPos) + ", " + revision + s.substring(closeBracketPos);
285                        }
286                }
287                if (jarMetadata != null) {
288                        if (jarMetadata.groupId!=null && jarMetadata.artifactId!=null) {
289                                if (highlight==HIGHLIGHT_HTML_WITH_CSS) {
290                                        s = s.substring(0, openBracketPos + 1) + "<span class=\"stackTrace-maven\">" + jarMetadata.groupId + ":" + jarMetadata.artifactId + ":" + jarMetadata.version + "</span> " + s.substring(openBracketPos + 1);
291                                } else {
292                                        s = s.substring(0, openBracketPos + 1) + jarMetadata.groupId + ":" + jarMetadata.artifactId + ":" + jarMetadata.version + ", " + s.substring(openBracketPos + 1);
293                                }
294                        }
295                        if (jarMetadata.gitRevision!=null) {
296                                if (highlight==HIGHLIGHT_HTML_WITH_CSS) {
297                                        s = s.substring(0, openBracketPos + 1) + "<span class=\"stackTrace-git\">" + jarMetadata.gitRevision + "</span> " + s.substring(openBracketPos + 1);
298                                } else {
299                                        s = s.substring(0, openBracketPos + 1) + jarMetadata.gitRevision.substring(0, NUM_CHARACTERS_GIT_REVISION) + ", " + s.substring(openBracketPos + 1);
300                                }
301                        }
302                }
303            } catch (Exception e2) {
304                // logger.error(e2);
305            } catch (NoClassDefFoundError ncdfe) {
306            }
307        }
308        return s;
309    }

311    /** Returns true if the provided className matches the highlightPrefix pattern supplied, 
312     * false otherwise
313     *  
314     * @param className The name of a class (i.e. the class contained in a stack trace element)
315     * @param highlightPrefix A prefix used to determine which stack trace elements are
316     *   rendered as being 'important'. (e.g. "<tt>com.randomnoun.common.</tt>"). Multiple
317     *   prefixes can be specified, separated by commas.
318     *   
319     * @return true if the provided className matches the highlightPrefix pattern supplied, 
320     *   false otherwise
321     */
322    private static boolean isHighlighted(String className, String highlightPrefix) {
323        if (highlightPrefix.contains(",")) {
324                String[] prefixes = highlightPrefix.split(",");
325                boolean highlighted = false; 
326                for (int i=0; i<prefixes.length && !highlighted; i++) {
327                        highlighted = className.startsWith(prefixes[i]);
328                }
329                return highlighted;
330        } else {
331                return className.startsWith(highlightPrefix);
332        }
333    }

354    /** Returns the JAR metadata (including the git revision and mvn groupId:artifactId co-ordinates) of the JAR containing a class. 
355     * 
356     * @param loader The classLoader to use
357     * @param className The class we wish to retrieve
358     * 
359     * @return A JarMetadata class, or null
360     * 
361     * @throws ClassNotFoundException
362     * @throws SecurityException
363     * @throws NoSuchFieldException
364     * @throws IllegalArgumentException
365     * @throws IllegalAccessException
366     */    
367    private static JarMetadata getJarMetadata(Map<String, JarMetadata> jarMetadataCache, ClassLoader loader, String className) 
368                throws ClassNotFoundException, SecurityException, NoSuchFieldException, 
369                IllegalArgumentException, IllegalAccessException 
370    {
371        if (className.indexOf('$')!=-1) {
372            className = className.substring(0, className.indexOf('$'));
373        }
374        
375        Class clazz = Class.forName(className, true, loader);
376        
377        // this is stored in the build.properties file, which we'll have to get from the JAR containing the class
378        // see http://stackoverflow.com/questions/1983839/determine-which-jar-file-a-class-is-from
379        int revisionMethod = 0;
380        
381        String uri = clazz.getResource('/' + clazz.getName().replace('.', '/') + ".class").toString();
382        if (uri.startsWith("file:")) {
383                // this happens in eclipse; e.g. 
384                //   uri=file:/C:/Users/knoxg/workspace-4.4.1-sts-3.6.2/.metadata/.plugins/org.eclipse.wst.server.core/tmp0/wtpwebapps/jacobi-web/WEB-INF/classes/com/jacobistrategies/web/struts/Struts1Shim.class
385                // use the current class loader's build.properties if available
386                revisionMethod = 1;
387        } else if (uri.startsWith("jar:file:")) {
388                revisionMethod = 2;
389        } else {
390            int idx = uri.indexOf(':');
391            String protocol = idx == -1 ? "(unknown)" : uri.substring(0, idx);
392            // logger.warn("unknown protocol " + protocol + " in classpath uri '" + uri + "'");
393            revisionMethod = -1;
394        }
395
396        JarMetadata md = null;
397        if (revisionMethod == 1) {
398                // get the revision from the supplied class loader's build.properties, if there is one
399                md = new JarMetadata();
400                InputStream is = loader.getResourceAsStream("/build.properties");
401                if (is==null) {
402                        // logger.warn("missing build.properties");
403                        return null;
404                } else {
405                        Properties props = new Properties();
406                        try {
407                                props.load(is);
408                                md.gitRevision = props.getProperty("git.buildNumber");
409                                if ("${buildNumber}".equals(md.gitRevision)) { md.gitRevision = null; }
410                        } catch (IOException ioe) {
411                                // ignore;
412                        }
413                }
414                
415        } else if (revisionMethod == 2) {
416                // get the revision from the containing JAR
417                
418                int idx = uri.indexOf('!');
419                //As far as I know, the if statement below can't ever trigger, so it's more of a sanity check thing.
420                if (idx == -1) {
421                        // logger.warn("unparseable classpath uri '" + uri + "'");
422                        return null;
423                }
424                String fileName;
425                try {
426                    fileName = URLDecoder.decode(uri.substring("jar:file:".length(), idx), Charset.defaultCharset().name());
427                } catch (UnsupportedEncodingException e) {
428                    throw new IllegalStateException("default charset doesn't exist. Your VM is borked.");
429                }
430                
431                // use the cache if it exists
432                if (jarMetadataCache!=null) {
433                        md = jarMetadataCache.get(fileName);
434                }
435                if (md==null) {
436                        md = new JarMetadata();
437                        jarMetadataCache.put(fileName, md);
438                        File f = new File(fileName); // .getAbsolutePath();
439                        
440                        // get the build.properties from this JAR
441                        // logger.debug("Getting build.properties for " + className + " from " + f.getPath());
442                        try {
443                                ZipInputStream zis = new ZipInputStream(new FileInputStream(f));
444                                try {
445                                        ZipEntry ze = zis.getNextEntry();
446                                        while (ze!=null && (md.gitRevision==null || md.version==null)) {
447                                                // logger.debug(ze.getName());
448                                                if (ze.getName().equals("build.properties") || ze.getName().equals("/build.properties")) {
449                                                        // logger.debug(ze.getName());
450                                                        Properties props = new Properties();
451                                                        props.load(zis);
452                                                        md.gitRevision = props.getProperty("git.buildNumber");
453                                                }
454                                                if (ze.getName().endsWith("/pom.properties")) { 
455                                                        // e.g. META-INF/maven/edu.ucla.cs.compilers/jtb/pom.xml, pom.properties
456                                                        // should probably check full path to prevent false positives
457                                                        Properties props = new Properties();
458                                                        props.load(zis);
459                                                        md.groupId = props.getProperty("groupId");
460                                                        md.artifactId = props.getProperty("artifactId");
461                                                        md.version = props.getProperty("version");
462                                                }
463                                                ze = zis.getNextEntry();
464                                        }
465                                } finally {
466                                        zis.close();
467                                }
468                        } catch (IOException ioe) {
469                                // ignore;
470                        }
471                }
472        }
473        return md;
474    }

476    /** Returns a string describing the CVS revision number of a class. 
477     * 
478     * <p>The class is initialised as part of this process, which may cause a number of 
479     * exceptions to be thrown. 
480     * 
481     * @param loader The classLoader to use
482     * @param className The class we wish to retrieve
483     * 
484     * @return The CVS revision string, in the form "ver 1.234"
485     * 
486     * @throws ClassNotFoundException
487     * @throws SecurityException
488     * @throws NoSuchFieldException
489     * @throws IllegalArgumentException
490     * @throws IllegalAccessException
491     */    
492    private static String getClassRevision(ClassLoader loader, String className) 
493                throws ClassNotFoundException, SecurityException, NoSuchFieldException, 
494                IllegalArgumentException, IllegalAccessException 
495    {
496        if (className.indexOf('$')!=-1) {
497            className = className.substring(0, className.indexOf('$'));
498        }
499        
500        // if this is in a CVS repository, each class has it's own _revision public final static String variable
501        Class clazz = Class.forName(className, true, loader);
502        Field field = null;
503        try {
504                field = clazz.getField("_revision");
505        } catch (NoSuchFieldException nsfe) {
506                
507        }
508        String classRevision = null;
509        if (field!=null) {
510                classRevision = (String) field.get(null);
511            
512                // remove rest of $Id$ text 
513                int pos = classRevision.indexOf(",v ");
514                if (pos != -1) {
515                    classRevision = classRevision.substring(pos + 3);
516                    pos = classRevision.indexOf(' ');
517                    classRevision = "ver " + classRevision.substring(0, pos);
518                }
519        }
520        return classRevision;
521    }

558    /** Returns a list of all the messages contained within a exception (following the causal
559     * chain of the supplied exception). This may be more useful to and end-user, since it
560     * should not contain any references to Java class names.
561     * 
562     * @param throwable an exception
563     * 
564     * @return a List of Strings
565     */
566    public static List<String> getStackTraceSummary(Throwable throwable) {
567        List<String> result = new ArrayList<String>();
568        while (throwable!=null) {
569                result.add(throwable.getMessage());
570                throwable = getCause(throwable);
571                // I think some RemoteExceptions have non-standard caused-by chains as well...
572                if (throwable==null) {
573                        if (throwable instanceof java.sql.SQLException) {
574                                throwable = ((java.sql.SQLException) throwable).getNextException();
575                        } 
576                }
577        }
578        return result;
579    }

581    /** Returns the cause of an exception, or null if not known.
582     * If the exception is a a <code>bsh.TargetError</code>, then the cause is determined
583     * by calling the <code>getTarget()</code> method, otherwise this method will
584     * return the same value returned by the standard Exception 
585     * <code>getCause()</code> method.
586     * 
587     * @param e the cause of an exception, or null if not known.
588     * 
589     * @return the cause of an exception, or null if not known.
590     */
591    private static Throwable getCause(Throwable e) {
592        Throwable cause = null;
593        if (e.getClass().getName().equals("bsh.TargetError")) {
594            try {
595                if (e.getClass().getName().equals("bsh.TargetError")) {
596                        Method m;
597                                m = e.getClass().getMethod("getTarget");
598                                cause = (Throwable) m.invoke(e);
599                }
600                } catch (SecurityException e1) {
601                        // ignore - just use original exception
602                } catch (NoSuchMethodException e1) {
603                        // ignore - just use original exception
604                } catch (IllegalArgumentException e1) {
605                        // ignore - just use original exception
606                } catch (IllegalAccessException e1) {
607                        // ignore - just use original exception
608                } catch (InvocationTargetException e1) {
609                        // ignore - just use original exception
610                }
611        } else {
612                cause = e.getCause();
613        }
614        return cause;
615    }

617    /**
618     * Returns the HTML-escaped form of a string. Any <tt>&amp;</tt>,
619     * <tt>&lt;</tt>, <tt>&gt;</tt>, and <tt>"</tt> characters are converted to
620     * <tt>&amp;amp;</tt>, <tt>&amp;lt;<tt>, <tt>&amp;gt;<tt>, and
621     * <tt>&amp;quot;</tt> respectively.
622     *
623     * @param string the string to convert
624     *
625     * @return the HTML-escaped form of the string
626     */
627    static public String escapeHtml(String string) {
628        if (string == null) {
629            return "";
630        }
631
632        char c;
633        StringBuffer sb = new StringBuffer(string.length());
634
635        for (int i = 0; i < string.length(); i++) {
636            c = string.charAt(i);
637
638            switch (c) {
639                case '&':
640                    sb.append("&amp;");
641                    break;
642                case '<':
643                    sb.append("&lt;");
644                    break;
645                case '>':
646                    sb.append("&gt;");
647                    break;
648                case '\"':
649                    sb.append("&quot;");
650                    break;
651                default:
652                    sb.append(c);
653            }
654        }
655
656        return sb.toString();
657    }
