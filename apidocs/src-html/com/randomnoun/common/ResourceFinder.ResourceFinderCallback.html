170        /** Tests a resource name against the search criteria specified in this object
171         * 
172         * @param resourceName the last component of a resource name
173         * 
174         * @return true if the resource passes the search criteria, false otherwise
175         */
176        public boolean matches(String resourceName) {
177                if (resourceName==null) { throw new NullPointerException("null string"); }
178                if (ignoreCase) {
179                        switch(matchType) {
180                                case MATCHTYPE_EXACT: return searchTerm.equalsIgnoreCase(resourceName);
181                                case MATCHTYPE_REGEX: return searchPattern.matcher(resourceName).find();
182                                case MATCHTYPE_STARTSWITH: return resourceName.toUpperCase().startsWith(searchTerm);
183                                case MATCHTYPE_CONTAINS: return resourceName.toUpperCase().contains(searchTerm);
184                                default: throw new IllegalStateException("Illegal matchType '" + matchType + "'");
185                        }
186                } else {
187                        switch(matchType) {
188                                case MATCHTYPE_EXACT: return searchTerm.equals(resourceName);
189                                case MATCHTYPE_REGEX: return searchPattern.matcher(resourceName).find();
190                                case MATCHTYPE_STARTSWITH: return resourceName.startsWith(searchTerm);
191                                case MATCHTYPE_CONTAINS: return resourceName.contains(searchTerm);
192                                default: throw new IllegalStateException("Illegal matchType '" + matchType + "'");
193                        }
194                }
195        }

198        /** An {@link java.io.InputStream} wrapper which updates an internal md5/sha1 digest
199         * as the stream is being read.
200         */
201        public static class HashGeneratingInputStream extends InputStream {
202                InputStream is;
203                MessageDigest algorithm1, algorithm2;
204                
205                public HashGeneratingInputStream(InputStream is) {
206                        this.is = is;
207                        try {
208                                algorithm1 = MessageDigest.getInstance("MD5");
209                                algorithm2 = MessageDigest.getInstance("SHA-1");
210                        } catch (NoSuchAlgorithmException nsae) {
211                                throw (IllegalStateException) new IllegalStateException(
212                                        "Invalid crypto config").initCause(nsae);
213                        }
214                algorithm1.reset();
215                algorithm2.reset();
216
217                }
218                
219                @Override
220                public int read() throws IOException {
221                        int result = is.read();
222                        if (result != -1) {
223                                algorithm1.update((byte) result);
224                                algorithm2.update((byte) result);
225                        }
226                        return result;
227                }
228                public int available() throws IOException {
229                        return is.available();
230                }
231                public void close() {
232                        // ignored;
233                }
234                public void mark(int readlimit) {
235                        is.mark(readlimit);
236                }
237        public boolean markSupported() {
238                return is.markSupported();
239        }
240        public int read(byte[] b)  throws IOException {
241                int result = is.read(b);
242                if (result!=-1) {
243                        algorithm1.update(b, 0, result);
244                        algorithm2.update(b, 0, result);
245                }
246                return result;
247        }
248        public int read(byte[] b, int off, int len)  throws IOException {
249                int result = is.read(b, off, len);
250                if (result!=-1) {
251                        algorithm1.update(b, off, result);
252                        algorithm2.update(b, off, result);
253                }
254                        return result;
255        }
256        public void reset() throws IOException {
257                is.reset();
258        }
259        public long skip(long n) throws IOException {
260                return is.skip(n);
261        }
262        /** Returns the MD5 digest of all input that has been read by this InputStream so far,
263         * in a hexadecimal String form */
264        public String getMd5() {
265                byte messageDigest[] = algorithm1.digest();
266                //System.err.println("md5 messageDigest is " + messageDigest.length + " bytes");
267                StringBuffer hexString = new StringBuffer();
268                for (int i=0;i<messageDigest.length;i++) {
269                        hexString.append(Integer.toString( ( messageDigest[i] & 0xff ) + 0x100, 16).substring( 1 ));
270                }
271                return hexString.toString();
272        }
273        /** Returns the SHA1 digest of all input that has been read by this InputStream so far,
274         * in a hexadecimal String form */
275                public String getSha1() {
276                byte messageDigest[] = algorithm2.digest();
277                //System.err.println("sha1 messageDigest is " + messageDigest.length + " bytes");
278                StringBuffer hexString = new StringBuffer();
279                for (int i=0;i<messageDigest.length;i++) {
280                        hexString.append(Integer.toString( ( messageDigest[i] & 0xff ) + 0x100, 16).substring( 1 ));
281                }
282                return hexString.toString();
283        }

336        /** Class which defines a callback which sends names and resource hashes to System.out.
337         */
338        public static class HashingResourceFinderCallback implements ResourceFinderCallback {
339                
340                int resourceIndex = 0;
341                boolean showHashes = false;
342        
343                public HashingResourceFinderCallback() {
344                }
345                
346                public ResourceFinderCallbackResult preProcess(String resourceName, long filesize, long timestamp, InputStream inputStream) throws IOException {
347                        ResourceFinderCallbackResult rfcbr = new ResourceFinderCallbackResult();
348                        rfcbr.replaceInputStream = new HashGeneratingInputStream(inputStream);
349                        return rfcbr;
350                }
351                
352                public ResourceFinderCallbackResult postProcess(String resourceName, long filesize, long timestamp, InputStream inputStream, ResourceFinderCallbackResult preProcessResult) throws IOException {
353                        ResourceFinderCallbackResult rfcbr = new ResourceFinderCallbackResult();
354                        try {
355                                SimpleDateFormat sdf = new SimpleDateFormat("dd/MM/yyyy HH:mm:ss");
356                                String md5, sha1;
357                                if (inputStream instanceof HashGeneratingInputStream) {
358                                        HashGeneratingInputStream hgis = (HashGeneratingInputStream) inputStream;
359                                        // pump the rest of the bits through this stream
360                                        byte[] buffer = new byte[4096];
361                                        while (inputStream.read(buffer) != -1) { /* nothing */ }
362                                        md5 = hgis.getMd5();
363                                        sha1 = hgis.getSha1();
364                                } else {
365                                        MessageDigest algorithm1, algorithm2;
366                                        try {
367                                                algorithm1 = MessageDigest.getInstance("MD5");
368                                                algorithm2 = MessageDigest.getInstance("SHA1");
369                                        } catch (NoSuchAlgorithmException nsae) {
370                                                throw (IllegalStateException) new IllegalStateException(
371                                                        "Invalid crypto config").initCause(nsae);
372                                        }
373                                algorithm1.reset();
374                                algorithm2.reset();
375                                        byte[] buffer = new byte[4096];
376                                        int bytesRead;
377                                        while ((bytesRead = inputStream.read(buffer)) != -1) {
378                                                algorithm1.update(buffer, 0, bytesRead);
379                                                algorithm2.update(buffer, 0, bytesRead);
380                                }
381                                byte messageDigest1[] = algorithm1.digest();
382                                byte messageDigest2[] = algorithm2.digest();
383                                StringBuffer hexString1 = new StringBuffer();
384                                StringBuffer hexString2 = new StringBuffer();
385                                for (int i=0; i<messageDigest1.length; i++) {
386                                        hexString1.append(Integer.toString( ( messageDigest1[i] & 0xff ) + 0x100, 16).substring( 1 ));
387                                }
388                                for (int i=0; i<messageDigest2.length; i++) {
389                                        hexString2.append(Integer.toString( ( messageDigest2[i] & 0xff ) + 0x100, 16).substring( 1 ));
390                                }
391                                md5 = hexString1.toString();
392                                sha1 = hexString2.toString();
393                                }
394                        System.out.println("[" + resourceIndex + "] " + resourceName + " " + (filesize==-1 ? "(unknown)" : String.valueOf(filesize)) + 
395                                " " + sdf.format(new Date(timestamp)) + " " + md5 + " " + sha1 );
396                        } catch (IllegalArgumentException iae) {
397                                // not sure if this is needed any more
398                                throw new IOException("IllegalArgumentException processing ZipInputStream", iae);
399                        }
400                        resourceIndex++;
401                        return rfcbr;
402                }
403        }

407         * 
408         * <p>This class uses '#' as a separator between the filesystem and files contained
409         * within archives; e.g. test.jar#abc.txt refers to abc.txt in test.jar.
410         * 
411         * <p>For comparison, Tangosol seems to use '!', includes a leading slash and 
412         * includes a protocol-like identifier at the beginning
413         * (e.g. jar:file:test.jar!/abc.txt). If a constructor is supplied which only provides
414         * a ZipInputStream (i.e. no filename is available), then resources will be returned 
415         * starting with the '#' character.
416         * 
417         */
418        public static class DisplayResourceFinderCallback implements ResourceFinderCallback {
419                
420                public final static int DUMP_NAMES = -1;
421                public final static int DUMP_NAMES_AND_RESOURCES = -2;
422                public final static int DUMP_RESOURCES = 0;
423        
424                int dumpType = 0;
425                List<Integer> dumpResourceNumbers;
426                int maxDumpResourceNumber = -1;
427                int resourceIndex = 0;
428                boolean verbose = false;
429                boolean manifests = false;
430                boolean decompile = false;
431                String searchContents = null;
432                boolean searchContentsIgnoreCase = false;
433
434                // with all the trimmings
435                public DisplayResourceFinderCallback(int dumpType, List<Integer> dumpResourceNumbers, boolean verbose, boolean manifests, boolean decompile, String searchContents, boolean searchContentsIgnoreCase) {
436                        // System.out.println("2 searchContents is " + searchContents);
437                        this.dumpResourceNumbers = dumpResourceNumbers;
438                        this.dumpType = dumpType;
439                        this.verbose = verbose;
440                        this.manifests = manifests;
441                        this.decompile = decompile;
442                        this.searchContents = searchContents;
443                        this.searchContentsIgnoreCase = searchContentsIgnoreCase;
444                        if (dumpResourceNumbers != null) {
445                                for (Integer drn : dumpResourceNumbers) {
446                                        maxDumpResourceNumber = Math.max(maxDumpResourceNumber, drn.intValue());

451                private ResourceFinderCallbackResult dump(String resourceName, long filesize, long timestamp, InputStream inputStream) throws IOException {
452                        ResourceFinderCallbackResult rfcr = new ResourceFinderCallbackResult(); 
453                        if ((dumpType == DUMP_NAMES_AND_RESOURCES || dumpType == DUMP_NAMES) &&
454                                (dumpResourceNumbers==null || dumpResourceNumbers.contains(new Integer(resourceIndex)))
455                           )
456                        {
457                                if (verbose) {
458                                        SimpleDateFormat sdf = new SimpleDateFormat("dd/MM/yyyy HH:mm:ss");
459                                        System.out.println("[" + resourceIndex + "] " + resourceName + " " + (filesize==-1 ? "(unknown)" : String.valueOf(filesize)) + " " + sdf.format(new Date(timestamp)));
460                                } else if (searchContents == null) {
461                                        System.out.println("[" + resourceIndex + "] " + resourceName);
462                                } else if (searchContents != null) {
463                                        int pos = -1;
464                                        // @TODO this assumes we never search for strings containing newlines
465                                        LineNumberReader lnr = new LineNumberReader(new InputStreamReader(inputStream));
466                                        if (searchContentsIgnoreCase) {
467                                                searchContents = searchContents.toLowerCase();
468                                                String line = lnr.readLine();
469                                                if (line!=null) {
470                                                        pos = line.toString().toLowerCase().indexOf(searchContents.toLowerCase());
471                                                        while (line!=null && pos==-1) {
472                                                                line = lnr.readLine();
473                                                                pos = line==null ? -1 : line.toString().toLowerCase().indexOf(searchContents.toLowerCase());
474                                                        }
475                                                }
476                                        } else {
477                                                String line = lnr.readLine();
478                                                if (line!=null) {
479                                                        pos = line.toString().indexOf(searchContents);
480                                                        while (line!=null && pos==-1) {
481                                                                line = lnr.readLine();
482                                                                pos = line==null ? -1 :  line.toString().indexOf(searchContents);
483                                                        }
484                                                }
485                                        }
486                                        if (pos!=-1) {
487                                                System.out.println("[" + resourceIndex + "] [line " + lnr.getLineNumber() + ", col " + pos + "] " + resourceName);
488                                        }
489                                }
490                        }  
491                        if ((dumpType == DUMP_NAMES_AND_RESOURCES || dumpType == DUMP_RESOURCES) &&
492                                (dumpResourceNumbers==null || dumpResourceNumbers.contains(new Integer(resourceIndex)))
493                                ) {
494                                // InputStream is = ResourceFinder.getResourceStream(resourceName);
495                                if (manifests) {
496                                        BufferedReader br = new BufferedReader(new InputStreamReader(inputStream));
497                                        String line = br.readLine();
498                                        while (line != null) {
499                                                int len = line.length();
500                                                while (line.length() > 0 && line.charAt(0) == ' ') {
501                                                        line = line.substring(1);
502                                                }
503                                                System.out.print(line);
504                                                if (len != 70) { System.out.println(); }
505                                                line = br.readLine();
506                                        }
507                                } else if (decompile) {
508                                        // hopefully this is deleted when the VM exits
509                                        // @TODO: we need to grab all inner classes for this class as well
510                                        File tmpFile = File.createTempFile("resourceFinder", ".class");
511                                        FileOutputStream fos = new FileOutputStream(tmpFile); 
512                                        StreamUtil.copyStream(inputStream, fos, 1024);
513                                        fos.close();
514                                        try {
515                                                String result = ProcessUtil.exec(new String[] { "jad", "-lnc", "-p", tmpFile.getCanonicalPath() });
516                                                System.out.println(result);
517                                        } catch (ProcessUtil.ProcessException pe) {
518                                                throw (IOException) new IOException("Problem executing jad").initCause(pe);
519                                        }
520                                        
521                                } else {
522                                        StreamUtil.copyStream(inputStream, System.out, 1024);
523                                }
524                                if (resourceIndex == maxDumpResourceNumber) {
525                                        // don't bother continuing this search if we're found the last resource being searched for
526                                        rfcr.abort = true;
527                                }
528
529                                // we don't insert an additional newline when dumping the contents of just one file
530                                // so that stdout redirection still does something useful
531                                if (dumpType == DUMP_NAMES_AND_RESOURCES && 
532                                        (dumpResourceNumbers==null || dumpResourceNumbers.size()>1)) {
533                                        System.out.println();
534                                }
535                        }
536                        resourceIndex++;
537                        return rfcr;
538                }

569        /** Common code to both the File and ZipInputStream constructors */
570        private void init(String searchTerm, int matchType, boolean ignoreCase, 
571                ResourceFinderCallback callback) 
572        {
573                this.matchType = matchType;
574                this.ignoreCase = ignoreCase;
575                this.showAll = false;
576                this.ignoreErrors = false;
577                this.callback = callback;
578
579                // @TODO clean this up a bit
580                if (ignoreCase) {
581                        switch (matchType) {
582                                case MATCHTYPE_EXACT: break;
583                                case MATCHTYPE_REGEX: searchPattern = Pattern.compile(searchTerm, Pattern.CASE_INSENSITIVE); break;
584                                case MATCHTYPE_STARTSWITH: searchTerm = searchTerm.toUpperCase(); break;
585                                case MATCHTYPE_CONTAINS: searchTerm = searchTerm.toUpperCase(); break;
586                                default: throw new IllegalStateException("Illegal matchType '" + matchType + "'");
587                        }
588                } else {
589                        switch (matchType) {
590                                case MATCHTYPE_EXACT: break;
591                                case MATCHTYPE_REGEX: searchPattern = Pattern.compile(searchTerm); break;
592                                case MATCHTYPE_STARTSWITH: break;
593                                case MATCHTYPE_CONTAINS: break;
594                                default: throw new IllegalStateException("Illegal matchType '" + matchType + "'");
595                        }
596                }
597                this.searchTerm = searchTerm;
598                
599        }

854        /** Returns a list of resources within the supplied archive, 
855         * and archives contained within this archive
856         * 
857         * @param zipInputStream the archive to search
858         * @param prefix a prefix which is included in any results returned by this
859         *   method. By convention, this prefix should end with a '#' to separate it
860         *   from resources found within the resource.
861         * 
862         * @return a List of Strings, in the syntax defined in the class javadoc
863         * 
864         * @throws IOException 
865         */
866
867        public void findResourceInZip(ZipInputStream zipInputStream, String prefix) throws IOException {
868                if (maxArchiveDepth!=-1 && currentArchiveDepth>=maxArchiveDepth) {
869                        return;
870                }
871                // System.err.println("findResourceInZip(" + prefix + "):" + currentArchiveDepth);
872                currentArchiveDepth++;
873                
874                // System.out.println("Searching in " + prefix);
875                ZipEntry zipEntry = null;
876                try {
877                        zipEntry = zipInputStream.getNextEntry();
878                } catch (EOFException oefe) {
879                        ignorableException("Error retrieving first entry in zip '" + prefix.substring(0, prefix.length()-1) + "'", oefe);
880                        currentArchiveDepth--;
881                        return;
882                } catch (ZipException ze) {
883                        ignorableException("Error retrieving first entry in zip '" + prefix.substring(0, prefix.length()-1) + "'", ze);
884                        currentArchiveDepth--;
885                        return;
886                } catch (IllegalArgumentException iae) {
887                        // can occur in ZipInputStream.getUTF8String
888                        ignorableException("Error retrieving first entry in zip '" + prefix.substring(0, prefix.length()-1) + "'", iae);
889                        currentArchiveDepth--;
890                        return;
891                }
892                while (zipEntry != null) {
893                        String name = zipEntry.getName();
894                        String shortName = name;
895
896                        // on unix, it's possible to get directory entries (trailing '/'s) within ZIPs; on windows this doesn't seem to happen
897                        while (shortName.endsWith("/")) { shortName = shortName.substring(0, shortName.length() - 1); }
898                        while (shortName.endsWith("\\")) { shortName = shortName.substring(0, shortName.length() - 1); }
899                        if (shortName.indexOf('/')!=-1) { shortName = shortName.substring(shortName.lastIndexOf('/') + 1); }
900                        if (shortName.indexOf('\\')!=-1) { shortName = shortName.substring(shortName.lastIndexOf('\\') + 1); }
901
902                        // may need to do case-sensitive match
903                        /* commenting this out temporarily
904
905                        if (showVersions && name.equalsIgnoreCase("META-INF/MANIFEST.MF")) {
906                                // treat this as a property file. Which is wrong, because it's got insane line breaks
907                                // but good enough for retrieving version data
908                                
909                                Properties props = new Properties();
910                                props.load(zipInputStream);
911                                if (props.getProperty("Specification-Version")!=null) {
912                                        // there's also an Implementation-Version, but this appears to be the same
913                                        // maven2 doesn't write these entries. perhaps.
914                                        // @TODO something
915                                }
916                        }
917                        */
918                        
919                        InputStream inputStreamToProcess = zipInputStream;
920                        
921                        // might just be easier to add .reset() to ZipInputStream
922                        ResourceFinderCallbackResult rfcbResult = null;
923                        if (isArchive(name)) {
924                                try {
925                                        if (matches(shortName) || showAll) {
926                                                rfcbResult = callback.preProcess(prefix + name, zipEntry.getSize(), zipEntry.getTime(), inputStreamToProcess);
927                                                if (rfcbResult!=null && rfcbResult.replaceInputStream!=null) {
928                                                        inputStreamToProcess = rfcbResult.replaceInputStream;
929                                                }
930                                                if (rfcbResult!=null && rfcbResult.abort) {
931                                                        this.abort = true; break;
932                                                }
933                                        }
934                                        ZipInputStream zis = new ZipInputStream(inputStreamToProcess);
935                                        findResourceInZip(zis, prefix + name + "#");
936                                        zipInputStream.closeEntry();
937                                } catch (EOFException eofe) {
938                                        // can trigger "java.io.EOFException: Unexpected end of ZLIB input stream" errors
939                                        ignorableException("Error reading zip '" + prefix + name + "'", eofe);
940                                } catch (ZipException ze) {
941                                        ignorableException("Error reading zip '" + prefix + name + "'", ze);
942                                }
943                        }
944                        if (matches(shortName) || showAll) {
945                                rfcbResult = callback.postProcess(prefix + name, zipEntry.getSize(), zipEntry.getTime(), inputStreamToProcess, rfcbResult);
946                                if (rfcbResult!=null && rfcbResult.abort) {
947                                        this.abort = true; break;
948                                }
949                        }
950                        
951                        try {
952                                zipEntry = zipInputStream.getNextEntry();
953                        } catch (EOFException oefe) {
954                                ignorableException("Error retrieving next entry in zip '" + prefix.substring(0, prefix.length()-1) + "'; after '"+ name + "'", oefe);
955                                break;
956                        } catch (ZipException ze) {
957                                ignorableException("Error retrieving next entry in zip '" + prefix.substring(0, prefix.length()-1) + "'; after '"+ name + "'", ze);
958                                break;
959                        } catch (IllegalArgumentException iae) {
960                                // can occur in ZipInputStream.getUTF8String
961                                ignorableException("Error retrieving next entry in zip '" + prefix.substring(0, prefix.length()-1) + "'; after '"+ name + "'", iae);
962                                break;
963                        } catch (IOException ioe) {
964                                // may occur after dodgy CRCs:
965                                // invalid entry CRC (expected 0xab633fa2 but got 0xc30a2df7)
966                                // Exception in thread "main" java.io.IOException: Push back buffer is full
967                                if (ioe.getMessage().contains("Push back buffer")) {
968                                        ignorableException("Error retrieving next entry in zip '" + prefix.substring(0, prefix.length()-1) + "'; after '"+ name + "'", ioe);
969                                        break;
970                                } else {
971                                        currentArchiveDepth--;
972                                        throw ioe;
973                                }
974                        }
975                        if (abort) { break ; }
976                }
977                currentArchiveDepth--;
978        }

981         * 
982         * @param resourceName a resource name, as defined by the class javadoc
983         * 
984         * @return the resource as an InputStream  
985         * 
986         * @throws FileNotFoundException the resource could not be found
987         * @throws IOException the resource could not be read

989        public static InputStream getResourceStream(String resourceName) throws IOException {
990                int pos = resourceName.indexOf("#");
991                if (pos == -1) {
992                        return new FileInputStream(resourceName);
993                } else {
994                        String filename = resourceName.substring(0, pos);
995                        String component = resourceName.substring(pos + 1);
996                        ZipInputStream zis = new ZipInputStream(new FileInputStream(filename));
997                        return getResourceComponent(zis, component, resourceName);
998                }
999        }

1001        /** Private method to recursively search within an archive for a file
1002         * 
1003         * @param zipInputStream input stream to search
1004         * @param component resource name fragment, separated by '#' characters
1005         * @param fullResource full resource name (only used in exception messages)  
1006         * 
1007         * @return the input stream 
1008         * 
1009         * @throws IOException the input stream cannot be read
1010         */
1011        public static InputStream getResourceComponent(ZipInputStream zipInputStream, String component, String fullResource) throws IOException {
1012                int pos = component.indexOf("#");
1013                String filename = component;
1014                String subComponent = null;
1015                if (pos != -1) {
1016                        filename = component.substring(0, pos);
1017                        subComponent = component.substring(pos + 1);
1018                }
1019                ZipEntry zipEntry = zipInputStream.getNextEntry();
1020                while (zipEntry != null) {
1021                        if (zipEntry.getName().equals(filename)) {
1022                                if (subComponent == null) {
1023                                        return zipInputStream;
1024                                } else {
1025                                        return getResourceComponent(new ZipInputStream(zipInputStream), subComponent, fullResource); 
1026                                }
1027                        }
1028                        zipEntry = zipInputStream.getNextEntry();
1029                }
1030                throw new FileNotFoundException("Could not find component '" + filename + "' in '" + fullResource + "'");
1031        }

1219
1220                if (showAll) {

1224                                System.out.println(usage());
1225                                throw new IllegalArgumentException("Expected resource search term");
1226                        }
1227                        searchTerm = args[argIndex++];
1228                }
1229

1231                if (showHashes) {

1233                } else {

1235                          searchContents, searchContentsIgnoreCase);

1237                
1238                ResourceFinder resourceFinder = new ResourceFinder(searchTerm, matchType, ignoreCase, new File("."), callback);
1239                resourceFinder.setFollowSymLinks(followSymlinks);
1240                resourceFinder.setShowAll(showAll);
1241                resourceFinder.setIgnoreErrors(ignoreErrors);
1242                if (maxArchiveDepth != -1) { resourceFinder.setMaxArchiveDepth(maxArchiveDepth); }
