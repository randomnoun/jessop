008/**
009 * An object representing a variable that can be passed in as a positional parameter.
010 *
011 * <p>See SqlGenerator for a description of how this class is intended to be used.
012 *
013 * 
014 * @author knoxg
015 */
016public class PositionalParameter
017{
018    /** The name of this positional parameter. */
019    private String name;
020
021    /** Create a new Positional Parameter
022     *
023     * @param name A name for this positional parameter, which is returned
024     *   in the VAR_PARAMETERS list when the parameter is referenced in generated SQL.
025     */
026    public PositionalParameter(String name)
027    {
028        this.name = name;
029    }
030
031    /** Retrieves the name of this positional parameter */
032    public String getName()
033    {
034        return name;
035    }
036}
