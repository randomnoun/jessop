020/**
021 * This class traverses (and evaluates) an expression AST graph. It
022 * traverses the AST in depth-first order (by extending the ObjectDepthFirst
023 * visitor class generated by jtb). This class implements the
024 * 'Visitor' design pattern.
025 *
026 * <p>Each visit() method in this class takes two parameters, the
027 * first being the node being visited, and the second contains
028 * an EvalContext object containing the variables and functions which can
029 * be referenced within the expression. The visitor (i.e. this class) is then
030 * passed to it's children nodes using the nodes' accept() method (generated
031 * by jtb).
032 *
033 * <p>Variables can be any wrappered primitive type (e.g. Long, Integer, ...),
034 * although mathematical evaluation (+, -, /, etc) are currently only
035 * implemented for Doubles and Longs.
036 *
037 * <p>This class can be considered thread-safe, since it holds no instance-wide state.
038 * (All methods that require access to a EvalContext have it passed in through a
039 * method parameter). i.e. many threads may use the same Evaluator instance to
040 * evaluate different expressions.
041 *
042 * <p>NB: In the javadocs below, "lhs" refers to the left-hand-side value of any binary
043 * operation, and "rhs" refers to the right-hand-side value; e.g. in "3 + 4",
044 * the lhs is "3", the rhs is "4" and the op is "+".
045 *
046 * 
047 * @author knoxg
048 */
049public class Evaluator
050    extends GJDepthFirst<Object, EvalContext> {

060    /** Binary operation interface; takes two parameters, returns a result. */
061    interface BinaryOp {
062        Object op(Object a, Object b);
063    }
064
065    /** A binary operation on two parameters whose types are guaranteed to be Long. */
066    abstract class LongOp implements BinaryOp {
067        public Object op(Object a, Object b) {
068            long longA = ((Long) a).longValue();
069            long longB = (b == null ? 0 : ((Long) b).longValue());
070            return new Long(longOp(longA, longB));
071        }
072        abstract long longOp(long a, long b);
073    }
074
075    /** A binary operation on two parameters whose types are guaranteed to be Double. */
076    abstract class DoubleOp implements BinaryOp {
077        public Object op(Object a, Object b) {
078            double doubleA = ((Double) a).doubleValue();
079            double doubleB = (b == null ? 0 : ((Double) b).doubleValue());
080            return new Double(doubleOp(doubleA, doubleB));
081        }
082        abstract double doubleOp(double a, double b);
083    }

085    /** Coerce parameter 'b' into a compatible type of parameter 'a', to be used in a
086     *  binary math op. If 'b' can't be coerced into 'a', then throw an EvalException.
087     *
088     * @param a First binary op parameter (the type of which we are casting to)
089     * @param b Second binary op parameter (the value we are casting)
090     * @return The value of 'b', coerced to be the same type of value 'a'
091     */
092    public Object coerceType(Object a, Object b) {
093        // don't coerce nulls
094        if (a == null) { return null; }
095
096        Class<? extends Object> clazz = a.getClass();
097        if (b == null || clazz.isInstance(b)) { return b; }
098
099        if (a instanceof Date) {
100            if (b instanceof String) {
101                // convert string to date
102                DateFormat sdf = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss");
103                try {
104                    return sdf.parse((String) b);
105                } catch (java.text.ParseException pe) {
106                    throw new EvalException("Could not parse date '" + b + "': " + pe.getMessage());
107                }
108            }
109        }
110
111        if (a instanceof Double) {
112            if (b instanceof Number) {
113                return new Double(((Number) b).doubleValue());
114            }
115        }
116
117        if (a instanceof Long) {
118            if (b instanceof Number) {
119                return new Long(((Number) b).longValue());
120            }
121        }
122
123                if (a instanceof BigDecimal) {
124                        if (b instanceof Number) {
125                                // not sure if this keeps enough precision, but good enough for testing 
126                                return new BigDecimal(((Number) b).toString());  
127                        }
128                }
129
130        throw new EvalException("cannot coerce " + b.getClass().getName() + " to " + clazz.getName());
131    }

133    /** Binary math operation. Takes two parameters, the name of the op (only used in
134     * exception text), and BinaryOp classes for all supported types. (We only
135     * support math operations on Longs and Doubles at the moment, should be easy to
136     * add new types if required). If we were aiming for completeness, I'd implement
137     * Short, Int, Byte, Char, BigInteger and BigDecimal.
138     *
139     * <p>The rhs of the expression is coerced into the type of the lhs.
140     *
141     * @param a lhs parameter
142     * @param b rhs parameter
143     * @param op name of op
144     * @param longOp operation to perform if lhs is Long
145     * @param doubleOp operation to perform if lhs in Double
146     * @return the evaluated result of the op
147     */
148    public Object mathOp(Object a, Object b, String op, BinaryOp longOp, BinaryOp doubleOp) {
149        b = coerceType(a, b);
150
151        if (a instanceof Long) {
152            return longOp.op(a, b);
153        } else if (a instanceof Double) {
154            return doubleOp.op(a, b);
155        }
156
157        throw new EvalException("'" + op + "' can only operator on numeric types (found '" + a + "' of type " + a.getClass().getName() + ")");
158    }

160    /** Evaluate a TopLevelExpression node. The PRE text in this javadoc corresponds to
161     * the javacc expansion of this node; e.g. in expression.jj, the rule for TopLevelExpression
162     * is
163     *
164     * <pre>
165     * void TopLevelExpression():
166     * {}
167     * {
168     *   Expression() <EOF>
169     * }
170     * </pre>
171     *
172     * <p>jtb then creates two fields in the TopLevelExpression object, "expression" (containing
173     * the Expression() node) and "nodeToken" (containing the EOF token). This is documented
174     * in the form below (this is included for all jtb-generated visit methods):
175     *
176     * <PRE>
177     * expression -> Expression()
178     * nodeToken -> &lt;EOF&gt;
179     * </PRE>
180     */
181    public Object visit(TopLevelExpression n, EvalContext context) {
182        return n.expression.accept(this, context);
183    }

185    /** Evaluate an Expression node.
186     *
187     * <PRE>
188     * conditionalAndExpression -> ConditionalAndExpression()
189     * nodeListOptional -> ( "||" ConditionalAndExpression() )*
190     * </PRE>
191     */
192    public Object visit(Expression n, EvalContext context) {
193        NodeSequence seq;
194        Object lhs;
195        Object rhs;
196
197        lhs = n.conditionalAndExpression.accept(this, context);
198
199        for (Enumeration<Node> e = n.nodeListOptional.elements(); e.hasMoreElements();) {
200            seq = (NodeSequence) e.nextElement();
201            rhs = seq.elementAt(1).accept(this, context);
202
203            if (!(lhs instanceof Boolean)) {
204                throw new EvalException("lhs must be boolean");
205            }
206
207            if (!(rhs instanceof Boolean)) {
208                throw new EvalException("rhs must be boolean");
209            }
210
211            // NB: performs short-cut evaluation 
212            boolean result = ((Boolean) lhs).booleanValue() || ((Boolean) rhs).booleanValue();
213
214            lhs = Boolean.valueOf(result);
215        }
216
217        return lhs;
218    }

220    /** Evaluate a ConditionalAndExpression node.
221     *
222     * <PRE>
223     * equalityExpression -> EqualityExpression()
224     * nodeListOptional -> ( "&&" EqualityExpression() )*
225     * </PRE>
226     */
227    public Object visit(ConditionalAndExpression n, EvalContext context) {
228        NodeSequence seq;
229        Object lhs;
230        Object rhs;
231
232        lhs = n.equalityExpression.accept(this, context);
233
234        for (Enumeration<Node> e = n.nodeListOptional.elements(); e.hasMoreElements();) {
235            seq = (NodeSequence) e.nextElement();
236            rhs = seq.elementAt(1).accept(this, context);
237            rhs = coerceType(lhs, rhs);
238
239            if (!(lhs instanceof Boolean)) {
240                throw new EvalException("lhs must be boolean");
241            }
242
243            if (!(rhs instanceof Boolean)) {
244                throw new EvalException("rhs must be boolean");
245            }
246
247            // NB: performs short-cut evaluation 
248            boolean result = ((Boolean) lhs).booleanValue() && ((Boolean) rhs).booleanValue();
249
250            lhs = Boolean.valueOf(result);
251        }
252
253        return lhs;
254    }

256    /** Evaluate a EqualityExpression node.
257     *
258     * <PRE>
259     * relationalExpression -> RelationalExpression()
260     * nodeListOptional -> ( ( "==" | "!=" ) RelationalExpression() )*
261     * </PRE>
262     */
263    @SuppressWarnings("unchecked")
264        public Object visit(EqualityExpression n, EvalContext context) {
265        NodeSequence seq;
266        Object lhs;
267        Object rhs;
268        // String[] ops = { "==", "!=" };
269
270        lhs = n.relationalExpression.accept(this, context);
271
272        for (Enumeration<Node> e = n.nodeListOptional.elements(); e.hasMoreElements();) {
273            seq = (NodeSequence) e.nextElement();
274            rhs = seq.elementAt(1).accept(this, context);
275            rhs = coerceType(lhs, rhs);
276
277            if (!(lhs==null || lhs instanceof Comparable)) {
278                throw new EvalException("Cannot compare lhs");
279            }
280
281            if (!(rhs==null || rhs instanceof Comparable)) {
282                // this may not be required (only lhs needs to be comparable to run .compareTo)
283                throw new EvalException("Cannot compare rhs");
284            }
285
286            int which = ((NodeChoice) seq.elementAt(0)).which;
287            
288            // this may throw an IllegalStateException if they're not comparable after all
289            int result = -1;
290            try {
291                // perform null checks first (allow nulls in either lhs or rhs of equality check)
292                if (lhs==null) {
293                        return (which==0) ? Boolean.valueOf(rhs==null) : Boolean.valueOf(rhs!=null);
294                } else if (rhs==null) {
295                                        return (which==0) ? Boolean.valueOf(lhs==null) : Boolean.valueOf(lhs!=null);
296                }
297                result = ((Comparable<Object>) lhs).compareTo(rhs);
298            } catch (IllegalStateException ise) {
299                throw (EvalException) new EvalException("Cannot perform evaluation").initCause(ise);
300            }
301
302            // logger.debug("Running op '" + ops[which] + "' on lhs:" + lhs + ", rhs:" + rhs);
303            switch (which) {
304                case 0: lhs = new Boolean(result == 0); break;
305                case 1: lhs = new Boolean(result != 0); break;
306                default:
307                    throw new EvalException("Internal error - unexpected relational operation");
308            }
309        }
310
311        return lhs;
312    }

314    /** Evaluate a RelationalExpression node.
315     *
316     * <PRE>
317     * additiveExpression -> AdditiveExpression()
318     * nodeListOptional -> ( ( "&lt;" | "&gt;" | "&lt;=" | "&gt;=" ) AdditiveExpression() )*
319     * </PRE>
320     */
321    @SuppressWarnings("unchecked")
322        public Object visit(RelationalExpression n, EvalContext context) {
323        NodeSequence seq;
324        Object lhs;
325        Object rhs;
326        // String[] ops = { "<", ">", "<=", ">=" };
327
328        lhs = n.additiveExpression.accept(this, context);
329
330        for (Enumeration<Node> e = n.nodeListOptional.elements(); e.hasMoreElements();) {
331            seq = (NodeSequence) e.nextElement();
332            rhs = seq.elementAt(1).accept(this, context);
333            rhs = coerceType(lhs, rhs);
334
335            if (!(lhs instanceof Comparable)) {
336                throw new EvalException("Cannot compare lhs (found type '" + lhs.getClass().getName() + "')");
337            }
338
339            if (!(rhs instanceof Comparable)) {
340                // this may not be required (only lhs needs to be comparable to run .compareTo)
341                throw new EvalException("Cannot compare rhs");
342            }
343
344            int which = ((NodeChoice) seq.elementAt(0)).which;
345            int result = -1;
346            try {
347                result = ((Comparable<Object>) lhs).compareTo(rhs);
348            } catch (IllegalStateException ise) {
349                throw (EvalException) new EvalException("Cannot perform evaluation").initCause(ise);
350            }
351
352            // logger.debug("Running op '" + ops[which] + "' on lhs:" + lhs + ", rhs:" + rhs);
353            switch (which) {
354                case 0: lhs = new Boolean(result < 0); break;
355                case 1: lhs = new Boolean(result > 0); break;
356                case 2: lhs = new Boolean(result <= 0); break;
357                case 3: lhs = new Boolean(result >= 0); break;
358                default:
359                    throw new EvalException("Internal error - unexpected relational operation");
360            }
361        }
362
363        return lhs;
364    }

366    /** Evaluate a AdditiveExpression node.
367     *
368     * <PRE>
369     * multiplicativeExpression -> MultiplicativeExpression()
370     * nodeListOptional -> ( ( "+" | "-" ) MultiplicativeExpression() )*
371     * </PRE>
372     */
373    public Object visit(AdditiveExpression n, EvalContext context) {
374        NodeSequence seq;
375        Object lhs;
376        Object rhs;
377        // String[] ops = { "+", "-" };
378
379        lhs = n.multiplicativeExpression.accept(this, context);
380
381        for (Enumeration<Node> e = n.nodeListOptional.elements(); e.hasMoreElements();) {
382            seq = (NodeSequence) e.nextElement();
383            rhs = seq.elementAt(1).accept(this, context);
384
385            int which = ((NodeChoice) seq.elementAt(0)).which;
386
387            // logger.debug("Running op '" + ops[which] + "' on lhs:" + lhs + ", rhs:" + rhs);
388            switch (which) {
389                case 0:
390                    if (lhs instanceof String) {
391                        // String addition
392                        lhs = ((String) lhs) + rhs.toString();
393                    } else {
394                        // Numeric addition
395                        lhs = mathOp(lhs, rhs, "+", 
396                          new LongOp() { public long longOp(long a, long b) { return a + b; } }, 
397                          new DoubleOp() { public double doubleOp(double a, double b) { return a + b; } });
398                    }
399                    break;
400                case 1:
401                    lhs = mathOp(lhs, rhs, "-", 
402                      new LongOp() { public long longOp(long a, long b) { return a - b; } }, 
403                      new DoubleOp() { public double doubleOp(double a, double b) { return a - b; } });
404                    break;
405                default:
406                    throw new EvalException("Internal error - unexpected additive operation");
407            }
408        }
409
410        return lhs;
411    }

417     * nodeListOptional -> ( ( "*" | "/" | "%" ) UnaryExpression() )*
418     * </PRE>
419     */

423        Object rhs;

430            rhs = seq.elementAt(1).accept(this, context);
431
432            int which = ((NodeChoice) seq.elementAt(0)).which;
433
434            // logger.debug("Running op '" + ops[which] + "' on lhs:" + lhs + ", rhs:" + rhs);
435            switch (which) {
436                case 0:
437                    lhs = mathOp(lhs, rhs, "*", 
438                      new LongOp() { public long longOp(long a, long b) { return a * b; } },

459    /** Evaluate a UnaryExpression node.
460     *
461     * <PRE>
462     * nodeChoice -> ( "~" | "!" | "-" ) UnaryExpression()
463     *       | PrimaryExpression()
464     * </PRE>
465     */
466    public Object visit(UnaryExpression n, EvalContext context) {
467        NodeSequence seq;
468        Object lhs;
469
470        if (n.nodeChoice.which == 0) {
471            seq = (NodeSequence) n.nodeChoice.choice;
472
473            int which = ((NodeChoice) seq.elementAt(0)).which;
474
475            // String op = ((NodeToken) ((NodeChoice) nl.nodes.get(0)).choice).tokenImage;
476            switch (which) {
477                case 0:
478                    // long rhs =  makeNumeric ( ((Node) nl.nodes.get(0)).accept(this, context) ); 
479                    // return new Long(~rhs); 
480                    throw new EvalException("~ not supported");
481                case 1:
482                    lhs = seq.elementAt(1).accept(this, context);
483                    if (!(lhs instanceof Boolean)) {
484                        throw new EvalException("boolean type expected");
485                    }
486                    return Boolean.valueOf(!((Boolean) lhs).booleanValue());
487                case 2:
488                    lhs = seq.elementAt(1).accept(this, context);
489                    if (!(lhs instanceof Number)) {
490                        throw new EvalException("numeric type expected");
491                    }
492                    return mathOp(lhs, null, "-", new 
493                      LongOp() { public long longOp(long a, long b) { return -a; } }, 
494                      new DoubleOp() { public double doubleOp(double a, double b) { return -a; } } );
495                default:
496                    throw new EvalException("Internal error - unexpected unary operation");
497            }
498        } else {
499            return n.nodeChoice.choice.accept(this, context);
500        }
501    }

503    /** Evaluate a PrimaryExpression node.
504     *
505     * <PRE>
506     * nodeChoice -> FunctionCall()
507     *       | Name()
508     *       | Literal()
509     *       | "(" Expression() ")"
510     * </PRE>
511     */
512    public Object visit(PrimaryExpression n, EvalContext context) {
513        NodeSequence seq;
514        int which = n.nodeChoice.which;
515
516        switch (which) {
517            case 0:
518            case 1:
519            case 2:
520                return n.nodeChoice.choice.accept(this, context);
521            case 3:
522                seq = (NodeSequence) n.nodeChoice.choice;
523                Object obj = seq.elementAt(1).accept(this, context);
524
525                ;
526
527                return obj;
528            default:
529                throw new EvalException("Internal parser error (PrimaryExpression)");
530        }
531    }

533    /** Evaluate a Name node.
534     *
535     * <PRE>
536     * nodeToken -> &lt;IDENTIFIER&gt;
537     * nodeListOptional -> ( "." &lt;IDENTIFIER&gt; )*
538     * </PRE>
539     */
540    public Object visit(Name n, EvalContext context) {
541        EvalContext evalContext = (EvalContext) context;
542        String varComponentName;
543        String varBaseName = n.nodeToken.tokenImage;
544
545        // logger.debug("Fetching var '" + varBaseName + "'");
546        if (evalContext == null) {
547            throw new EvalException("Cannot retrieve variable '" + varBaseName + "' with a null evalContext");
548        }
549
550        if (!evalContext.hasVariable(varBaseName)) {
551            throw new EvalException("Unknown variable '" + varBaseName + "'");
552        }
553
554        Object value = evalContext.getVariable(varBaseName);
555        NodeSequence seq;
556
557        /*
558           if (value == null) {
559               logger.debug(" = null");
560           } else {
561               logger.debug(" = " + value.getClass().getName() + " with value '" + value + "'");
562           }
563         */
564        for (Enumeration<Node> e = n.nodeListOptional.elements(); e.hasMoreElements();) {
565            seq = (NodeSequence) e.nextElement();
566            varComponentName = ((NodeToken) seq.elementAt(1)).tokenImage;
567
568            //logger.debug("Fetching component '" + varComponentName + "' from var '" + varBaseName + "'");
569            if (!evalContext.hasVariableComponent(value, varBaseName, varComponentName)) {
570                throw new EvalException("Unknown variable component '" + varComponentName + "' in variable '" + varBaseName + "'");
571            }
572
573            value = evalContext.getVariableComponent(value, varBaseName, varComponentName);
574            varBaseName = varBaseName + "." + varComponentName;
575        }
576
577        return value;
578    }

580    /** Evaluate a FunctionCall node.
581     *
582     * <PRE>
583     * nodeToken -> &lt;IDENTIFIER&gt;
584     * arguments -> Arguments()
585     * </PRE>
586     */
587    public Object visit(FunctionCall n, EvalContext context) {
588        @SuppressWarnings("unchecked")
589                List<Object> argumentList = (List<Object>) n.arguments.accept(this, context);
590        String functionName = n.nodeToken.tokenImage;
591        EvalContext evalContext = (EvalContext) context;
592        EvalFunction function = (EvalFunction) evalContext.getFunction(functionName);
593        if (function == null) {
594            throw new EvalException("Unknown function '" + functionName + "'");
595        }
596        /*
597        if (evalContext.getVariable(VAR_EVALUATOR) != null) {
598            evalContext.setVariable(VAR_EVALUATOR, this);
599        }
600        */
601        return function.evaluate(functionName, evalContext, argumentList);
602    }

621     *
622     * <PRE>
623     * expression -> Expression()
624     * nodeListOptional -> ( "," Expression() )*
625     * </PRE>
626     */
627    public Object visit(ArgumentList n, EvalContext context) {

635            arguments.add(seq.elementAt(1).accept(this, context));

641    /** Evaluate a Literal node.
642     *
643     * <PRE>
644     * nodeChoice -> &lt;INTEGER_LITERAL&gt;
645     *       | &lt;FLOATING_POINT_LITERAL&gt;
646     *       | &lt;CHARACTER_LITERAL&gt;
647     *       | &lt;STRING_LITERAL&gt;
648     *       | BooleanLiteral()
649     *       | NullLiteral()
650     * </PRE>
651     *
652     */
653    public Object visit(Literal n, EvalContext context) {
654        String token = null;
655
656        if (n.nodeChoice.choice instanceof NodeToken) {
657            token = ((NodeToken) n.nodeChoice.choice).tokenImage;
658        }
659
660        switch (n.nodeChoice.which) {
661            case 0:
662                return new Long(token);
663            case 1:
664                return new Double(token);
665            case 2:
666                return new Character(token.charAt(1));
667            case 3:
668                return Text.unescapeJava(token.substring(1, token.length() - 1));
669        }
670
671        // must be 'true', 'false', or 'null'
672        return n.nodeChoice.accept(this, context);
673    }

690    /** Evaluate a NullLiteral node.
691     *
692     * <PRE>
693     * nodeToken -> "null"
694     * </PRE>
695     */
