007import java.util.*;
008
009import com.randomnoun.common.jexl.eval.EvalException;
010import com.randomnoun.common.jexl.eval.EvalFunction;
011
012/**
013 * Provides variables and functions for an expression to be evaluated with
014 *
015 * @see EvalTest
016 *
017 * 
018 * @author knoxg
019 */
020public class EvalContext
021{
022    /** Contains a map of variable names to variable values */
023    private Map<String, Object> variables = new HashMap<String, Object>();
024
025    /** Contains a map of function names to EvalFunction values */
026    private Map<String, EvalFunction> functions = new HashMap<String, EvalFunction>();
027
028    /** Sets all variables accessable to an expression. */
029    public void setVariables(Map<String, Object> variables)
030    {
031        this.variables = variables;
032    }
033
034    /** Set a specific variable binding */
035    public void setVariable(String name, Object value)
036    {
037        variables.put(name, value);
038    }
039
040    /** Removes a specific variable binding */
041    public void unsetVariable(String name)
042    {
043        variables.remove(name);
044    }
045
046    /** Returns the value of a variable (used in the Evaluator class). */
047    public Object getVariable(String varName)
048    {
049        return variables.get(varName);
050    }
051
052    /** Returns true if the variable exists. */
053    public boolean hasVariable(String varName)
054    {
055        return variables.containsKey(varName);
056    }
057
058    /** Sets all variables accessible to an expression. */
059    public void setFunctions(Map<String, EvalFunction> functions)
060    {
061        this.functions = functions;
062    }
063
064    /** Set a specific function */
065    public void setFunction(String name, EvalFunction function)
066    {
067        functions.put(name, function);
068    }
069
070    /** Returns the value of a variable (used in the Evaluator class). */
071    public Object getFunction(String function)
072    {
073        return functions.get(function);
074    }
075
076    /** Returns a component of a compound variable (which must currently be a Map object).
077     * The component name is the key of the map. 
078     * e.g. if a Map variable called 'x' contains a key 'y' with the value
079     * 'z', then the following code could be used to retrieve this value:
080     *
081     * <pre>
082     *   x = evalContext.getVariable("x")
083     *   value = evalContext.getVariableComponent(x, "x", "y")  // evaluates to 'z'
084     * </pre>
085     *
086     * <p>Note that this is only useful in the Evaluator to retrieve compound variables,
087     * outside the evaluator you should be just be evaluating the value "x.y" directly.
088     *
089     * @param varObj the compound variable value that is being evaluated (as returned
090     *   by the getVariable() method)
091     * @param baseName the name of the variable
092     * @param componentName the name of the component in varObj to return.
093     *
094     * */
095    public Object getVariableComponent(Object varObj, String baseName,
096        String componentName)
097        throws EvalException
098    {
099        if (varObj == null) {
100            throw new EvalException("Can not retrieve component '" + componentName +
101                "' from '" + baseName + "'; base is null");
102        }
103
104        if (varObj instanceof Map) {
105            return ((Map<?, ?>) varObj).get(componentName);
106        } else {
107            throw new EvalException("Can not retrieve component '" + componentName +
108                "' from '" + baseName + "'; base is of type '" +
109                varObj.getClass().getName() + "'");
110        }
111    }
112
113    /**
114     * DOCUMENT ME!
115     *
116     * @param varObj DOCUMENT ME!
117     * @param baseName DOCUMENT ME!
118     * @param componentName DOCUMENT ME!
119     *
120     * @return DOCUMENT ME!
121     */
122    public boolean hasVariableComponent(Object varObj, String baseName,
123        String componentName)
124    {
125        if (varObj == null) {
126            throw new EvalException("Can not retrieve component '" + componentName +
127                "' from '" + baseName + "'; base is null");
128        }
129
130        if (varObj instanceof Map) {
131            return ((Map<?, ?>) varObj).containsKey(componentName);
132        } else {
133            throw new EvalException("Can not retrieve component '" + componentName +
134                "' from '" + baseName + "'; base is of type '" +
135                varObj.getClass().getName() + "'");
136        }
137    }
138
139}
