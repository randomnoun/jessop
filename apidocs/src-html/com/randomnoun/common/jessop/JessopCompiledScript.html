007import java.io.PrintWriter;
008import java.io.Writer;
009
010import javax.script.Bindings;
011import javax.script.Compilable;
012import javax.script.CompiledScript;
013import javax.script.ScriptContext;
014import javax.script.ScriptEngine;
015import javax.script.ScriptException;
016
017/** Container class for the target script generated by the JessopScriptBuilder from the jessop source.
018 * 
019 * <p>If the implementation language supports compilation then this class also contains
020 * the compiled form of the generated script.
021 * 
022 * @author knoxg
023 * @version $Id$
024 */
025public class JessopCompiledScript extends CompiledScript {
026
027    /** A revision marker to be used in exception stack traces. */
028    public static final String _revision = "$Id$";
029        
030        /** target implementation engine (e.g. rhino, jython) */
031        ScriptEngine engine;    
032        
033        /** generated script in target language */
034        String source;         
035
036        /** filename */
037        String filename;         
038
039        /** compiled target language source, if the implementation engine supports it */
040        CompiledScript compiledSource;
041        
042        /** the JessopExceptionConverter to use to convert runtime exceptions */
043        JessopExceptionConverter jec;
044        
045        /** the JessopBindingConverter to use to convert bindings for this engine */
046        JessopBindingsConverter jbc;
047        
048        public JessopCompiledScript(ScriptEngine engine,
049                boolean isCompileTarget,
050                String filename, String source,
051                JessopExceptionConverter jec,
052                JessopBindingsConverter jbc) throws ScriptException {
053                if (engine==null) { throw new NullPointerException("null engine"); }
054                this.filename = filename;
055                this.engine = engine;
056                this.source = source;
057                this.jec = jec;
058                this.jbc = jbc;
059                if (isCompileTarget && (engine instanceof Compilable)) {
060                        try {
061                                // could have another declaration to suppress compilation here
062                                // for rhino, the FILENAME needs to be defined at compilation time, not execution time
063                                if (filename!=null) {
064                                        // this is the default context, not the context used at runtime
065                                        ScriptContext defaultContext = engine.getContext();
066                                        defaultContext.setAttribute(ScriptEngine.FILENAME, filename, ScriptContext.ENGINE_SCOPE);
067                                        // this is the target engine, not the jessop engine
068                                        engine.put(ScriptEngine.FILENAME, filename); 
069                                }
070                                this.compiledSource = ((Compilable) engine).compile(source);
071                        } catch (Error e) {
072                                // bsh says it implements Compilable, but then throws an 'unimplemented' Error.
073                                if ("unimplemented".equals(e.getMessage())) { 
074                                        /* ignore */ 
075                                } else {
076                                        throw e;
077                                }
078                        }
079                }
080        }
081
082        /** {@inheritDoc} */
083        @Override
084        public Object eval(ScriptContext context) throws ScriptException {
085                // NOTE TO FUTURE SELF
086                // it is around this point that I think the rhino initStandardObjects() is called.
087                // it's impossible to farging step through because I don't have the src for the oracle rhino impl.
088                // what I would dearly love is to have it call initSafeStandardObjects() instead, so that
089                // people can't escape the rhino sandbox. Is that too much to ask ? Is it ?
090                // may be able to wrap this entire method in a ContextFactory.call() to set an alternate Context
091                // see http://mozilla.github.io/rhino/javadoc/org/mozilla/javascript/ContextFactory.html#call(org.mozilla.javascript.ContextAction)
092        // the rhino code is a intricate maze of twisty passages, all of which look alike.
093                
094                if (context==null) { 
095                        context = engine.getContext();
096                } else {
097                        // some engines may require us to wrap/unwrap maps and lists to 
098                        // other data structures in order to treat them as native maps/dicts and arrays 
099                        // in that engine's language
100                        
101                        // may have to convert this context to whatever this engine expects (here's looking at you, lua)
102                        if (jbc!=null) {
103                                ScriptContext newContext = engine.getContext();
104                                Bindings gs = context.getBindings(ScriptContext.GLOBAL_SCOPE);
105                        if (gs != null) {
106                                Bindings newGs = jbc.toScriptBindings(engine, newContext, gs, ScriptContext.GLOBAL_SCOPE); 
107                            newContext.setBindings(newGs, ScriptContext.GLOBAL_SCOPE);
108                        }
109                        // this should transfer the filename for lua, but doesn't. Ah. filename may not be in the source context yet.
110                        Bindings es = context.getBindings(ScriptContext.ENGINE_SCOPE);
111                        Bindings newEs = jbc.toScriptBindings(engine, newContext, es, ScriptContext.ENGINE_SCOPE); 
112                    newContext.setBindings(newEs, ScriptContext.ENGINE_SCOPE);
113                        newContext.setReader(context.getReader());
114                        newContext.setWriter(context.getWriter());
115                        newContext.setErrorWriter(context.getErrorWriter());
116                        context = newContext;
117                        } else {
118                                // see if the class changes
119                                // (could probably put this into a LuaBindingConverter instead now)
120                                
121                                ScriptContext newContext = engine.getContext();
122                                if (newContext.getClass().equals(context.getClass())) {
123                                        // it's fine
124                                } else {
125                                Bindings gs = context.getBindings(ScriptContext.GLOBAL_SCOPE);
126                                if (gs != null) {
127                                    newContext.setBindings(gs, ScriptContext.GLOBAL_SCOPE);
128                                }
129                                // this should transfer the filename for lua, but doesn't. Ah. filename may not be in the source context yet.
130                            newContext.setBindings(context.getBindings(ScriptContext.ENGINE_SCOPE), ScriptContext.ENGINE_SCOPE);
131                                newContext.setReader(context.getReader());
132                                newContext.setWriter(context.getWriter());
133                                newContext.setErrorWriter(context.getErrorWriter());
134                                context = newContext;
135                                }
136                        }
137                }
138                // get this from the jessop declaration eventually, but for now
139                PrintWriter out;
140                if (context.getWriter()==null) {  // jruby has a 'null' default writer
141                        out = new PrintWriter(System.out, true);
142                } else {
143                        out = new PrintWriter(context.getWriter(), true);
144                }
145                context.setAttribute("out",  out, ScriptContext.ENGINE_SCOPE); // should be something like SCRIPT_SCOPE, really
146                context.setAttribute(ScriptEngine.FILENAME,  filename, ScriptContext.ENGINE_SCOPE); // for engines that need this at runtime
147                Object result;
148                try {
149                        if (compiledSource!=null) {
150                                result = compiledSource.eval(context); // these throw LuaErrors instead of ScriptExceptions
151                        } else {
152                                result = engine.eval(source, context);
153                        }
154                } catch (Throwable t) {
155                        if (jec==null) {
156                                throw t;
157                        } else {
158                                throw jec.toScriptException(context, t);
159                        }
160                }
161                out.flush();
162                return result;
163        }
164
165        /** {@inheritDoc} */
166        @Override
167        public ScriptEngine getEngine() {
168                return engine;
169        }
170        
171        /** Return the generated source; can be used for debugging.
172         * 
173         * @return the generated source
174         */
175        public String getSource() {
176                return source;
177        }
