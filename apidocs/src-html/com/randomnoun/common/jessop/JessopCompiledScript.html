007import java.io.PrintWriter;
008
009import javax.script.Bindings;
010import javax.script.Compilable;
011import javax.script.CompiledScript;
012import javax.script.ScriptContext;
013import javax.script.ScriptEngine;
014import javax.script.ScriptException;
015
016/** Container class for the target script generated by the JessopScriptBuilder from the jessop source.
017 * 
018 * <p>If the implementation language supports compilation then this class also contains
019 * the compiled form of the generated script.
020 * 
021 * @author knoxg
022 */
023public class JessopCompiledScript extends CompiledScript {
024
025        
026        /** target implementation engine (e.g. rhino, jython) */
027        ScriptEngine engine;    
028        
029        /** generated script in target language */
030        String source;         
031
032        /** filename */
033        String filename;         
034
035        /** compiled target language source, if the implementation engine supports it */
036        CompiledScript compiledSource;
037        
038        /** the JessopExceptionConverter to use to convert runtime exceptions */
039        JessopExceptionConverter jec;
040        
041        /** the JessopBindingConverter to use to convert bindings for this engine */
042        JessopBindingsConverter jbc;
043        
044        public JessopCompiledScript(ScriptEngine engine,
045                boolean isCompileTarget,
046                String filename, String source,
047                JessopExceptionConverter jec,
048                JessopBindingsConverter jbc) throws ScriptException {
049                if (engine==null) { throw new NullPointerException("null engine"); }
050                this.filename = filename;
051                this.engine = engine;
052                this.source = source;
053                this.jec = jec;
054                this.jbc = jbc;
055                if (isCompileTarget && (engine instanceof Compilable)) {
056                        try {
057                                // could have another declaration to suppress compilation here
058                                // for rhino, the FILENAME needs to be defined at compilation time, not execution time
059                                if (filename!=null) {
060                                        // this is the default context, not the context used at runtime
061                                        ScriptContext defaultContext = engine.getContext();
062                                        defaultContext.setAttribute(ScriptEngine.FILENAME, filename, ScriptContext.ENGINE_SCOPE);
063                                        // this is the target engine, not the jessop engine
064                                        engine.put(ScriptEngine.FILENAME, filename); 
065                                }
066                                this.compiledSource = ((Compilable) engine).compile(source);
067                        } catch (Error e) {
068                                // bsh says it implements Compilable, but then throws an 'unimplemented' Error.
069                                if ("unimplemented".equals(e.getMessage())) { 
070                                        /* ignore */ 
071                                } else {
072                                        throw e;
073                                }
074                        }
075                }
076        }
077
078        /** {@inheritDoc} */
079        @Override
080        public Object eval(ScriptContext context) throws ScriptException {
081                // NOTE TO FUTURE SELF
082                // it is around this point that I think the rhino initStandardObjects() is called.
083                // it's impossible to farging step through because I don't have the src for the oracle rhino impl.
084                // what I would dearly love is to have it call initSafeStandardObjects() instead, so that
085                // people can't escape the rhino sandbox. Is that too much to ask ? Is it ?
086                // may be able to wrap this entire method in a ContextFactory.call() to set an alternate Context
087                // see http://mozilla.github.io/rhino/javadoc/org/mozilla/javascript/ContextFactory.html#call(org.mozilla.javascript.ContextAction)
088        // the rhino code is a intricate maze of twisty passages, all of which look alike.
089                
090                if (context==null) { 
091                        context = engine.getContext();
092                }
093                
094                // some engines may require us to wrap/unwrap maps and lists to 
095                // other data structures in order to treat them as native maps/dicts and arrays 
096                // in that engine's language
097                
098                // may have to convert this context to whatever this engine expects (here's looking at you, lua)
099                if (jbc!=null) {
100                        ScriptContext newContext = engine.getContext();
101                        Bindings gs = context.getBindings(ScriptContext.GLOBAL_SCOPE);
102                if (gs != null) {
103                        Bindings newGs = jbc.toScriptBindings(engine, newContext, gs, ScriptContext.GLOBAL_SCOPE); 
104                    newContext.setBindings(newGs, ScriptContext.GLOBAL_SCOPE);
105                }
106                // this should transfer the filename for lua, but doesn't. Ah. filename may not be in the source context yet.
107                Bindings es = context.getBindings(ScriptContext.ENGINE_SCOPE);
108                Bindings newEs = jbc.toScriptBindings(engine, newContext, es, ScriptContext.ENGINE_SCOPE); 
109            newContext.setBindings(newEs, ScriptContext.ENGINE_SCOPE);
110                newContext.setReader(context.getReader());
111                newContext.setWriter(context.getWriter());
112                newContext.setErrorWriter(context.getErrorWriter());
113                context = newContext;
114                } else {
115                        // see if the class changes
116                        // (could probably put this into a LuaBindingConverter instead now)
117                        
118                        ScriptContext newContext = engine.getContext();
119                        if (newContext.getClass().equals(context.getClass())) {
120                                // it's fine
121                        } else {
122                        Bindings gs = context.getBindings(ScriptContext.GLOBAL_SCOPE);
123                        if (gs != null) {
124                            newContext.setBindings(gs, ScriptContext.GLOBAL_SCOPE);
125                        }
126                        // this should transfer the filename for lua, but doesn't. Ah. filename may not be in the source context yet.
127                    newContext.setBindings(context.getBindings(ScriptContext.ENGINE_SCOPE), ScriptContext.ENGINE_SCOPE);
128                        newContext.setReader(context.getReader());
129                        newContext.setWriter(context.getWriter());
130                        newContext.setErrorWriter(context.getErrorWriter());
131                        context = newContext;
132                        }
133                }
134                
135                // get this from the jessop declaration eventually, but for now
136                PrintWriter out;
137                if (context.getWriter()==null) {  // jruby has a 'null' default writer
138                        out = new PrintWriter(System.out, true);
139                } else {
140                        out = new PrintWriter(context.getWriter(), true);
141                }
142                context.setAttribute("out",  out, ScriptContext.ENGINE_SCOPE); // should be something like SCRIPT_SCOPE, really
143                context.setAttribute(ScriptEngine.FILENAME,  filename, ScriptContext.ENGINE_SCOPE); // for engines that need this at runtime
144                Object result;
145                try {
146                        if (compiledSource!=null) {
147                                result = compiledSource.eval(context); // these throw LuaErrors instead of ScriptExceptions
148                        } else {
149                                result = engine.eval(source, context);
150                        }
151                } catch (Throwable t) {
152                        if (jec==null) {
153                                throw t;
154                        } else {
155                                throw jec.toScriptException(context, t);
156                        }
157                }
158                out.flush();
159                return result;
160        }
161
162        /** {@inheritDoc} */
163        @Override
164        public ScriptEngine getEngine() {
165                return engine;
166        }
167        
168        /** Return the generated source; can be used for debugging.
169         * 
170         * @return the generated source
171         */
172        public String getSource() {
173                return source;
174        }
175
176}
