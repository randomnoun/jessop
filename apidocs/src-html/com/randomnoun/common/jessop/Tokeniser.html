007import javax.script.ScriptException;
008
009import org.apache.log4j.Logger;
010
011// could have some kind of lineCountingPrintWriter, but let's just keep that in the JSB class
012// going to use unix EOLs for everything for now
013// so should I have a Lexer here as well ? hmm. skip it for now.
014
015public class Tokeniser {
016
017    /** A revision marker to be used in exception stack traces. */
018    public static final String _revision = "$Id$";
019        Logger logger = Logger.getLogger(Tokeniser.class);
020        int state;         // parse state
021        int charOffset;    // character number (from start of file); starts at 0
022        int line;          // source line number; starts at 1
023        int eline;         // expression start line. Whenever we emit anything, reset the eline to line
024        String unclosed;   // used in the error message if we hit EOF in an invalid state
025        StringBuilder sb;  // output stringBuilder
026        StringBuilder esb; // expression (or directive) stringBuilder
027
028        JessopScriptEngine jse;  // only used to access the registry of JessopScriptBuilders
029        JessopScriptBuilder jsb; // emit() methods are called on this during parsing
030        public Tokeniser(JessopScriptEngine jse, JessopScriptBuilder jsb) {
031                state = 0;
032                line = 1; eline = 1; 
033                charOffset = 0;
034                this.jse = jse;
035                this.jsb = jsb;
036                sb = new StringBuilder();
037                esb = new StringBuilder();
038        }
039        public void setJessopScriptBuilder(JessopScriptBuilder jsb) {
040                // use this to switch languages within the tokeniser
041                this.jsb = jsb;
042        }
043        public void parseChar(char ch) throws ScriptException {
044                charOffset++;
045                // logger.debug("state " + state + " ch " + ch );
046                switch (state) {
047                        case 0:  // initial state; parsing text to display
048                                if (ch=='<') {
049                                        state = 1;
050                                } else {
051                                        sb.append(ch);
052                                }
053                                break;
054                                
055                        case 1:  // parsed initial '<'
056                                if (ch=='%') {  // <% ... %> or <%= ... %>
057                                        if (sb.length()>0) {
058                                                jsb.emitText(eline, sb.toString());
059                                                sb.setLength(0);
060                                                eline = line;
061                                        }
062                                        unclosed = "<%";
063                                        state = 2;
064                                } else if (ch=='<') {
065                                        // normal '<' followed by an possible initial '<'
066                                        sb.append(ch);
067                                        unclosed = null;
068                                        state = 1;
069                                } else {
070                                        // just a normal tag
071                                        sb.append('<');
072                                        sb.append(ch);
073                                        unclosed = null;
074                                        state = 0;
075                                }
076                                break;
077                                
078                        case 2: // parsed initial '<%'
079                                if (ch == '=') {  // <%= ... %>
080                                        unclosed = "<%=";
081                                        state = 3;
082                                } else if (ch=='@') { // <%@ ... %> declaration 
083                                        unclosed = "<%@";
084                                        state = 5;
085                                } else if (ch=='!') { // <%! ... %> block
086                                        unclosed = "<%!";
087                                        state = 6;
088                                } else if (ch=='-') { // <%-- ... --%> block
089                                        unclosed = "<%--";
090                                        state = 7;
091                                } else {        // <%  ... %>   NB: no space required after '<%'
092                                        esb.append(ch);
093                                        state = 4;
094                                }
095                                break;
096                                
097                        case 3:
098                                if (ch=='%') {
099                                        state = 13;  // possibly closing % of <%= ... %> 
100                                } else {
101                                        esb.append(ch);
102                                }
103                                break;
104                                
105                        case 4:
106                                if (ch=='%') {
107                                        state = 14;  // possibly closing % of <% ... %> 
108                                } else {
109                                        esb.append(ch);
110                                }
111                                break;
112                                
113                        case 5:
114                                if (ch=='"') {
115                                        state = 16;  // start of directive attribute
116                                        esb.append(ch);
117                                } else if (ch=='%') { // closing % of <%@ ... %>
118                                        state = 15;
119                                } else {
120                                        esb.append(ch);
121                                }
122                                break;
123
124                        case 6:
125                                if (ch=='%') {
126                                        state = 16;  // possibly closing % of <%! ... %> 
127                                } else {
128                                        esb.append(ch);
129                                }
130                                break;
131                                
132                        case 7:
133                                if (ch=='-') {
134                                        state = 8;   // second '-' of starting <%-- ... --%>
135                                } else {
136                                        // could say that this is in state 4; e.g. <%-someFunction%>
137                                        // but I'm going to chuck an exception
138                                        throw new ScriptException("'<%-' can only start a '<%--' comment block", null, line);  // charOffset
139                                }
140                                break;
141                                
142                        case 8:
143                                if (ch=='-') {
144                                        state = 9;   // possibly close '-' of <%-- ... --%>
145                                } else {
146                                        // stay in state 8
147                                        // ignore comments
148                                }
149                                break;
150                                
151                        case 9:
152                                if (ch=='-') {
153                                        state = 10;  // possibly closing '--' of <%-- ... --%>
154                                } else {
155                                        state = 8;
156                                        // ignore comments
157                                }
158                                break;
159                        
160                        case 10:
161                                if (ch=='%') {   // possibly closing '--%' of <%-- ... --%>
162                                        state = 11;
163                                } else {
164                                        state = 8;
165                                        // ignore comments
166                                }
167                                
168                        case 11:
169                                if (ch=='>') {   // closing '--%>' of <%-- ... --%>
170                                        unclosed = null;
171                                        state = 0;
172                                } else {
173                                        state = 8; 
174                                }
175                                break;
176                                
177                        case 13:
178                                if (ch=='>') {   // closing '%>' of <%= ... %>
179                                        jsb.emitExpression(eline, esb.toString());
180                                        esb.setLength(0);
181                                        eline = line;
182                                        unclosed = null;
183                                        state = 0;
184                                } else {
185                                        esb.append(ch);
186                                        state = 3; 
187                                }
188                                break;
189                        
190                        case 14:
191                                if (ch=='>') {   // closing '%>' of <% ... %>
192                                        jsb.emitScriptlet(eline, esb.toString());
193                                        esb.setLength(0);
194                                        eline = line;
195                                        unclosed = null;
196                                        state = 0;
197                                } else {
198                                        esb.append(ch);
199                                        state = 4; 
200                                }
201                                break;
202                                
203                        case 15:
204                                if (ch=='>') {   // closing '%>' of <%@ ... %> declaration
205                                        jsb.emitDeclaration(eline, esb.toString());
206                                        esb.setLength(0);
207                                        eline = line;
208                                        unclosed = null;
209                                        state = 0;
210                                } else {
211                                        esb.append(ch);
212                                        state = 5; 
213                                }
214                                break;
215                        
216                        case 16:
217                                if (ch=='"') {   // closing quote of <%@ ... %> declaration attribute
218                                        esb.append(ch);
219                                        state = 5;
220                                } else {
221                                        esb.append(ch);
222                                        // stay in state 16
223                                }
224                                break;
225                }
226                
227                if (ch=='\n') { line++; }                       
228        }
229        
230        public void parseEndOfFile() throws ScriptException {
231                // emit anything that's left, raise exceptions if in invalid state
232                // logger.debug("state " + state + " EOF");
233                if (state != 0 && unclosed != null) {
234                        throw new ScriptException("unexpected EOF (unclosed '" + unclosed + "')", null, line); // charOffset
235                }
236                if (sb.length()>0) {
237                        jsb.emitText(eline, sb.toString());
238                        sb.setLength(0);
239                        eline = line;
240                }
241        }
242}
