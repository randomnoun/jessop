163        /** {@inheritDoc} */
164        @Override
165        public CompiledScript compile(Reader script) throws ScriptException {
166                try {
167                        // if (initialLanguage == null) { initialLanguage = "javascript"; }
168
169                        JessopDeclarations declarations = new JessopDeclarations();
170                        // jessop defaults
171                        declarations.setEngine(JESSOP_DEFAULT_ENGINE);
172                        declarations.setExceptionConverter(JESSOP_DEFAULT_EXCEPTION_CONVERTER);
173                        declarations.setBindingsConverter(JESSOP_DEFAULT_BINDINGS_CONVERTER);
174                        declarations.setCompileTarget(Boolean.valueOf(JESSOP_DEFAULT_COMPILE_TARGET));
175                        declarations.setSuppressEol(Boolean.valueOf(JESSOP_DEFAULT_SUPPRESS_EOL));
176
177                        // ScriptEngine defaults
178                        String filename = (String) get(ScriptEngine.FILENAME);
179                        String initialLanguage = (String) get(JessopScriptEngine.JESSOP_LANGUAGE);
180                        String initialEngine = (String) get(JessopScriptEngine.JESSOP_ENGINE);
181                        String initialExceptionConverter = (String) get(JessopScriptEngine.JESSOP_EXCEPTION_CONVERTER);
182                        String initialBindingsConverter = (String) get(JessopScriptEngine.JESSOP_BINDINGS_CONVERTER);
183                        String initialCompileTarget = (String) get(JessopScriptEngine.JESSOP_COMPILE_TARGET);
184                        String initialSuppressEol = (String) get(JessopScriptEngine.JESSOP_SUPPRESS_EOL);
185                        if (initialLanguage==null) { initialLanguage = JESSOP_DEFAULT_LANGUAGE; }
186                        if (filename!=null) { declarations.setFilename(filename); }
187                        if (initialEngine!=null) { declarations.setEngine(initialEngine); }
188                        if (initialExceptionConverter!=null) { declarations.setExceptionConverter(initialExceptionConverter); }
189                        if (initialBindingsConverter!=null) { declarations.setBindingsConverter(initialBindingsConverter); }
190                        if (initialCompileTarget!=null) { declarations.setCompileTarget(Boolean.valueOf(initialCompileTarget)); }
191                        if (initialSuppressEol!=null) { declarations.setSuppressEol(Boolean.valueOf(initialSuppressEol)); }
192                        
193                        ByteArrayOutputStream baos = new ByteArrayOutputStream(); 
194                        PrintWriter pw = new PrintWriter(baos);
195                    // new JavascriptJessopScriptBuilder(); // default for now
196                        JessopScriptBuilder jsb = ((JessopScriptEngineFactory) getFactory()).getJessopScriptBuilderForLanguage(initialLanguage);
197                        jsb.setPrintWriter(pw);
198                        Tokeniser t = new Tokeniser(this, jsb);
199                        jsb.setTokeniserAndDeclarations(t, declarations);
200                        
201                        // tokenise the script
202                        int ch = script.read();
203                        while (ch!=-1) {
204                                t.parseChar((char) ch);
205                                ch = script.read();
206                        }
207                        t.parseEndOfFile();
208                        pw.flush();
209                        
210                        // get the output from the PrintWriter
211                        String newScript = baos.toString();
212                        
213                        // the final JSB contains the final declarations that were in effect 
214                        declarations = t.jsb.getDeclarations();
215                        
216                        // get this from the jessop declaration eventally, but for now:
217                        // if the underlying engine supports compilation, then compile that here, otherwise just store the source
218                        ScriptEngine engine = new ScriptEngineManager().getEngineByName(declarations.engine);  // nashorn in JDK9
219                        if (engine==null) {
220                                throw new ScriptException("java.scriptx engine '" + declarations.engine + "' not found");
221                        }
222                        
223                        JessopBindingsConverter jbc = null;
224                        if (declarations.bindingsConverter !=null && !declarations.bindingsConverter.equals("")) {
225                                try {
226                                        jbc = (JessopBindingsConverter) 
227                                                Class.forName(declarations.bindingsConverter).newInstance();
228                                } catch (Exception e) {
229                                        throw (ScriptException) new ScriptException(
230                                          "bindingsConverter '" + declarations.bindingsConverter + "' not loaded").initCause(e);
231                                }
232                        }
233                        
234                        JessopExceptionConverter jec = null;
235                        if (declarations.exceptionConverter!=null && !declarations.exceptionConverter.equals("")) {
236                                try {
237                                        jec = (JessopExceptionConverter) 
238                                                Class.forName(declarations.exceptionConverter).newInstance();
239                                } catch (Exception e) {
240                                        throw (ScriptException) new ScriptException(
241                                          "exceptionConverter '" + declarations.exceptionConverter + "' not loaded").initCause(e);
242                                }
243                        }
244                         
245                        // com.sun.script.javascript.RhinoScriptEngine m = (com.sun.script.javascript.RhinoScriptEngine) engine;
246                        // the newScript is compiled here, if the engine supports it
247                        return new JessopCompiledScript(engine, declarations.isCompileTarget(), 
248                                declarations.getFilename(), newScript, jec, jbc);
249                        
250                } catch (IOException ioe) {
251                        throw new ScriptException(ioe);
252                }
253                
254                
255                
256        }
