007import org.apache.log4j.Logger;
008
009import com.randomnoun.common.jessop.AbstractJessopScriptBuilder;
010import com.randomnoun.common.jessop.JessopScriptBuilder;
011
012public class Python2JessopScriptBuilder extends AbstractJessopScriptBuilder implements JessopScriptBuilder {
013
014    /** A revision marker to be used in exception stack traces. */
015    public static final String _revision = "$Id$";
016        Logger logger = Logger.getLogger(Python2JessopScriptBuilder.class);
017        int outputLine = 1;        // current line number in the target script;
018        int outputCol = 1;         // current output column
019        int lastScriptletLine = 1; // the last line number of the last scriptlet (used for suppressEol)
020        int indent = 0;            // current number of spaces at start of line (we use 4-space indents)
021        public Python2JessopScriptBuilder() {
022        }
023        private void skipToLine(int line, int indent) {
024                if (outputLine > line) {
025                        // could allow, but then that'll open another can of worms 
026                        // throw new IllegalArgumentException("cannot generate output on same line as starting new python block");
027                        logger.warn("can't go back to line " + line + " (outputLine=" + outputLine + "); line numbers may be inaccurate");
028                }
029                while (outputLine < line) { print("\n"); }
030                while (outputCol < indent) { print(" "); }
031                // for (int i=0; i<indent; i++) { print(" "); }
032        }
033        private void print(String s) {
034                // logger.info("** PRINT " + s);
035                pw.print(s);
036                for (int i=0; i<s.length(); i++) {
037                        if (s.charAt(i)=='\n') { outputLine++; outputCol = 1; }
038                        else { outputCol++; }
039                }
040        }
041        private static String escapePython(String string) {
042                /* valid escapes ( https://docs.python.org/2.0/ref/strings.html )
043\a      bell
044\b      back space
045\f      form feed
046\n      newline
047\r      carriage return
048\t      horizontal tab
049\v      vertical tab
050\\      backslash
051\"      double quote
052\'      single quote
053                 */
054                
055        StringBuilder sb = new StringBuilder(string.length());
056        String escapeChars = "\u0007" + "\u0008" + "\u000f" + "\n" + "\r" + "\u0009" + "\u000b" + "\\" + "\"" + "'";
057        String backslashChars = "abfnrtv\\\"'";
058                for (int i = 0; i<string.length(); i++) {
059                        char ch = string.charAt(i);
060                        int pos = escapeChars.indexOf(ch);
061                        if (pos !=- 1) {
062                           sb.append("\\" + backslashChars.charAt(pos));
063                        
064                        } else if (ch<32 || (ch>126 && ch <= 255)) {
065                                String hex = Integer.toString(ch, 16);
066                                sb.append("\\x" + "00".substring(0, 2-hex.length()) + hex);
067                                sb.append(ch);
068                                
069                        } else if (ch<=255) {
070                                sb.append(ch);
071                                
072                        } else {
073                                throw new IllegalArgumentException("Cannot escape characters > 0xFF in python2 (found char '" + ch + "'; code=" + ((int) ch) + ")");
074                        }
075                }
076        return sb.toString();
077    }
078        
079        @Override
080        public void emitText(int line, String s) {
081                skipToLine(line, indent);
082                s = suppressEol(s, declarations.isSuppressEol() && lastScriptletLine == line);
083                print("out.write(\"" + escapePython(s) + "\");");
084                lastScriptletLine = 0; // don't suppress eols on this line
085        }
086        @Override
087        public void emitExpression(int line, String s) {
088                skipToLine(line, indent);
089                print("out.write((str) (" + s + "));"); // coerce to String
090                lastScriptletLine = 0; // don't suppress eols on this line
091        }
092        @Override
093        public void emitScriptlet(int line, String s) {
094                if (outputLine > line) {
095                        throw new IllegalArgumentException("cannot generate scriptlet output on same line as starting new python block");
096                }
097
098                skipToLine(line, indent);
099                // if there's content before the first newline, remove whitespace from the beginning
100                // so that we maintain our line/indentation position
101                int pos = s.indexOf("\n"); if (pos==-1) { pos = s.length(); }
102                int i=0;
103                while (i < s.length() && s.charAt(i)==' ') { i++; }
104                if (s.charAt(i)=='\t') { throw new IllegalArgumentException("tab indentation for python scriplets not supported"); }
105                s = s.substring(i);
106                
107                logger.debug("scriptlet is '" + s + "'");
108                print(s);
109                /* 
110                  <%
111                      for i=1..10:
112                        something
113                        somethingElse
114                  %>no idea whether this is in the for loop or not
115                  
116                  <%
117                      for i=1..10:
118                  %>presumably this is in the for loop. not sure how to terminate it though.
119                  <%
120                      pass;  # empty statement on a line could indicate end of block
121                  %>
122                   
123                      
124                 * This sort of thing is why whitespace-significant languages should burn in hellfire,
125                 * (I'm looking at you, yaml), and why things like jinja are apparently necessary
126                 */
127                
128                // get the amount of indentation on the last non-blank line
129                int endLinePos = s.length();
130                int startLinePos = s.lastIndexOf("\n");
131                while (s.substring(startLinePos+1).trim().equals("")) { 
132                        endLinePos = startLinePos; 
133                        startLinePos = s.lastIndexOf("\n", startLinePos-1); 
134                }
135                startLinePos++; // don't include the '\n'
136                
137                i = 0;
138                while (startLinePos + i < endLinePos && s.charAt(i)==' ') { i++; }
139                logger.debug("python last newline pos=" + startLinePos + ", indent on last line=" + i);
140                if (s.charAt(startLinePos + i)=='\t') { 
141                        // could support this later, perhaps
142                        throw new IllegalArgumentException("tab indentation for python scriplets not supported");
143                }
144                indent = i;
145                if (s.substring(startLinePos, endLinePos).trim().endsWith(":")) {
146                        logger.debug("last non-blank line endsWith ':', indenting by 4");
147                        // this may mean some loop constructs will now have inaccurate line numbers
148                        // e.g. <% for i in range (0,10): %><%= something %>
149                        print("\n"); 
150                        indent += 4;
151                }
152
153                // this should probably go before the indent processing above. maybe.
154                lastScriptletLine = line;
155                for (i=0; i<s.length(); i++) { if (s.charAt(i)=='\n') { lastScriptletLine++; } }
156        }
157        @Override
158        public String getLanguage() {
159                return "python2";
160        }
161        @Override
162        public String getDefaultScriptEngineName() {
163                return "jython";
164        }
