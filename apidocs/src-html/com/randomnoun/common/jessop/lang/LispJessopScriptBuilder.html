007import java.util.ArrayList;
008import java.util.List;
009
010import org.apache.log4j.Logger;
011
012import com.randomnoun.common.jessop.AbstractJessopScriptBuilder;
013import com.randomnoun.common.jessop.JessopScriptBuilder;
014
015public class LispJessopScriptBuilder extends AbstractJessopScriptBuilder implements JessopScriptBuilder {
016
017    /** A revision marker to be used in exception stack traces. */
018    public static final String _revision = "$Id$";
019        Logger logger = Logger.getLogger(LispJessopScriptBuilder.class);
020        int outputLine = 1;        // current line number in the target script;
021        int lastScriptletLine = 1; // the last line number of the last scriptlet (used for suppressEol)
022
023        public LispJessopScriptBuilder() {
024        }
025        private void skipToLine(int line) {
026                while (outputLine < line) { print("\n"); }
027        }
028        private void print(String s) {
029                // logger.info("** PRINT " + s);
030                pw.print(s);
031                for (int i=0; i<s.length(); i++) {
032                        if (s.charAt(i)=='\n') { outputLine++; } 
033                }
034        }
035        
036        
037        private static List<String> escapeLisp(String string) {
038                // lisp doesn't appear to have an inline escape syntax. so that's terrific, isn't it.
039                // instead, you need to put in ~a placeholders which are then filled by 
040                // (code-char #x000d) parameters to the format function, or in abcl, a possibly-nonstandard
041                // '#\\U000D parameter.         
042
043                // also, '~' characters need to be doubled-up.
044                List<String> result = new ArrayList<String>();
045                result.add("");
046                
047        StringBuilder sb = new StringBuilder(string.length());
048        //String escapeChars = "\u0007" + "\u0008" + "\u000f" + "\n" + "\r" + "\u0009" + "\u000b" + "\u001b" + "\\" + "\"";
049        //String backslashChars = "abfnrtve\\\"";
050                for (int i = 0; i<string.length(); i++) {
051                        char ch = string.charAt(i);
052                        if (ch=='~') {
053                                sb.append("~~");
054                        } else if (ch=='\\') {
055                                sb.append("\\\\");
056                        } else if (ch=='"') {
057                                sb.append("\\\"");
058                        } else if (ch>=32 && ch<=127) {
059                                sb.append(ch);
060                        } else {
061                                sb.append("~a");
062                                String hex = Long.toString(ch, 16);
063                                result.add("'#\\U" + "0000".substring(0, 4-hex.length()) + hex);
064                        }
065                }
066                result.set(0, sb.toString());
067                return result;
068    }
069        
070        @Override
071        public void emitText(int line, String s) {
072                //if (outputLine == line) { print ("; "); }
073                skipToLine(line);
074                s = suppressEol(s, declarations.isSuppressEol() && lastScriptletLine == line);
075                List<String> formatParams = escapeLisp(s);
076                print("(format t \"" + formatParams.get(0) + "\"");
077                for (int i=1; i<formatParams.size(); i++) {
078                        print(" " + formatParams.get(i));
079                }
080                print(")");
081                lastScriptletLine = 0; // don't suppress eols on this line
082        }
083        @Override
084        public void emitExpression(int line, String s) {
085                //if (outputLine == line) { print ("; "); }
086                skipToLine(line);
087                print("(format t \"~a\" " + s + ")");  // 'aesthetically' print the value of s
088                lastScriptletLine = 0; // don't suppress eols on this line
089        }
090        @Override
091        public void emitScriptlet(int line, String s) {
092                // if (outputLine == line) { print ("; "); }
093                skipToLine(line);
094                print(s);
095                lastScriptletLine = line;
096                for (int i=0; i<s.length(); i++) { if (s.charAt(i)=='\n') { lastScriptletLine++; } }
097        }
098        @Override
099        public String getLanguage() {
100                return "lisp";
101        }
102        @Override
103        public String getDefaultScriptEngineName() {
104                return "ABCL";
105        }
106        @Override
107        public String getDefaultExceptionConverterClassName() {
108                // return "com.randomnoun.common.jessop.engine.JRubyExceptionConverter";
109                return null;
110        }
