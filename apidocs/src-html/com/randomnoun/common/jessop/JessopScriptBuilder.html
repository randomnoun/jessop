007import java.io.PrintWriter;
008
009import javax.script.ScriptException;
010
011/** Each target language we intend to support within jessop must have an implementation of this interface. 
012 * Developers wishing to implement this interface should use the {@link AbstractJessopScriptBuilder} abstract class.
013 * 
014 * <p>Implementations of this class should return the language it supports (and the default script engine name) via the
015 * {@link #getLanguage()} and {@link #getDefaultScriptEngineName()} methods.
016 * 
017 * <p>These values will be used to select the correct JessopScriptBuilder identified in the 
018 * <tt>&lt;%@ jessop language="xxx"%&gt;</tt>
019 * declaration in the jessop source file.
020 * 
021 * <p>When this class is instantiated, the caller will invoke {@link #setPrintWriter(PrintWriter)} and 
022 * {@link #setTokeniserAndDeclarations(Tokeniser, JessopDeclarations)}.
023 * 
024 * <p>As the Tokeniser parses the jessop source file, it will invoke emit() methods on this class.
025 * As these methods are called, this class should generate code in the target language via the printWriter.
026 * 
027 * <p>Care should be taken to ensure that code in the target language script is on the same line number
028 * as the corresponding code on the input script, to make error messages more developer-friendly.
029 * 
030 * @author knoxg
031 * @version $Id$
032 */
033public interface JessopScriptBuilder {
034        // lines are source line numbers; try to keep these intact in the generated script
035        
036        /** Returns the name of the language that this scriptBuilder can parse; e.g. "javascript" or "python2".
037         * 
038         * <p>This is used to register this JessopScriptBuilder in the registry and is used to lookup the JessopScriptBuilder
039         * from the language defined in the jessop script declaration.
040         * 
041         * @return the name of the language that this scriptBuilder can parse; e.g. "javascript" or "python2".
042         */
043        String getLanguage();                                             
044        
045        /** Returns the name of the script engine that is used to evaluate this script; e.g. "rhino" or "jython"
046         * 
047         * @return the name of the script engine that is used to evaluate this script; e.g. "rhino" or "jython"
048         */
049        String getDefaultScriptEngineName();                             
050
051        /** Returns the name of the default exceptionConverter class that should be used for the default script engine,
052         * or null if no converter is required.
053         * 
054         * <p>Changing the engine in the jessop declaration will reset the the converter to null. 
055         * 
056         * @return the name of the default exceptionConverter class that should be used for the default script engine
057         */
058        String getDefaultExceptionConverterClassName();                             
059
060        /** Returns the name of the default bindingsConverter class that should be used for the default script engine,
061         * or null if no converter is required.
062         * 
063         * <p>Changing the engine in the jessop declaration will reset the the converter to null. 
064         * 
065         * @return the name of the default bindingsConverter class that should be used for the default script engine
066         */
067        String getDefaultBindingsConverterClassName();                             
068
069        
070        /** Sets the printWriter that this class will write to during emit() methods
071         * 
072         * @param pw the printWriter that this class will write to during emit() methods
073         */
074        void setPrintWriter(PrintWriter pw);
075        
076        /** Sets the tokeniser and declaration that are in effect at the start of parsing. This allows this
077         * ScriptBuilder to switch ScriptBuilders.
078         * 
079         * @param t tokeniser that is processing the jessop source
080         * @param declarations jessop declarations that are in effect
081         */
082        void setTokeniserAndDeclarations(Tokeniser t, JessopDeclarations declarations);
083        
084        /** Called by the tokeniser and requests that this JessopScriptBuilder generate code to generate some text output.
085         * 
086         * <p>This method should escape the output and generate code that outputs this text in the implementation language.
087         *  
088         * @param line jessop source line number that begins this text output
089         * @param s the text to output; may include newlines.
090         */
091        void emitText(int line, String s);        
092        
093        /** Called by the tokeniser and requests that this JessopScriptBuilder generate code to generate some evaluated output.
094         * i.e. process a <tt>&lt;%= ... %&gt;</tt> expression.
095         * 
096         * <p>This method should generate the code to evaluate and output the contents of the <tt>&lt;%= ... %&gt;</tt> expression.
097         * 
098         * <p>This method should attempt to preserve line numbers between the jessop source file and the generated code file.
099         * 
100         * @param line jessop source line number that begins this text output.
101         * @param s the contents of the <tt>&lt;%= ... %&gt;</tt> expression.
102         */
103        void emitExpression(int line, String s);  // <%= ... %>
104        
105
106        /** Called by the tokeniser and requests that this JessopScriptBuilder copy the included code into the generated script.
107         * i.e. process a <tt>&lt;% ... %&gt;</tt> scriptlet. (The 'scriptlet' term is the term used in the JSP specification)
108         * 
109         * <p>This method should generate the same code that is contained within the <tt>&lt;% ... %&gt;</tt> scriptlet.
110         * 
111         * <p>This method should attempt to preserve line numbers between the jessop source file and the generated code file.
112         * 
113         * <p>This scriptlet may not be syntactically correct by itself and may require further scriptlets to function, e.g.
114         * <pre>
115         * &lt;% if (a>b) { %&gt;
116         * something
117         * &lt;% } %&gt;
118         * </pre>
119         * 
120         * @param line jessop source line number that begins this scriptlet.
121         * @param s the contents of the <tt>&lt;%= ... %&gt;</tt> expression.
122         */
123        void emitScriptlet(int line, String s);
124
125        /** Called by the tokeniser and requests that this JessopScriptBuilder process a <tt>&lt;%@ ... %&gt;</tt> directive.
126         * 
127         * <p>The exact semantics of these declarations can be defined per-language, but 'jessop' declarations 
128         * should be processed by the AbstractJessopScriptBuilder, i.e. declarations that are in the form
129         * <tt>&lt;%@ jessop language="xxx" %&gt;</tt>
130         * 
131         * <p>Ideally, there's only one declaration, at the top of the source script.
132         * 
133         * @param line jessop source line number that begins this directive
134         * @param s the contents of the <tt>&lt;%@ ... %&gt;</tt> directive
135         * @throws ScriptException
136         */
137        void emitDeclaration(int line, String s) throws ScriptException; 
138
139        /** Returns the value of any jessop declarations that are in effect.
140         * 
141         * <p>Used to maintain declarations when switching JessopScriptBuilders, and to retrieve declarations after the script is built.
142         */
143        JessopDeclarations getDeclarations();
144        
145}
