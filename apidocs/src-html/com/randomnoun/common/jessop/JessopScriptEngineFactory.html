007import java.io.IOException;
008import java.io.InputStream;
009import java.util.Collections;
010import java.util.HashMap;
011import java.util.List;
012import java.util.Map;
013import java.util.Properties;
014import java.util.ServiceLoader;
015
016import javax.script.ScriptEngine;
017import javax.script.ScriptEngineFactory;
018
019/** A class to implement the {@link javax.script.ScriptEngineFactory} interface for the jessop templating language.
020 * 
021 * @author knoxg
022 */
023// was thinking of creating one of these per jessop language; (e.g. jessop-javascript), but 
024// will just use declarations to define the language (and default to javascript if the declaration is missing)
025// actually getNames() could return jessop-xxx variants based on what's in the registry, couldn't it.
026// can't see how to get the value of that name though, so that's unfortunate
027public class JessopScriptEngineFactory implements ScriptEngineFactory {
028
029        /** The version of the jessop 'language' that this script engine implements */
030    // may lag the release version if the syntax doesn't change
031    // 1.0    - initial release
032    // 1.0.2  - declarations can now have a bindingsConverter; syntax is still the same though
033        public static final String JESSOP_LANGUAGE_VERSION = "1.0";
034        
035        @Override
036        public String getEngineName() {
037                return "jessop";
038        }
039
040        @Override
041        public String getEngineVersion() {
042                return getMavenVersion();
043        }
044
045        @Override
046        public List<String> getExtensions() {
047                return Collections.singletonList("jessop");
048        }
049
050        @Override
051        public List<String> getMimeTypes() {
052                return Collections.singletonList("application/x.jessop");
053        }
054
055        @Override
056        public List<String> getNames() {
057                return Collections.singletonList("jessop");
058        }
059
060        @Override
061        public String getLanguageName() {
062                return "jessop";
063        }
064
065        @Override
066        public String getLanguageVersion() {
067                return JESSOP_LANGUAGE_VERSION;
068        }
069
070        @Override
071        public Object getParameter(String key) {
072                if (key.equals(ScriptEngine.ENGINE)) { return getEngineName(); }
073                else if (key.equals(ScriptEngine.ENGINE_VERSION)) { return getEngineVersion(); }
074                else if (key.equals(ScriptEngine.LANGUAGE)) { return getLanguageName(); }
075                else if (key.equals(ScriptEngine.LANGUAGE_VERSION)) { return getLanguageVersion(); }
076                else if (key.equals("THREADING")) { return "MULTITHREADED"; } // probably depends on impl language
077                else { 
078                        throw new IllegalArgumentException("Invalid key"); 
079                }
080                
081        }
082
083        // this is language dependent
084        @Override
085        public String getMethodCallSyntax(String obj, String m, String... args) {
086                 String ret = obj;
087             ret += "." + m + "(";
088             for (int i = 0; i < args.length; i++) {
089                 ret += args[i];
090                 if (i < args.length - 1) {
091                     ret += ",";
092                 }
093             }
094             ret += ")";
095             return ret;
096        }
097
098        // this is language dependent
099        @Override
100        public String getOutputStatement(String toDisplay) {
101                return "print(" + toDisplay + ")";
102        }
103
104        // this is language dependent
105        @Override
106        public String getProgram(String... statements) {
107                String retval = "<?\n";
108            int len = statements.length;
109            for (int i = 0; i < len; i++) {
110                retval += statements[i] + ";\n";
111            }
112            return retval += "?>";
113        }
114
115        @Override
116        public ScriptEngine getScriptEngine() {
117                JessopScriptEngine se = new JessopScriptEngine();
118                se.setEngineFactory(this);
119                return se;
120        }
121        
122        // versioning
123        private static Properties getBuildProperties() throws IOException {
124                InputStream is = JessopScriptEngineFactory.class.getClassLoader().getResourceAsStream("build.properties");
125        Properties props = new Properties();
126        if (is==null) {
127                return null;
128        } else {
129                props.load(is);
130                is.close();
131        }
132        return props;
133        }
134        
135        private static String getMavenVersion() {
136                try {
137                        Properties buildProperties = getBuildProperties();
138                        String mavenRelease = "(local build)";
139                        if (buildProperties!=null) {
140                                mavenRelease = buildProperties.getProperty("maven.pom.version");
141                                if (mavenRelease.equals("${pom.version}")) { mavenRelease="(local build)"; }
142                        }
143                        return mavenRelease;
144                } catch (IOException ioe) {
145                        return "(unknown)";
146                }
147        }
148
149        // could make this static, but there aren't going to be too many ScriptEngineFactory's around, surely.
150        private Map<String, Class<? extends JessopScriptBuilder>> registry = null;
151        
152        // these methods are not part of the ScriptEngineFactory interface
153        private Map<String, Class<? extends JessopScriptBuilder>> getRegistry() {
154                if (registry!=null) { return registry; }
155                
156                Map<String, Class<? extends JessopScriptBuilder>> newRegistry = new HashMap<String, Class<? extends JessopScriptBuilder>>();
157                ServiceLoader<JessopScriptBuilder> jsbLoader = ServiceLoader.load(JessopScriptBuilder.class);
158                for (JessopScriptBuilder jsb : jsbLoader) {
159                        // System.out.println(jsb.getLanguage());
160                        newRegistry.put(jsb.getLanguage(), jsb.getClass()); // may not be able to re-use these objects, so just store the Class
161                        // it'd be nice if we could dynamically register new JessopScriptEngineFactories here as well.
162                        // will set the default language through a ScriptEngine var instead
163                }
164
165                registry = newRegistry;
166                return registry;
167        }
168        
169        public JessopScriptBuilder getJessopScriptBuilderForLanguage(String language) {
170                Map<String, Class<? extends JessopScriptBuilder>> registry = getRegistry();
171                Class<? extends JessopScriptBuilder> c = registry.get(language);
172                if (c == null) { throw new IllegalArgumentException("No JessopScriptBuilder registered for language '" + language + "'"); }
173                try {
174                        return c.newInstance();
175                } catch (InstantiationException e) {
176                        throw new IllegalStateException("Could not instantiate JessopScriptBuilder for language '" + language + "'", e);
177                } catch (IllegalAccessException e) {
178                        throw new IllegalStateException("Could not instantiate JessopScriptBuilder for language '" + language + "'", e);
179                }
180        }
181
