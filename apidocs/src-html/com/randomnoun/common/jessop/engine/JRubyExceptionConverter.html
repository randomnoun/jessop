007import javax.script.ScriptContext;
008import javax.script.ScriptEngine;
009import javax.script.ScriptException;
010
011import com.randomnoun.common.jessop.JessopExceptionConverter;
012
013public class JRubyExceptionConverter implements JessopExceptionConverter {
014
015    /** A revision marker to be used in exception stack traces. */
016    public static final String _revision = "$Id$";
017
018        /* jruby doesn't return filenames or line numbers in the ScriptException object, 
019         * but these are contained in the jruby RaiseException object */
020    /* example stacktrace
021  NameError: undefined local variable or method `floob' for main:Object
022  <top> at test.jessop:6
023     */
024        @Override
025        public ScriptException toScriptException(ScriptContext scriptContext, Throwable t) {
026                if (t instanceof ScriptException) { 
027                        ScriptException se = (ScriptException) t;
028                        StackTraceElement[] stes = getFirstStackTrace(se.getCause());
029                        if (stes!=null && stes.length > 0) {
030                                return (ScriptException) new ScriptException(se.getMessage(), 
031                                        stes[0].getFileName(), stes[0].getLineNumber()).initCause(t);   
032                        }
033                }
034                return (ScriptException) new ScriptException(t.getMessage()).initCause(t);
035        }
036        
037        private StackTraceElement[] getFirstStackTrace(Throwable e) {
038                if (e instanceof org.jruby.exceptions.RaiseException) {
039                        return e.getStackTrace();
040                }
041                if (e.getCause()!=null) { return getFirstStackTrace(e.getCause()); }
042                // no RaiseException in stacktrace
043                return e.getStackTrace();
044        }
045
046}
