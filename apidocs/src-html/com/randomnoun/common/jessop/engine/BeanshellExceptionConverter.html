007import javax.script.ScriptContext;
008import javax.script.ScriptEngine;
009import javax.script.ScriptException;
010
011import com.randomnoun.common.jessop.JessopExceptionConverter;
012
013public class BeanshellExceptionConverter implements JessopExceptionConverter {
014
015    /** A revision marker to be used in exception stack traces. */
016    public static final String _revision = "$Id$";
017
018        /* bsh doesn't return filenames properly */ 
019        @Override
020        public ScriptException toScriptException(ScriptContext scriptContext, Throwable t) {
021                if (t instanceof ScriptException) { 
022                        ScriptException se = (ScriptException) t;
023                        String filename = (String) scriptContext.getAttribute(ScriptEngine.FILENAME, ScriptContext.ENGINE_SCOPE);
024                        return (ScriptException) new ScriptException(se.getMessage(), 
025                          filename, se.getLineNumber()).initCause(t);
026                }
027                return (ScriptException) new ScriptException(t.getMessage()).initCause(t);
028        }
029
030}
