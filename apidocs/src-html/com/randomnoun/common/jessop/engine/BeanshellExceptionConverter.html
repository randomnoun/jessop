007import javax.script.ScriptContext;
008import javax.script.ScriptEngine;
009import javax.script.ScriptException;
010
011import com.randomnoun.common.jessop.JessopExceptionConverter;
012
013public class BeanshellExceptionConverter implements JessopExceptionConverter {
014
015        /* bsh doesn't return filenames properly */ 
016        @Override
017        public ScriptException toScriptException(ScriptContext scriptContext, Throwable t) {
018                if (t instanceof ScriptException) { 
019                        ScriptException se = (ScriptException) t;
020                        String filename = (String) scriptContext.getAttribute(ScriptEngine.FILENAME, ScriptContext.ENGINE_SCOPE);
021                        return (ScriptException) new ScriptException(se.getMessage(), 
022                          filename, se.getLineNumber()).initCause(t);
023                }
024                return (ScriptException) new ScriptException(t.getMessage()).initCause(t);
025        }
026
027}
