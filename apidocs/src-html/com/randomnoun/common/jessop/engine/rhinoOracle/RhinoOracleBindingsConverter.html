014        @SuppressWarnings("restriction")
015        @Override
016        public Bindings toScriptBindings(ScriptEngine engine,
017                        ScriptContext newContext, Bindings bindings, int engineScope) {
018                Context cx = Context.enter();
019                try {
020                        // so hopefully initStandardObjects() is less expensive than cloning all these maps/lists
021                        // this will be a different scope to that used at runtime, which hopefully won't be a problem later
022                        ScriptableObject newScope = cx.initStandardObjects();
023                        Bindings newBindings = engine.createBindings();
024                        for (String k : bindings.keySet()) {
025                                Object v = bindings.get(k);
026                                newBindings.put(k, ScriptUtils.javaToJS(v, newScope)); 
027                                // newBindings.put(k, toScriptObject(newScope, v));
028                        }
029                        return newBindings;
030                } finally {
031                        Context.exit();
032                }
033        }
