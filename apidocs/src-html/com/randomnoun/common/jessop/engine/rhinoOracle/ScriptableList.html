022// import org.mozilla.javascript.*;

024import sun.org.mozilla.javascript.internal.*;

026import java.util.List;
027import java.util.ArrayList;
028import java.util.Collection;
029import java.util.Map;
030
031/**
032 * ScriptableList is a wrapper for java.util.List instances that allows developers
033 * to interact with them like it was a native JavaScript array.
034 */
035public class ScriptableList extends NativeJavaObject {
036
037    List<Object> list;
038    static final String CLASSNAME = "ScriptableList";
039
040    // Set up a custom constructor, for this class is somewhere between a host class and
041    // a native wrapper, for which no standard constructor class exists
042    public static void init(Scriptable scope) throws NoSuchMethodException {
043        BaseFunction ctor = new BaseFunction(scope, ScriptableObject.getFunctionPrototype(scope)) {
044            @Override
045            public Scriptable construct(Context cx, Scriptable scope, Object[] args) {
046                if (args.length > 1) {
047                    throw new EvaluatorException("ScriptableList() requires a java.util.List argument");
048                }
049                return new ScriptableList(scope, args.length == 0 ? null : args[0]);
050            }
051            @Override
052            public Object call(Context cx, Scriptable scope, Scriptable thisObj, Object[] args) {
053                return construct(cx, scope, args);
054            }
055        };
056        ScriptableObject.defineProperty(scope, CLASSNAME, ctor,
057                ScriptableObject.DONTENUM | ScriptableObject.READONLY);
058    }
059
060    /**
061     * Create a ScriptableList wrapper around a java.util.List
062     * @param scope the scope
063     * @param obj the list, possibly wrapped
064     */
065    @SuppressWarnings("unchecked")
066    private ScriptableList(Scriptable scope, Object obj) {
067        this.parent = scope;
068        if (obj instanceof Wrapper) {
069            obj = ((Wrapper) obj).unwrap();
070        }
071        if (obj instanceof List) {
072            this.javaObject = this.list = (List) obj;
073        } else if (obj instanceof Collection) {
074            this.javaObject = this.list = new ArrayList<Object>((Collection<?>) obj);
075        } else if (obj instanceof Map) {
076            this.javaObject = this.list = new ArrayList<Object>(((Map<?,?>)obj).values());
077        } else if (obj == null || obj == Undefined.instance) {
078            this.javaObject = this.list = new ArrayList<Object>();
079        } else {
080            throw new EvaluatorException("Invalid argument to ScriptableList(): " + obj);
081        }
082        this.staticType = this.list.getClass();
083        initMembers();
084        initPrototype(scope);
085    }
086
087
088    /**
089     * Create a ScriptableList wrapper around a java.util.List.
090     * @param scope the scope
091     * @param list the list instance
092     */
093    @SuppressWarnings("unchecked")
094    public ScriptableList(Scriptable scope, List list) {
095        super(scope, list, list.getClass());
096        this.list = list;
097        initPrototype(scope);
098    }
099
100    /**
101     * Set the prototype to the Array prototype so we can use array methds such as
102     * push, pop, shift, slice etc.
103     * @param scope the global scope for looking up the Array constructor
104     */
105    protected void initPrototype(Scriptable scope) {
106        Scriptable arrayProto = ScriptableObject.getClassPrototype(scope, "Array");
107        if (arrayProto != null) {
108            this.setPrototype(arrayProto);
109        }
110    }
111
112    public void delete(int index) {
113        if (list != null) {
114            try {
115                list.remove(index);
116            } catch (RuntimeException e) {
117                throw Context.throwAsScriptRuntimeEx(e);
118            }
119        } else {
120            super.delete(index);
121        }
122    }
123
124    public Object get(int index, Scriptable start) {
125        if (list == null)
126            return super.get(index, start);
127        try {
128            if (index < 0 || index >= list.size()) {
129                return Undefined.instance;
130            } else {
131                return ScriptUtils.javaToJS(list.get(index), getParentScope());
132            }
133        } catch (RuntimeException e) {
134            throw Context.throwAsScriptRuntimeEx(e);
135        }
136    }
137
138    public boolean has(int index, Scriptable start) {
139        if (list == null)
140            return super.has(index, start);
141        return index >= 0 && index < list.size();
142    }
143
144    public void put(String name, Scriptable start, Object value) {
145        if (list != null && "length".equals(name)) {
146            double d = ScriptRuntime.toNumber(value);
147            long longVal = ScriptRuntime.toUint32(d);
148            if (longVal != d) {
149                String msg = ScriptRuntime.getMessage0("msg.arraylength.bad");
150                throw ScriptRuntime.constructError("RangeError", msg);
151            }
152            int size = list.size();
153            if (longVal > size) {
154                for (int i = size; i < longVal; i++) {
155                    // push nulls as undefined is probably meaningless to java code
156                    list.add(null);
157                }
158            } else if (longVal < size) {
159                for (int i = size - 1; i >= longVal; i--) {
160                    list.remove(i);
161                }
162            }
163        } else {
164            super.put(name, start, value);
165        }
166    }
167
168    public void put(int index, Scriptable start, Object value) {
169        if (list != null) {
170            try {
171                if (index == list.size()) {
172                    list.add(ScriptUtils.jsToJava(value));
173                } else {
174                    list.set(index, ScriptUtils.jsToJava(value));
175                }
176            } catch (RuntimeException e) {
177                Context.throwAsScriptRuntimeEx(e);
178            }
179        } else {
180            super.put(index, start, value);
181        }
182    }
183
184    public Object get(String name, Scriptable start) {
185        if ("length".equals(name) && list != null) {
186            return new Integer(list.size());
187        }
188        return super.get(name, start);
189    }
190
191    public Object[] getIds() {
192        if (list == null)
193            return super.getIds();
194        int size = list.size();
195        Object[] ids = new Object[size];
196        for (int i = 0; i < size; ++i) {
197            ids[i] = new Integer(i);
198        }
199        return ids;
200    }
201
202    public String toString() {
203        if (list == null)
204            return super.toString();
205        return list.toString();
206    }
207
208    public Object getDefaultValue(Class typeHint) {
209        return toString();
210    }
211
212    public Object unwrap() {
213        return list;
214    }
215
216    public List getList() {
217        return list;
218    }
219
220    public String getClassName() {
221        return CLASSNAME;
222    }
223}
