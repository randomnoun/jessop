022// import com.sun.script.javascript.*;
023import sun.org.mozilla.javascript.internal.*;

025import java.util.Map;
026import java.util.HashMap;
027
028/**
029 * ScriptableMap is a wrapper for java.util.Map instances that allows developers
030 * to interact with them as if it were a native JavaScript object.
031 */
032public class ScriptableMap extends NativeJavaObject {
033
034    boolean reflect;
035    Map map;
036    final static String CLASSNAME = "ScriptableMap";
037
038    // Set up a custom constructor, for this class is somewhere between a host class and
039    // a native wrapper, for which no standard constructor class exists
040    public static void init(Scriptable scope) throws NoSuchMethodException {
041        BaseFunction ctor = new BaseFunction(scope, ScriptableObject.getFunctionPrototype(scope)) {
042            @Override
043            public Scriptable construct(Context cx, Scriptable scope, Object[] args) {
044                boolean reflect = false;
045                if (args.length > 2) {
046                    throw new EvaluatorException("ScriptableMap() called with too many arguments");
047                } if (args.length == 2) {
048                    reflect = ScriptRuntime.toBoolean(args[1]);
049                }
050                return new ScriptableMap(scope, args.length == 0 ? null : args[0], reflect);
051            }
052            @Override
053            public Object call(Context cx, Scriptable scope, Scriptable thisObj, Object[] args) {
054                return construct(cx, scope, args);
055            }
056        };
057        ScriptableObject.defineProperty(scope, CLASSNAME, ctor,
058                ScriptableObject.DONTENUM | ScriptableObject.READONLY);
059    }
060
061    @SuppressWarnings("unchecked")
062    private ScriptableMap(Scriptable scope, Object obj, boolean reflect) {
063        this.parent = scope;
064        this.reflect = reflect;
065        if (obj instanceof Wrapper) {
066            obj = ((Wrapper) obj).unwrap();
067        }
068        if (obj instanceof Map) {
069            this.map = (Map) obj;
070        } else if (obj == null || obj == Undefined.instance) {
071            this.map = new HashMap();
072        } else if (obj instanceof Scriptable) {
073            this.map = new HashMap();
074            Scriptable s = (Scriptable) obj;
075            Object[] ids = s.getIds();
076            for (Object id: ids) {
077                if (id instanceof String) {
078                    map.put(id, s.get((String)id, s));
079                } else if (id instanceof Number) {
080                    map.put(id, s.get(((Number)id).intValue(), s));
081                }
082            }
083        } else {
084            throw new EvaluatorException("Invalid argument to ScriptableMap(): " + obj);
085        }
086        this.javaObject = this.map;
087        this.staticType = this.map.getClass();
088        initMembers();
089        initPrototype(scope);
090
091    }
092
093    public ScriptableMap(Scriptable scope, Map map) {
094        super(scope, map, map.getClass());
095        this.map = map;
096        initPrototype(scope);
097    }
098
099    /**
100     * Set the prototype to the Object prototype so we can use object methods such as
101     * getOwnPropertyNames, hasOwnProperty, keys etc.
102     * @param scope the global scope for looking up the Object constructor
103     */
104    protected void initPrototype(Scriptable scope) {
105        Scriptable objectProto = ScriptableObject.getClassPrototype(scope, "Object");
106        if (objectProto != null) {
107            this.setPrototype(objectProto);
108        }
109    }
110
111    public Object get(String name, Scriptable start) {
112        if (map == null || (reflect && super.has(name, start))) {
113            return super.get(name, start);
114        }
115        return getInternal(name);
116    }
117
118    public Object get(int index, Scriptable start) {
119        if (map == null) {
120            return super.get(index, start);
121        }
122        return getInternal(new Integer(index));
123    }
124
125    private Object getInternal(Object key) {
126        Object value = map.get(key);
127        if (value == null) {
128            return Scriptable.NOT_FOUND;
129        }
130        return ScriptUtils.javaToJS(value, getParentScope());
131    }
132
133    public boolean has(String name, Scriptable start) {
134        if (map == null || (reflect && super.has(name, start))) {
135            return super.has(name, start);
136        } else {
137            return map.containsKey(name);
138        }
139    }
140
141    public boolean has(int index, Scriptable start) {
142        if (map == null) {
143            return super.has(index, start);
144        } else {
145            return map.containsKey(new Integer(index));
146        }
147    }
148
149    public void put(String name, Scriptable start, Object value) {
150        if (map == null || (reflect && super.has(name, start))) {
151            super.put(name, start, value);
152        } else {
153            putInternal(name, value);
154        }
155    }
156
157    public void put(int index, Scriptable start, Object value) {
158        if (map == null) {
159             super.put(index, start, value);
160         } else {
161             putInternal(new Integer(index), value);
162        }
163    }
164
165    @SuppressWarnings("unchecked")
166    private void putInternal(Object key, Object value) {
167        try {
168            map.put(key, ScriptUtils.jsToJava(value));
169        } catch (RuntimeException e) {
170            Context.throwAsScriptRuntimeEx(e);
171        }
172    }
173
174    public void delete(String name) {
175        if (map != null) {
176            try {
177                map.remove(name);
178            } catch (RuntimeException e) {
179                Context.throwAsScriptRuntimeEx(e);
180            }
181        } else {
182            super.delete(name);
183        }
184    }
185
186    public void delete(int index) {
187        if (map != null) {
188            try {
189                map.remove(new Integer(index));
190            } catch (RuntimeException e) {
191                Context.throwAsScriptRuntimeEx(e);
192            }
193        } else {
194            super.delete(index);
195        }
196    }
197
198    public Object[] getIds() {
199        if (map == null) {
200            return super.getIds();
201        } else {
202            return map.keySet().toArray();
203        }
204    }
205
206    public String toString() {
207        if (map == null)
208            return super.toString();
209        return map.toString();
210    }
211
212    public Object getDefaultValue(Class typeHint) {
213        return toString();
214    }
215
216    public Object unwrap() {
217        return map;
218    }
219
220    public Map getMap() {
221        return map;
222    }
223
224    public String getClassName() {
225        return CLASSNAME;
226    }
227}
