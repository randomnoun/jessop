001package com.randomnoun.common.jessop.engine.graaljs;
002
003import javax.script.ScriptContext;
004import javax.script.ScriptException;
005
006import com.randomnoun.common.jessop.JessopExceptionConverter;
007
008public class GraalJsExceptionConverter implements JessopExceptionConverter {
009
010        @Override
011        public ScriptException toScriptException(ScriptContext scriptContext, Throwable t) {
012                String filename = null;
013                Integer lineNumber = null;
014                String innerMessage = null;
015                if (t instanceof ScriptException) {
016                        // try to find a stack trace element with "<js>" in it
017                        Throwable chain = t;
018                        while (chain != null && filename == null) {
019                                for (StackTraceElement ste : chain.getStackTrace()) {
020                                        if (ste.getClassName().equals("<js>")) {
021                                                innerMessage = chain.getMessage();
022                                                filename = ste.getFileName();
023                                                lineNumber = ste.getLineNumber();
024                                                break;
025                                        }
026                                }
027                                chain = chain.getCause();
028                        }
029                        if (filename != null) {
030                                return (ScriptException) new ScriptException(innerMessage, filename, lineNumber == null ? -1 : lineNumber).initCause(t);
031                        } else {
032                                return (ScriptException) t;
033                        }
034                }
035                return (ScriptException) new ScriptException(t.getMessage()).initCause(t);
036        }
037
038}
