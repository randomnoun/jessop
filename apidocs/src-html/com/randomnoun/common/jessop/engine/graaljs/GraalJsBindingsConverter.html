010import javax.script.Bindings;

015import org.graalvm.polyglot.proxy.ProxyObject;
016
017import com.randomnoun.common.jessop.JessopBindingsConverter;
018
019public class GraalJsBindingsConverter implements JessopBindingsConverter {
020
021        @Override
022        public Bindings toScriptBindings(ScriptEngine engine,
023                        ScriptContext newContext, Bindings bindings, int engineScope) {
024
025                // return the same bindings and add a few graaljs-specific properties
026                // see https://github.com/graalvm/graaljs/blob/master/docs/user/ScriptEngine.md
027                bindings.put("polyglot.js.allowAllAccess", true);
028                
029                // return bindings;
030                // Bindings newBindings = engine.createBindings();
031                // bindings.put("polyglot.js.allowAllAccess", true);
032                // this fails with 'failed to set graal-js option "polyglot.js.allowAllAccess": js context is already initialized'
033                
034                Set<String> keySet = new TreeSet<String>(bindings.keySet());
035                for (String k : keySet) {
036                        Object v = bindings.get(k);
037                        bindings.put(k, toScriptObject(v)); 
038                }
039                return bindings;
040        }
041        
042        // wrap Maps and Lists in ProxyObject
043        @SuppressWarnings({ "rawtypes", "unchecked" })
044        public Object toScriptObject(Object o) {
045                Object result;
046                
047                if (o == null) {
048                        return null;
049                
050                } else if (o instanceof Map) {
051                        Map javaMap = (Map) o;
052                        Map jsMap = new HashMap<String, Object>();
053                        
054                        for (Object entryObj : javaMap.entrySet()) {
055                                Map.Entry entry = (Map.Entry) entryObj;
056                                jsMap.put((String) entry.getKey(), toScriptObject(entry.getValue()));
057                        }
058                        result = ProxyObject.fromMap(jsMap);
059
060                } else if (o instanceof List) {
061                        // return o;
062                        List javaList = (List) o;
063                        List newList = new ArrayList(javaList.size()); 
064                        for (int i=0; i < javaList.size(); i++) {
065                                newList.add(toScriptObject(javaList.get(i)));
066                        }
067                        result = newList;
068
069                // possibly array types here
070                } else if (o instanceof String) { 
071                        result = o;
072                
073                } else {
074                        result = o; 
075                }
076                return result;
077        }
