007import java.util.regex.Matcher;
008import java.util.regex.Pattern;
009
010import javax.script.ScriptContext;
011import javax.script.ScriptEngine;
012import javax.script.ScriptException;
013
014import org.luaj.vm2.LuaError;
015
016import com.randomnoun.common.jessop.JessopExceptionConverter;
017
018public class LuajExceptionConverter implements JessopExceptionConverter {
019
020        /* because lua's special */ 
021        @Override
022        public ScriptException toScriptException(ScriptContext scriptContext, Throwable t) {
023                if (t instanceof ScriptException) { 
024                        return (ScriptException) t;
025                } else if (t instanceof LuaError) {
026                        // there's a 'fileline' attribute in the LuaError, which isn't public, but is
027                        // set in the LuaClosure class to
028                        // le.fileline = (p.source != null? p.source.tojstring(): "?") + ":" 
029                        //  + (p.lineinfo != null && pc >= 0 && pc < p.lineinfo.length? String.valueOf(p.lineinfo[pc]): "?");
030                        // it's prepended to the error message
031                        
032                        LuaError le = (LuaError) t;
033                        String msg = le.getMessage();
034                        Pattern p = Pattern.compile("^(.*):([0-9+]|\\?) (.*)$", Pattern.DOTALL);
035                        Matcher m = p.matcher(msg);
036                        if (m.matches()) {
037                                // just to make things even more annoying, the
038                                //   public CompiledScript compile(Reader script) throws ScriptException
039                                // method in LuaScriptEngine hardcodes the name 'script' as the script filename
040                                // String filename = m.group(1);
041                                String filename = (String) scriptContext.getAttribute(ScriptEngine.FILENAME, ScriptContext.ENGINE_SCOPE);
042                                return (ScriptException) new ScriptException(m.group(3), filename, 
043                                        m.group(2).equals("?") ? -1 : Integer.parseInt(m.group(2))).initCause(t);
044                        } 
045                }
046                return (ScriptException) new ScriptException(t.getMessage()).initCause(t);
047        }
048
049}
