014        @Override
015        public Bindings toScriptBindings(ScriptEngine engine,
016                        ScriptContext newContext, Bindings bindings, int engineScope) {
017                Context cx = Context.enter();
018                try {
019                        // so hopefully initStandardObjects() is less expensive than cloning all these maps/lists
020                        // this will be a different scope to that used at runtime, which hopefully won't be a problem later
021                        ScriptableObject newScope = cx.initStandardObjects();
022                        Bindings newBindings = engine.createBindings();
023                        for (String k : bindings.keySet()) {
024                                Object v = bindings.get(k);
025                                newBindings.put(k, ScriptUtils.javaToJS(v, newScope)); 
026                                // newBindings.put(k, toScriptObject(newScope, v));
027                        }
028                        return newBindings;
029                } finally {
030                        Context.exit();
031                }
032        }
