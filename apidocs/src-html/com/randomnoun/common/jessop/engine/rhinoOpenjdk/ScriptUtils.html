034public class ScriptUtils {

036    /**
037     * Coerce/wrap a java object to a JS object, and mask Lists and Maps
038     * as native JS objects.
039     * @param obj the object to coerce/wrap
040     * @param scope the scope
041     * @return the wrapped/masked java object
042     */
043    @SuppressWarnings("unchecked")
044    public static Object javaToJS(Object obj, Scriptable scope) {
045        if (obj instanceof Scriptable) {
046            if (obj instanceof ScriptableObject
047                    && ((Scriptable) obj).getParentScope() == null
048                    && ((Scriptable) obj).getPrototype() == null) {
049                ScriptRuntime.setObjectProtoAndParent((ScriptableObject) obj, scope);
050            }
051            return obj;
052        } else if (obj instanceof List) {
053            return new ScriptableList(scope, (List) obj);
054        } else if (obj instanceof Map) {
055            return new ScriptableMap(scope, (Map) obj);
056        } else {
057            return Context.javaToJS(obj, scope);
058        }
059    }
060
061    /**
062     * Unwrap a JS object to a java object. This is much more conservative than
063     * Context.jsToJava in that it will preserve undefined values.
064     * @param obj a JavaScript value
065     * @return a Java object corresponding to obj
066     */
067    public static Object jsToJava(Object obj) {
068        while (obj instanceof Wrapper) {
069            obj = ((Wrapper) obj).unwrap();
070        }
071        return obj;
072    }
073
074    /**
075     * Return a class prototype, or the object prototype if the class
076     * is not defined.
077     * @param scope the scope
078     * @param className the class name
079     * @return the class or object prototype
080     */
081    public static Scriptable getClassOrObjectProto(Scriptable scope, String className) {
082        Scriptable proto = ScriptableObject.getClassPrototype(scope, className);
083        if (proto == null) {
084            proto = ScriptableObject.getObjectPrototype(scope);
085        }
086        return proto;
087    }
088
089    /**
090     * Make sure that number of arguments is valid.
091     * @param args the argument array
092     * @param min the minimum number of arguments
093     * @param max the maximum number of arguments
094     * @throws IllegalArgumentException if the number of arguments is not valid
095     */
096    public static void checkArguments(Object[] args, int min, int max) {
097        if (min > -1 && args.length < min)
098            throw new IllegalArgumentException();
099        if (max > -1 && args.length > max)
100            throw new IllegalArgumentException();
101    }
102
103    /**
104     * Get an argument as ScriptableObject
105     * @param args the argument array
106     * @param pos the position of the requested argument
107     * @return the argument as ScriptableObject
108     * @throws IllegalArgumentException if the argument can't be converted to a map
109     */
110    public static ScriptableObject getScriptableArgument(Object[] args, int pos, boolean allowNull)
111            throws IllegalArgumentException {
112        if (pos >= args.length || args[pos] == null || args[pos] == Undefined.instance) {
113            if (allowNull) return null;
114            throw ScriptRuntime.constructError("Error", "Argument " + (pos + 1) + " must not be null");
115        } if (args[pos] instanceof ScriptableObject) {
116            return (ScriptableObject) args[pos];
117        }
118        throw ScriptRuntime.constructError("Error", "Can't convert to ScriptableObject: " + args[pos]);
119    }

121    /**
122     * Get an argument as string
123     * @param args the argument array
124     * @param pos the position of the requested argument
125     * @return the argument as string
126     */
127    public static String getStringArgument(Object[] args, int pos, boolean allowNull) {
128        if (pos >= args.length || args[pos] == null || args[pos] == Undefined.instance) {
129            if (allowNull) return null;
130            throw ScriptRuntime.constructError("Error", "Argument " + (pos + 1) + " must not be null");

167    /**
168     * Try to convert an object to an int value, returning the default value if conversion fails.
169     * @param obj the value
170     * @param defaultValue the default value
171     * @return the converted value
172     */
173    public static int toInt(Object obj, int defaultValue) {
174        double d = ScriptRuntime.toNumber(obj);
175        if (d == ScriptRuntime.NaN || (int)d != d) {
176            return defaultValue;
177        }
178        return (int) d;
179    }

182    /**
183     * Get a snapshot of the current JavaScript evaluation state by creating
184     * an Error object and invoke the function on it passing along any arguments.
185     * Used to invoke console.trace() and friends because implementing this
186     * in JavaScript would mess with the evaluation state.
187     * @param function the function to call
188     * @param args optional arguments to pass to the function.
189     */
190    public static void traceHelper(Function function, Object... args) {
191        Context cx = Context.getCurrentContext();
192        Scriptable scope = ScriptableObject.getTopLevelScope(function);
193        EcmaError error = ScriptRuntime.constructError("Trace", "");
194        WrapFactory wrapFactory = cx.getWrapFactory();
195        Scriptable thisObj = wrapFactory.wrapAsJavaObject(cx, scope, error, null);
196        for (int i = 0; i < args.length; i++) {
197            args[i] = wrapFactory.wrap(cx, scope, args[i], null);
198        }
199        function.call(cx, scope, thisObj, args);
200    }

202    /**
203     * Helper for console.assert(). Implemented in Java in order not to
204     * modify the JavaScript stack.
205     * @param condition the condition to test
206     * @param args one or more message parts
207     */
208    public static void assertHelper(Object condition, Object... args) {
209        if (ScriptRuntime.toBoolean(condition)) {
210            return;
211        }
212        // assertion failed
213        String msg = "";
214        if (args.length > 0) {
215            msg = ScriptRuntime.toString(args[0]);
216            Pattern pattern = Pattern.compile("%[sdifo]");
217            for (int i = 1; i < args.length; i++) {
218                Matcher matcher = pattern.matcher(msg);
219                if (matcher.find()) {
220                    msg = matcher.replaceFirst(ScriptRuntime.toString(args[i]));
221                } else {
222                    msg = msg + " " + ScriptRuntime.toString(args[i]);
223                }
224            }
225        }
226        throw ScriptRuntime.constructError("AssertionError", msg);
227    }
