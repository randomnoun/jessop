030/** A class to convert the Win32 windows tree into a DOM object
031 * 
032 * @blog http://www.randomnoun.com/wp/2012/12/26/automating-windows-from-java-and-windowtreedom/
033 * @author knoxg
034 * 
035 */
036public class WindowTreeDom {
037
038        // the User32 functions we invoke from this class
039        public interface User32 extends StdCallLibrary {
040      User32 INSTANCE = (User32) Native.loadLibrary("user32", User32.class, 
041        W32APIOptions.DEFAULT_OPTIONS);
042      
043      public static final DWORD GW_OWNER = new DWORD(4);
044      boolean EnumWindows(WinUser.WNDENUMPROC lpEnumFunc, Pointer arg);
045      boolean EnumChildWindows(HWND hWnd, WNDENUMPROC lpEnumFunc, Pointer data);
046      int GetWindowText(HWND hWnd, char[] lpString, int nMaxCount);
047      int GetClassName(HWND hWnd, char[] lpClassName, int nMaxCount);
048      public HWND GetWindow(HWND hWnd, DWORD cmd);
049      HWND GetParent(HWND hWnd);
050    }
051        
052        /** JNA interface to USER32.DLL */
053        final static User32 lib = User32.INSTANCE;
054
055        /** Logger instance for this class */
056        static Logger logger = Logger.getLogger(WindowTreeDom.class);
057        
058        /** WindowTreeDom constructor.
059         * 
060         * @see #getDom()
061         */
062        public WindowTreeDom() {
063                
064        }
065        
066        /** This callback is invoked for each window found. It generates XML 
067         * {#link org.w3c.Element}s for each window, and attaches them to the supplied 
068         * {#link org.w3c.Document}.
069         * 
070         */
071        private static class WindowCallback implements WinUser.WNDENUMPROC {
072                Document doc;
073                Element documentElement;
074                Element topLevelWindow; 
075                Map<String, Element> hwndMap = new HashMap<String, Element>();
076                
077                /** Creates a new window callback
078                 * 
079                 * @param doc The XML document populated by this callback.
080                 * @param topLevelHWND If non-null, the windows being returned should all be
081                 *   child windows of this HWND (via EnumChildWindows), otherwise it is
082                 *   assumed toplevel windows are returned (via EnumWindows)
083                 * @param topLevelWindow If non-null, the document Element within <tt>doc</tt>
084                 *   which will contain new child elements.
085                 */
086                public WindowCallback(Document doc, HWND topLevelHWND, Element topLevelWindow) {
087                        this.doc = doc;
088                        this.topLevelWindow = topLevelWindow;
089                        if (topLevelWindow != null) {
090                                hwndMap.put(topLevelHWND.getPointer().toString(), topLevelWindow);
091                        }
092                        documentElement = doc.getDocumentElement();
093                }
094                
095                public boolean callback(HWND hWnd, Pointer data) {
096                        
097                        char[] buffer = new char[512];
098                        User32.INSTANCE.GetWindowText(hWnd, buffer, 512);
099                        
100                        char[] buffer2 = new char[1026];
101                        int classLen = User32.INSTANCE.GetClassName(hWnd, buffer2, 1026);
102                    
103                        String windowTitle = Native.toString(buffer);
104                        String className = Native.toString(buffer2);
105
106                        HWND parent = User32.INSTANCE.GetParent(hWnd);
107                        HWND owner = User32.INSTANCE.GetWindow(hWnd, User32.GW_OWNER);
108                        
109                    // check if this has already been created in the DOM
110                        Element el = hwndMap.get(hWnd.getPointer().toString());
111                        if (el==null) {
112                                el = doc.createElement("window");
113                        } else {
114                                el.removeAttribute("pwindow");
115                        }
116                        el.setAttribute("hwnd", hWnd.getPointer().toString());
117                        if (owner!=null) {
118                                el.setAttribute("owner", owner.getPointer().toString());
119                        }
120                        el.setAttribute("title", windowTitle);
121                        el.setAttribute("class", className);
122                        
123                        hwndMap.put(hWnd.getPointer().toString(), el);
124                        if (topLevelWindow==null) { 
125                                // this is a real top level element, so enumerate its children
126                                WindowCallback childDommer = new WindowCallback(doc, hWnd, el);
127                                // this code relies on being able to enum child windows whilst enumming toplevel windows
128                                lib.EnumChildWindows (hWnd, childDommer, new Pointer(0));
129                                try {
130                                        childDommer.checkForOrphanedWindows();
131                                } catch (TransformerException e) {
132                                        logger.error("Problem serialising orphaned windows to XML", e);
133                                }
134                        }
135                        
136                        if (parent==null) {
137                                documentElement.appendChild(el);
138                                if (topLevelWindow!=null) {
139                                        // have seen VMDragDetectWndClass'es here, presumably a vmware thing
140                                        // (note that this window won't be in the parent callback's hwndMap)
141                                        try {
142                                                logger.warn("Toplevel child window found: " + XmlUtil.getXmlString(el, true));
143                                        } catch (TransformerException e) {
144                                                logger.error("Toplevel child window found, problem serialising toplevel windows to XML", e);
145                                        }
146                                }
147                                
148                        } else {
149                                Element parentEl = hwndMap.get(parent.getPointer().toString());
150                                if (parentEl==null) {
151                                        // throw new IllegalStateException("Unknown parent window '" + parent.getPointer().toString() + "'");
152                                        // it appears that we can get IME child windows being returned 
153                                        // by EnumWindows, even though they're not top-level
154                                        parentEl = doc.createElement("window");
155                                        parentEl.setAttribute("pwindow", "true");
156                                        parentEl.setAttribute("hwnd", parent.getPointer().toString());
157                                        hwndMap.put(parent.getPointer().toString(), parentEl); 
158                                }
159                                parentEl.appendChild(el);
160                        }
161                        
162                        return true;
163                }
164                
165                /** Lists any window nodes that were generated via enumeration, whose 
166                 * parent nodes were not generated.
167                 * 
168                 * @throws TransformerException
169                 */
170                public void checkForOrphanedWindows() throws TransformerException {
171                        for (Element e : hwndMap.values()) {
172                                if (!e.getAttribute("pwindow").equals("")) {
173                                        // the desktop window isn't in the enumeration
174                                        logger.warn("Parent window found that was not in enumeration: " + XmlUtil.getXmlString(e, true));
175                                        // throw new IllegalStateException("Window found without parent window");
176                                }
177                        }
178                }
179
180        }
181        
182        /** Generate an XML document from the Win32 window tree */
183        public Document getDom() throws ParserConfigurationException, TransformerException {
184                DocumentBuilderFactory docBuilderFactory = DocumentBuilderFactory.newInstance();
185                DocumentBuilder docBuilder = docBuilderFactory.newDocumentBuilder();
186                Document doc = docBuilder.newDocument();
187                Element topElement = doc.createElement("windows");
188                doc.appendChild(topElement);
189                
190                WindowCallback dommer = new WindowCallback(doc, null, null);
191                lib.EnumWindows (dommer, new Pointer(0));
192                dommer.checkForOrphanedWindows();
193                
194                return doc;
195        }
196        
197        /** Return the hwnd of an element, as a pointer represented as a long 
198         * 
199         * @param windowEl a window element returned from getDom()
200         * 
201         * @return the hwnd of the element.
202         */
203        public HWND getHwnd(Element windowEl) {
204        String hwndString = windowEl.getAttribute("hwnd");
205        if (hwndString.startsWith("native@0x")) {
206                return new HWND(new Pointer(Long.parseLong(hwndString.substring(9), 16)));
207        } else {
208                throw new IllegalStateException("Could not determine HWND of window element: found '" + hwndString + "'");
209        }
210        }
