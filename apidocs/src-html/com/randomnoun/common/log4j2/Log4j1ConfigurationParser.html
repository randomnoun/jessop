066/**
067 * You know how log4j 1 can be configured via XML or property files, and
068 * how log4j 2 can be configured via maliciously and arbitrarily different XML or property files ?
069 * 
070 * <p>This class allows you to configure log4j2 using log4j1-style property configuration files,
071 * which allows you to slowly migrate your clusterfuck of logging frameworks from 
072 * log4j1 to log4j2 without having to redo your entire logging configuration, which by
073 * now is Turing-complete and giving recipe suggestions.
074 * 
075 * <p>It's based on the org.apache.log4j.config.Log4j1ConfigurationParser that's bundled in the 
076 * log4j-1.2-api artifact ( specifically, org.apache.logging.log4j:log4j-1.2-api:2.14.0 ), but:
077 * 
078 * <ul><li>buildConfigurationBuilder() can take a Properties object (as well as an InputStream)
079 *     <li>Unknown log4j1 appenders are constructed (and wrapped in some kind of log4j2 adapter) instead of ignored
080 *     <li>Unknown log4j2 appenders are constructed instead of ignored
081 * </ul> 
082 *
083 * <p>The original Log4j1ConfigurationParser also converts a handful of "well-known" appenders
084 * into their log4j2 equivalents, which still happens in this class.
085 *
086 * <p>To use, run the following:
087 * <pre>
088Log4j1ConfigurationParser lcp = new Log4j1ConfigurationParser();
089ConfigurationBuilder<?> builder = lcp.buildConfigurationBuilder(props);
090Configuration config = builder.build();
091Configurator.reconfigure(config); 
092 * </pre>
093 *
094 * <hr/>
095 * Original comment block:
096 * 
097 * Experimental parser for Log4j 1.2 properties configuration files.
098 *
099 * This class is not thread-safe.
100 * 
101 * <p>
102 * From the Log4j 1.2 Javadocs:
103 * </p>
104 * <p>
105 * All option values admit variable substitution. The syntax of variable substitution is similar to that of Unix shells. The string between
106 * an opening "${" and closing "}" is interpreted as a key. The value of the substituted variable can be defined as a system property or in
107 * the configuration file itself. The value of the key is first searched in the system properties, and if not found there, it is then
108 * searched in the configuration file being parsed. The corresponding value replaces the ${variableName} sequence. For example, if java.home
109 * system property is set to /home/xyz, then every occurrence of the sequence ${java.home} will be interpreted as /home/xyz.
110 * </p>
111 * 
112 * <p>Changes from the Log4j1ConfigurationParser in org.apache.logging.log4j:log4j-1.2-api:2.14.0
113 * <ul><li>buildConfigurationBuilder() can take a Properties object (as well as an InputStream)
114 *     <li>Unknown log4j2 appenders are constructed instead of ignored
115 * </ul> 
116 */

132    /**
133     * Parses a Log4j 1.2 properties configuration file in ISO 8859-1 encoding into a ConfigurationBuilder.
134     *
135     * @param input
136     *            InputStream to read from is assumed to be ISO 8859-1, and will not be closed.
137     * @return the populated ConfigurationBuilder, never {@literal null}
138     * @throws IOException
139     *             if unable to read the input
140     * @throws ConfigurationException
141     *             if the input does not contain a valid configuration
142     */
143    public ConfigurationBuilder<BuiltConfiguration> buildConfigurationBuilder(final InputStream input)
144        throws IOException 
145    {
146        try {
147                properties.load(input);
148                _buildConfigurationBuilder();
149                return builder;
150        } catch (final IllegalArgumentException e) {
151            throw new ConfigurationException(e);
152        }
153                
154    }

156    public ConfigurationBuilder<BuiltConfiguration> buildConfigurationBuilder(Properties properties) {
157        try {
158                this.properties.putAll(properties);
159                _buildConfigurationBuilder();
160                return builder;
161        } catch (final IllegalArgumentException e) {
162            throw new ConfigurationException(e);
163        }
164    }

166    protected ConfigurationBuilder<BuiltConfiguration> _buildConfigurationBuilder() {
167        try {
168            strSubstitutorProperties = new StrSubstitutor(properties);
169            strSubstitutorSystem = new StrSubstitutor(System.getProperties());
170            final String rootCategoryValue = getLog4jValue(ROOTCATEGORY);
171            final String rootLoggerValue = getLog4jValue(ROOTLOGGER);
172            if (rootCategoryValue == null && rootLoggerValue == null) {
173                // This is not a Log4j 1 properties configuration file.
174                warn("Missing " + ROOTCATEGORY + " or " + ROOTLOGGER + " in properties");
175                // throw new ConfigurationException(
176                // "Missing " + ROOTCATEGORY + " or " + ROOTLOGGER + " in " + input);
177            }
178            builder.setConfigurationName("Log4j1");
179            // DEBUG
180            final String debugValue = getLog4jValue("debug");
181            if (Boolean.valueOf(debugValue)) {
182                builder.setStatusLevel(Level.DEBUG);
183            }
184            // Root
185            buildRootLogger(getLog4jValue(ROOTCATEGORY));
186            buildRootLogger(getLog4jValue(ROOTLOGGER));
187            // Appenders
188            final Map<String, String> appenderNameToClassName = buildClassToPropertyPrefixMap();
189            for (final Map.Entry<String, String> entry : appenderNameToClassName.entrySet()) {
190                final String appenderName = entry.getKey();
191                final String appenderClass = entry.getValue();
192                buildAppender(appenderName, appenderClass);
193            }
194            // Loggers
195            buildLoggers("log4j.category.");
196            buildLoggers("log4j.logger.");
197            buildProperties();
198            return builder;
199        } catch (final IllegalArgumentException e) {
200            throw new ConfigurationException(e);
201        }
202    }

205    private void buildProperties() {
206        for (final Map.Entry<Object, Object> entry : new TreeMap<Object, Object>(properties).entrySet()) {
207            final String key = entry.getKey().toString();
208            if (!key.startsWith("log4j.") && !key.equals(ROOTCATEGORY) && !key.equals(ROOTLOGGER)) {
209                builder.addProperty(key, Objects.toString(entry.getValue(), Strings.EMPTY));
210            }
211        }
212    }

218    private Map<String, String> buildClassToPropertyPrefixMap() {
219        final String prefix = "log4j.appender.";
220        final int preLength = prefix.length();
221        final Map<String, String> map = new HashMap<String, String>();
222        for (final Map.Entry<Object, Object> entry : properties.entrySet()) {
223            final Object keyObj = entry.getKey();
224            if (keyObj != null) {
225                final String key = keyObj.toString();
226                if (key.startsWith(prefix)) {
227                    if (key.indexOf('.', preLength) < 0) {
228                        final String name = key.substring(preLength);
229                        final Object value = entry.getValue();
230                        if (value != null) {
231                            map.put(name, value.toString());
232                        }
233                    }
234                }
235            }
236        }
237        return map;
238    }

240    private void buildAppender(final String appenderName, final String appenderClass) {
241        if ("org.apache.log4j.ConsoleAppender".equals(appenderClass)) {
242            buildConsoleAppender(appenderName);
243        } else if ("org.apache.log4j.FileAppender".equals(appenderClass)) {
244            buildFileAppender(appenderName);
245        } else if ("org.apache.log4j.DailyRollingFileAppender".equals(appenderClass)) {
246            buildDailyRollingFileAppender(appenderName);
247        } else if ("org.apache.log4j.RollingFileAppender".equals(appenderClass)) {
248            buildRollingFileAppender(appenderName);
249        } else if ("org.apache.log4j.varia.NullAppender".equals(appenderClass)) {
250            buildNullAppender(appenderName);
251        } else {
252                // old Log4j1ConfigurationParser did this:
253            //   reportWarning("Unknown appender class: " + appenderClass + "; ignoring appender: " + appenderName);
254                // which isn't helpful
255                
256                Class<?> c;
257                try {
258                                c = Class.forName(appenderClass);
259                        } catch (ClassNotFoundException e) {
260                                throw new ConfigurationException("Cannot find appender class '" + appenderClass + "'", e);
261                        }
262                // org.apache.log4j.ConsoleAppender c1;
263                if (org.apache.log4j.Appender.class.isAssignableFrom(c)) {
264                        /* log4j1 appender, use an adapter once I fathom how that's supposed to work */
265                        // throw new ConfigurationException("Cannot handle log4j1 appender class '" + appenderClass + "'"); 
266                        
267                        /*
268                        // construct the log4j1 appender. this is soooooooooooo going to work.
269                final String prefix = "log4j.appender." + appenderName;
270                final String layoutPrefix = "log4j.appender." + appenderName + ".layout";
271                final String filterPrefix = "log4j.appender." + appenderName + ".filter.";
272                final org.apache.log4j.Appender log4j1Appender = buildLog4j1Appender(appenderName, c.getName(), prefix, layoutPrefix, filterPrefix, properties);
273                        
274                // this should work but doesn't because DefaultComponentBuilder.addAttribute() converts every value it receives into a string, 
275                // when then has to be converted back into an object using a TypeConverter (which needs it's own TypeConverter registry),
276                // and a log4j1 appender won't round-trip through that properly.
277
278                                AppenderComponentBuilder appenderBuilder = builder.newAppender(appenderName, Log4j1WrapperAppender.PLUGIN_NAME);
279                                appenderBuilder.addAttribute("log4j1Appender", log4j1Appender);
280                                builder.add(appenderBuilder);
281                                // ^ the attribute collection in that builder is a Map<String, String>
282                
283                // so not only that, you can't create your own "non-default" DefaultComponentBuilder 
284                // because the Components that these builders are building are also composed of Strings.
285                // I suppose you could create a Component which consists of the entire property file that we're configuring this with
286                // and then extract the appender properties out of that, but frankly, this is the point at which I'm giving up.
287                        */
288                        
289                        // ok, so now the Log4jWrapperAppender constructs the appender, we don't construct it here
290                        
291                        StringWriter w = new StringWriter();
292                        try {
293                                        properties.store(w,  null);
294                                } catch (IOException e) {
295                                        throw new IllegalStateException("IOException in StringWriter");
296                                }
297                        
298                        AppenderComponentBuilder appenderBuilder = builder.newAppender(appenderName, Log4j1WrapperAppender.PLUGIN_NAME);
299                        appenderBuilder.addAttribute("className", c.getName());
300                        appenderBuilder.addAttribute("properties", w.toString());
301                        builder.add(appenderBuilder);
302                        
303                        
304                } else if (org.apache.logging.log4j.core.Appender.class.isAssignableFrom(c)) {
305                        /* log4j2 appender, because wheels aren't going to reinvent themselves */
306
307                        // no idea what these are for or why they're necessary, if they are, which they probably aren't
308                        Plugin log4jPlugin = c.getAnnotation(org.apache.logging.log4j.core.config.plugins.Plugin.class);
309                        String pluginName = log4jPlugin.name();
310                        if (pluginName == null) {
311                                pluginName = "generatedPluginName_" + c.getName();
312                        }
313                        final AppenderComponentBuilder appenderBuilder = builder.newAppender(appenderName, pluginName);
314
315                        // set the props ( from "log4j.appender." + appenderName + "." + attributeName );
316                        // we're checking for getter/setter methods on the Appender to work out what's settable
317                        // even though we end up using a Builder class to construct the thing for some reason
318                                BeanInfo b;
319                                try {
320                                        b = Introspector.getBeanInfo(c);
321                                } catch (IntrospectionException e1) {
322                                        throw new ConfigurationException("Could not determine properties of appender class '" + appenderClass + "'", e1);
323                                }
324                                PropertyDescriptor[] pds = b.getPropertyDescriptors();
325                                Map<String, PropertyDescriptor> pdMap = new HashMap<String, PropertyDescriptor>();
326                                for (PropertyDescriptor pd : pds) {
327                                        pdMap.put(pd.getName().toLowerCase(), pd); // case-insensitive match 
328                                }
329                                String prefix = "log4j.appender." + appenderName + ".";
330                        for (Entry<Object, Object> e : properties.entrySet()) {
331                                String k = (String) e.getKey();
332                                String v = (String) e.getValue().toString();
333                                if (k.startsWith(prefix)) {
334                                        String rest = k.substring(prefix.length());
335                                        PropertyDescriptor pd = pdMap.get(rest.toLowerCase());
336                                        if (pd == null) {
337                                                throw new ConfigurationException("Appender class '" + appenderClass + "' has no property '" + rest + "'");
338                                        } else if (pd.getPropertyType().equals(String.class)) {
339                                                appenderBuilder.addAttribute(pd.getName(), v);
340                                        } else if (pd.getPropertyType().equals(Integer.class) || pd.getPropertyType().equals(int.class)) {
341                                                appenderBuilder.addAttribute(pd.getName(), Integer.valueOf(v).intValue());
342                                        } else {
343                                                // all the other types, but let's see if Strings work, which they won't.
344                                                appenderBuilder.addAttribute(pd.getName(), v);
345                                        }
346                                }
347                        }
348                        
349                        builder.add(appenderBuilder);
350                }
351        }
352    }

355    private void buildConsoleAppender(final String appenderName) {
356        final AppenderComponentBuilder appenderBuilder = builder.newAppender(appenderName, ConsoleAppender.PLUGIN_NAME);
357        final String targetValue = getLog4jAppenderValue(appenderName, "Target", "System.out");
358        if (targetValue != null) {
359            final ConsoleAppender.Target target;
360            if (targetValue.equals("System.out")) {
361                target = ConsoleAppender.Target.SYSTEM_OUT;
362            } else if (targetValue.equals("System.err")) {
363                target = ConsoleAppender.Target.SYSTEM_ERR;
364            } else {
365                reportWarning("Unknown value for console Target: " + targetValue);
366                target = null;
367            }
368            if (target != null) {
369                appenderBuilder.addAttribute("target", target);
370            }
371        }
372        buildAttribute(appenderName, appenderBuilder, "Follow", "follow");
373        if (FALSE.equalsIgnoreCase(getLog4jAppenderValue(appenderName, "ImmediateFlush"))) {
374            reportWarning("ImmediateFlush=false is not supported on Console appender");
375        }
376        buildAppenderLayout(appenderName, appenderBuilder);
377        builder.add(appenderBuilder);
378    }

395    private void buildDailyRollingFileAppender(final String appenderName) {
396        final AppenderComponentBuilder appenderBuilder = builder.newAppender(appenderName,
397                RollingFileAppender.PLUGIN_NAME);
398        buildFileAppender(appenderName, appenderBuilder);
399        final String fileName = getLog4jAppenderValue(appenderName, "File");
400        final String datePattern = getLog4jAppenderValue(appenderName, "DatePattern", fileName + "'.'yyyy-MM-dd");
401        appenderBuilder.addAttribute("filePattern", fileName + "%d{" + datePattern + "}");
402        final ComponentBuilder<?> triggeringPolicy = builder.newComponent("Policies")
403                .addComponent(builder.newComponent("TimeBasedTriggeringPolicy").addAttribute("modulate", true));
404        appenderBuilder.addComponent(triggeringPolicy);
405        appenderBuilder
406                .addComponent(builder.newComponent("DefaultRolloverStrategy").addAttribute("max", Integer.MAX_VALUE));
407        builder.add(appenderBuilder);
408    }

455    private void buildAppenderLayout(final String name, final AppenderComponentBuilder appenderBuilder) {
456        final String layoutClass = getLog4jAppenderValue(name, "layout", null);
457        if (layoutClass != null) {
458            if ("org.apache.log4j.PatternLayout".equals(layoutClass) ||
459                "org.apache.log4j.EnhancedPatternLayout".equals(layoutClass)) {
460                final String pattern = getLog4jAppenderValue(name, "layout.ConversionPattern", null)
461
462                        // Log4j 2's %x (NDC) is not compatible with Log4j 1's
463                        // %x
464                        // Log4j 1: "foo bar baz"
465                        // Log4j 2: "[foo, bar, baz]"
466                        // Use %ndc to get the Log4j 1 format
467                        .replace("%x", "%ndc")
468
469                        // Log4j 2's %X (MDC) is not compatible with Log4j 1's
470                        // %X
471                        // Log4j 1: "{{foo,bar}{hoo,boo}}"
472                        // Log4j 2: "{foo=bar,hoo=boo}"
473                        // Use %properties to get the Log4j 1 format
474                        .replace("%X", "%properties");
475
476                appenderBuilder.add(newPatternLayout(pattern));
477            } else if ("org.apache.log4j.SimpleLayout".equals(layoutClass)) {
478                appenderBuilder.add(newPatternLayout("%level - %m%n"));
479            } else if ("org.apache.log4j.TTCCLayout".equals(layoutClass)) {
480                String pattern = "%r ";
481                if (Boolean.parseBoolean(getLog4jAppenderValue(name, "layout.ThreadPrinting", TRUE))) {
482                    pattern += "[%t] ";
483                }
484                pattern += "%p ";
485                if (Boolean.parseBoolean(getLog4jAppenderValue(name, "layout.CategoryPrefixing", TRUE))) {
486                    pattern += "%c ";
487                }
488                if (Boolean.parseBoolean(getLog4jAppenderValue(name, "layout.ContextPrinting", TRUE))) {
489                    pattern += "%notEmpty{%ndc }";
490                }
491                pattern += "- %m%n";
492                appenderBuilder.add(newPatternLayout(pattern));
493            
494            } else if ("org.apache.log4j.HTMLLayout".equals(layoutClass)) {
495                final LayoutComponentBuilder htmlLayout = builder.newLayout("HtmlLayout");
496                htmlLayout.addAttribute("title", getLog4jAppenderValue(name, "layout.Title", "Log4J Log Messages"));
497                htmlLayout.addAttribute("locationInfo",
498                        Boolean.parseBoolean(getLog4jAppenderValue(name, "layout.LocationInfo", FALSE)));
499                appenderBuilder.add(htmlLayout);
500            } else if ("org.apache.log4j.xml.XMLLayout".equals(layoutClass)) {
501                final LayoutComponentBuilder xmlLayout = builder.newLayout("Log4j1XmlLayout");
502                xmlLayout.addAttribute("locationInfo",
503                        Boolean.parseBoolean(getLog4jAppenderValue(name, "layout.LocationInfo", FALSE)));
504                xmlLayout.addAttribute("properties",
505                        Boolean.parseBoolean(getLog4jAppenderValue(name, "layout.Properties", FALSE)));
506                appenderBuilder.add(xmlLayout);
507            } else {
508                reportWarning("Unknown layout class: " + layoutClass);
509            }
510        }
511    }

521    private void buildRootLogger(final String rootLoggerValue) {
522        if (rootLoggerValue == null) {
523            return;
524        }
525        final String[] rootLoggerParts = rootLoggerValue.split(COMMA_DELIMITED_RE);
526        final String rootLoggerLevel = getLevelString(rootLoggerParts, Level.ERROR.name());
527        final RootLoggerComponentBuilder loggerBuilder = builder.newRootLogger(rootLoggerLevel);
528        //
529        final String[] sortedAppenderNames = Arrays.copyOfRange(rootLoggerParts, 1, rootLoggerParts.length);
530        Arrays.sort(sortedAppenderNames);
531        for (final String appender : sortedAppenderNames) {
532            loggerBuilder.add(builder.newAppenderRef(appender));
533        }
534        builder.add(loggerBuilder);
535    }

537    private String getLevelString(final String[] loggerParts, final String defaultLevel) {
538        return loggerParts.length > 0 ? loggerParts[0] : defaultLevel;
539    }

541    private void buildLoggers(final String prefix) {
542        final int preLength = prefix.length();
543        for (final Map.Entry<Object, Object> entry : properties.entrySet()) {
544            final Object keyObj = entry.getKey();
545            if (keyObj != null) {
546                final String key = keyObj.toString();
547                if (key.startsWith(prefix)) {
548                    final String name = key.substring(preLength);
549                    final Object value = entry.getValue();
550                    if (value != null) {
551                        // a Level may be followed by a list of Appender refs.
552                        final String valueStr = value.toString();
553                        final String[] split = valueStr.split(COMMA_DELIMITED_RE);
554                        final String level = getLevelString(split, null);
555                        if (level == null) {
556                            warn("Level is missing for entry " + entry);
557                        } else {
558                            final LoggerComponentBuilder newLogger = builder.newLogger(name, level);
559                            if (split.length > 1) {
560                                // Add Appenders to this logger
561                                final String[] sortedAppenderNames = Arrays.copyOfRange(split, 1, split.length);
562                                Arrays.sort(sortedAppenderNames);
563                                for (final String appenderName : sortedAppenderNames) {
564                                    newLogger.add(builder.newAppenderRef(appenderName));
565                                }
566                            }
567                            builder.add(newLogger);
568                        }
569                    }
570                }
571            }
572        }
573    }
