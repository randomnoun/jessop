017package com.randomnoun.common.log4j2;
018
019import org.apache.log4j.Category;
020import org.apache.log4j.Level;
021import org.apache.log4j.spi.LocationInfo;
022import org.apache.log4j.spi.LoggingEvent;
023import org.apache.log4j.spi.ThrowableInformation;
024import org.apache.logging.log4j.core.LogEvent;
025import org.apache.logging.log4j.core.util.Loader;
026import org.apache.logging.log4j.core.util.Throwables;
027import org.apache.logging.log4j.spi.StandardLevel;
028import org.apache.logging.log4j.status.StatusLogger;
029
030import java.lang.reflect.Method;
031import java.util.Map;
032import java.util.Set;
033
034// a modified form of the LogEventAdapter in log4j2, which uses our own LoggingEvent class, which has a non-final getTimestamp()
035
036/**
037 * Converts a Log4j 2 LogEvent into the components needed by a Log4j 1.x LoggingEvent.
038 * This class requires Log4j 2.
039 */
040public class LogEventAdapter extends LoggingEvent {
041
042    private static final long JVM_START_TIME = initStartTime();
043
044    private final LogEvent event;
045
046    public LogEventAdapter(LogEvent event) {
047        this.event = event;
048    }
049
050    /**
051     * Returns the time when the application started, in milliseconds
052     * elapsed since 01.01.1970.
053     * @return the time when the JVM started.
054     */
055    public static long getStartTime() {
056        return JVM_START_TIME;
057    }
058
059    /**
060     * Returns the result of {@code ManagementFactory.getRuntimeMXBean().getStartTime()},
061     * or the current system time if JMX is not available.
062     */
063    private static long initStartTime() {
064        // We'd like to call ManagementFactory.getRuntimeMXBean().getStartTime(),
065        // but Google App Engine throws a java.lang.NoClassDefFoundError
066        // "java.lang.management.ManagementFactory is a restricted class".
067        // The reflection is necessary because without it, Google App Engine
068        // will refuse to initialize this class.
069        try {
070            final Class<?> factoryClass = Loader.loadSystemClass("java.lang.management.ManagementFactory");
071            final Method getRuntimeMXBean = factoryClass.getMethod("getRuntimeMXBean");
072            final Object runtimeMXBean = getRuntimeMXBean.invoke(null);
073
074            final Class<?> runtimeMXBeanClass = Loader.loadSystemClass("java.lang.management.RuntimeMXBean");
075            final Method getStartTime = runtimeMXBeanClass.getMethod("getStartTime");
076            return (Long) getStartTime.invoke(runtimeMXBean);
077        } catch (final Throwable t) {
078            StatusLogger.getLogger().error("Unable to call ManagementFactory.getRuntimeMXBean().getStartTime(), "
079                    + "using system time for OnStartupTriggeringPolicy", t);
080            // We have little option but to declare "now" as the beginning of time.
081            return System.currentTimeMillis();
082        }
083    }
084
085    public LogEvent getEvent() {
086        return this.event;
087    }
088
089    /**
090     * Set the location information for this logging event. The collected
091     * information is cached for future use.
092     */
093    @Override
094    public LocationInfo getLocationInformation() {
095        return new LocationInfo(event.getSource());
096    }
097
098    /**
099     * Return the level of this event. Use this form instead of directly
100     * accessing the <code>level</code> field.
101     */
102    @Override
103    public Level getLevel() {
104        switch (StandardLevel.getStandardLevel(event.getLevel().intLevel())) {
105            case TRACE:
106                return Level.TRACE;
107            case DEBUG:
108                return Level.DEBUG;
109            case INFO:
110                return Level.INFO;
111            case WARN:
112                return Level.WARN;
113            case FATAL:
114                return Level.FATAL;
115            case OFF:
116                return Level.OFF;
117            case ALL:
118                return Level.ALL;
119            default:
120                return Level.ERROR;
121        }
122    }
123
124    /**
125     * Return the name of the logger. Use this form instead of directly
126     * accessing the <code>categoryName</code> field.
127     */
128    @Override
129    public String getLoggerName() {
130        return event.getLoggerName();
131    }
132
133    /**
134     * Gets the logger of the event.
135     */
136    @Override
137    public Category getLogger() {
138        return Category.getInstance(event.getLoggerName());
139    }
140
141    /*
142     Return the message for this logging event.
143    */
144    @Override
145    public Object getMessage() {
146        return event.getMessage();
147    }
148
149    /*
150     * This method returns the NDC for this event.
151     */
152    @Override
153    public String getNDC() {
154        return event.getContextStack().toString();
155    }
156
157    /*
158     Returns the the context corresponding to the <code>key</code> parameter.
159     */
160    @Override
161    public Object getMDC(String key) {
162        if (event.getContextData() != null) {
163            return event.getContextData().getValue(key);
164        } else {
165            return null;
166        }
167    }
168
169    /**
170     * Obtain a copy of this thread's MDC prior to serialization or
171     * asynchronous logging.
172     */
173    @Override
174    public void getMDCCopy() {
175    }
176
177    @Override
178    public String getRenderedMessage() {
179        return event.getMessage().getFormattedMessage();
180    }
181
182    @Override
183    public String getThreadName() {
184        return event.getThreadName();
185    }
186
187    /**
188     * Returns the throwable information contained within this
189     * event. May be <code>null</code> if there is no such information.
190     *
191     * <p>Note that the {@link Throwable} object contained within a
192     * {@link ThrowableInformation} does not survive serialization.
193     *
194     * @since 1.1
195     */
196    @Override
197    public ThrowableInformation getThrowableInformation() {
198        if (event.getThrown() != null) {
199            return new ThrowableInformation(event.getThrown());
200        }
201        return null;
202    }
203
204    /**
205     * Return this event's throwable's string[] representaion.
206     */
207    @Override
208    public String[] getThrowableStrRep() {
209        if (event.getThrown() != null) {
210            return Throwables.toStringList(event.getThrown()).toArray(new String[0]);
211        }
212        return null;
213    }
214
215    @Override
216    public String getProperty(final String key) {
217        return event.getContextData().getValue(key);
218    }
219
220    @Override
221    public Set getPropertyKeySet() {
222        return event.getContextData().toMap().keySet();
223    }
224
225    @Override
226    public Map getProperties() {
227        return event.getContextData().toMap();
228    }
229    
230    @Override
231    public long getTimeStamp() {
232        return event.getTimeMillis();
233    }
234    
235}
