001package com.randomnoun.common.log4j2;
002
003import java.io.IOException;
004import java.io.InputStreamReader;
005import java.io.OutputStream;
006import java.io.PipedInputStream;
007import java.io.PipedOutputStream;
008import java.nio.charset.Charset;
009
010import org.apache.logging.log4j.Logger;
011
012/** An outputstream that writes to a log4j Logger
013 *
014 * @author knoxg
015 */
016public class LoggingOutputStream extends OutputStream {
017        Logger logger;
018        StringBuffer line = new StringBuffer();
019        PipedOutputStream pos;
020        PipedInputStream pis;
021        InputStreamReader isr;
022        
023        public LoggingOutputStream(Logger logger, Charset charset) throws IOException {
024                this.logger = logger;
025                this.pos = new PipedOutputStream();
026                this.pis = new PipedInputStream(pos);
027                this.isr = new InputStreamReader(pis, charset);
028        }
029        public LoggingOutputStream(Logger logger, String charsetName) throws IOException {
030                this(logger, Charset.forName(charsetName));
031        }
032        public LoggingOutputStream(Logger logger) throws IOException {
033                this(logger, Charset.defaultCharset());
034        }
035        
036        @Override
037        public void write(int b) throws IOException {
038                
039                pos.write(b);
040                int ch = isr.read();
041                if (ch == -1) {
042                        // not a full character
043                } else if (ch == 10) {
044                        logger.info(line); line.setLength(0);
045                } else { 
046                        line.append((char) b);
047                }
048        }
049}
