001package com.randomnoun.common.log4j2;
002
003import java.util.Properties;
004
005import org.apache.log4j.PropertyConfigurator;
006import org.apache.logging.log4j.LogManager;
007import org.apache.logging.log4j.core.LifeCycle;
008import org.apache.logging.log4j.core.config.Configuration;
009import org.apache.logging.log4j.core.config.Configurator;
010import org.apache.logging.log4j.core.config.builder.api.ConfigurationBuilder;
011
012/** Configures log4j for command-line interface programs.
013 * 
014 * <p>This class configured the log4j2 (two!) framework, using log4j1 (one!) properties.
015 * 
016 * <p>It uses a modified Log4j1ConfigurationParser (from the log4j2 framework) to do this, 
017 * which maps some well-known log4j1 appenders to log4j2 appenders, and does a pretty ham-fisted job of everything else.
018 * 
019 * <p>By default, everything's sent to stdout, using the following log4j initialisation properties:
020 * 
021 * <pre>
022log4j.rootCategory=DEBUG, CONSOLE
023log4j.appender.CONSOLE=org.apache.log4j.ConsoleAppender
024log4j.appender.CONSOLE.layout=org.apache.log4j.PatternLayout
025log4j.appender.CONSOLE.layout.ConversionPattern=%d{ABSOLUTE} %-5p %c - %m %n
026
027log4j.logger.org.springframework=INFO
028
029# log4j.appender.FILE=com.randomnoun.common.log4j.CustomRollingFileAppender
030# log4j.appender.FILE.File=c:\\another.log
031# log4j.appender.FILE.MaxBackupIndex=100
032# log4j.appender.FILE.layout=org.apache.log4j.PatternLayout
033# log4j.appender.FILE.layout.ConversionPattern=%d{dd/MM/yy HH:mm:ss.SSS} %-5p %c - %m %n
034   </pre>
035 * 
036 * <p>with an optional line prefix before the <code>%d{ABSOLUTE}</code> in the ConversionPattern.
037 * 
038 * @see http://logging.apache.org/log4j/1.2/manual.html
039 * 
040 * @blog http://www.randomnoun.com/wp/2013/01/13/logging/
041 * @author knoxg
042 * 
043 */
044public class Log4j2CliConfiguration {
045    
046        /** Create a new Log4jCli2Configuration instance */
047        public Log4j2CliConfiguration() {
048        }
049        
050        /** Initialise log4j.
051         *
052         * <p>The properties file supplied is in log4j format.
053         * 
054         * @param logFormatPrefix a string prefixed to each log. Useful for program identifiers;
055         *   e.g. "[programName] "
056         * @param override if non-null, additional log4j properties. Any properties contained in 
057         *   this object will override the defaults.
058         * 
059         */
060        public void init(String logFormatPrefix, Properties override) {
061                
062                if (logFormatPrefix==null) { 
063                        logFormatPrefix = ""; 
064                } else {
065                        logFormatPrefix += " ";
066                }
067                
068                Properties props = new Properties();
069                props.put("log4j.rootCategory", "DEBUG, CONSOLE");
070                
071                // log4j1 class names - the Log4j1ConfigurationParser recognises a limited set of these
072                // and converts them into the log4j2 equivalents. 
073                // Unknown appenders are ignored, which is obviously what people want to happen.
074                props.put("log4j.appender.CONSOLE", "org.apache.log4j.ConsoleAppender");
075                props.put("log4j.appender.CONSOLE.layout", "org.apache.log4j.PatternLayout");
076
077                // log4j2 class names - the Log4j1ConfigurationParser doesn't recognise these
078                // props.put("log4j.appender.CONSOLE", "org.apache.logging.log4j.core.appender.ConsoleAppender");
079                // props.put("log4j.appender.CONSOLE.layout", "org.apache.logging.log4j.core.layout.PatternLayout");
080                props.put("log4j.appender.CONSOLE.layout.ConversionPattern", logFormatPrefix + "%d{ABSOLUTE} %-5p %c - %m%n");
081                props.put("log4j.logger.org.springframework", "INFO"); // since Spring is a bit too verbose for my liking at DEBUG level
082                if (override!=null) { 
083                        props.putAll(override);
084                }
085
086                /*
087                // to use the bundled Log41ConfigurationParser, had to do this:
088                try {
089                        ByteArrayOutputStream baos = new ByteArrayOutputStream();
090                        props.store(baos,  null);
091                        ByteArrayInputStream bais = new ByteArrayInputStream(baos.toByteArray());
092                        
093                        // in log4j-1.2.jar 
094                        // Log4j1ConfigurationParser lcp = new Log4j1ConfigurationParser();
095                        
096                        Log4j1ConfigurationParser lcp = new Log4j1ConfigurationParser();
097                        ConfigurationBuilder<?> builder = lcp.buildConfigurationBuilder(bais);
098                } catch (IOException e) {
099                        throw new IllegalStateException("IOException in log4j initiialisation", e);
100                }
101                */
102
103                // our Log4j1ConfigurationParser can take a Properties object instead
104                Log4j1ConfigurationParser lcp = new Log4j1ConfigurationParser();
105                ConfigurationBuilder<?> builder = lcp.buildConfigurationBuilder(props);
106                
107                // this needs to happen before Configurator.initialize()
108                // LogManager.shutdown();
109                Configuration config = builder.build();
110
111                // so this doesn't work, because the (ctx.getState() == LifeCycle.State.INITIALIZED)) condition
112                // in Log4jContextFactory.getContext() prevents the configuration from taking effect 
113                /*LoggerContext ctx; =*/ // Configurator.initialize(config);
114                
115                // so you need to do this,
116                // which is not what https://logging.apache.org/log4j/2.x/manual/customconfig.html 
117                // fucking tells you to do. This took about two hours to figure out.
118                /*LoggerContext ctx; =*/  Configurator.reconfigure(config);
119
120                
121                // incidentally, did you know that every time you create a Logger, it's introspecting the stack for no good reason ?
122                // *slow hand clap*, log4j.
123                
124                
125                
126                // log4j doesn't have ThrowableRenderers, it has ExtendedStackTraceElements instead
127                // which I can't find a working example of anywhere
128                /*
129                String highlightPrefix = props.getProperty("log4j.revisionedThrowableRenderer.highlightPrefix");
130                if (highlightPrefix != null) {
131                        LoggerRepository repo = LogManager.getLoggerRepository();
132                        if (repo instanceof ThrowableRendererSupport) {
133                    // if null, log4j will use a DefaultThrowableRenderer
134                    // ThrowableRenderer renderer = ((ThrowableRendererSupport) repo).getThrowableRenderer();
135                    ((ThrowableRendererSupport) repo).setThrowableRenderer(new RevisionedThrowableRenderer(highlightPrefix));
136                }
137                }
138                
139                PropertyConfigurator.configure(props);
140        */
141        }
142}
