007import java.io.File;
008import java.io.FileInputStream;
009import java.io.IOException;
010import java.io.InputStream;
011import java.util.Properties;
012
013import javax.servlet.ServletException;
014import javax.servlet.http.HttpServlet;
015import javax.servlet.http.HttpServletRequest;
016import javax.servlet.http.HttpServletResponse;
017
018import org.apache.log4j.Logger;
019import org.json.JSONObject;
020
021/**
022 * Reads the contents of the build.properties file in the 
023 * current application and sends it to the browser in a 
024 * variety of forms. Probably JSON for starters.
025 *
026 * <p>The properties file must be located on the classpath at the
027 * location "/build.properties".
028 * 
029 * <p>For this to work, the application needs to have been built
030 * through maven (and possibly bamboo), and there should be a 
031 * <tt>build.properties</tt> file in the application being created, 
032 * with the contents:
033 * 
034 * <pre>
035 *  bamboo.buildKey=${bambooBuildKey}
036        bamboo.buildNumber=${bambooBuildNumber}
037        bamboo.buildPlanName=${bambooBuildPlanName}
038        bamboo.buildTimeStamp=${bambooBuildTimeStamp}
039        bamboo.buildForceCleanCheckout=${bambooForceCleanCheckout}
040        bamboo.custom.cvs.last.update.time=${bambooCustomCvsLastUpdateTime}
041        bamboo.custom.cvs.last.update.time.label=${bambooCustomCvsLastUpdateTimeLabel}
042        
043        maven.pom.name=${pom.name}
044    maven.pom.version=${pom.version}
045    maven.pom.groupId=${pom.groupId}
046    maven.pom.artifactId=${pom.artifactId}
047   </pre>
048 *
049 * and the bamboo plan should have the following in the bamboo plan's
050 * stage's job's task's maven goal specification:
051 * 
052 * <pre>
053 *  "-DbambooBuildKey=${bamboo.buildKey}" 
054    "-DbambooBuildNumber=${bamboo.buildNumber}" 
055    "-DbambooBuildPlanName=${bamboo.buildPlanName}" 
056    "-DbambooBuildTimeStamp=${bamboo.buildTimeStamp}" 
057    "-DbambooBuildForceCleanCheckout=${bamboo.buildForceCleanCheckout}" 
058    "-DbambooCustomCvsLastUpdateTime=${bamboo.custom.cvs.last.update.time}" 
059    "-DbambooCustomCvsLastUpdateTimeLabel=${bamboo.custom.cvs.last.update.time.label}" 
060    -DuniqueVersion=false 
061 * </pre>
062 *
063 * <hr/>
064 * 
065 * This script will also look for a build.properties file in /etc/build.properties; 
066 * this should exist on docker containers built by bamboo. If this exists, it is also 
067 * included in the JSON output under the "/etc/build.properties" key.
068 * 
069 * 
070 * @author knoxg
071 * @blog http://www.randomnoun.com/wp/2013/09/24/webapp-versions-v1-0/
072 * 
073 */
074@SuppressWarnings("serial")
075public class VersionServlet extends HttpServlet {
076    
077        
078    
079
080        /** Logger for this class */
081    public static final Logger logger = Logger.getLogger(VersionServlet.class);
082
083    
084        /** Post method; just defers to get
085         * 
086         * @see javax.servlet.http.HttpServlet#doPost(HttpServletRequest request, HttpServletResponse response)
087         */
088        protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
089                doGet(request, response);
090        }
091
092        /** If a property has not been populated through the maven resource filter
093         * mechanism, then remove it from the Properties object.
094         *   
095         * @param props Properties object
096         * @param key key to check
097         * @param value default value
098         */
099        private void removeDefaultProperty(Properties props, String key, String value) {
100                if (("${" + value + "}").equals(props.get(key))) {
101                        props.remove(key);
102                }
103        }
104        
105        /** Removes build properties that were not set during the build process
106         *  (i.e. are still set to '${xxx}' placeholders).
107         *   
108         * @param props Properties object
109         */
110        private void removeDefaultProperties(Properties props) {
111                // projects not built in bamboo won't set these
112        removeDefaultProperty(props, "bamboo.buildKey", "bambooBuildKey");
113        removeDefaultProperty(props, "bamboo.buildNumber", "bambooBuildNumber");
114        removeDefaultProperty(props, "bamboo.buildPlanName", "bambooBuildPlanName");
115        removeDefaultProperty(props, "bamboo.buildTimeStamp", "bambooBuildTimeStamp");
116        removeDefaultProperty(props, "bamboo.buildForceCleanCheckout", "bambooForceCleanCheckout");
117        // bamboo.custom.cvs.last.update.time == "${bambooCustomCvsLastUpdateTime}" 
118        // if there is no -Dbamboo.custom.cvs.last.update.time property on mvn build cmdline
119        removeDefaultProperty(props, "bamboo.custom.cvs.last.update.time", "bambooCustomCvsLastUpdateTime");
120        removeDefaultProperty(props, "bamboo.custom.cvs.last.update.time.label", "bambooCustomCvsLastUpdateTimeLabel");
121        // bamboo.custom.cvs.last.update.time == "${bamboo.custom.cvs.last.update.time}"
122        // if there IS a -Dbamboo.custom.cvs.last.update.time property on mvn build cmdline
123        // but it's not set by bamboo (because it's now a git project, but the mvn cmdline hasn't been changed)  
124        removeDefaultProperty(props, "bamboo.custom.cvs.last.update.time", "bamboo.custom.cvs.last.update.time");
125        removeDefaultProperty(props, "bamboo.custom.cvs.last.update.time.label", "bamboo.custom.cvs.last.update.time.label");
126        
127        
128        // bamboo git projects have some extra properties we could include here; 
129        //   bamboo.repository.git.branch
130        //   bamboo.repository.git.repositoryUrl
131        // see https://confluence.atlassian.com/bamboo/bamboo-variables-289277087.html
132        
133        // projects in cvs repositories won't set this
134        removeDefaultProperty(props, "git.buildNumber", "buildNumber");
135        }
136        
137        /** See class documentation
138         * 
139         * @see javax.servlet.http.HttpServlet#doGet(HttpServletRequest request, HttpServletResponse response)
140         */
141    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException 
142    {
143        InputStream is = VersionServlet.class.getClassLoader().getResourceAsStream("/build.properties");
144        Properties props = new Properties();
145        if (is==null) {
146                props.put("error", "Missing build.properties");
147        } else {
148                props.load(is);
149                is.close();
150        }
151        
152        removeDefaultProperties(props);
153        File etcPropsFile = new File("/etc/build.properties");
154        if (etcPropsFile.exists()) {
155                Properties etcProps = new Properties();
156                is = new FileInputStream(etcPropsFile);
157                etcProps.load(is);
158                is.close();
159                removeDefaultProperties(etcProps);
160                props.put("/etc/build.properties", etcProps);
161        }
162        
163        response.setHeader("Content-Type", "application/json");
164        response.setStatus(HttpServletResponse.SC_OK);
165        response.getWriter().println(new JSONObject(props).toString());
166    }
167}
