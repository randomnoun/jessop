019/**
020 * The PropertyParser class parses a property definition text file into
021 * a Properties object.
022 *
023 * <p>This parser differs from the standard Properties parser in that
024 * sections can be marked off for particular regions (e.g. properties
025 * that only take effect in development, or when run on certain machines).
026 * The current region
027 * is specified by the <tt>com.randomnoun.common.mode</tt> system property
028 * set on the VM commandline. If this system property is not set,
029 * the region defaults to the value "<tt>localhost-dev-unknown</tt>".
030 * An alternate constructor exists if you
031 * wish to specify the region manually.
032 *
033 * <p>Environments are specified in '<i>machine-release-subsystem</i>'
034 * format, with each segment set as follows:
035 * <attributes>
036 * machine - the hostname of the system (in lowercase)
037 * release - the development phase of the system (either set to <tt>dev</tt>,
038 *   <tt>xpt</tt>, <tt>prd</tt> for development, acceptance or production
039 * subsystem - the subsystem that this VM represents. 
040 * </attributes>
041 *
042 * <p>Properties are specified in the standard "propertyName=propertyValue" method.
043 * Whitespace is removed from either side of the '=' character. New-lines
044 * can be specified in the property value by using the escape sequence '\n'.
045 * Lines can be continued over a single line by placing the character '\' at
046 * at the end of the line to be continued; e.g.
047 *
048 * <pre style="code">
049 *   property1=value1
050 *   property2=this is a very long value for property2, which spans \
051 *             over a single line.
052 * </pre>
053 *
054 * <p>Properties that are specific to a particular region should be surrounded
055 * by the lines "STARTENVIRONMENT environmentMask" and "ENDENVIRONMENT environmentMask".
056 * Individual properties can be defined for a region by prefixing the line
057 * with "ENV environmentMask"; e.g.
058 *
059 * <pre style="code">
060 *   property1=all regions
061 *
062 *   STARTENVIRONMENT *-xpt-*
063 *     property2=this property only set in acceptance region
064 *     property3=same for this property
065 *   ENDENVIRONMENT
066 *
067 *   STARTENVIRONMENT dtp11523-dev-*
068 *     property2=these properties only set in the development region
069 *     property3=running on the host dtp11523
070 *   ENDENVIRONMENT
071 *
072 *   ENV *-prd-* property4=this property only visible in production
073 * </pre>
074 *
075 * <p>As shown above, the '<tt>*</tt>' character can be used to specify a property
076 * across multiple regions. The keywords 'STARTENVIRONMENT', 'ENDENVIRONMENT' and
077 * 'ENV' are case-insensitive
078 * 
079 * <p>You can now also specify environments based on the values of previously-defined
080 * properties; this allows a simple <tt>#ifdef</tt> style facility. There are 
081 * two types of syntax, which use regex matching or simple string matching; e.g.
082 * 
083 * <pre style="code">
084 *   enable.fileAct=true
085 *   compound.property=123-456-789
086 * 
087 *   STARTENVIRONMENT enable.fileAct = true
088 *     # these properties are only set when enable.fileAct is set to true
089 *   ENDENVIRONMENT
090 *   STARTENVIRONMENT compound.property =~ *-789
091 *     # these properties are only set when compound.property ends with -789
092 *   ENDENVIRONMENT
093 * </pre>
094 * 
095 * <p>Property files can contain any number of blank lines, or comments (lines
096 * starting with the '#' character), which will be ignored by the parser.
097 *
098 * <p>Any occurences of the string "<tt>\n</tt>" in a property value will be
099 * replaced by a newline character.
100 *
101 * <p>A property make also contain a List, rather than a string, by including the
102 * index of the list item in square brackets in the property key; e.g.
103 *
104 * <pre style="code">
105 *   listname[1]=first element
106 *   listname[3]=third element
107 * </pre>
108 *
109 * <p>This implementation returns an ArrayList of Strings for these types of declarations.
110 * Undeclared array elements that appear before the last index will return null.
111 *
112 * <p>If you want to create a list, but the index of the elements is unknown (they are 
113 * dependant on other properties, for example), then you can use a "*" to denote the
114 * next available list index, or leave it empty to denote the last used list index; e.g.
115 * 
116 * <pre style="code">
117 * testList[0].a=a-value 0
118 * testList[0].b=b-value 0
119 * testList[0].c=a-value 0
120 *  
121 * testList[1].a=a-value 1
122 * testList[1].b=b-value 1
123 * testList[1].c=a-value 1
124 *  
125 * testList[*].a=a-value 2
126 * testList[].b=b-value 2
127 * testList[].c=a-value 2
128 * </pre>
129 * 
130 * will generate a three-element list, each of which contains a map with three key/value pairs.
131 *
132 * <p>Properties that appear multiple times will take the value of the last-specified
133 * value.
134 *
135 *
136 * @author  knoxg
137 * 
138 */

187    /**
188     * Generates a Properties object from the input stream.
189     *
190     * @return A valid Properties object.
191     *
192     * @throws IOException
193     * @throws ParseException
194     */
195    public Properties parse()
196        throws ParseException, IOException {
197        String line = ""; // current line
198        String token = ""; // current token
199
200        // ensure that all class fields have been reset
201        properties = new Properties();
202
203        line = lineReader.readLine();
204
205        while (line != null) {
206            line = line.trim();
207
208            // line-continuation (a line ending with '\' is appended with the next)
209            // could prove hazardous if we get a \ on the last line
210            while (line != null && line.endsWith("\\")) {
211                line = line.substring(0, line.length() - 1);
212
213                try {
214                    line = line + lineReader.readLine().trim();
215                } catch (NullPointerException npe) {
216                    // end of file reached; ignore
217                }
218            }
219
220            // true = don't return delimiters
221            thisline = new StringTokenizer(line, " =\t\n\r", true);
222
223            if (thisline.hasMoreTokens()) {
224                token = parseToken("keyword");
225
226                // are we in the correct environment ?
227                if (correctEnvironment || token.toLowerCase().equals("endenvironment")) {
228                    parseLine(token);
229                }
230            }
231
232            line = lineReader.readLine();
233        }
234
235        return properties;
236    }

238    /**
239     * Parses a single line of the site definition file.
240     *
241     * @param token The token that was at the beginning of this line
242     *
243     * @throws ParseException
244     */
245    @SuppressWarnings("unchecked")
246        private void parseLine(String token)
247        throws ParseException {
248        String lowerCaseToken;
249        String nextToken = null;
250        String value = null;
251
252        // System.out.println("P " + token + "(" + line + ")");
253        lowerCaseToken = token.toLowerCase();
254
255        if (token.startsWith("#")) {
256            // comment
257            // call the appropriate method depending on the keyword...
258        } else if (lowerCaseToken.equals("startenvironment")) {
259            parseStartEnvironment();
260        } else if (lowerCaseToken.equals("endenvironment")) {
261            parseEndEnvironment();
262        } else if (lowerCaseToken.equals("env")) {
263            parseEnv();
264                } else if (lowerCaseToken.equals("includeresource")) {
265                        parseIncludeResource();
266        } else {
267            // see if this is a key=value property assignment
268            try {
269                nextToken = parseToken("token");
270            } catch (ParseException pe) {
271                newParseException("Unknown keyword '" + token + "', '=' expected");
272            }
273
274            if (nextToken.equals("=")) {
275                // must be in a key=value assignment
276                value = null;
277
278                try {
279                    value = parseTokenToEOL("property value");
280                } catch (ParseException pe) {
281                }
282
283                if (value == null) {
284                    value = "";
285                }
286
287                // check if this is an array element    
288                int lb = token.indexOf('[');
289                int rb = token.indexOf(']');
290                
291
292                if (lb != -1 || rb != -1) {
293                    if (lb == -1 || rb == -1) {
294                        newParseException("Invalid list property key '" + token + "'");
295                    }
296                    List<Object> list = null;
297                    String keyPart = token.substring(0, lb);
298                    String listPart = token.substring(lb);
299                    try {
300                        list = (List<Object>) properties.get(keyPart);
301                    } catch (ClassCastException cce) {
302                        newParseException("Cannot create list '" + token + "', this property already exists");
303                    }
304                    if (list == null) {
305                        list = new ArrayList<Object>();
306                        properties.put(keyPart, list);
307                    }
308                    // if list index is "*" then create a new list index; if it's "-", then use the last index
309                    String index = token.substring(lb+1, rb);
310                    if (index.equals("*")) {
311                        index = String.valueOf(list.size()); // 0-based list, so next index is set to the size
312                        listPart = "[" + index + "]" + listPart.substring(listPart.indexOf("]")+1);
313                    } else if (index.equals("")) {
314                        index = String.valueOf(list.size()-1);
315                                                listPart = "[" + index + "]" + listPart.substring(listPart.indexOf("]")+1);
316                    }
317                    
318                    Struct.setValue(list, listPart, value, true, true, true);
319                } else {
320                    // no index supplied, just set the property 
321                    properties.setProperty(token, value);
322                }
323            } else {
324                newParseException("Unknown token '" + nextToken + "', '=' expected");
325            }
326        }
327    }

329    /**
330     * Creates and throws a new parse exception, which includes the current parse
331     * position within the Reader.
332     *
333     * @param s The additional text to be included in this exception
334     *
335     * @throws ParseException The parse exception requested
336     */
337    private void newParseException(String s)
338        throws ParseException {
339        throw new ParseException("line " + lineReader.getLineNumber() + ": " + s, 0);
340    }

342    /**
343     * Returns the next token.
344     *
345     * @param what The type of token we are expecting. This string is only used
346     *   in any parseExceptions which are thrown.
347     *
348     * @return The next token on the line.
349     *
350     * @throws ParseException A parsing exception has occurred.
351     */
352    private String parseToken(String what)
353        throws ParseException {
354        String result = null;
355
356        // keep grabbing tokens, until we hit something that isn't considered whitespace
357        while (result == null || result.equals(" ") || result.equals("\r") || result.equals("\n") || result.equals("\t")) {
358            if (!thisline.hasMoreTokens()) {
359                newParseException("Expecting " + what);
360            }
361
362            result = thisline.nextToken(" =\t\n\r");
363        }
364
365        // convert \n's to newlines.
366        result = result.replaceAll("\\\\n", "\n");
367
368        if (verbose) {
369            logger.debug("parsed token: '" + result + "'");
370        }
371
372        return result;
373    }

375    /**
376     * Grabs every token until the end of line, and returns it as a string. If the
377     * debugging variable 'verbose' is set to true, then each token is sent
378     * to System.out as it is read. Enclosing single or double-quotes are removed.
379     *
380     * @param what The type of token we are expecting. This text is used
381     *   in any parseExceptions which are thrown.
382     *
383     * @return The remaining text.
384     *
385     * @throws ParseException
386     */
387    private String parseTokenToEOL(String what)
388        throws ParseException {
389        String token;
390
391        if (!thisline.hasMoreTokens()) {
392            newParseException("Expecting " + what);
393        }
394
395        token = thisline.nextToken("\n").trim();
396        token = token.replaceAll("\\\\n", "\n");
397
398        if (verbose) {
399            System.out.println("parsed token: " + token);
400        }
401
402        return token;
403    }

405    /**
406     * Processes an ENV rule
407     *
408     * @throws ParseException
409     */
410    private void parseEnv()
411        throws ParseException {
412        String selectedenvironmentID;
413        String token;
414
415        selectedenvironmentID = parseToken("Environment ID");
416
417        // @TODO this should really use the same rules as parseStartEnvironment
418        if (selectedenvironmentID.equals(environmentID)) {
419            token = parseToken("keyword").toLowerCase();
420            parseLine(token);
421        }
422    }

424        private void parseIncludeResource() throws ParseException {
425                // include another properties file in here. This has a very
426                // good chance of creating a infinite loop if one property file
427                // includes another one, which in turn includes the first one. 
428                // - this will manifest itself in some kind of OutOfMemoryException
429                // or a StackOverflowException
430                
431                String resourceName = parseTokenToEOL("resource name");
432                logger.debug("Including property resource '" + resourceName + "'");
433
434                // load properties from classpath (in .EAR)
435                InputStream inputStream = PropertyParser.class.getClassLoader().getResourceAsStream(resourceName);
436                if (inputStream==null) {
437                        throw new ParseException("Could not find included resource '" + resourceName + "'", 0);
438                }
439                PropertyParser propertyParser = new PropertyParser(new InputStreamReader(inputStream), environmentID);
440                Properties includedProperties = new Properties();
441                try {
442                        includedProperties = propertyParser.parse();
443                } catch (Exception e) {
444                        throw (ParseException) new ParseException("Could not load included resource '" +
445                                resourceName + "'", 0).initCause(e);
446                }
447                
448                // should merge lists/maps, rather than replacing them.
449                // (maybe this should be configurable ?)
450                // @TODO code below only merges lists
451                for (Iterator<Entry<Object, Object>> i = includedProperties.entrySet().iterator(); i.hasNext(); ) {
452                        Map.Entry<Object, Object> entry = i.next();
453                        if (entry.getValue() instanceof List) {
454                                Object existingObj = properties.get(entry.getKey());
455                                if (existingObj==null || !(existingObj instanceof List)) {
456                                        properties.put(entry.getKey(), entry.getValue());
457                                } else {
458                                        @SuppressWarnings("unchecked")
459                                        List<Object> existingList = (List<Object>) existingObj;
460                                        @SuppressWarnings("unchecked")
461                                        List<Object> listValue = (List<Object>) entry.getValue();
462                                        for (int j=0; j<listValue.size(); j++) {
463                                                if (listValue.get(j)!=null) {
464                                                        Struct.setListElement(existingList, j, listValue.get(j));
465                                                }
466                                        }
467                                }
468                        } else {
469                                properties.put(entry.getKey(), entry.getValue());
470                        }
471                }
472                // properties.putAll(includedProperties);
473        }

475    /** Begins per-environment parsing rules.
476     *  @throws ParseException
477     */
478    private void parseStartEnvironment()
479        throws ParseException {
480        if (inEnvironment) {
481            newParseException("attempted to nest environment areas");
482        }
483        String propertyName = "environmentId";
484        String propertyMatch = null;
485        String propertyValue = null;
486        String envSpec = parseTokenToEOL("Environment specification");
487        if (envSpec.indexOf("=~")!=-1) {
488                propertyName = envSpec.substring(0, envSpec.indexOf("=~")).trim();
489                propertyMatch = envSpec.substring(envSpec.indexOf("=~")+2).trim();
490                        propertyMatch = propertyMatch.replaceAll("\\*", ".*");
491                propertyValue = propertyName.equals("environmentId") ? environmentID : properties.getProperty(propertyName);
492                if (propertyValue==null) { propertyValue = ""; }
493                        correctEnvironment = propertyValue.matches(propertyMatch);      
494        } else if (envSpec.indexOf("=")!=-1) {
495                        propertyName = envSpec.substring(0, envSpec.indexOf("=")).trim();
496                        propertyMatch = envSpec.substring(envSpec.indexOf("=")+1).trim();
497                        propertyValue = propertyName.equals("environmentId") ? environmentID : properties.getProperty(propertyName);
498                        if (propertyValue==null) { propertyValue = ""; }
499                        correctEnvironment = propertyValue.equals(propertyMatch);       
500        } else {
501                // not on 'xxx=xxx' form, default to old behaviour (match on envId)
502                        propertyMatch = envSpec.replaceAll("\\*", ".*");
503                        correctEnvironment = environmentID.toLowerCase().matches(propertyMatch.toLowerCase());
504        }
505        
506        // test for "property=value" style environments
507        //selectedenvironmentID = selectedenvironmentID.replaceAll("\\*", ".*");
508        //correctEnvironment = environmentID.matches(selectedenvironmentID);
509        inEnvironment = true;
510    }

512    /**
513     * Completes per-environment parsing rules.
514     *
515     * @throws ParseException
516     */
517    private void parseEndEnvironment()
518        throws ParseException {
519        if (!inEnvironment) {
520            newParseException("'endenvironment' without matching 'startenvironment'");
521        }
522
523        inEnvironment = false;
524        correctEnvironment = true;
525    }

527    /**
528     * Returns a subset of a Properties object. The subset is determined by only
529     * returning those key/value pairs whose keys begin with a set prefix. e.g.
530     * if 'a' contains the properties
531     *
532     * <attributes>
533     *   customer.1.name=fish
534     *   customer.1.description=Patagonian toothfish
535     *   customer.2.name=hunter
536     *   customer.2.description=Patagonian toothfish hunter
537     *   customer.11.name=spear
538     *   customer.11.description=Patagonian toothfish hunter's spear
539     * </attributes>
540     *
541     * <p>then calling <code>restrict(a, "customer.1", false)</code> will return:
542     *
543     * <attributes>
544     *   customer.1.name=fish
545     *   customer.1.description=Patagonian toothfish
546     * </attributes>
547     *
548     * <p>setting the 'removePrefix' to true will remove the initial prefix from
549     * the returned property list; <code>restrict(a, "customer.1", true)</code> in
550     * this case will then return:
551     *
552     * <attributes>
553     *   name=fish
554     *   description=Patagonian toothfish
555     * </attributes>
556     *
557     * <p>Note that the prefix passed in to this method has no trailing period,
558     * but each property must contain that period (to prevent <tt>customer.11</tt>
559     * from being returned in the example above).
560     *
561     * <p>If 'properties' is set to null, then this method will return null.
562     * If 'prefix' is set to null, then this method will return the entire property list.
563     *
564     * @param properties The initial property list that we wish to restrict
565     * @param prefix     The prefix used to restrict the property list
566     * @param removePrefix  If set to true, the result keys will be stripped of the initial prefix text
567     *
568     * @return A restricted property list.
569     */
570    public static Map<? extends Object, ? extends Object> restrict(Map<Object, Object> properties, String prefix, boolean removePrefix) {
571        if (properties == null) {
572            return null;
573        }
574
575        if (prefix == null) {
576            return properties;
577        }
578
579        Properties result = new Properties();
580
581        // this could possibly break existing, yet weird code
582        // if (!prefix.endsWith(".")) { prefix = prefix + "."; };
583        prefix = prefix + ".";
584        
585        Map.Entry<Object, Object> entry;
586
587        for (Iterator<Entry<Object, Object>> i = properties.entrySet().iterator(); i.hasNext();) {
588            entry = i.next();
589
590            String key = (String) entry.getKey();
591
592            if (key.startsWith(prefix)) {
593                if (removePrefix) {
594                    key = key.substring(prefix.length());
595                }
596
597                result.put(key, entry.getValue());
598            }
599        }
600
601        return result;
602    }

605     * Method used to test the parser from the command line. The file to parse is
606     * specified on the command line; if missing, then it uses 'test.properties'
607     * as the default.
608     *
609     * @param args
610     *
611     * @throws IOException
612     * @throws ParseException
613     */
614    public static void main(String[] args)
615        throws IOException, ParseException {

617        PropertyParser propertyParser;
618
619        if (args.length != 1) {
620            System.out.println("Reading from '" + filename + "' by default...");
621        } else {
622            filename = args[0];
623        }
624

626            propertyParser = new PropertyParser(new FileReader(filename));

628            System.out.println("Parse OK");
