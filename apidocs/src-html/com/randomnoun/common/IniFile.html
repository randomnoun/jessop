007import java.io.File;
008import java.io.InputStream;
009import java.io.FileInputStream;
010import java.io.IOException;
011import java.io.LineNumberReader;
012import java.io.InputStreamReader;
013import java.text.ParseException;
014
015import java.util.HashMap;
016import java.util.Map;
017import java.util.Set;
018
019
020/** Windows .ini file parser
021 * 
022 * Properties without explicit groups are recorded with a null group
023 * 
024 * 
025 */
026public class IniFile {
027
028
029        /** Map of property groups to property name-value pairs */
030        private Map<String, Map<String, String>> properties;
031        
032        public IniFile() {
033                properties = new HashMap<String, Map<String, String>>();
034        
035        }
036        
037        public void load(File file) throws IOException, ParseException {
038                InputStream is = new FileInputStream(file);
039                load(is);
040        }
041        
042        public void load(InputStream is) throws IOException, ParseException {
043                LineNumberReader lnr = new LineNumberReader(new InputStreamReader(is));
044                try {
045                        String line = lnr.readLine();
046                        String group = null;
047                        while(line!=null) {
048                                // @TODO handle newlines 
049                                line = line.trim();
050                                if (line.startsWith(";")) {
051                                        // comment; ignore
052                                } else if (line.startsWith("[")) {
053                                        if (line.endsWith("]")) {
054                                                group = line.substring(1, line.length()-1);
055                                        } else {
056                                                throw new ParseException("Line " + lnr.getLineNumber() + ": invalid .ini group declaration", 0);
057                                        }
058                                } else if (line.contains("=")) {
059                                        int pos = line.indexOf("=");
060                                        String name = line.substring(0, pos);
061                                        String value = line.substring(pos+1);
062                                        Map<String, String> propMap = properties.get(group);
063                                        if (propMap==null) {
064                                                propMap = new HashMap<String, String>();
065                                                properties.put(group, propMap);
066                                        }
067                                        propMap.put(name, value);
068                                }
069                                
070                                line=lnr.readLine();
071                        }
072                } finally {
073                        try {
074                                is.close();     
075                        } catch (IOException ioe) {
076                                // ignore
077                        }
078                }
079        }
080        
081        public String get(String group, String key) {
082                Map<String, String> propMap = properties.get(group);
083                if (propMap==null) { return null; }
084                return propMap.get(key);
085        }
086        
087        public Set<String> getGroups() {
088                return properties.keySet();
089        }
090        
091        public Set<String> getGroupKeys(String group) {
092                Map<String, String> propMap = properties.get(group);
093                if (propMap==null) { return null; }
094                return propMap.keySet();
095        }
096        
097}
