007import java.util.*;
008
009import org.apache.log4j.Logger;
010
011/**
012 * Most-recently used cache class. 
013 * 
014 * <p>You probably want to use something in the guava Collections library these days instead.
015 * 
016 * <p>This class can be used to implement serverside data caches. The constructor
017 * takes three parameters: maximum size, expiry time (in milliseconds) and a 
018 * 'RetrievalCallback' object (defined as an interface in this class). 
019 * 
020 * <p>As items are put into this object, they are timestamped; if a key is applied
021 * to retrieve an object that has 'expired', then this object will return null
022 * (if no callback has been supplied), or will execute the callback to refresh
023 * the value. A callback can (optionally) be supplied with every get() invocation
024 * so that this object knows precisely how to generate the requested object. (This
025 * works since it is usually computationally cheaper to generate the Callback object
026 * than to generate the result of the callback, which generally needs to fetch
027 * information from a database).
028 * 
029 * <p>There are two possible expiry mechanisms: either an item is expired from the
030 * time it is entered into the cache, or expired from the time it is last retrieved
031 * from the cache. The former makes sense for 'dynamic' data that may change over time
032 * (e.g. rowcounts); the latter makes sense for 'static' data that is just used to
033 * cache large objects that don't need to be in memory at all times 
034 * (e.g. message definitions). By default this object will operate as per 'static'
035 * rules; the other method can be selected by calling the setDynamicCaching() after
036 * construction.
037 *
038 * <p><b>Implementation notes</b>
039 *    
040 * <p>It may be possible to extend this class to use WeakReferences, so that it can
041 * shrink in size as memory constraints within the VM become increased. This is 
042 * unlikely to be useful in practice, however, since this is probably an indication
043 * of a memory leak or over-utilised server. It may make more sense to just reduce the
044 * maximum size in these cases. Performance tuning would be required to know 
045 * for sure.
046 * 
047 * <p>Also note that if an object exceeds it's expiryTime, it is *not* automatically
048 * removed from the cache. This would involve having an additional Timer thread
049 * running through these objects and scanning for expired objects, which is unlikely
050 * to be useful in practice. An additional method could be implemented to trigger these
051 * cleanups if this becomes necessary.
052 * 
053 * @author knoxg
054 * 
055 *
056 */
057
058// this probably isn't as efficient as it could be (it contains a mruList AND a 
059// HashMap of expiryTimes, which I'm sure is overkill. To do this properly
060// would involve extending Map.Entry to include a timestamp value, however,
061// which would require a fair amount of re-engineering. (would need to subclass 
062// Map.Entry, and use "this.putAll()" to insert subclasses into this instance. 
063// you'd still need a way of keeping a sorted list of timestamped entries anyway,
064// now that I think about it (to choose the next expiry candidate), so this may not 
065// be so bad after all. 
066public class MRUCache
067    extends HashMap
068{
069    
070    
071    
072    /** Logger instance for this class */
073    public static Logger logger = Logger.getLogger(MRUCache.class);
074    
075    /** The callback interface. Caches should provide one of these classes to 
076     * perform any time-consuming tasks required to generate values in this cache.
077     * If the time taken to generate a value isn't time-consuming, then perhaps
078     * you shouldn't be using a cache to store them.
079     */
080    public static interface RetrievalCallback {
081        
082        /** Dynamically generates the value of an entity in a cache
083         * 
084         * @param key The key used to reference the cached object
085         * 
086         * @return The newly generated (or updated) cached object
087         */ 
088        public Object get(Object key);
089    }
090    
091    /** If set to true, expires data from the time it was <i>entered</i> into the cache,
092     * rather than the time if was <i>fetched</i> from the cache. Suitable when we 
093     * wish to apply a strict expiry time for dynamically-updating data, e.g.
094     * rowcounts.
095     */
096    private boolean dynamicCaching = false;
097
098    /** An ordered list of the objects in this cache. Objects at position 0 on the list
099     *  have been the most recently accessed (i.e. objects will be expired from the
100     * end of this list). */
101    private LinkedList mruList;
102
103    /** The maximum number of elements that this cache can hold. If &lt;= 0, then the
104     * size of the map is unlimited. */
105    private int cacheSize;
106
107    /** A callback which can be used to populate entries in the cache that are unknown,
108     *  or have expired. */
109    private RetrievalCallback callbackInstance;
110
111    /** A Map whose keys are elements in this cache, and who's entries are the last time
112     *  that entry was 'touched' (i.e. added to or retrieved from the cache) */
113    private Map lastUpdatedTimes;
114
115    /** The amount of time an entry can remain valid, measured from when the element
116     *  is first added to the cache. If expiryTime is set to &lt;= 0, then items never expire. */
117    private int expiryTime;
118
119    /**
120     * Creates a new MRUCache object.
121     *
122     * @param cacheSize  The maximum number of elements that this cache can hold. A value &lt;=0 means 
123     *    that the size of this map is not limited.
124     * @param expiryTime The amount of time (in ms) an entry can remain valid, measured from when the element
125     *    is first added to the cache (or last retrieved if {@link #setDynamicCaching()} is 
126     *    in effect. A value &lt;=0 means that items do not have an
127     *    expiry period.
128     * @param callback   A callback which can be used to populate entries in the cache that are unknown,
129     *    or have expired. This parameter can be set to null <b>if and only if</b> the
130     *    two-parameter {@link #get(String, RetrievalCallback)} method is used to retrieve
131     *    elements from the map.
132     */
133    public MRUCache(int cacheSize, int expiryTime, RetrievalCallback callback)
134    {
135        mruList = new LinkedList();
136        lastUpdatedTimes = new HashMap();
137        this.cacheSize = cacheSize;
138        this.callbackInstance = callback;
139        this.expiryTime = expiryTime;
140    }
141
142    /**
143     * Retrieve an element from the cache. If a callback was defined in the cache
144     * constructor, then this will be used to refresh
145     * values in the cache if they are missing, or have expired.
146     *
147     * @param key key whose associated value is to be returned
148     *
149     * @return the value to which this map maps the specified key, or
150     *   null if the map contains no mapping for this key.
151     */
152    public synchronized Object get(Object key)
153    {
154        return get(key, callbackInstance);
155    }
156
157    /**
158     * Retrieves an element from the cache. The supplied callback will be used to
159     * refresh the value in the cache if it is missing, or has expired. If the
160     * requested element is missing and the callback is set to null, then
161     * null is returned.
162     *
163     * @param key key whose associated value is to be returned
164     * @param callback a callback used to populate the cache if necessary
165     *
166     * @return the value to which this map maps the specified key, or
167     *   null if the map contains no mapping for this key.
168     */
169    public synchronized Object get(Object key, RetrievalCallback callback)
170    {
171        long now = System.currentTimeMillis();
172
173        if (!this.containsKey(key))
174        {
175            if (callback == null) {
176                return null;
177            } else {
178                // cache is missing key; retrieve from callback and update access times
179                synchronized (this) {
180                    if (cacheSize > 0 && mruList.size() >= cacheSize) {
181                        this.remove(mruList.getLast());
182                        mruList.removeLast();
183                    }
184
185                    super.put(key, callback.get(key));
186                    lastUpdatedTimes.put(key, new Long(now));
187                    mruList.addFirst(key);
188                }
189            }
190            
191        } else {
192            
193            // cache contains key
194            if (expiryTime > 0) {
195                
196                // expire this element if it's old
197                Long lastUpdatedTime = (Long)lastUpdatedTimes.get(key);
198                if (lastUpdatedTime==null) {
199                        logger.warn("MRUCache contains key '" + key + "' with no lastUpdatedTime set");
200                }
201                if (lastUpdatedTime == null || (lastUpdatedTime.longValue() + expiryTime < now)) {
202                    synchronized (this) {
203                        if (callback == null) {
204                            // no callback, we just remove it
205                            super.remove(key);
206                            lastUpdatedTimes.remove(key);
207                            mruList.remove(key);
208
209                            return null;
210                        } else {
211                            super.put(key, callback.get(key));
212                            lastUpdatedTimes.put(key, new Long(now));
213                        }
214                    }
215                }
216            }
217
218            // we have an element and it has been retrieved, move it to the
219            // top of the mruList. (we may not necessarily want to do this;
220            // this should be a constructor preference). There are pros & cons 
221            // to this approach:
222            // pros: if we just got an element, we probably don't want it expiring out
223            //   of the cache
224            // cons: this moves 'stale' objects to the top of the mruList
225            //   stack; this means that when we remove elements from mruList
226            //   to fit in more elements, we may be expiring older elements.
227            // this should probably be a decision made by the developer, rather than
228            // this code, but I'd like to keep the interface simple.
229            if (!dynamicCaching) {
230                synchronized (this) {
231                    mruList.remove(key);
232                    mruList.addFirst(key);
233                }
234            }
235        }
236
237        return super.get(key);
238    }
239
240    /**
241     * Returns the object if it is in the cache and has not expired.
242     * This method will return null if the the key is not in cache, or
243     * if it is in the cache but has expired.
244     *
245     * @param key key whose associated value is to be returned
246     *
247     * @return the value to which this map maps the specified key, or
248     *   null if the map contains no mapping for this key.
249     */
250    public synchronized Object getNoCallback(Object key) {
251        if (!this.containsKey(key)) {
252            return null;
253        }
254
255        if (expiryTime > 0) {
256            if (((Long)lastUpdatedTimes.get(key)).longValue() + expiryTime < 
257              System.currentTimeMillis()) 
258            {
259                return null;
260            }
261        }
262
263        return super.get(key);
264    }
265
266    /**
267     * Add an element into the cache.
268     *
269     * @param key  key with which the specified value is to be associated.
270     * @param value value to be associated with the specified key.
271     * 
272     * @return the previous value of this map entry, if one exists, otherwise null
273     */
274    public synchronized Object put(Object key, Object value) {
275        
276        synchronized (this) {
277                        Object lastValue = super.put(key, value);
278            lastUpdatedTimes.put(key, new Long(System.currentTimeMillis()));
279            return lastValue;
280        }
281    }
282    
283    /** Enforces dynamic caching rules. When called, data will be expired from the time it 
284     * is <i>entered</i> into the cache, rather than the time if is <i>fetched</i> from 
285     * the cache. Suitable when we wish to apply a strict expiry time for 
286     * dynamically-updating data, e.g. rowcounts.
287     */
288    public void setDynamicCaching() {
289        dynamicCaching = true;
290    }
291}
