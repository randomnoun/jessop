021/**
022 * This class manages users, roles, resources and permissions for an application.
023 * 
024 * <p>Most of the code that adds/deletes/maintains these objects has been removed, that is
025 * now the responsibility of the security implementation code. 
026 * 
027 * <p>Methods to read user and permission data are delegated to the SecurityLoader, and
028 * methods that authenticate users are delegated to the SecurityAuthenticator.
029 * 
030 * <p>This class now mostly acts as a cache for user and role data, and can perform
031 * simple and complex permission checks for users against resources.
032 * 
033 * <p>The following properties can be passed to the SecurityContext during construction;
034 * property keys are defined as static public final Strings in this class.
035 *
036 * <attributes>
037 *   INIT_CASE_INSENSITIVE - make the security cache case-insensitive, typically when interfacing with 
038 *     Active Directory. Defaults to false.
039 *   INIT_USER_CACHE_SIZE - maximum size of user cache
040 *   INIT_USER_CACHE_EXPIRY - expiry time of users from the user cache (in milliseconds). If this
041 *     property is not set, user caching is disabled.
042 * </attributes>
043 *
044 * <p>Additional properties may also be required based on the SecurityLoader implementation used.
045 * 
046 * @author knoxg
047 *
048 * @see com.randomnoun.common.security.SecurityLoader
049 */
050public class SecurityContext {

118        /** this method is only called if the required value is not in the cache,
119         * or the value in the cache has expired.
120         *
121         * @param key The username of the User to return
122         */
123        public Object get(Object key) {
124            if (key==null) { throw new NullPointerException("null key"); }
125            if (!(key instanceof User)) {
126                throw new IllegalArgumentException("Expected user as User, found " + key.getClass().getName());
127            }
128
129            User user = (User) key;
130                        
131                        Map<String, Permission> result = new HashMap<String, Permission>();
132            try {
133                List<Permission> permissions = loader.loadUserPermissions(user);
134                for (Iterator<Permission> i = permissions.iterator(); i.hasNext(); ) {
135                        Permission perm = (Permission) i.next();
136                        if (result.containsKey(perm.getActivity() + "." + perm.getResource())) {
137                                throw new IllegalStateException("User '" + key + "' contains two versions of permission '" + 
138                                  perm.getActivity() + "." + perm.getResource() + "'; please check the database");
139                        }
140                        result.put(perm.getActivity() + "." + perm.getResource(), perm);
141                }
142            } catch (IOException ioe) {
143                throw new RuntimeException("IOException reading user permissions", ioe);
144            }
145            return result;
146        }
147    }

457    /** Returns true if a user is allowed to perform the permission supplied. The permission
458     *  is expressed in 'activity.resourceType' format, e.g. 'update.message'. No expression
459     *  context is supplied; this method will not evaluate any conditional resource
460     *  restrictions. This is useful in cases where the full resource context is not known,
461     *  for example when a message is first created by a user.
462     *
463     *  <p>In this case, the 'create.message' permission can be checked using this method
464     *  before the user starts entering information, and 'create.message' can be
465     *  checked with an expression context after the header fields have been populated.
466     *
467     * <p>If a permission is supplied that is not known by the application, this
468     * method will return false.
469     *
470     * @param user The user we are determining
471     * @param permission The permission we are testing for. Permissions are expressed in
472     *   'activity.resourceType' format.
473     * @return true if the permission is allowed, false is the permission is denied.
474     *
475     * @throws NullPointerException if either parameter to this method is null
476     * @throws IllegalArgumentException if the permission supplied is formatted incorrectly.
477     */
478    public boolean hasPermission(User user, String permission) {
479        return hasPermission(user, permission, null);
480    }

482    /**
483     * Returns true if a user is allowed to perform the permission supplied, with
484     * given resource context. If a permission is assigned to both the user
485     * and the role, then the user permission is evaluated first.
486     *
487     * @param user The user we are determining
488     * @param permission The permission we are testing for. Permissions are expressed in
489     *   'activity.resourceType' format.
490     * @param context The resource context used to evaluate against the resource expression
491     *
492     * @return true if the permission is allowed, false is the permission is denied.
493     *
494     * @throws NullPointerException if either parameter to this method is null
495     * @throws IllegalArgumentException if the permission supplied is formatted incorrectly.
496     */
497    public boolean hasPermission(User user, String permission, Map<String, Object> context) {
498        // @TODO should get this User out of our userCache, keyed by id
499        if (permission == null) { throw new NullPointerException("Null permission"); }
500        if (user == null) { throw new NullPointerException("Null user"); }
501
502        int pos = permission.indexOf('.');
503        if (pos == -1) {
504            throw new IllegalArgumentException("Illegal permission value '" + permission + "'");
505        }
506
507                //  try per-user permissions...
508                Map<String, Permission> userPermissions = userPermissionCache.get(user);
509                if (userPermissions == null) {
510                        // @TODO - load from DB ? should be handled by MRUCache
511                        throw new IllegalStateException("Unknown user '" + user.getUsername() + "'");
512                } else {
513                        Permission userPermission = (Permission) userPermissions.get(permission);
514                        if (userPermission!=null) {
515                                if (context == null) {
516                                        return true;
517                                } else {
518                                        ResourceCriteria criteria = userPermission.getResourceCriteria();
519                                        if (criteria == null || criteria.evaluate(context)) {
520                                                return true;
521                                        }
522                                }
523                        }
524                }
525                
526                // then try per-role permissions ...
527                List<String> roles = userRoleCache.get(user);
528                if (roles == null) {
529                        // @TODO - load from DB ? should be handled by MRUCache
530                        throw new IllegalStateException("Unknown user '" + user.getUsername() + "'");
531                } else {
532                        for (Iterator<String> i = roles.iterator(); i.hasNext(); ) {
533                                String rolename = (String) i.next();
534                                Map<String, Permission> rolePermissions = rolePermissionCache.get(rolename);
535                                if (rolePermissions == null) {
536                                        // @TODO - load from DB ? should be handled by MRUCache
537                                        throw new IllegalStateException("Unknown role '" + rolename + "'");
538                                } else {
539                                        // this is a strange data structure
540                                        Permission rolePermission = (Permission) rolePermissions.get(permission);
541                                        if (rolePermission!=null) {
542                                                if (context == null) {
543                                                        return true;
544                                                } else {
545                                                        ResourceCriteria criteria = rolePermission.getResourceCriteria();
546                                                        if (criteria == null || criteria.evaluate(context)) {
547                                                                return true;
548                                                        }
549                                                }       
550                                        }
551                                }
552                        }
553                }
554
555        return false;
556    }

579                        throw new IllegalArgumentException("Illegal permission value '" + permission + "'");

598    /**
599     * Returns a list of all Permission objects assigned to a user and all the 
600     * roles that the user is a member of. This allows multiple permission conditions
601     * to be applied to a user, one for each role.
602     *
603     * @param user The user we are determining
604     * @param permission The permission we are testing for. Permissions are expressed in
605     *   'activity.resourceType' format.
606     *
607     * @return a List of Permission objects, or an empty list if the user 
608     *   (and none of their roles) contains this permission
609     *
610     * @throws NullPointerException if either parameter to this method is null
611     * @throws IllegalArgumentException if the permission supplied is formatted incorrectly.
612     */
613    public List<Permission> getPermissions(User user, String permission) {
614        // @TODO should get this User out of our userCache, keyed by id
615        if (permission == null) { throw new NullPointerException("Null permission"); }
616        if (user == null) { throw new NullPointerException("Null user"); }
617        List<Permission> result = new ArrayList<Permission>();
618
619        int pos = permission.indexOf('.');
620        if (pos == -1) {
621            throw new IllegalArgumentException("Illegal permission value '" + permission + "'");
622        }
623
624                //  try per-user permissions...
625                Map<String, Permission> userPermissions = userPermissionCache.get(user);
626                if (userPermissions == null) {
627                        // @TODO - load from DB ? should be handled by MRUCache
628                        throw new IllegalStateException("Unknown user '" + user.getUsername() + "'");
629                } else {
630                        Permission userPermission = (Permission) userPermissions.get(permission);
631                        if (userPermission != null) {
632                                result.add(userPermission);
633                        }
634                }
635                
636                // then try per-role permissions ...
637                List<String> roles = userRoleCache.get(user);
638                if (roles == null) {
639                        throw new IllegalStateException("Unknown user '" + user.getUsername() + "'");
640                } else {
641                        for (Iterator<String> i = roles.iterator(); i.hasNext(); ) {
642                                String rolename = (String) i.next();
643                                Map<String, Permission> rolePermissions = rolePermissionCache.get(rolename);
644                                if (rolePermissions == null) {
645                                        throw new IllegalStateException("Unknown role '" + rolename + "'");
646                                } else {
647                                        Permission rolePermission = (Permission) rolePermissions.get(permission);
648                                        if (rolePermission!=null) {
649                                                result.add(rolePermission);
650                                        }
651                                }
652                        }
653                }
654                return result;
655    }

667    /** Clear all caches and re-initialises this security context (as defined 
668     * in this instance's initial initialisation properties). 
669     * This method also resets this security context's loader.
670     */
671    @SuppressWarnings("unchecked")
672        public void resetSecurityContext() {
673        // logger.debug("Security context properties: " + properties.toString());
674        
675                int cacheSize = Integer.MAX_VALUE;
676                int cacheExpiry = Integer.MAX_VALUE;
677        if (properties.get(INIT_USER_CACHE_SIZE) != null && properties.get(INIT_USER_CACHE_EXPIRY) != null) {
678            cacheSize = Integer.parseInt((String) properties.get(INIT_USER_CACHE_SIZE));
679            cacheExpiry = Integer.parseInt((String) properties.get(INIT_USER_CACHE_EXPIRY));
680        }
681
682        UserPermissionCallback userPermissionCallback = new UserPermissionCallback(this.securityLoader);
683        userPermissionCache = new MRUCache(cacheSize, cacheExpiry, userPermissionCallback );
684
685                UserRoleCallback userRoleCallback = new UserRoleCallback (this.securityLoader);
686                userRoleCache = new MRUCache(cacheSize, cacheExpiry, userRoleCallback );
687
688                RolePermissionCallback rolePermissionCallback = new RolePermissionCallback (this.securityLoader);
689                rolePermissionCache = new MRUCache(cacheSize, cacheExpiry, rolePermissionCallback );
690
691                UserCallback userCallback = new UserCallback (this.securityLoader);
692                userCache = new MRUCache(cacheSize, cacheExpiry, userCallback );
693
694                try {
695                        securityLoader.resetSecurityContext();  
696                } catch (IOException ioe) {
697                        throw (IllegalArgumentException) new IllegalArgumentException(
698                          "Cannot initialise security Context").initCause(ioe);
699                }
700    }
