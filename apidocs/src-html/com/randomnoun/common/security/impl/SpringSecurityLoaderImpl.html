027/**
028 * An implementation of the {@link com.randomnoun.common.security.SecurityLoader}
029 * class, using the Spring framework to populate the SecurityContext from a JDBC
030 * datasource.
031 * 
032 * <p>This security context used to optionally take a customerId (used to partition
033 * users across separate SaaS contexts), or an applicationId (used to partition
034 * permissions across separate software products). If not defined, then these 
035 * columns did not need to be present in the database. This functionality
036 * has been deprecated.
037 * 
038 * <p>I think we had String userids at one point as well, so look out for that.
039 *
040 * <p>This class has a number of initialisation properties that are specific
041 * to this class (in addition to those initialisation properties that are
042 * set by the SecurityContext itself):
043 *
044 * <attributes>
045 * {@link #INIT_JDBCTEMPLATE} - The Spring JdbcTemplate class used to retrieve
046 *   information from a database.
047 * {@link #INIT_USERNAME_MASK} - Used to mangle usernames before they are looked
048 *   up in an LDAP repository. A mask does a simple text replacment, with the
049 *   text "<tt>{username}</tt>" being replaced by the real username being passed
050 *   to the authenticate method. <i>e.g.</i> a username mask of
051 *   "{username}@abc.com" will convert the username "bob" to "bob@abc.com"
052 *   before authentication is performed.
053 * {@link #INIT_TABLE_SUFFIX} - A String appended to each table used in queries
054 *   against the database. Used to separate the 'LIVE' area from the 'WORK' staging
055 *   area.
056 * {@link #INIT_DATABASE_VENDOR} - Set to one of the SqlGenerator.DATABASE_* constants,
057 *   which specifies what syntax of SQL to use (DB2, Oracle or SqlServer).
058 * {@link #INIT_AUDIT_ENABLED} - (Boolean) If false, then disables auditting security changes
059 * </attributes>
060 * 
061 * @TODO this class does not internationalise text added to the audit log
062 *   (it should be in a form that can be parsed and internationalised depending on the
063 *   user viewing it).
064 *
065 * 
066 * @author knoxg
067 */

168        /** Returns true if this security context is case-insensitive, false otherwise
169         * 
170         * @return true if this security context is case-insensitive, false otherwise
171         */
172        private boolean getCaseInsensitive() {
173                // @TODO: cache this
174                return ((Boolean) properties.get(INIT_CASE_INSENSITIVE_OBJ)).booleanValue();
175        }

177        /** Return the sql for a string comparison for a username field, taking
178         * case sensitivity into account.
179         * 
180         * @param lhs expression to lowercase (if required) 
181         *
182         * @return
183         */
184        private String lowerSql(String lhs) {
185                if (getCaseInsensitive()) {
186                        String vendor = getDatabaseVendor();
187                        if (vendor.equals(SqlGenerator.DATABASE_DB2)) {
188                                return "LOWER(" + lhs + ")";
189                        } else if (vendor.equals(SqlGenerator.DATABASE_ORACLE)) {
190                                return "LOWER(" + lhs + ")";
191                        } else if (vendor.equals(SqlGenerator.DATABASE_SQLSERVER)) {
192                                return "LOWER(" + lhs + ")";
193                        } else if (vendor.equals(SqlGenerator.DATABASE_MYSQL)) {
194                                return "LOWER(" + lhs + ")";
195                        } else {
196                                throw new IllegalStateException("Case-insensitivity security contexts" +
197                                  " not supported for database type '" + vendor + "'");
198                        }
199                } else {
200                        return lhs;
201                }
202        }
203        
204        /** Converts a value to lowercase, but only if case insensitivity is required */
205        private String lower(String value) {
206                if (getCaseInsensitive()) {
207                        return value.toLowerCase();
208                } else {
209                        return value;
210                }
211        }

213        /** Converts a long into an object suitable to be passed to the database
214         * as a roleId or permissionId. 
215         * 
216         * @param var object to convert
217         * 
218         * @return a Long representation of the object (or Integer for JET DBs)
219         */
220        private Object toSequenceType(long var) {
221                String vendor = getDatabaseVendor();
222                if (vendor.equals(SqlGenerator.DATABASE_JET)) {
223                        return new Integer((int) var);
224                } else {
225                        return new Long(var);
226                }

283        /** Load all role permissions.
284         *
285         * {@inheritDoc}
286         *
287         * @return A List of Permission objects for all roles in the current application. Does
288         * not return Permissions that are not explicitly associated with a role in the security table.
289         *
290         * @throws IOException if an error occured loading from the database. This IOException
291         *   will always contain a spring DataAccessException which can be accessed in its .getCause()
292         *   method.
293         *
294         * @see com.randomnoun.common.security.SecurityLoader#loadAllRolePermissions()
295         */
296        public List<Permission> loadAllRolePermissions()
297                throws IOException
298        {
299                JdbcTemplate jt = (JdbcTemplate)properties.get("jdbcTemplate");
300                String loadAllRolePermissionsSql = 
301                  "SELECT roleName, activityName, resourceName, resourceCriteria " +
302                  "FROM " + roleTable() + ", " + securityTable() + ", permission, resources " + 
303                  "WHERE " +
304                  " (" + securityTable() + ".roleId = " + roleTable() + ".roleId) " +
305                  " AND " + "(" + securityTable() + ".permissionId = permission.permissionId) " +
306                  " AND " + "(permission.resourceId = resources.resourceId)";
307                List<Object> sqlParams = new ArrayList<Object>(2);  
308                List<Map<String, Object>> list = jt.query(loadAllRolePermissionsSql,
309                  sqlParams.toArray(),
310                  new ColumnMapRowMapper());
311                if (logger.isDebugEnabled()) {
312                        logger.debug(Struct.structuredListToString("loadAllRolePermissions", list));
313                }
314                permissionCamelCaser.renameList(list);
315                return convertPermissionList(list, PERMISSION_ROLE);
316        }

466        /** Private method to convert a list of PERMISSION rows (as returned by Spring)
467         *  into Permission Objects.
468         *
469         * <p>On reflection, probably should have used a Spring RowMapper for this. Oh well. It works.
470         *
471         * @param list  The list to convert
472         * @return  A List of Permission objects
473         */
474        private List<Permission> convertPermissionList(List<Map<String, Object>> list, int permissionType)
475        {
476                String resourceType;
477                String expressionString;
478                Permission permission;
479                List<Permission> result = new ArrayList<Permission>(list.size());
480        
481                for (Iterator<Map<String, Object>> i = list.iterator(); i.hasNext(); ) {
482                        Map<String, Object> map = i.next();
483                        resourceType = (String) map.get("resourceName");
484                        expressionString = (String) map.get("resourceCriteria");
485                        ResourceCriteriaImpl resourceCriteriaImpl = null;
486                        if (expressionString != null && !expressionString.equals("")) {
487                                try {
488                                        resourceCriteriaImpl = new ResourceCriteriaImpl(expressionString);
489                                } catch (Exception ce) {
490                                        throw new DataIntegrityViolationException(
491                                                "Invalid criteria found in SECURITY.RESOURCECRITERIA: '" +
492                                                expressionString + "'", ce);
493                                }
494                        }
495                        
496                        if (permissionType == PERMISSION_ROLE) {
497                                permission = new Permission((String)map.get("roleName"),
498                                  (String)map.get("activityName"), resourceType, resourceCriteriaImpl);
499                        } else if (permissionType == PERMISSION_USER) {
500                                User user = new User();
501                                user.setUsername((String) map.get("userId"));
502                                permission = new Permission( user,
503                                  (String)map.get("activityName"), resourceType, resourceCriteriaImpl);
504                        } else if (permissionType == PERMISSION_NONE) {
505                                permission = new Permission((String)map.get("activityName"), resourceType);
506                                
507                        } else {
508                                throw new IllegalArgumentException("Unknown permission type '" + permissionType + "'");
509                        }
510                        result.add(permission);
511                }
512                return result;
513        }
