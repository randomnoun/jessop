013/**
014 * An object that performs time benchmarking.
015 *
016 * <p>This object contains a reference to all benchmark instances
017 * that are currently executing in the VM, referenced by <code>benchId</code>.
018 * It is encouraged that each individual request has a unique benchId, although
019 * this is not required unless you need to use the {@link #getBenchmark} method.
020 *
021 * (if multiple instances are created using the same <code>benchId</code>,
022 * only the most recent instance will be retrievable via getBenchmark()).
023 *
024 * <p>Multiple checkpoints can be set during the course of a benchmark, which
025 * will be written to disk after completion.
026 *
027 * <p>Output is generated only after the benchmark has completed. All output is
028 * buffered, and references to any open benchmark files are kept within this
029 * object in order to minimise the amount of overhead that benchmarking will
030 * impose.
031 *
032 * <p>This class does not expire benchmarks, and therefore may cause memory leaks
033 * if benchmarks are not closed correctly.
034 *
035 * <p>Output is of the form:
036 * <pre style="code">
037 *   benchId1,begin,timestamp,pointID1,timestamp1,pointID1,timestamp2,[...],end,timestampn,duration
038 *   benchId2,begin,timestamp,pointID1,timestamp1,pointID1,timestamp2,[...],end,timestampn,duration
039 *   :
040 *   :
041 * </pre>
042 *
043 * <p>where <i>benchId</i> is the benchId indentifying this benchmark, and
044 * <i>pointIDn</i> are the individual checkpoint identifiers. Each line finishes
045 * with the text ",duration," followed by the duration of the entire benchmark, in
046 * milliseconds. Timestamps are displayed in the format defined by the
047 * {@link #setDateFormat(String)} method.
048 *
049 * <p>There is no current way of following a benchmark through the EJB boundary layer;
050 * to do so would currently involve overriding the User object, which I'm not terribly keen to do.
051 * It is possible to set up a new Benchmark on the EJB side, and track performance
052 * independently over there. It should then be a simple matter to heuristically match up
053 * individual HTTP requests/struts actions and the EJB methods that they invoke
054 * (under normal load conditions). It's important, however, that two Benchmark engines in the
055 * same VM do not write to the same file, as they will corrupt each other's output.
056 *
057 * @author knoxg
058 * 
059 */
060public class Benchmark {
061    /** A string to use in stack traces. */
062    public static String _revision = "$Id$";
063
064
065    /** A Comparator which performs comparisons between two Benchmarks.
066     */
067    public static class BenchmarkComparator implements Comparator<Benchmark> {        
068
069        /** Create a new BenchmarkComparator object */
070        public BenchmarkComparator() {
071        }
072
073        /** Compare two structured list elements
074         *
075         * @see java.util.Comparator#compare(java.lang.Object, java.lang.Object)
076         */
077        public int compare(Benchmark a, Benchmark b) throws IllegalArgumentException {
078            Benchmark benchmarkA = (Benchmark) a;
079            Benchmark benchmarkB = (Benchmark) b;
080
081            return benchmarkA.startTime < benchmarkB.startTime ? -1 : 
082              (benchmarkA.startTime > benchmarkB.startTime ? 1 : 0);
083        }
084    }
085
086    /** A class containing the checkpoints for a single benchmark */
087    public class CheckpointList {
088        /** List of Checkpoints, containing the checkpoints for a single
089         *  benchmark */
090        List<Checkpoint> list = new ArrayList<Checkpoint>();
091
092        /** Return an iterator for the checkpoint list */
093        public Iterator<Checkpoint> iterator() {
094            return list.iterator();
095        }
096
097        /** Retrieve an individual checkpoint */
098        public Checkpoint get(int n) {
099            return (Checkpoint) list.get(n);
100        }
101
102        /** Adds a checkpoint to this set. The checkpoint is stamped using
103         *  the current system time. */
104        public void addCheckPoint(String id) {
105            list.add(new Checkpoint(id));
106        }
107
108        /** Adds an annotation to this set. */
109        public void addAnnotation(String annotation) {
110            list.add(new Checkpoint(annotation, -1));
111        }
112        
113        
114        /** Returns a string version of this set. Timestamps are returned
115         *  using the format specified by the Benchmark class */
116        public String toString() {
117            int i;
118            int size = list.size();
119            StringBuffer sb = new StringBuffer();
120            Checkpoint checkpoint;
121            SimpleDateFormat dateFormat;
122
123            dateFormat = new SimpleDateFormat(Benchmark.dateFormatText);
124
125            for (i = 0; i < size; i++) {
126                checkpoint = (Checkpoint) list.get(i);
127                if (checkpoint.getTimestamp()==-1) {
128                        sb.append(',');
129                        sb.append(Text.escapeCsv("\u0087" + checkpoint.id)); // ASCII 135, Hex 87
130                } else {
131                        sb.append(',');
132                        sb.append(Text.escapeCsv(dateFormat.format(checkpoint.getDate())));
133                        sb.append(',');
134                        sb.append(Text.escapeCsv(checkpoint.id));
135                }
136            }
137
138            if (list.size() > 0) {
139                sb.append(",t,");
140                checkpoint = (Checkpoint) list.get(list.size() - 1);
141                sb.append(HiResTimer.getElapsedMillis(((Checkpoint) list.get(0)).timestamp, checkpoint.timestamp));
142            }
143
144            return sb.toString();
145        }
146    }
147
148    /** A single benchmark point. May be either a begin, end, or checkpoint.
149     * 
150     * A checkpoint with a timestamp of -1 is an annotation.
151     */
152    public class Checkpoint {
153        /** The string ID for this benchmark point */
154        String id;
155
156        /** The time at which this point was created to the benchmark */
157        long timestamp;
158
159        public Checkpoint(String id) {
160            this.id = id;
161            this.timestamp = HiResTimer.getTimestamp();
162        }
163
164        public Checkpoint(String id, long timestamp) {
165            this.id = id;
166            this.timestamp = timestamp;
167        }
168
169        public String getId() {
170            return id;
171        }
172
173        public long getTimestamp() {
174            return timestamp;
175        }
176
177        public Date getDate() {
178            // return date;
179            return new Date(startTime + HiResTimer.getElapsedMillis(startHiresTimestamp, timestamp));
180        }
181    }
182
183    /** Hashtable mapping filenames to java.io.Writer objects */
184    private static Hashtable<String, Writer> writers = null;
185
186    /** Hashtable mapping benchmark IDs to Benchmark objects */
187    private static Hashtable<String, Benchmark> benchmarks = null;
188
189    /** date format for timestamps. The default date format is
190     *  the SimpleDateFormat <tt>"yyyy-dd-MM HH:mm:ss.SSS"</tt>. Note that
191     *  better-than-millisecond precision is unavailable using the
192     *  standard Java-supplied timer functions. */
193    private static String dateFormatText = null;
194
195    /** the set of checkpoints for this benchmark instance */
196    protected CheckpointList checkpointList = null;
197
198    /** the string ID for this benchmark */
199    private String benchId = null;
200
201    /** Start time, in milliseconds */
202    protected long startTime;
203
204    /** HiRes Timer token for start of benchmark */
205    protected long startHiresTimestamp;
206
207    /** True if benchmark is active, false if the benchmark has completed. */
208    protected boolean active;
209
210    /** the file where the results of this benchmark will be written */
211    private String filename = null;
212
213    /** Returns a PrintWriter that can be written to in order to
214     *  commit a benchmark to disk. */
215    private static PrintWriter open(String filename)
216        throws IOException {
217        PrintWriter out;
218
219        // use existing Writer if it is already open
220        if (writers.get(filename) != null) {
221            return (PrintWriter) writers.get(filename);
222        }
223
224        // 'true' flag indicates files will be appended to
225        out = new PrintWriter(new BufferedWriter(new FileWriter(filename, true)));
226        writers.put(filename, out);
227
228        return out;
229    }
230
231    /** Instantiates a new benchmark. Automatically creates a 'begin' checkpoint
232     *  for this new instance. */
233    public Benchmark(String benchId, String filename) {
234        startTime = System.currentTimeMillis();
235        startHiresTimestamp = HiResTimer.getTimestamp();
236
237        checkpointList = new CheckpointList();
238        benchmarks.put(benchId, this);
239        checkpointList.addCheckPoint("begin");
240        this.benchId = benchId;
241        this.filename = filename;
242        this.active = true;
243    }
244
245    /** Alters the ID for a benchmark that is already running. */
246    public void setBenchId(String benchId) {
247        benchmarks.remove(this.benchId);
248        this.benchId = benchId;
249        benchmarks.put(benchId, this);
250    }
251
252    /** Performs checkpoint. */
253    public void checkpoint(String pointID) {
254        checkpointList.addCheckPoint(pointID);
255    }
256    
257    /** Creates a benchmark annotation */
258    public void annotate(String annotation) {
259        checkpointList.addAnnotation(annotation);
260    }
261
262
263    /** Returns a string representation of this benchmark */
264    public String toString() {
265        return benchId + checkpointList.toString();
266    }
267
268    /** Completes benchmark and writes to disk */
269    public void end()
270        throws IOException {
271        PrintWriter printWriter;
272
273        checkpointList.addCheckPoint("end");
274        if (filename != null) {
275            printWriter = (PrintWriter) writers.get(filename);
276            if (printWriter == null) {
277                printWriter = open(filename);
278            }
279            printWriter.println(this.toString());
280            printWriter.flush();
281        }
282        active = false;
283        benchmarks.remove(benchId);
284    }
285
286    /** Returns whether this benchmark is currently active */
287    public boolean isActive() {
288        return active;
289    }
290
291
292    /** Cancel this benchmark */
293    public void cancel() {
294        benchmarks.remove(benchId);
295    }
296
297    /** Returns all checkpoints stored in this benchmark */
298    public CheckpointList getCheckpointList() {
299        return checkpointList;
300    }
301
302    /** Returns the benchmark ID */
303    public String getId() {
304        return benchId;
305    }
306
307    /** Returns the benchmark with the supplied ID */
308    public static Benchmark getBenchmark(String benchId) {
309        return (Benchmark) benchmarks.get(benchId);
310    }
311
312    /** Sets the date format for all output of this class. The date format
313     * is represented as a string, rather than a DateFormat object, since
314     * most SimpleDateFormats are not thread-safe. */
315    public static void setDateFormat(String dateFormatText) {
316        Benchmark.dateFormatText = dateFormatText;
317    }
318
319    /** Flushes all writers within this benchmark object (commits any
320     *  unwritten data to disk). Files are still kept open. */
321    public static void flushWriters()
322        throws IOException {
323        Enumeration<Writer> e;
324
325        for (e = writers.elements(); e.hasMoreElements();) {
326            ((Writer) e.nextElement()).flush();
327        }
328    }
329
330    /** Closes all writers within this benchmark object. Files will be
331     *  automatically reopened for any running benchmarks.  */
332    public static void closeWriters()
333        throws IOException {
334        Enumeration<Writer> e;
335
336        for (e = writers.elements(); e.hasMoreElements();) {
337            ((Writer) e.nextElement()).close();
338        }
339
340        // remove references to old writers
341        writers = new Hashtable<String, Writer>();
342    }
343    
344    public String toJson() {
345        String result = "[";
346                   Iterator<Checkpoint> i = getCheckpointList().iterator();
347                   // Benchmark.Checkpoint cpStart = i.hasNext() ? (Benchmark.Checkpoint) i.next() : null;
348                   while (i.hasNext()) {
349                           Benchmark.Checkpoint cpNext = (Benchmark.Checkpoint) i.next();
350                           result += "{\"id\":\"" + Text.escapeJavascript(cpNext.getId()) + "\",\"timestamp\":" + cpNext.getTimestamp() + "}";
351                           if (i.hasNext()) { result += ","; }
352                   }
353                   result += "]";
354                   return result;
355    }
356    
357
358    static {
359        writers = new Hashtable<String, Writer>();
360        benchmarks = new Hashtable<String, Benchmark>();
361
362        // default date format
363        dateFormatText = "yyyy-MM-dd HH:mm:ss.SSS";
364    }
