007import org.apache.log4j.Logger;
008
009import com.randomnoun.common.timer.HiResTimer;
010
011/**
012 * This is some increasingly ancient code to provide microsecond timer resolution in Java 
013 * using JNI on Windows.
014 * 
015 * <p>Java now provides this natively via System.nanoTime(), so all of this can and should be thrown out
016 * some day. 
017 * 
018 * <p>Historical background: Java previously supplied only millisecond
019 * resolution, and on Windows this was only around 10-20 millisecond precision;
020 * approx 16ms (Win3.1), 55ms (ME/98/95) or 10ms (others)). Other platforms (Linux/Solaris)
021 * appear to provide real millisecond resolution.
022 *
023 * <p>This class uses JNI to access the Windows QueryPerformanceCounter API. It
024 * requires the HiResTimer.dll to be located in the VM's java.library.path. In order to
025 * work in WSAD, it will also examine the contents of the 'workspace.root' System
026 * property, and attempt to reference it from within the developer's build tree.
027 *
028 * <p>If the native library cannot be loaded, this class falls back to Java's
029 * standard timer implementation (System.currentTimeMillis).
030 *
031 * <p>Note that the Websphere reloading ClassLoader cannot deal with JNI,
032 * so this class must exist in an external classpath or JAR which is referenced from
033 * the system classpath. The Ant build task 'buildStartupJar' in ext/build.xml
034 * can compile the JNI and create this startup JAR.
035 *
036 * <hr/>
037 *
038 * <p>This file was based on the discussion at
039 * <br/><a href="http://www.fawcette.com/archives/premier/mgznarch/javapro/2001/08aug01/km0108/km0108-1.asp">
040 * http://www.fawcette.com/archives/premier/mgznarch/javapro/2001/08aug01/km0108/km0108-1.asp</a>
041 *
042 * <p>Some instructions on building JNI in a cygwin/gcc environment was found at
043 * <br/><a href="http://www.inonit.com/cygwin/jni/helloWorld/">
044 * http://www.inonit.com/cygwin/jni/helloWorld/</a>
045 * 
046 * <p>Documentation on Windows APIs can be found in Windows 2003 Core API download, found at
047 * <br/><a href="http://www.microsoft.com/msdownload/platformsdk/sdkupdate/">
048 * http://www.microsoft.com/msdownload/platformsdk/sdkupdate/</a>
049 *
050 * @author  knoxg
051 * 
052 */
053public class HiResTimer
054{
055    
056    
057 
058    /** Logger instance for this class */
059    private static Logger logger = Logger.getLogger(HiResTimer.class);
060    
061    /** Set to true if the HiResTimer DLL can be found in the java library path */
062    private static boolean foundLibrary = false;
063
064    /** A HiResTimer instance; used since I want to be able to provide these methods
065     *  as static, which I found difficult in JNI directly. 
066     */
067    private static HiResTimer instance = new HiResTimer();
068
069
070    private native boolean native_isHighResTimerAvailable();
071
072    /**
073     * JNI method to return the frequency of the system clock (ticks per second)
074     *
075     * @return the frequency of the system clock (ticks per second)
076     */
077    private native long native_getFrequency();
078
079    /**
080     * JNI method to return the current timestamp (in frequency units)
081     *
082     * @return the current timestamp (in frequency units)
083     */
084    private native long native_getTimestamp();
085
086    /**
087     * Return the frequency of the available timer.
088     *
089     * @return The granularity of the timer, measured in ticks per second.
090     */
091    public static long getFrequency()
092    {
093        if (foundLibrary) {
094            return instance.native_getFrequency();
095        } else {
096            // System ostensibly supplies us with 1ms precision (although it doesn't, really)
097            return 1000;
098        }
099        
100    }
101
102    /**
103     * Returns the current value of the timer, as measured in the frequency supplied
104     * by {@link #getFrequency()}. The starting time for returned values is undefined
105     * (i.e. there is no fixed point for timestamp = 0); this value can only be used
106     * for relative timing only, or must be correlated with System.currentTimeMillis().
107     *
108     * @return The current value of the timer.
109     */
110    public static long getTimestamp()
111    {
112        if (foundLibrary) {
113            return instance.native_getTimestamp();
114        } else {
115            return System.currentTimeMillis();
116        }
117    }
118
119    /** Returns true if we can use a natively-supplied timer, false otherwise
120     *
121     * @return true if we can use a natively-supplied timer, false otherwise
122     */
123    public static boolean isNativeTimerAvailable()
124    {
125        return foundLibrary;
126    }
127
128    /**
129     * Returns true if the Windows HAL (Hardware Abstraction Layer) supports
130     * a high-resolution performance counter. Note that even if this returns false,
131     * we can fall back to a native counter, which is still more precise than
132     * the one that Java supplies.
133     *
134     * @return True if high resolution performance is available, false if not.
135     */
136    public static boolean isHiResTimerAvailable()
137    {
138        // NB: short-cut boolean logic
139        return foundLibrary && instance.native_isHighResTimerAvailable();
140    }
141
142    /** Returns the elapsed time between two timestamps, as a integer number of
143     *  milliseconds.
144     *
145     * @return the elapsed time between two timestamps, as a integer number of
146     *  milliseconds.
147     */
148    public static long getElapsedMillis(long timestamp1, long timestamp2)
149    {
150        return ((timestamp2 - timestamp1) * 1000) / getFrequency();
151    }
152
153    /** Returns the elapsed time between two timestamps, as a integer number of
154     *  microseconds.
155     *
156     * @return the elapsed time between two timestamps, as a integer number of
157     *  microseconds.
158     */
159    public static long getElapsedNanos(long timestamp1, long timestamp2)
160    {
161        return ((timestamp2 - timestamp1) * 1000000) / getFrequency();
162    }
163
164    /**
165     * Test program.
166     *
167     * @param args Command-line options
168     *
169     * @throws InterruptedException if the Thread.sleep() call was interrupted
170     */
171    public static void main(String[] args)
172        throws InterruptedException
173    {
174        System.out.println("Native timer: " +
175            (HiResTimer.isNativeTimerAvailable() ? "available" : "unavailable"));
176        System.out.println("Hi-res timer: " +
177            (HiResTimer.isHiResTimerAvailable() ? "available" : "unavailable"));
178
179        java.text.NumberFormat nf = new java.text.DecimalFormat("0.#########");
180
181        long freq = HiResTimer.getFrequency();
182
183        System.out.println("Timer has frequency of approx " + freq + " ticks/sec");
184
185        // this isn't really the granularity, you know. It's just the *maximum* precision 
186        // we can ever expect to get out of the timer. Whether the timer actually reaches
187        // this theoretical maximum is a completely different matter.
188        System.out.println("Timer has granularity of approx " +
189            nf.format(1 / (double)freq) + " seconds.");
190
191        long dStart = HiResTimer.getTimestamp();
192
193        Thread.sleep(1000);
194
195        long dEnd = HiResTimer.getTimestamp();
196
197        System.out.println("Thread.sleep() test=" +
198            HiResTimer.getElapsedMillis(dStart, dEnd) +
199            " milliseconds (should be around 1000); = " +
200            HiResTimer.getElapsedNanos(dStart, dEnd) + " microseconds");
201
202        dStart = HiResTimer.getTimestamp();
203        Thread.sleep(2000);
204        dEnd = HiResTimer.getTimestamp();
205        System.out.println("Thread.sleep() test=" +
206            HiResTimer.getElapsedMillis(dStart, dEnd) +
207            " milliseconds (should be around 2000); = " +
208            HiResTimer.getElapsedNanos(dStart, dEnd) + " microseconds");
209    }
210
211    static
212    {
213        try
214        {
215            // use the java.library.path variable first
216            System.loadLibrary("HiResTimer");
217            foundLibrary = true;
218        }
219        catch (UnsatisfiedLinkError ule)
220        {
221            // exception intentionally ignored
222        }
223
224        if (!foundLibrary)
225        {
226            // when run in WSAD, workspace.root is something like
227            //   C:/Documents and Settings/knoxg/My Documents/IBM/wsappdev51/workspace/Servers/QueryBuilderSvr.wsc
228            // ... so we can try looking in here
229            String workspaceRoot = System.getProperty("workspace.root");
230            
231            // NB, in eclipse wtp.deploy is set to something like
232            //   C:\Documents and Settings\knoxg\workspace-3.5sr2\.metadata\.plugins\org.eclipse.wst.server.core\tmp0\wtpwebapps
233            // so could try something relative to that as well
234
235            if (workspaceRoot != null)
236            {
237                String location = workspaceRoot +
238                    "/../../common/ext/bootstrap/HiResTimer.dll";
239
240                try {
241                    System.load(location);
242                    foundLibrary = true;
243                } catch (UnsatisfiedLinkError ule) {
244                    logger.debug(
245                        "Could not load HiResTimer from java.library.path or '" +
246                        location + "'");
247
248                    // ule.printStackTrace();
249                    // NB: we still can't use a DLL even if it's loaded in another ClassLoader
250                    // exception intentionally ignored
251                }
252            }
253        }
254
255        if (foundLibrary) {
256            HiResTimer timer = new HiResTimer();
257            logger.info("Native timer library available: high resolution timer " +
258                (timer.native_isHighResTimerAvailable() ? "available" : "unavailable"));
259        } else {
260                logger.info("Native timer library unavailable");
261        }
262    }
263}
