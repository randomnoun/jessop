118                ch = text.charAt(i);

128                // leading negative symbol OK
129            } else if (ch < '0' || ch > '9') {
130                return false;
131            }
132            }
133            return true;

137    /** Ensures that a string returned from a browser (on any platform) conforms
138     * to unix line-EOF conventions. Any instances of consecutive CRs (<tt>0xD</tt>) 
139     * and LFs (<tt>0xA</tt>) in a string will be reduced to a series of CRs (the number of CRs will be the
140     * maximum number of CRs or LFs found in a row).  
141     * 
142     * @param input the input string
143     * 
144     * @return the canonicalised string, as described above
145     */
146    public static String reduceNewlines(String input) {
147        StringBuilder sb = new StringBuilder();
148        int len = input.length();
149        int crCount = 0;
150                int lfCount = 0;
151                boolean insertNewline = false;
152                char ch;
153        for (int i=0; i<len; i++) {
154                ch = input.charAt(i);
155                if (ch == (char) 0xA) {
156                        lfCount ++; insertNewline = true;
157                } else if (ch == (char) 0xD) {
158                        crCount ++; insertNewline = true;
159                } else if (insertNewline) {
160                                for (int j=0; j<Math.max(lfCount, crCount); j++) {
161                                        sb.append((char) 0xA);
162                                }
163                                insertNewline = false; lfCount=0; crCount=0;
164                                sb.append(ch);
165                } else {
166                        sb.append(ch);
167                }
168        }
169        if (insertNewline) {
170                        for (int j=0; j<Math.max(lfCount, crCount); j++) {
171                                sb.append((char) 0xA);
172                        }
173        }
174        
175        return sb.toString();
176    }

180     * Returns the HTML-escaped form of a string. The <tt>&amp;</tt>,
181     * <tt>&lt;</tt>, <tt>&gt;</tt>, and <tt>"</tt> characters are converted to
182     * <tt>&amp;amp;</tt>, <tt>&amp;lt;<tt>, <tt>&amp;gt;<tt>, and
183     * <tt>&amp;quot;</tt> respectively.
184     * 
185     * <p>Characters in the unicode control code blocks ( apart from \t, \n and \r ) are converted to &xfffd;
186     * <p>Characters outside of the ASCII printable range are converted into &xnnnn; form
187     *
188     * @param string the string to convert
189     *
190     * @return the HTML-escaped form of the string
191     */
192    static public String escapeHtml(String string) {
193        if (string == null) {
194            return "";
195        }
196        char c;
197        String hex;
198        StringBuilder sb = new StringBuilder(string.length());
199        for (int i = 0; i < string.length(); i++) {
200            c = string.charAt(i);
201            // check for illegal characters
202            switch (c) {
203                case '&':
204                    sb.append("&amp;");
205                    break;
206                case '<':
207                    sb.append("&lt;");
208                    break;
209                case '>':
210                    sb.append("&gt;");
211                    break;
212                case '\"':
213                    // interestingly, &quote; (with the e) works fine for HTML display,
214                    // but not inside hidden field values
215                    sb.append("&quot;");
216                    break;
217                default:
218                        // 'illegal characters' according to ESAPI. 7f to 9f are control characters in unicode 
219                        if ( ( c <= 0x1f && c != '\t' && c != '\n' && c != '\r' ) || ( c >= 0x7f && c <= 0x9f ) ) {
220                                sb.append("&#xfffd;"); // REPLACEMENT_HEX in ESAPI's HtmlEntityCodec
221                        } else if ( c > 0x1f && c <= 0x7f ) {
222                                // safe printable
223                                sb.append(c);
224                        } else {
225                                // ESAPI didn't have the else block above, which was causing it escape everything 
226                                hex = getHexForNonAlphanumeric(c);
227                                sb.append("&#x" + hex + ";");
228                        }
229                        
230            }
231        }
232
233        return sb.toString();
234    }

300    /**
301     * Returns the csv-escaped form of a string. A csv-escaped string is
302     * used when writing to a CSV (comma-separated-value) file. It ensures
303     * that commas included within a string are quoted. We use the Microsoft-Excel
304     * quoting rules, so that our CSV files can be imported into that. These rules
305     * (derived from experimentation) are:
306     *
307     * <ul>
308     * <li>Strings without commas (,) inverted commas ("), or newlines (\n) are returned as-is.
309     * <li>Otherwise, the string is surrounded by inverted commas, and any
310     *   inverted commas within the string are doubled-up (i.e. '"' becomes '""').
311     * <li>A value that starts with any of "=", "@", "+" or "-" has a leading single apostrophe added
312     *   to prevent the value being evaluated in Excel. The leading quote is visible to the user when the
313     *   csv is opened, which may mean that it will have to be removed when roundtripping data.
314     *   This may complicate things if the user actually wants a leading single quote in their CSV value.   
315     * </ul>
316     *
317     * <p>Embedded newlines are inserted as-is, as per Excel. This will require
318     * some care whilst parsing if we want to be able to read these files.
319     *
320     * @param string the string to convert
321     *
322     * @return the csv-escaped form of the string
323     */
324    static public String escapeCsv(String string) {
325        if (string == null) {
326            return "";
327        }
328
329        boolean quoted = false;
330        // from https://www.contextis.com/en/blog/comma-separated-vulnerabilities
331        // prefix cells that start with ‘=’ , '@', '+' or '-' with an apostrophe 
332        // This will ensure that the cell isn’t interpreted as a formula, and as a bonus in Microsoft Excel the apostrophe itself will not be displayed.
333        if (string.startsWith("=") || 
334          string.startsWith("@") || 
335          string.indexOf('%')!=-1) {
336            // prefix the string with an a single quote char to escape it
337            string = "'" + string;
338            quoted = true; // not sure need to quote here, but doesn't hurt
339        } else if ((string.startsWith("+") || string.startsWith("-")) && 
340                (string.length() == 1 || !Text.isNumericDecimalExp(string))) {
341                // numbers can legitimately start with '+' or '-' but anything else should be escaped
342                string = "'" + string;
343            quoted = true; 
344        }
345
346        
347        if (string.indexOf(',') == -1 && string.indexOf('"') == -1 && string.indexOf('\n') == -1 && !quoted) {
348                return string;
349        }
350        string = Text.replaceString(string, "\"", "\"\"");
351        string = "\"" + string + "\"";
352
353        return string;
354    }

356    /** Given a csv-encoded string (as produced by the rules in {@see #escapeCsv},
357     *  produces a List of Strings which represent the individual values in the string.
358     *  Note that this method is *not* equivalent to calling <tt>Arrays.asList(astring.split(","))</tt>.
359     *
360     * <p>Setting the whitespaceSensitive parameter to false allows leading and trailing
361     * whitespace in *non-quoted* values to be removed, e.g. if the input string <tt>text</tt> is:
362     *
363     * <pre class="code">
364     * abc,def,  ghi, j k ,"lmn"," op "," q,r","""hello""", "another"
365     * </pre>
366     *
367     * then <tt>parseCsv(text, <b>false</b>)</tt> will return the strings:
368     * <pre class="code">
369     * abc
370     * def
371     * ghi
372     * j k
373     * lmn
374     *  op        <i>(this String has one leading space, and a trailing space after 'p')</i>
375     *  q,r       <i>(this String has one leading space)</i>
376     * "hello"
377     * another
378     * </pre>
379     *
380     * and <tt>parseCsv(text, <b>true</b>)</tt> would throw a ParseException (since the
381     * final element is a quoted value, but begins with a space).
382     *
383     * If the <tt>, "another"</tt> text is removed, however, then
384     * <tt>parseCsv(text, true)</tt> would return the following:
385     *
386     * and <tt>parseCsv(text, true)</tt> will return the string
387     * <pre>
388     * abc
389     * def
390     *   ghi      <i>(this String has two leading spaces)</i>
391     *  j k       <i>(this String has one leading space and a trailing space after the 'k' character)</i>
392     * lmn
393     *  op        <i>(this String has one leading space, and a trailing space after 'p')</i>
394     *  q,r       <i>(this String has one leading space)</i>
395     * "hello"
396     * </pre>
397     *
398     * <p>Most applications would want to use the 'whiteSpaceSensitive=false' form of this function, since
399     * (a) less chance of a ParseException, and (b) it's what an end-user would normally
400     * expect. This can be performed by calling the {@link #parseCsv(String)} method.
401     *
402     * <p>Whitespace is determined by using the <tt>Character.isSpaceChar()</tt> method,
403     * which is Unicode-aware.
404     *
405     * @param text   The CSV-encoded string to parse
406     * @param trim   If set to true, will trim leading and trailing whitespace in *non-quoted* values.
407     *
408     * @return a List of Strings. The returned List is guaranteed to always contain at least one element.
409     *
410     * @throws NullPointerException if the text passed to this method is null
411     * @throws ParseException if a quoted value contains leading whitespace before the
412     *  opening quote, or after the trailing quote.
413     * @throws ParseException if a quoted value has a start quote, but no end quote, or
414     *   if a value has additional text after a quoted value (before the next comma or EOL).
415     */
416    static public List<String> parseCsv(String text, boolean whitespaceSensitive)
417        throws ParseException {
418        if (text == null) {
419            throw new NullPointerException("null text");
420        }
421
422        // parse state: 
423        //   0=searching for new value (at start of line or after comma) 
424        //   1=consuming non-quoted values
425        //   2=consuming quoted value
426        //   3=consumed first quote within a quoted value (may be termining quote or a "" sequence)
427        //   4=consuming whitespace up to next comma/EOL (after quoted value, not whitespaceSensitive)
428        int parseState = 0;
429        int length = text.length();
430        String element;
431        List<String> result = new ArrayList<String>();
432        char ch;
433        StringBuilder buffer = new StringBuilder();
434
435        for (int pos = 0; pos < length; pos++) {
436            ch = text.charAt(pos);
437
438            // System.out.println("pos " + pos + ", state=" + parseState + ", nextchar=" + ch + ", buf=" + buffer);
439            switch (parseState) {
440                case 0:
441                    if (Character.isSpaceChar(ch)) {
442                        if (whitespaceSensitive) {
443                            buffer.append(ch);
444                            parseState = 1;
445                        } else {
446                            // ignore
447                        }
448                    } else if (ch == '"') {
449                        parseState = 2;
450                    } else if (ch == ',') {
451                        result.add(""); // add an empty element; state remains unchanged
452                    } else {
453                        buffer.append(ch);
454                        parseState = 1;
455                    }
456                    break;
457                case 1:
458                    if (ch == ',') {
459                        element = buffer.toString();
460                        if (!whitespaceSensitive) {
461                            element = element.trim();
462                        }
463                        result.add(element);
464                        buffer.setLength(0);
465                        parseState = 0;
466                    } else {
467                        buffer.append(ch);
468                    }
469                    break;
470                case 2:
471                    if (ch == '"') {
472                        parseState = 3;
473                    } else {
474                        buffer.append(ch);
475                    }
476                    break;
477                case 3:
478                    if (ch == '"') {
479                        buffer.append('"');
480                        parseState = 2;
481                    } else if (ch == ',') {
482                        result.add(buffer.toString());
483                        buffer.setLength(0);
484                        parseState = 0;
485                    } else if (Character.isSpaceChar(ch)) {
486                        if (whitespaceSensitive) {
487                            throw new ParseException("Cannot have trailing whitespace after close quote character", pos);
488                        }
489                        parseState = 4;
490                    } else {
491                        throw new ParseException("Cannot have trailing data after close quote character", pos);
492                    }
493                    break;
494                case 4:
495                    if (Character.isSpaceChar(ch)) {
496                        // consume and ignore
497                    } else if (ch == ',') {
498                        result.add(buffer.toString());
499                        buffer.setLength(0);
500                        parseState = 0;
501                    } else {
502                        throw new ParseException("Cannot have trailing data after close quote character", pos);
503                    }
504                    break;
505                    
506                default:
507                    throw new IllegalStateException("Illegal state '" + parseState + "' in parseCsv");
508            }
509        }
510
511        // if state is 2, we are in the middle of a quoted value
512        if (parseState == 2) {
513            throw new ParseException("Missing endquote in csv text", length);
514        }
515
516        // otherwise we still need to add what's left in the buffer into the result list
517        element = buffer.toString();
518        if (parseState == 1 && !whitespaceSensitive) {
519            element = element.trim();
520        }
521        result.add(element);
522        return result;
523    }

1444    /** Natural compare operation. Stolen from 
1445     * http://www.eekboom.com/java/compareNatural/src/com/eekboom/utils/Strings.java
1446     * (source file is under BSD license). 
1447     * 
1448     * @param s             first string
1449     * @param t             second string
1450     * @param caseSensitive treat characters differing in case only as equal - will be ignored if a collator is given
1451     * @param collator      used to compare subwords that aren't numbers - if null, characters will be compared
1452     *                      individually based on their Unicode value
1453     * @return zero iff <code>s</code> and <code>t</code> are equal,
1454     *         a value less than zero iff <code>s</code> lexicographically precedes <code>t</code>
1455     *         and a value larger than zero iff <code>s</code> lexicographically follows <code>t</code>
1456     */
1457    private static int compareNatural(String s, String t, boolean caseSensitive, Collator collator) {
1458        int sIndex = 0;
1459        int tIndex = 0;
1460
1461        int sLength = s.length();
1462        int tLength = t.length();
1463
1464        while(true) {
1465            // both character indices are after a subword (or at zero)
1466
1467            // Check if one string is at end
1468            if(sIndex == sLength && tIndex == tLength) {
1469                return 0;
1470            }
1471            if(sIndex == sLength) {
1472                return -1;
1473            }
1474            if(tIndex == tLength) {
1475                return 1;
1476            }
1477
1478            // Compare sub word
1479            char sChar = s.charAt(sIndex);
1480            char tChar = t.charAt(tIndex);
1481
1482            boolean sCharIsDigit = Character.isDigit(sChar);
1483            boolean tCharIsDigit = Character.isDigit(tChar);
1484
1485            if(sCharIsDigit && tCharIsDigit) {
1486                // Compare numbers
1487
1488                // skip leading 0s
1489                int sLeadingZeroCount = 0;
1490                while(sChar == '0') {
1491                    ++sLeadingZeroCount;
1492                    ++sIndex;
1493                    if(sIndex == sLength) {
1494                        break;
1495                    }
1496                    sChar = s.charAt(sIndex);
1497                }
1498                int tLeadingZeroCount = 0;
1499                while(tChar == '0') {
1500                    ++tLeadingZeroCount;
1501                    ++tIndex;
1502                    if(tIndex == tLength) {
1503                        break;
1504                    }
1505                    tChar = t.charAt(tIndex);
1506                }
1507                boolean sAllZero = sIndex == sLength || !Character.isDigit(sChar);
1508                boolean tAllZero = tIndex == tLength || !Character.isDigit(tChar);
1509                if(sAllZero && tAllZero) {
1510                    continue;
1511                }
1512                if(sAllZero && !tAllZero) {
1513                    return -1;
1514                }
1515                if(tAllZero) {
1516                    return 1;
1517                }
1518
1519                int diff = 0;
1520                do {
1521                    if(diff == 0) {
1522                        diff = sChar - tChar;
1523                    }
1524                    ++sIndex;
1525                    ++tIndex;
1526                    if(sIndex == sLength && tIndex == tLength) {
1527                        return diff != 0 ? diff : sLeadingZeroCount - tLeadingZeroCount;
1528                    }
1529                    if(sIndex == sLength) {
1530                        if(diff == 0) {
1531                            return -1;
1532                        }
1533                        return Character.isDigit(t.charAt(tIndex)) ? -1 : diff;
1534                    }
1535                    if(tIndex == tLength) {
1536                        if(diff == 0) {
1537                            return 1;
1538                        }
1539                        return Character.isDigit(s.charAt(sIndex)) ? 1 : diff;
1540                    }
1541                    sChar = s.charAt(sIndex);
1542                    tChar = t.charAt(tIndex);
1543                    sCharIsDigit = Character.isDigit(sChar);
1544                    tCharIsDigit = Character.isDigit(tChar);
1545                    if(!sCharIsDigit && !tCharIsDigit) {
1546                        // both number sub words have the same length
1547                        if(diff != 0) {
1548                            return diff;
1549                        }
1550                        break;
1551                    }
1552                    if(!sCharIsDigit) {
1553                        return -1;
1554                    }
1555                    if(!tCharIsDigit) {
1556                        return 1;
1557                    }
1558                } while(true);
1559            }
1560            else {
1561                // Compare words
1562                if(collator != null) {
1563                    // To use the collator the whole subwords have to be compared - character-by-character comparision
1564                    // is not possible. So find the two subwords first
1565                    int aw = sIndex;
1566                    int bw = tIndex;
1567                    do {
1568                        ++sIndex;
1569                    } while(sIndex < sLength && !Character.isDigit(s.charAt(sIndex)));
1570                    do {
1571                        ++tIndex;
1572                    } while(tIndex < tLength && !Character.isDigit(t.charAt(tIndex)));
1573
1574                    String as = s.substring(aw, sIndex);
1575                    String bs = t.substring(bw, tIndex);
1576                    int subwordResult = collator.compare(as, bs);
1577                    if(subwordResult != 0) {
1578                        return subwordResult;
1579                    }
1580                }
1581                else {
1582                    // No collator specified. All characters should be ascii only. Compare character-by-character.
1583                    do {
1584                        if(sChar != tChar) {
1585                            if(caseSensitive) {
1586                                return sChar - tChar;
1587                            }
1588                            sChar = Character.toUpperCase(sChar);
1589                            tChar = Character.toUpperCase(tChar);
1590                            if(sChar != tChar) {
1591                                sChar = Character.toLowerCase(sChar);
1592                                tChar = Character.toLowerCase(tChar);
1593                                if(sChar != tChar) {
1594                                    return sChar - tChar;
1595                                }
1596                            }
1597                        }
1598                        ++sIndex;
1599                        ++tIndex;
1600                        if(sIndex == sLength && tIndex == tLength) {
1601                            return 0;
1602                        }
1603                        if(sIndex == sLength) {
1604                            return -1;
1605                        }
1606                        if(tIndex == tLength) {
1607                            return 1;
1608                        }
1609                        sChar = s.charAt(sIndex);
1610                        tChar = t.charAt(tIndex);
1611                        sCharIsDigit = Character.isDigit(sChar);
1612                        tCharIsDigit = Character.isDigit(tChar);
1613                    } while(!sCharIsDigit && !tCharIsDigit);
1614                }
1615            }
1616        }
1617    }

1705    /** Number of character edits between two strings; taken from  
1706         * http://www.merriampark.com/ldjava.htm. There's a version in commongs-lang,
1707         * apparently, but according to the comments on that page, it uses O(n^2) memory,
1708         * which can't be good.
1709         * 
1710         * @param s string 1
1711         * @param t string 2
1712         *  
1713         * @return the smallest number of edits required to convert s into t 
1714         */
1715        public static int getLevenshteinDistance (String s, String t) {
1716                  if (s == null || t == null) {
1717                    throw new IllegalArgumentException("Strings must not be null");
1718                  }
1719                                
1720                  /*
1721                    The difference between this impl. and the previous is that, rather 
1722                     than creating and retaining a matrix of size s.length()+1 by t.length()+1, 
1723                     we maintain two single-dimensional arrays of length s.length()+1.  The first, d,
1724                     is the 'current working' distance array that maintains the newest distance cost
1725                     counts as we iterate through the characters of String s.  Each time we increment
1726                     the index of String t we are comparing, d is copied to p, the second int[].  Doing so
1727                     allows us to retain the previous cost counts as required by the algorithm (taking 
1728                     the minimum of the cost count to the left, up one, and diagonally up and to the left
1729                     of the current cost count being calculated).  (Note that the arrays aren't really 
1730                     copied anymore, just switched...this is clearly much better than cloning an array 
1731                     or doing a System.arraycopy() each time  through the outer loop.)
1732
1733                     Effectively, the difference between the two implementations is this one does not 
1734                     cause an out of memory condition when calculating the LD over two very large strings.              
1735                  */            
1736                                
1737                  int n = s.length(); // length of s
1738                  int m = t.length(); // length of t
1739                                
1740                  if (n == 0) {
1741                    return m;
1742                  } else if (m == 0) {
1743                    return n;
1744                  }
1745
1746                  int p[] = new int[n+1]; //'previous' cost array, horizontally
1747                  int d[] = new int[n+1]; // cost array, horizontally
1748                  int _d[]; //placeholder to assist in swapping p and d
1749
1750                  // indexes into strings s and t
1751                  int i; // iterates through s
1752                  int j; // iterates through t
1753
1754                  char t_j; // jth character of t
1755
1756                  int cost; // cost
1757
1758                  for (i = 0; i<=n; i++) {
1759                     p[i] = i;
1760                  }
1761                                
1762                  for (j = 1; j<=m; j++) {
1763                     t_j = t.charAt(j-1);
1764                     d[0] = j;
1765                                
1766                     for (i=1; i<=n; i++) {
1767                        cost = s.charAt(i-1)==t_j ? 0 : 1;
1768                        // minimum of cell to the left+1, to the top+1, diagonally left and up +cost                            
1769                        d[i] = Math.min(Math.min(d[i-1]+1, p[i]+1),  p[i-1]+cost);  
1770                     }
1771
1772                     // copy current distance counts to 'previous row' distance counts
1773                     _d = p;
1774                     p = d;
1775                     d = _d;
1776                  } 
1777                                
1778                  // our last action in the above loop was to switch d and p, so p now 
1779                  // actually has the most recent cost counts
1780                  return p[n];
1781        }

1830         * <attributes>

1835         * will be "xxxxdefyyyy${def}zzzz"
1836         * 
1837         * <p><tt>$</tt> followed by any other character will be left as-is. 
1838         * 
1839         * @param variables a set of variable names and values, used in the substitution 
1840         * @param text the text to be substituted.
1841         * 
1842         * @return text, with placeholders replaced with values in the variables parameter
1843         */
1844        public static String substitutePlaceholders(Map variables, String text) {
1845                // escaped version of (\$\{.*?\}|[^$]+|\$.)
1846                Pattern p = Pattern.compile("(\\$\\{.*?\\}|[^$]+|\\$)"); // modified regex
1847                Matcher m = p.matcher(text);
1848                String result = "";
1849                while (m.find()) {
1850                        String token = m.group(1);
1851                        if (token.startsWith("${") && token.endsWith("}")) {
1852                                Object value = variables.get(token.substring(2, token.length()-1));
1853                                if (value == null) {
1854                                        result = result + token;
1855                                } else {
1856                                        result = result + value.toString();
1857                                }
1858                        } else {
1859                                result = result + token;
1860                        }
1861                }
1862                return result;
1863        }
