099    /** Ensure that the database drivers are accessible, for simple and dbcp connection types 
100     * 
101     * @throws RuntimeExcption if the database drivers are not on the classpath of this webapp
102     */
103    protected void initDatabase() {
104        logger.info("Initialising database...");
105        try {
106                String connectionType = getProperty("database.connectionType");
107                if (connectionType==null || 
108                  connectionType.equals("simple") || 
109                  connectionType.equals("dbcp")) { 
110                        Class.forName(getProperty("database.driver")).newInstance();
111                }
112        } catch (Exception e) {
113            initialisationFailure = e;
114            throw new RuntimeException("Could not load database driver '" + getProperty("database.driver") + "'", e);
115        }
116    }

118    /** Initialises log4j 
119     */
120    protected void initLogger() {
121        System.out.println("Initialising log4j...");
122        
123        Properties props = new Properties();
124        props.putAll(PropertyParser.restrict(this, "log4j", false));
125    
126        // replace ${xxxx}-style placeholders
127        String key, value, varName, varValue;
128        int startIdx, endIdx;
129        for (Enumeration<Object> e = props.keys(); e.hasMoreElements(); ) {
130                key = (String) e.nextElement();
131                value = (String) props.get(key);
132                startIdx = 0;
133                startIdx = value.indexOf("${", startIdx);
134                while (startIdx > -1) {
135                        endIdx = value.indexOf("}", startIdx+2);
136                        if (endIdx>-1) {
137                                varName = value.substring(startIdx+2, endIdx-startIdx-2);
138                                varValue = (String) this.get(varName);
139                                if (varValue == null) { varValue = ""; }
140                                value = value.substring(0, startIdx) + varValue + value.substring(endIdx + 1);
141                                startIdx = value.indexOf("${", startIdx); 
142                        } else {
143                                startIdx = -1;
144                        }
145                }
146        }
147        
148        PropertyConfigurator.configure(props);
149        logger.debug("log4j initialised");
150    }

152    protected void initSecurityContext() {
153        
154        if (securityContext==null) {
155                        logger.info("Initialising security context...");
156                        if (!"false".equals(getProperty("auth.enableSecurityContext"))) {
157                                Map<String, Object> securityProperties  = new HashMap<String, Object>();
158                                // no customerIds, applicationIds, auditUsernames or table suffixes ! Yay !
159                                String dbVendor = (String) this.get("database.vendor");
160                                if (Text.isBlank(dbVendor)) { dbVendor = SqlGenerator.DATABASE_MYSQL; }
161                                
162                                securityProperties.put(SpringSecurityLoaderImpl.INIT_DATABASE_VENDOR, dbVendor);
163                                securityProperties.put(SpringSecurityLoaderImpl.INIT_JDBCTEMPLATE, getJdbcTemplate());
164                                
165                                securityProperties.put(SecurityContext.INIT_CASE_INSENSITIVE, this.get("securityContext.caseInsensitive"));
166                                securityProperties.put(SecurityContext.INIT_USER_CACHE_SIZE, this.get("securityContext.userCacheSize"));
167                                securityProperties.put(SecurityContext.INIT_USER_CACHE_EXPIRY, this.get("securityContext.userCacheExpiry"));
168                                
169                                SecurityLoader securityLoader = new SpringSecurityLoaderImpl();
170                                SecurityAuthenticator securityAuthenticator = new SpringSecurityAuthenticatorImpl();
171                                
172                                securityContext = new SecurityContext(securityProperties, securityLoader, securityAuthenticator);
173                        } else {
174                                Map<String, Object> securityProperties  = new HashMap<String, Object>();
175                                SecurityLoader securityLoader = new NullSecurityLoaderImpl();
176                                SecurityAuthenticator securityAuthenticator = new NullSecurityAuthenticatorImpl();
177                                securityContext = new SecurityContext(securityProperties, securityLoader, securityAuthenticator);
178                        }
179        }
180    }

182    /** Sets the hostname application property. Should be called before {@link #loadProperties()},
183     * which may have ENVIRONMENT settings that rely on the hostname.
184     */
185    protected void initHostname() {
186        String hostname;
187        try {
188            java.net.InetAddress localMachine = java.net.InetAddress.getLocalHost();
189            hostname = localMachine.getHostName();
190        } catch(java.net.UnknownHostException uhe) {
191            hostname = "localhost";
192        }
193        this.put("hostname", hostname);
194    }

196    /** Returns a bundle to be used for internationalisation. The bundle will initially be
197     * searched for using the standard ResourceBundle.getBundle(name) method (i.e. in the
198     * server's classpath); and, if this fails, will return 
199     * ResourceBundle("resources.i18n." + name); which will search the i18n tree of the
200     * eomail project. 
201     * 
202     * @param i18nBundleName The bundle name to search for (e.g. "user")
203     * @param user The user for whom we are retrieving i18ned messages for (the locale of this
204     *   user will be used to determine which resource bundle to use)
205     * @return The ResourceBundle requested.
206     */    
207    public ResourceBundle getBundle(String i18nBundleName, User user) {
208        if (i18nBundleName.startsWith("resources")) {
209            return ResourceBundle.getBundle(i18nBundleName, user.getLocale());    
210        } else {
211            return ResourceBundle.getBundle("resources.i18n." + i18nBundleName, user.getLocale());
212        }

217    /** Load the configuration from both the classpath and the
218     * filesystem. Properties defined on the filesystem will override any contained within
219     * the web application. 
220     * 
221     * <p>Multiple environments can be specified in properties files using the 
222     * STARTENVIRONMENT/ENDENVIRONMENT tags. The tags use the machine's hostname as an
223     * environment specifier. 
224     */
225    protected void loadProperties() {
226        String configPath = System.getProperty(getSystemPropertyKeyConfigPath());
227        if (configPath==null) { configPath = "."; }
228        File configFile = new File(configPath + getConfigResourceLocation());
229        try {
230            // load from resources
231            System.out.println("Loading properties file for environment '" + this.getProperty("hostname") + "'");
232            InputStream is = this.getClass().getClassLoader().getResourceAsStream(getConfigResourceLocation());
233            if (is==null && getConfigResourceLocation().charAt(0)=='/') {
234                // command-line support
235                is = this.getClass().getClassLoader().getResourceAsStream(getConfigResourceLocation().substring(1));
236            }
237            if (is==null) {
238                System.out.println("Could not find internal property file");
239            } else {
240                PropertyParser parser = new PropertyParser(new InputStreamReader(is), this.getProperty("hostname"));
241                Properties props = parser.parse();
242                is.close();
243                this.putAll(props);
244            }
245            
246            // then load from file (if it exists)
247            if (configFile.exists()) {
248                System.out.println("Loading properties file from '" + configFile.getCanonicalPath() + "'");
249                is = new BufferedInputStream(new FileInputStream(configFile));
250                PropertyParser parser = new PropertyParser(new InputStreamReader(is), this.getProperty("hostname"));
251                Properties props = parser.parse();
252                is.close();
253                this.putAll(props);
254            } else {
255                System.out.println("Properties file not found at '" + configFile.getCanonicalPath() + "' ... using defaults");
256                
257            }
258        } catch (Exception e) {
259            this.initialisationFailure = e;
260            throw new RuntimeException("Could not load " + getConfigResourceLocation(), e);
261        }
262    }

273    /** Returns a named jdbcTemplate object that can be used to query the database
274     *  
275     * @param connectionName The name of the connection
276     * 
277     * @return a jdbcTemplate to the database
278     */
279    public synchronized JdbcTemplate getJdbcTemplate(String connectionName) {
280        String prefix = connectionName == null ? "database." : "databases." + connectionName + ".";
281        
282        JdbcTemplate jt = jdbcTemplates.get(connectionName);
283        if (jt==null) {
284                String connectionType = getProperty(prefix + "connectionType");
285                if (connectionType==null) { connectionType = "simple"; }
286                String driver = getProperty(prefix + "driver");
287                String url = getProperty(prefix + "url");
288                String username = getProperty(prefix + "username");
289                String password = getProperty(prefix + "password");
290                logger.info("Retrieving " + connectionType + " connection for '" + username + "'/'" + password + "' @ '" + url + "'");
291                DataSource ds = null;
292                if (!Text.isBlank(driver)) { 
293                        try {
294                                Class.forName(driver); 
295                        } catch (ClassNotFoundException e) {
296                                        logger.error("Error loading class '" + driver + "' for database '" + connectionName + "'", e);
297                                } 
298                }
299                if (connectionType.equals("simple")) {
300                        if (url==null) { throw new NullPointerException(prefix + "url has not been initialised"); }
301                        Connection connection;
302                        try {
303                            if (Text.isBlank(username)) {
304                                connection = DriverManager.getConnection (url);
305                            } else {
306                                connection = DriverManager.getConnection (url, username, password);
307                            }
308                        } catch (SQLException sqle) {
309                            throw new DataAccessResourceFailureException("Could not open connection to database", sqle); 
310                        }
311                        ds = new SingleConnectionDataSource(connection, false);
312                        
313                } else if (connectionType.equals("dbcp")) {
314                        if (url==null) { throw new NullPointerException(prefix + "url has not been initialised"); }
315                        BasicDataSource bds = new BasicDataSource();
316                        bds.setDriverClassName(driver);
317                        Map<? extends Object, ? extends Object> dbcpProps = PropertyParser.restrict(this, prefix + "dbcp", true);
318                        Struct.setFromMap(bds, dbcpProps, false, true, false);
319                        bds.setUrl(url);
320                        if (!Text.isBlank(username)) {
321                                bds.setUsername(username);
322                                bds.setPassword(password);
323                        }
324                        ds = bds;
325                        
326                } else if (connectionType.equals("c3p0")) {
327                        if (url==null) { throw new NullPointerException(prefix + "url has not been initialised"); }
328                        ComboPooledDataSource  cpds = new ComboPooledDataSource ();
329                        try {
330                                        cpds.setDriverClass(driver);
331                                } catch (PropertyVetoException e) {
332                                        throw new IllegalStateException("Could not set driverClass '" + driver + "' for c3p0 datasource", e);
333                                }
334                        @SuppressWarnings("unchecked")
335                                Map<String,String> c3poProps = (Map<String, String>) PropertyParser.restrict(this, prefix + "c3p0", true);
336                        // remove extensions from props
337                        for (Iterator<Map.Entry<String,String>> i = c3poProps.entrySet().iterator(); i.hasNext(); ) { 
338                                Map.Entry<String,String> e=i.next(); 
339                                if (e.getKey().startsWith("extensions.")) { i.remove(); }
340                        }
341                        Struct.setFromMap(cpds, c3poProps, false, true, false);
342                        
343                        // set extensions separately
344                        @SuppressWarnings("unchecked")
345                                Map<String,String> c3poExtensionProps = (Map<String, String>) PropertyParser.restrict(this, prefix + "c3p0.extensions", true);
346                        cpds.setExtensions(c3poExtensionProps);
347                        cpds.setJdbcUrl(url);
348                        if (!Text.isBlank(username)) {
349                                cpds.setUser(username);
350                                cpds.setPassword(password);
351                        }
352                        ds = cpds;
353                        
354                } else if (connectionType.equals("jndi")) {
355                        String jndiName = getProperty(prefix + "jndiName");
356                                try {
357                                        InitialContext ctx = new InitialContext();
358                                        Context envContext  = (Context) ctx.lookup("java:/comp/env");
359                                        ds = (DataSource) envContext.lookup(jndiName);
360                                        // could fallback to global datasource if comp/env is not here
361                                } catch (NamingException e) {
362                                        throw new IllegalStateException("Could not retrieve datasource '" + jndiName + "' from JNDI", e);
363                                }
364                } else {
365                        throw new IllegalStateException("Unknown " + prefix + "connectionType property '" + connectionType + "')");
366                }
367                
368                jt = new JdbcTemplate(ds);
369                dataSources.put(connectionName, ds);
370                jdbcTemplates.put(connectionName, jt);
371        }
372        return jt;
373    }

388        /** Determines whether a user 'may have' a permission on the application. 
389         * See the hasPermission method in SecurityContext for more details.
390         *
391         * @param user The user we are checking permissions for. 
392         * @param permission The permission we are checking.
393         * @return 'true' if the user may be authorised to perform the permission,
394         *   false otherwise
395         */
396        public boolean hasPermission(User user, String permission)
397        {
398                if ("false".equals(getProperty("auth.enableSecurityContext"))) { return true; }
399                SecurityContext securityContext = getSecurityContext();
400                return securityContext.hasPermission(user, permission);
401        }
402
403        /** Determines whether a user 'may have' a permission on the application. 
404         * See the hasPermission method in SecurityContext for more details. If the user
405         * does not have the permission, then a SecurityException is thrown
406         *
407         * @see SecurityContext#hasPermission(com.randomnoun.common.security.User, String)
408         *
409         * @param user The user we are checking permissions for. 
410         * @param permission The permission we are checking.
411         * 
412         * @throws SecurityException if the user does not have the supplied permission 
413         */
414        public void checkPermission(User user, String permission) throws SecurityException {
415                if (!hasPermission(user,  permission)) {
416                        throw new SecurityException("User does not have '" + permission + "' permission");
417                }
418    }

420        /** Determines whether a user has a permission on the application. 
421         * See the hasPermission method in SecurityContext for more details.
422         *
423         * @param user The user we are checking permissions for.
424         * @param permission The permission we are checking.
425         * @param resourceContext Fine-grained resource context
426         * @return 'true' if the user is authorised to perform the permission,
427         *   false otherwise
428         */
429        public boolean hasPermission(User user, String permission, Map<String, Object> resourceContext)
430        {
431                if ("false".equals(getProperty("auth.enableSecurityContext"))) { return true; }
432                SecurityContext securityContext = getSecurityContext();
433                return securityContext.hasPermission(user, permission, resourceContext);
434        }
