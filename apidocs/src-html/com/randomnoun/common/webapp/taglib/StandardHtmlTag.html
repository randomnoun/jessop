011/**
012 * This is an abstract taglib, which can be used to propagate standard HTML attributes
013 * through to a HTML page. It provides handlers for all the 'general' attributes
014 * that are present for most HTML tags (e.g. onclick, style, ...).
015 *
016 * <p>Other taglibs that are used to generate HTML elements (e.g. SelectTag), can
017 * extend this class to take advantage of this functionality.
018 *
019 * <p>Superclasses may override any method defined in this tag to provide their own
020 * special handling.
021 *
022 * <b>The attributes handled by this tag</b> are
023      accesskey, styleClass, contenteditable, dir, disabled, lang, language, maxlength,
024      size, style, title, width, onactivate, onafterupdate, onbeforeactvate,
025      onbeforecut, onbeforedeactivate, onbeforeeditfocus, onbeforepaste,
026      onbeforeupdate, onblur, onchange, onclick, oncontextmenu, oncontrolselect,
027      oncut, ondblclick, ondeactivate, ondrag, ondragend, ondragenter, ondragleave,
028      ondragover, ondragstart, ondrop, onerrorupdate, onfilterchange, onfocus,
029      onfocusin, onfocusout, onhelp, onkeydown, onkeypress, onkeyup, onlosecapture,
030      onmousedown, onmouseenter, onmouseleave, onmousemove, onmouseout, onmouseover,
031      onmouseup, onmousewheel, onmove, onmoveend, onmovestart, onpaste,
032      onpropertychange, onreadystatechange, onresize, onresizeend, onresizestart,
033      onselect, onselectstart, ontimeerror, hidefocus, readonly, tabindex and
034      unselectable.
035 *
036 * <p>Note that the HTML attribute 'class' is now called 'styleClass', since the
037 * Object.getClass() method interferes with standard JavaBean reflection.
038 *
039 * <p>All attributes are lower-case, except extraAttributes (see below).
040 *
041 * <p>Note that the attributes 'name', 'type' and 'value' are intentionally not on this list,
042 * as they would normally be generated by the specific tag libary itself.
043 * 
044 * <p>The 'disabled' attributes has some special handling -- if it's set to the value 
045 * "false", then the attribute is removed. This is due to IE treating 'disabled="false"' as
046 * actually disabling the control. Which is odd.
047 *
048 * <p>The IE-only attributes hidefocus, readonly, tabindex and unselectable are included
049 * on this list, because they seem reasonably useful. The IE-only attributes
050 * begin, end, datafld, datasrc, autocomplete, syncmasterm, timecontainer and vcard_name
051 * are not on this list, because they don't seem as useful. You can always add them if
052 * you think they're needed.
053 *
054 * <p>Values are EL-evaluated (so you can have JSTL-style <tt>${xxxx}</tt> values).
055 *
056 * <p>Additional attributes can be expressed using the extraAttributes attribute, e.g.
057 * to put a non-standard attribute 'onmousewheelclick' into the element, you could
058 * use, for example:
059 * <pre style="code">
060 * &lt;mm:something extraAttribute="onmousewheelclick='doSomethingInJavascript()'" />
061 * </pre>

353    /** Returns a string with contains all attributes assigned to this tag, in key="value"
354     * format.
355     * 
356     * <p>(NB: values are not currently quote-escaped here) 
357     *
358     * @return All attributes assigned to this tag.
359     */
360    protected String getAttributeString() {
361        StringBuffer sb = new StringBuffer();
362        for (Iterator<Map.Entry<String, String>> i = attributes.entrySet().iterator(); i.hasNext();) {
363            Map.Entry<String, String> entry = i.next();
364            sb.append(entry.getKey() + "=\"" + entry.getValue() + "\" ");
365        }
366        return sb.toString() + (extraAttributes == null ? "" : extraAttributes);
367    }
