038        public static String windowsPathToCygwin(String winPath) {
039                winPath = winPath.replace('\\', '/');
040                if (winPath.length() > 2 && winPath.charAt(1) == ':') {
041                        return "/cygdrive/" + Character.toLowerCase(winPath.charAt(0)) + "/" +
042                          winPath.substring(2);
043                } else if (winPath.length() > 2 && winPath.startsWith("//")) {
044                        return winPath;
045                } else {
046                        return winPath;
047                }
048        }

050        /** Encapsulates an error from executing a command through System.exec()
051         * 
052         */
053        public static class ProcessException extends Exception {
054                
055                private String command;
056                private String hostname;
057                private int exitCode;
058                private String stdout;
059                private String stderr;
060                private String exitCause;
061
062                /** Create a new executable exception
063                 * 
064                 * @param command Command being executed
065                 * @param hostname The host the command was being executed on
066                 * @param exitCode The exit code of the program
067                 * @param stdout The standard output of the program
068                 * @param stderr The error output of the program
069                 */
070                public ProcessException(String command, String hostname, String exitCause, int exitCode, String stdout, String stderr) {
071                        super("Error executing '" + command + "'" + (hostname==null ? "" : " on '" + hostname + "'") + ", cause='" + exitCause + "'");
072                        this.hostname = hostname;
073                        this.command = command;
074                        this.exitCode = exitCode;
075                        this.exitCause = exitCause;
076                        this.stdout = stdout;
077                        this.stderr = stderr;
078                }
079                public String getStdout() { return stdout; }
080                public String getStderr() { return stderr; }
081                public String getCommand() { return command; }
082                public String getHostname() { return hostname; }
083                public int getExitCode() { return exitCode; }
084                public String getExitCause() { return exitCause; }
085                public String getMessage() {
086                        // trim stdout/stderr
087                        if (maxOutputChars>0) { stdout = Text.getDisplayString("stdout", stdout, 8000); }
088                        if (maxOutputChars>0) { stderr = Text.getDisplayString("stderr", stderr, 8000); }
089                        return super.getMessage() +
090                          "; " + 
091                          (stdout==null ? "" : ", stdout='" + stdout + "'\n") +
092                      (stderr==null ? "" : ", stderr='" + stderr + "'\n") +
093                      "exitCode=" + exitCode;
094                }
095        }

098        public static String sshExec(String remoteUser, String hostname, String command, long timeout) throws ProcessException {
099                Process process;
100                try {
101                        process = Runtime.getRuntime().exec(new String[] {
102                                        "ssh", "-T", "-o", "StrictHostKeyChecking=no", remoteUser + "@" + hostname, command });
103                } catch (IOException ioe) {
104                        throw (ProcessException) new ProcessException(command, hostname, "IOException", 0, "", "").initCause(ioe);
105                }
106                InputStream stdout = process.getInputStream();
107                InputStream stderr = process.getErrorStream();
108                ByteArrayOutputStream stdoutByteArrayStream = new ByteArrayOutputStream();
109                ByteArrayOutputStream stderrByteArrayStream = new ByteArrayOutputStream();
110                Thread copyThread = StreamUtil.copyThread(stdout, stdoutByteArrayStream, 1024);
111                Thread copyErrorThread = StreamUtil.copyThread(stderr, stderrByteArrayStream, 1024);
112                copyThread.start();
113                copyErrorThread.start();
114                int exitCode = -1;
115                boolean throwException = false;
116                long interval = 100;
117                String cause = "";
118                
119                try {
120                        // exitCode = process.waitFor();
121                        // if (exitCode != 0) { throwException = true; }
122                        // copyThread.join(); // wait for copy thread to complete
123                        long    timeWaiting = 0;
124                        boolean processFinished = false;
125                        
126                        while (timeWaiting < timeout && !processFinished) {
127                                processFinished = true;
128                                Thread.sleep(interval);
129                                try {
130                                        exitCode = process.exitValue();
131                                } catch (IllegalThreadStateException e) {
132                                        processFinished = false;
133                                }
134                                timeWaiting += interval;
135                        }
136                        
137                        if (processFinished) {
138                                // wait for  copy threads to complete
139                                copyThread.join();
140                                copyErrorThread.join();  
141                                if (exitCode != 0) {
142                                        cause = "non-0 exitCode";
143                                        throwException = true;
144                                }
145                        } else {
146                                cause = "timeout";
147                                throwException = true;
148                                process.destroy();
149                                copyErrorThread.join();
150                                copyThread.join();
151                        }                       
152                        
153                } catch (InterruptedException ie) {
154                        cause = "InterruptedException";
155                        throwException = true;
156                        process.destroy();
157                        copyErrorThread.stop();
158                        copyThread.stop();
159                }
160                
161                if (throwException) {
162                        throw new ProcessException(
163                                command, hostname, cause, exitCode,  
164                                stdoutByteArrayStream.toString(), stderrByteArrayStream.toString());
165                }
166                return stdoutByteArrayStream.toString();
167        }

171        public static String sshExec(String remoteUser, String hostname, String command) throws ProcessException {
172                Process process;
173                try {
174                        process = Runtime.getRuntime().exec(new String[] {
175                                        "ssh", "-T", "-o", "StrictHostKeyChecking=no", remoteUser + "@" + hostname, command });
176                } catch (IOException ioe) {
177                        throw (ProcessException) new ProcessException(command, hostname, "IOException", 0, "", "").initCause(ioe);
178                }
179                InputStream stdout = process.getInputStream();
180                InputStream stderr = process.getErrorStream();
181                ByteArrayOutputStream baos = new ByteArrayOutputStream();
182                ByteArrayOutputStream baos2 = new ByteArrayOutputStream();
183                Thread copyThread = StreamUtil.copyThread(stdout, baos, 1024);
184                Thread copyErrorThread = StreamUtil.copyThread(stderr, baos2, 1024);
185                copyThread.start();
186                copyErrorThread.start();
187                int exitCode = 0;
188                boolean throwException = false;
189                String cause = "";
190                
191                try {
192                        exitCode = process.waitFor(); 
193                        if (exitCode != 0) { throwException = true; cause = "non-0 exitCode"; }
194                        copyThread.join(); 
195                        copyErrorThread.join();
196                } catch (InterruptedException ie) {
197                        throwException = true;
198                        cause = "InterruptedException";
199                        process.destroy();
200                        copyErrorThread.stop();
201                        copyThread.stop();
202                }
203                if (throwException) {
204                        throw new ProcessException(
205                                command, hostname, cause, exitCode,  
206                                baos.toString(), baos2.toString());
207                }
208                return baos.toString();
209        }

236        // see https://docs.oracle.com/javase/7/docs/api/java/lang/ProcessBuilder.html
237        private static String sshExec(String remoteUser, String hostname, String command, long timeout, InputStream stdin, String[] env) throws ProcessException {
238                Process process;
239                try {
240                        if (hostname == null) {
241                                if (env==null) {
242                                        process = Runtime.getRuntime().exec(command);
243                                } else {
244                                        process = Runtime.getRuntime().exec(command, env);
245                                }
246                        } else {
247                                if (env != null) {
248                                        throw new IllegalArgumentException("Cannot pass environment to remote process");
249                                }
250                                process = Runtime.getRuntime().exec(new String[] {
251                                                "ssh", "-T", "-o", "StrictHostKeyChecking=no", remoteUser + "@" + hostname, command }); 
252                        }
253                } catch (IOException ioe) {
254                        throw (ProcessException) new ProcessException(command, hostname, "IOException", 0, "", "").initCause(ioe);
255                }
256                InputStream stdout = process.getInputStream();
257                InputStream stderr = process.getErrorStream();
258                OutputStream processStdin = process.getOutputStream();
259                ByteArrayOutputStream stdoutByteArrayStream = new ByteArrayOutputStream();
260                ByteArrayOutputStream stderrByteArrayStream = new ByteArrayOutputStream();
261                Thread copyStdoutThread = StreamUtil.copyThread(stdout, stdoutByteArrayStream, 1024);
262                Thread copyStderrThread = StreamUtil.copyThread(stderr, stderrByteArrayStream, 1024);
263                Thread copyStdinThread = null;
264                if (stdin!=null) { 
265                        copyStdinThread = StreamUtil.copyAndCloseThread(stdin, processStdin, 1024);
266                        copyStdinThread.start();
267                }
268                
269                copyStdoutThread.start();
270                copyStderrThread.start();
271                int exitCode = -1;
272                boolean throwException = false;
273                long interval = 100;
274                String cause = "";
275                
276                try {
277                        // yes, I've tried putting a timeout in here
278                        // exitCode = process.waitFor();  // @TODO something with this
279                        // if (exitCode != 0) { throwException = true; }
280                        // copyThread.join(); // wait for copy thread to complete
281                        long    timeWaiting = 0;
282                        boolean processFinished = false;
283                        
284                        while ((timeout == -1 || timeWaiting < timeout) && !processFinished) {
285                                processFinished = true;
286                                Thread.sleep(interval);  // caught below
287                                
288                                try {
289                                        exitCode = process.exitValue();
290                                } catch (IllegalThreadStateException e) {
291                                        // process hasn't finished yet
292                                        processFinished = false;
293                                }
294                                timeWaiting += interval;
295                        }
296                        
297                        if (processFinished) {
298                                // wait for  copy threads to complete
299                                copyStdoutThread.join();
300                                copyStderrThread.join();  
301                                if (copyStdinThread!=null) {
302                                        copyStdinThread.stop();   // can't do much about stdin
303                                }
304                                if (exitCode != 0) {
305                                        cause = "non-0 exitCode";
306                                        throwException = true;
307                                }
308                        } else {
309                                cause = "timeout";
310                                throwException = true;
311                                process.destroy(); 
312                                if (copyStdinThread!=null) {
313                                        copyStdinThread.stop();
314                                }
315                                copyStderrThread.stop(); // .join() here hangs. should probably set a flag here instead
316                                copyStdoutThread.stop(); // maybe stop these ?
317                        }                       
318                        
319                } catch (InterruptedException ie) {
320                        // should probably clean things up here.
321                        cause = "InterruptedException";
322                        throwException = true;
323                        process.destroy();
324                        if (copyStdinThread!=null) {
325                                copyStdinThread.stop();
326                        }
327                        copyStderrThread.stop();
328                        copyStdoutThread.stop(); // maybe allow these to continue to completion ?                       
329                }
330                
331                if (throwException) {
332                        throw new ProcessException(
333                                command, hostname, cause, exitCode,  
334                                stdoutByteArrayStream.toString(), stderrByteArrayStream.toString());
335                }
336                return stdoutByteArrayStream.toString();
337        }
