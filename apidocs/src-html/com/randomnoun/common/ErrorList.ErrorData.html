012/**
013 * The ErrorList class is a a fairly generic class for containing validation errors for
014 * input forms, similar to the struts ActionErrors class. Each error within this class can
015 * contain a 'short' and 'long' description, where the short description is normally a 
016 * categorisation of the error and the longer description describes what has happened and
017 * how to fix the problem; (e.g. shortText="Missing field", 
018 * longText="The field 'id' is mandatory. Please enter a value for this field."). The short
019 * description is normally rendered by <tt>errorHeader.jsp</tt> in bold before the long
020 * description. An individual error also may contain a list of fields that it applies to
021 * (e.g. a 'mandatory inclusive' error may apply to several fields at once), and a
022 * severity (normally set to {@link #SEVERITY_INVALID}. Errors are displayed by the
023 * <tt>errorHeader.jsp</tt> JSP, which is typically included at the top of any page
024 * that contains an input form. An error without a severity supplied is assumed to be 
025 * of SEVERITY_INVALID, and an error without any fields supplied is assumed to be
026 * associated with the entire form, rather than a specific set of fields.
027 * 
028 * <p>Errors are inserted into an ErrorList by using one of the addError methods:
029 * <ul>
030 * <li> {@link #addError(String, String)} - add an invalid form error
031 * <li> {@link #addError(String, String, int)} - add an form error with a specific severity
032 * <li> {@link #addError(String, String, String)} - add an invalid field error  
033 * <li> {@link #addError(String, String, String, int)} - add an invalid field error with a specific severity 
034 * </ul>
035 * 
036 * <p>Field-level errors are detected by the <tt>&lt;mm:input&gt;</tt> and <tt>&lt;mm:select&gt;</tt>
037 * tag libraries to alert the user that the fields are in error.
038 * 
039 * <p>Code that uses an ErrorList to perform validation may 'attach' an object to the
040 * errorList instance to perform standard validations and to localise error messages. This
041 * information can be used by other helper classes to extract field names and data from
042 * complex objects. In common, methods to perform standard validation are contained 
043 * in the {@link com.randomnoun.common.struts.ActionBase} class (which can be
044 * used as a superclass of any struts action). 
045 * 
046 * <p>EJBs that wish to pass information back to the presentation layer should use
047 * ValidationExceptions (which take an ErrorList as an argument).
048 *
049 * @author knoxg
050 * @version <tt>$Id$</tt>
051 */
052public class ErrorList extends ArrayList<ErrorList.ErrorData>
053    implements java.io.Serializable
054{

182    /**
183     * Adds an error. This is the "real" addError() method; all others delegate to this
184     * one.
185     *
186     * @param errorField  a comma-separated list of field names that caused this error
187     * @param shortText a short string conveying the error type (e.g. 'Missing field')
188     * @param longText  a longer string describing the nature of the error and how to resolve it
189     *   (e.g. 'The field 'id' is mandatory. Please enter a value for this field.')
190     * @param severity    the severity of this error (one of the SEVERITY_* constants of this class).
191     *
192     * @see SEVERITY_INVALID
193     */
194    public void addError(String errorField, String shortText, String longText,
195        int severity)
196    {
197        // if we have an attachedFieldFormat, then format each field to this format
198        if (attachedFieldFormat != null && errorField != null) {
199            try {
200                List fields = Text.parseCsv(errorField);
201                errorField = "";
202                for (Iterator i = fields.iterator(); i.hasNext();) {
203                    String field = (String)i.next();
204                    errorField = errorField + getFieldName(field) +
205                        (i.hasNext() ? "," : "");
206                }
207            } catch (ParseException pe) {
208                throw (IllegalArgumentException)new IllegalArgumentException(
209                    "Invalid errorField list '" + errorField + "'").initCause(pe);
210            }
211        }
212        super.add(new ErrorData(shortText, longText, errorField, severity));
213    }

264    /**
265     * Appends the errors contained within another ErrorData into this one.
266     *
267     * @param errorList
268     *
269     * @return <b>true</b> if there were any additional errors to embed and at least
270     *   one of the errors had a severity of SEVERITY_ERROR or higher (worse),
271     *   <b>false</b> otherwise.
272     */
273    public boolean addErrors(ErrorList errorList)
274    {
275        if (errorList == null) {
276            return false;
277        }
278        if (errorList.size() == 0) {
279            return false;
280        }
281        int maxSeverity = Integer.MIN_VALUE;
282        int severity;
283
284        for (int i = 0; i < errorList.size(); i++) {
285            severity = errorList.getSeverityAt(i);
286            add(errorList.get(i));
287            if (severity > maxSeverity) {
288                maxSeverity = severity;
289            }
290        }
291
292        if (maxSeverity >= SEVERITY_ERROR) {
293            return true;
294        } else {
295            return false;
296        }
297    }

299    /**
300     * Returns true if an error has occured on the CGI field
301     * passed as a parameter to this method
302     *
303     * @return true if an error occured, false if not.
304     */
305    public boolean hasErrorOn(String field)
306    {
307        ErrorData errorInfo;
308        boolean hasError = false;
309
310        for (Iterator i = super.iterator(); i.hasNext(); ) {
311            errorInfo = (ErrorData)i.next();
312            if (errorInfo.getField()!=null) {
313                    String[] errorFields = errorInfo.getField().split(",");
314                    for (int j = 0; j < errorFields.length; j++) {
315                        if (errorFields[j].equals(field)) {
316                            return true;
317                        }
318                    }
319            }
320        }
321        return false;
322    }

354     * Returns the maximum severity of all current errors.
355     *
356     * @return the severity ranking of the most severe error, or -1 if
357     *   there are no errors contained within this ErrorData object.
358     */
359    public int maxErrorSeverity()
360    {
361        int maxSeverity = -1;
362        int curSeverity;
363
364        for (int i = 0; i < size(); i++) {
365            curSeverity = getSeverityAt(i);
366            if (curSeverity > maxSeverity) {
367                maxSeverity = curSeverity;
368            }
369        }
370        return maxSeverity;
371    }

424     */
425    public String toString() {
426        StringBuffer sb = new StringBuffer();
427        ErrorData errorData;
428        for (Iterator i = super.iterator(); i.hasNext(); ) {
429            errorData = (ErrorData)i.next();
430
431            sb.append("#");
432            sb.append(errorData.getShortText() + " - ");
433            sb.append(errorData.getLongText());
434            if (!Text.isBlank(errorData.getField())) {
435                sb.append(" [" + errorData.getField() + "]\n");
436            }
437        }
438        return sb.toString();
439    }

441    /** Attaches an object to be validated to this ErrorList.
442     *
443     * @param attachedObject a POJO, a HttpServletRequest, or a Map
444     *   (e.g. request.getParameterMap()). It identifies  where the validated data
445     *   is coming from.
446     * @param attachedBundle a bundle where the field names for this object are
447     *   to be retrieved from.
448     * @param bundleFormat a MessageFormat used to to retrieve field names from
449     *   attachedBundle. The "{0}" placeholder in this String is replaced with
450     *   the name of the field being validated.
451     */
452
453    // first parameter of this method can be 
454    // second method 
455    public void setValidatedObject(Object attachedObject, String attachedFieldFormat,
456        ResourceBundle attachedBundle, String bundleFormat, Locale locale)
457    {
458        if (attachedObject==null) { throw new NullPointerException("null attachedObject"); }
459        if (attachedFieldFormat==null) { throw new NullPointerException("null fieldFormat"); }
460        if (attachedBundle==null) { throw new NullPointerException("null attachedBundle"); } 
461        if (bundleFormat==null) { throw new NullPointerException("null bundleFormat"); }
462        if (locale==null) { throw new NullPointerException("null locale"); }
463
464        this.attachedObject = attachedObject;
465        this.attachedFieldFormat = attachedFieldFormat;
466        this.attachedBundle = attachedBundle;
467        this.bundleFormat = bundleFormat;
468        this.locale = locale;
469    }

499    /** Returns the value of a field from the attached object
500     * 
501     * @param name field name
502     * 
503     * @return field value
504     */
505    public String getFieldValue(String name) {
506        // use reflection
507        if (attachedObject instanceof HttpServletRequest) {
508            return ((HttpServletRequest)attachedObject).getParameter(name);
509        }
510
511        Object obj = Struct.getValue(attachedObject, name);
512        if (obj == null) {
513            return null;
514        } else if (obj instanceof String[]) {
515            return ((String[])obj)[0];
516        } else if (obj instanceof String) {
517            return (String)obj;
518        } else {
519            throw new IllegalStateException("Cannot retrieve non-string value '" + name +
520                "' (found class " + obj.getClass().getName() + ")");
521        }
522    }

568    /** Return the JSON representation of this object
569     *  
570     * @return the JSON representation of this object
571     */
572    public String toJSON() {
573        StringBuffer sb = new StringBuffer();
574        ErrorList.ErrorData errorData;
575        sb.append("[");
576        for (int i=0; i<this.size(); i++) {
577                errorData = (ErrorData) get(i);
578                if (i>0) { sb.append(","); }
579                sb.append(
580                  "{\"shortText\":\"" + Text.escapeJavascript2(errorData.getShortText()) + "\"," +
581                  "\"longText\":\"" + Text.escapeJavascript2(errorData.getLongText()) + "\"," +
582                  "\"fields\":\"" + Text.escapeJavascript2(errorData.getField()) + "\"," +
583                  "\"severity\":" + errorData.getSeverity() + "}");
584        }
585        sb.append("]");
586        return sb.toString();
