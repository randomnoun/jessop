017package org.apache.log4j.spi;
018
019import org.apache.log4j.Category;
020import org.apache.log4j.Level;
021import org.apache.log4j.bridge.LogEventAdapter;
022
023import java.util.Map;
024import java.util.Set;
025
026// a LoggingEvent that doesn't have a final getTimestamp()
027
028/**
029 *  No-op version of Log4j 1.2 LoggingEvent. This class is not directly used by Log4j 1.x clients but is used by
030 *  the Log4j 2 LogEvent adapter to be compatible with Log4j 1.x components.
031 */
032public class LoggingEvent {
033
034    /**
035     Set the location information for this logging event. The collected
036     information is cached for future use.
037     @return Always returns null.
038     */
039    public LocationInfo getLocationInformation() {
040        return null;
041    }
042
043    /**
044     * Return the level of this event. Use this form instead of directly
045     * accessing the <code>level</code> field.
046     * @return Always returns null.
047     */
048    public Level getLevel() {
049        return null;
050    }
051
052    /**
053     * Return the name of the logger. Use this form instead of directly
054     * accessing the <code>categoryName</code> field.
055     * @return Always returns null.
056     */
057    public String getLoggerName() {
058        return null;
059    }
060
061    public String getFQNOfLoggerClass() {
062        return null;
063    }
064
065    
066    // why the fuck is this final ?
067    public long getTimeStamp() {
068        return 0;
069    }
070
071    /**
072     * Gets the logger of the event.
073     * Use should be restricted to cloning events.
074     * @return Always returns null.
075     * @since 1.2.15
076     */
077    public Category getLogger() {
078        return null;
079    }
080
081    /**
082     Return the message for this logging event.
083
084     <p>Before serialization, the returned object is the message
085     passed by the user to generate the logging event. After
086     serialization, the returned value equals the String form of the
087     message possibly after object rendering.
088     @return Always returns null.
089     @since 1.1 */
090    public
091    Object getMessage() {
092        return null;
093    }
094
095    public
096    String getNDC() {
097        return null;
098    }
099
100    public
101    Object getMDC(String key) {
102        return null;
103    }
104
105    /**
106     Obtain a copy of this thread's MDC prior to serialization or
107     asynchronous logging.
108     */
109    public void getMDCCopy() {
110    }
111
112    public String getRenderedMessage() {
113        return null;
114    }
115
116    /**
117     Returns the time when the application started, in milliseconds
118     elapsed since 01.01.1970.
119     @return the JVM start time.
120     */
121    public static long getStartTime() {
122        return LogEventAdapter.getStartTime();
123    }
124
125    public String getThreadName() {
126        return null;
127    }
128
129    /**
130     Returns the throwable information contained within this
131     event. May be <code>null</code> if there is no such information.
132
133     <p>Note that the {@link Throwable} object contained within a
134     {@link ThrowableInformation} does not survive serialization.
135     @return Always returns null.
136     @since 1.1 */
137    public ThrowableInformation getThrowableInformation() {
138        return null;
139    }
140
141    /**
142     Return this event's throwable's string[] representaion.
143     @return Always returns null.
144     */
145    public String[] getThrowableStrRep() {
146        return null;
147    }
148
149    public void setProperty(final String propName,
150            final String propValue) {
151
152    }
153
154    public String getProperty(final String key) {
155        return null;
156    }
157
158    public Set getPropertyKeySet() {
159        return null;
160    }
161
162    public Map getProperties() {
163        return null;
164    }
165
166    public Object removeProperty(String propName) {
167        return null;
168    }
169}
